127 int mutt_protect (HEADER *msg, char *keylist)
128 {
129  BODY *pbody = NULL, *tmp_pbody = NULL;
130  BODY *tmp_smime_pbody = NULL;
131  BODY *tmp_pgp_pbody = NULL;
132  int flags = (WithCrypto & APPLICATION_PGP)? msg->security: 0;
133 
134  if (!WithCrypto)
135  return -1;
136 
137  if (!(msg->security & (ENCRYPT | SIGN)))
138  return 0;
139 
140  if ((msg->security & SIGN) && !crypt_valid_passphrase (msg->security))
141  return (-1);
142 
143  if ((WithCrypto & APPLICATION_PGP) && ((msg->security & PGPINLINE) == PGPINLINE))
144  {
145  if ((msg->content->type != TYPETEXT) ||
146  ascii_strcasecmp (msg->content->subtype, "plain"))
147  {
148  if (query_quadoption (OPT_PGPMIMEAUTO,
149  _("Inline PGP can't be used with attachments. Revert to PGP/MIME?")) != MUTT_YES)
150  {
151  mutt_error _("Mail not sent: inline PGP can't be used with attachments.");
152  return -1;
153  }
154  }
155  else
156  {
157  /* they really want to send it inline... go for it */
158  if (!isendwin ()) mutt_endwin _("Invoking PGP...");
159  pbody = crypt_pgp_traditional_encryptsign (msg->content, flags, keylist);
160  if (pbody)
161  {
162  msg->content = pbody;
163  return 0;
164  }
165 
166  /* otherwise inline won't work...ask for revert */
167  if (query_quadoption (OPT_PGPMIMEAUTO, _("Message can't be sent inline. Revert to using PGP/MIME?")) != MUTT_YES)
168  {
169  mutt_error _("Mail not sent.");
170  return -1;
171  }
172  }
173 
174  /* go ahead with PGP/MIME */
175  }
176 
177  if (!isendwin ()) mutt_endwin (NULL);
178 
179  if ((WithCrypto & APPLICATION_SMIME))
180  tmp_smime_pbody = msg->content;
181  if ((WithCrypto & APPLICATION_PGP))
182  tmp_pgp_pbody = msg->content;
183 
184  if (option (OPTCRYPTUSEPKA) && (msg->security & SIGN))
185  {
186  /* Set sender (necessary for e.g. PKA). */
187 
188  if ((WithCrypto & APPLICATION_SMIME)
189  && (msg->security & APPLICATION_SMIME))
190  crypt_smime_set_sender (msg->env->from->mailbox);
191  else if ((WithCrypto & APPLICATION_PGP)
192  && (msg->security & APPLICATION_PGP))
193  crypt_pgp_set_sender (msg->env->from->mailbox);
194  }
195 
196  if (msg->security & SIGN)
197  {
198  if ((WithCrypto & APPLICATION_SMIME)
199  && (msg->security & APPLICATION_SMIME))
200  {
201  if (!(tmp_pbody = crypt_smime_sign_message (msg->content)))
202  return -1;
203  pbody = tmp_smime_pbody = tmp_pbody;
204  }
205 
206  if ((WithCrypto & APPLICATION_PGP)
207  && (msg->security & APPLICATION_PGP)
208  && (!(flags & ENCRYPT) || option (OPTPGPRETAINABLESIG)))
209  {
210  if (!(tmp_pbody = crypt_pgp_sign_message (msg->content)))
211  return -1;
212 
213  flags &= ~SIGN;
214  pbody = tmp_pgp_pbody = tmp_pbody;
215  }
216 
217  if (WithCrypto
218  && (msg->security & APPLICATION_SMIME)
219  && (msg->security & APPLICATION_PGP))
220  {
221  /* here comes the draft ;-) */
222  }
223  }
224 
225 
226  if (msg->security & ENCRYPT)
227  {
228  if ((WithCrypto & APPLICATION_SMIME)
229  && (msg->security & APPLICATION_SMIME))
230  {
231  char *new_keylist = keylist;
232 
233  if (SmimeDefaultKey && query_quadoption(OPT_SMIMEENCRYPTSELF, _("Encrypt message to S/MIME Default Key also?")) == MUTT_YES)
234  {
235  /* +1 for NULL, +1 for \n */
236  int size = mutt_strlen(keylist) + mutt_strlen(SmimeDefaultKey) + 2;
237  new_keylist = safe_malloc(size);
238  snprintf(new_keylist, size, "%s%s\n", keylist, SmimeDefaultKey);
239  }
240 
241  tmp_pbody = crypt_smime_build_smime_entity (tmp_smime_pbody, new_keylist);
242  if (new_keylist != keylist)
243  FREE (&new_keylist);
244 
245  if (!tmp_pbody)
246  {
247  /* signed ? free it! */
248  return (-1);
249  }
250  /* free tmp_body if messages was signed AND encrypted ... */
251  if (tmp_smime_pbody != msg->content && tmp_smime_pbody != tmp_pbody)
252  {
253  /* detatch and don't delete msg->content,
254  which tmp_smime_pbody->parts after signing. */
255  tmp_smime_pbody->parts = tmp_smime_pbody->parts->next;
256  msg->content->next = NULL;
257  mutt_free_body (&tmp_smime_pbody);
258  }
259  pbody = tmp_pbody;
260  }
261 
262  if ((WithCrypto & APPLICATION_PGP)
263  && (msg->security & APPLICATION_PGP))
264  {
265  char *new_keylist = keylist;
266 
267  if (PgpSignAs && query_quadoption (OPT_PGPENCRYPTSELF, _("Encrypt message to PGP Default Key also?")) == MUTT_YES)
268  {
269  /* +1 for SPACE, +1 for NULL */
270  int size = mutt_strlen (keylist) + mutt_strlen (PgpSignAs) + 2;
271  new_keylist = safe_malloc (size);
272  snprintf (new_keylist, size, "%s %s", keylist, PgpSignAs);
273  }
274 
275  pbody = crypt_pgp_encrypt_message (tmp_pgp_pbody, new_keylist,
276  flags & SIGN);
277  if (new_keylist != keylist)
278  FREE(&new_keylist);
279  if (!pbody)
280  {
281  /* did we perform a retainable signature? */
282  if (flags != msg->security)
283  {
284  /* remove the outer multipart layer */
285  tmp_pgp_pbody = mutt_remove_multipart (tmp_pgp_pbody);
286  /* get rid of the signature */
287  mutt_free_body (&tmp_pgp_pbody->next);
288  }
289 
290  return (-1);
291  }
292 
293  /* destroy temporary signature envelope when doing retainable
294  * signatures.
295 
296  */
297  if (flags != msg->security)
298  {
299  tmp_pgp_pbody = mutt_remove_multipart (tmp_pgp_pbody);
300  mutt_free_body (&tmp_pgp_pbody->next);
301  }
302  }
303  }
304 
305  if(pbody)
306  msg->content = pbody;
307 
308  return 0;
309 }

314 int mutt_is_multipart_signed (BODY *b)
315 {
316  char *p;
317 
318  if (!b || !(b->type == TYPEMULTIPART) ||
319  !b->subtype || ascii_strcasecmp(b->subtype, "signed"))
320  return 0;
321 
322  if (!(p = mutt_get_parameter("protocol", b->parameter)))
323  return 0;
324 
325  if (!(ascii_strcasecmp (p, "multipart/mixed")))
326  return SIGN;
327 
328  if ((WithCrypto & APPLICATION_PGP)
329  && !(ascii_strcasecmp (p, "application/pgp-signature")))
330  return PGPSIGN;
331 
332  if ((WithCrypto & APPLICATION_SMIME)
333  && !(ascii_strcasecmp (p, "application/x-pkcs7-signature")))
334  return SMIMESIGN;
335  if ((WithCrypto & APPLICATION_SMIME)
336  && !(ascii_strcasecmp (p, "application/pkcs7-signature")))
337  return SMIMESIGN;
338 
339  return 0;
340 }

423 int mutt_is_application_pgp (BODY *m)
424 {
425  int t = 0;
426  char *p;
427 
428  if (m->type == TYPEAPPLICATION)
429  {
430  if (!ascii_strcasecmp (m->subtype, "pgp") || !ascii_strcasecmp (m->subtype, "x-pgp-message"))
431  {
432  if ((p = mutt_get_parameter ("x-action", m->parameter))
433  && (!ascii_strcasecmp (p, "sign") || !ascii_strcasecmp (p, "signclear")))
434  t |= PGPSIGN;
435 
436  if ((p = mutt_get_parameter ("format", m->parameter)) &&
437  !ascii_strcasecmp (p, "keys-only"))
438  t |= PGPKEY;
439 
440  if(!t) t |= PGPENCRYPT; /* not necessarily correct, but... */
441  }
442 
443  if (!ascii_strcasecmp (m->subtype, "pgp-signed"))
444  t |= PGPSIGN;
445 
446  if (!ascii_strcasecmp (m->subtype, "pgp-keys"))
447  t |= PGPKEY;
448  }
449  else if (m->type == TYPETEXT && ascii_strcasecmp ("plain", m->subtype) == 0)
450  {
451  if (((p = mutt_get_parameter ("x-mutt-action", m->parameter))
452  || (p = mutt_get_parameter ("x-action", m->parameter))
453  || (p = mutt_get_parameter ("action", m->parameter)))
454  && !ascii_strncasecmp ("pgp-sign", p, 8))
455  t |= PGPSIGN;
456  else if (p && !ascii_strncasecmp ("pgp-encrypt", p, 11))
457  t |= PGPENCRYPT;
458  else if (p && !ascii_strncasecmp ("pgp-keys", p, 7))
459  t |= PGPKEY;
460  }
461  if (t)
462  t |= PGPINLINE;
463 
464  return t;
465 }

467 int mutt_is_application_smime (BODY *m)
468 {
469  char *t=NULL;
470  int len, complain=0;
471 
472  if(!m)
473  return 0;
474 
475  if ((m->type & TYPEAPPLICATION) && m->subtype)
476  {
477  /* S/MIME MIME types don't need x- anymore, see RFC2311 */
478  if (!ascii_strcasecmp (m->subtype, "x-pkcs7-mime") ||
479  !ascii_strcasecmp (m->subtype, "pkcs7-mime"))
480  {
481  if ((t = mutt_get_parameter ("smime-type", m->parameter)))
482  {
483  if (!ascii_strcasecmp (t, "enveloped-data"))
484  return SMIMEENCRYPT;
485  else if (!ascii_strcasecmp (t, "signed-data"))
486  return (SMIMESIGN|SMIMEOPAQUE);
487  else return 0;
488  }
489  /* Netscape 4.7 uses
490  * Content-Description: S/MIME Encrypted Message
491  * instead of Content-Type parameter
492  */
493  if (!ascii_strcasecmp (m->description, "S/MIME Encrypted Message"))
494  return SMIMEENCRYPT;
495  complain = 1;
496  }
497  else if (ascii_strcasecmp (m->subtype, "octet-stream"))
498  return 0;
499 
500  t = mutt_get_parameter ("name", m->parameter);
501 
502  if (!t) t = m->d_filename;
503  if (!t) t = m->filename;
504  if (!t)
505  {
506  if (complain)
507  mutt_message (_("S/MIME messages with no hints on content are unsupported."));
508  return 0;
509  }
510 
511  /* no .p7c, .p10 support yet. */
512 
513  len = mutt_strlen (t) - 4;
514  if (len > 0 && *(t+len) == '.')
515  {
516  len++;
517  if (!ascii_strcasecmp ((t+len), "p7m"))
518 #if 0
519  return SMIMEENCRYPT;
520 #else
521  /* Not sure if this is the correct thing to do, but
522  it's required for compatibility with Outlook */
523  return (SMIMESIGN|SMIMEOPAQUE);
524 #endif
525  else if (!ascii_strcasecmp ((t+len), "p7s"))
526  return (SMIMESIGN|SMIMEOPAQUE);
527  }
528  }
529 
530  return 0;
531 }

538 int crypt_query (BODY *m)
539 {
540  int t = 0;
541 
542  if (!WithCrypto)
543  return 0;
544 
545  if (!m)
546  return 0;
547 
548  if (m->type == TYPEAPPLICATION)
549  {
550  if ((WithCrypto & APPLICATION_PGP))
551  t |= mutt_is_application_pgp(m);
552 
553  if ((WithCrypto & APPLICATION_SMIME))
554  {
555  t |= mutt_is_application_smime(m);
556  if (t && m->goodsig) t |= GOODSIGN;
557  if (t && m->badsig) t |= BADSIGN;
558  }
559  }
560  else if ((WithCrypto & APPLICATION_PGP) && m->type == TYPETEXT)
561  {
562  t |= mutt_is_application_pgp (m);
563  if (t && m->goodsig)
564  t |= GOODSIGN;
565  }
566 
567  if (m->type == TYPEMULTIPART)
568  {
569  t |= mutt_is_multipart_encrypted(m);
570  t |= mutt_is_multipart_signed (m);
571  t |= mutt_is_malformed_multipart_pgp_encrypted (m);
572 
573  if (t && m->goodsig)
574  t |= GOODSIGN;
575  }
576 
577  if (m->type == TYPEMULTIPART || m->type == TYPEMESSAGE)
578  {
579  BODY *p;
580  int u, v, w;
581 
582  u = m->parts ? 0xffffffff : 0; /* Bits set in all parts */
583  w = 0; /* Bits set in any part */
584 
585  for (p = m->parts; p; p = p->next)
586  {
587  v = crypt_query (p);
588  u &= v; w |= v;
589  }
590  t |= u | (w & ~GOODSIGN);
591 
592  if ((w & GOODSIGN) && !(u & GOODSIGN))
593  t |= PARTSIGN;
594  }
595 
596  return t;
597 }

602 int crypt_write_signed(BODY *a, STATE *s, const char *tempfile)
603 {
604  FILE *fp;
605  int c;
606  short hadcr;
607  size_t bytes;
608 
609  if (!WithCrypto)
610  return -1;
611 
612  if (!(fp = safe_fopen (tempfile, "w")))
613  {
614  mutt_perror (tempfile);
615  return -1;
616  }
617 
618  fseeko (s->fpin, a->hdr_offset, 0);
619  bytes = a->length + a->offset - a->hdr_offset;
620  hadcr = 0;
621  while (bytes > 0)
622  {
623  if ((c = fgetc (s->fpin)) == EOF)
624  break;
625 
626  bytes--;
627 
628  if (c == '\r')
629  hadcr = 1;
630  else
631  {
632  if (c == '\n' && !hadcr)
633  fputc ('\r', fp);
634 
635  hadcr = 0;
636  }
637 
638  fputc (c, fp);
639 
640  }
641  safe_fclose (&fp);
642 
643  return 0;
644 }

649 {
650  if (!WithCrypto)
651  return;
652 
653  while (a)
654  {
655  if (a->type == TYPEMULTIPART)
656  {
657  if (a->encoding != ENC7BIT)

660  convert_to_7bit(a->parts);
661  }
662  else if ((WithCrypto & APPLICATION_PGP) && option (OPTPGPSTRICTENC))
663  convert_to_7bit (a->parts);
664  }
665  else if (a->type == TYPEMESSAGE &&
666  ascii_strcasecmp(a->subtype, "delivery-status"))
667  {
668  if(a->encoding != ENC7BIT)
669  mutt_message_to_7bit (a, NULL);
670  }
671  else if (a->encoding == ENC8BIT)
672  a->encoding = ENCQUOTEDPRINTABLE;
673  else if (a->encoding == ENCBINARY)
674  a->encoding = ENCBASE64;
675  else if (a->content && a->encoding != ENCBASE64 &&
676  (a->content->from || (a->content->space &&
677  option (OPTPGPSTRICTENC))))
678  a->encoding = ENCQUOTEDPRINTABLE;
679  a = a->next;
680  }

686 void crypt_extract_keys_from_messages (HEADER * h)
687 {
688  int i;
689  char tempfname[_POSIX_PATH_MAX], *mbox;
690  ADDRESS *tmp = NULL;
691  FILE *fpout;
692 
693  if (!WithCrypto)
694  return;
695 
696  mutt_mktemp (tempfname, sizeof (tempfname));
697  if (!(fpout = safe_fopen (tempfname, "w")))
698  {
699  mutt_perror (tempfname);
700  return;
701  }
702 
703  if ((WithCrypto & APPLICATION_PGP))
704  set_option (OPTDONTHANDLEPGPKEYS);
705 
706  if (!h)
707  {
708  for (i = 0; i < Context->vcount; i++)
709  {
710  if (Context->hdrs[Context->v2r[i]]->tagged)
711  {
712  mutt_parse_mime_message (Context, Context->hdrs[Context->v2r[i]]);
713  if (Context->hdrs[Context->v2r[i]]->security & ENCRYPT &&
714  !crypt_valid_passphrase (Context->hdrs[Context->v2r[i]]->security))
715  {
716  safe_fclose (&fpout);
717  break;
718  }
719 
720  if ((WithCrypto & APPLICATION_PGP)
721  && (Context->hdrs[Context->v2r[i]]->security & APPLICATION_PGP))
722  {
723  mutt_copy_message (fpout, Context, Context->hdrs[Context->v2r[i]],
724  MUTT_CM_DECODE|MUTT_CM_CHARCONV, 0);
725  fflush(fpout);
726

917 int mutt_signed_handler (BODY *a, STATE *s)
918 {
919  char tempfile[_POSIX_PATH_MAX];
920  int signed_type;
921  int inconsistent = 0;
922 
923  BODY *b = a;
924  BODY **signatures = NULL;
925  int sigcnt = 0;
926  int i;
927  short goodsig = 1;
928  int rc = 0;
929 
930  if (!WithCrypto)
931  return -1;
932 
933  a = a->parts;
934  signed_type = mutt_is_multipart_signed (b);
935  if (!signed_type)
936  {
937  /* A null protocol value is already checked for in mutt_body_handler() */
938  state_printf (s, _("[-- Error: "
939  "Unknown multipart/signed protocol %s! --]\n\n"),
940  mutt_get_parameter ("protocol", b->parameter));
941  return mutt_body_handler (a, s);
942  }
943 
944  if (!(a && a->next))
945  inconsistent = 1;
946  else
947  {
948  switch (signed_type)
949  {
950  case SIGN:
951  if (a->next->type != TYPEMULTIPART ||
952  ascii_strcasecmp (a->next->subtype, "mixed"))
953  inconsistent = 1;
954  break;
955  case PGPSIGN:
956  if (a->next->type != TYPEAPPLICATION ||
957  ascii_strcasecmp (a->next->subtype, "pgp-signature"))
958  inconsistent = 1;
959  break;
960  case SMIMESIGN:
961  if (a->next->type != TYPEAPPLICATION ||
962  (ascii_strcasecmp (a->next->subtype, "x-pkcs7-signature") &&
963  ascii_strcasecmp (a->next->subtype, "pkcs7-signature")))
964  inconsistent = 1;
965  break;
966  default:
967  inconsistent = 1;
968  }
969  }
970  if (inconsistent)
971  {
972  state_attach_puts (_("[-- Error: "
973  "Inconsistent multipart/signed structure! --]\n\n"),
974  s);
975  return mutt_body_handler (a, s);
976  }
977 
978  if (s->flags & MUTT_DISPLAY)
979  {
980 
981  crypt_fetch_signatures (&signatures, a->next, &sigcnt);
982 
983  if (sigcnt)
984  {
985  mutt_mktemp (tempfile, sizeof (tempfile));
986  if (crypt_write_signed (a, s, tempfile) == 0)
987  {
988  for (i = 0; i < sigcnt; i++)
989  {
990  if ((WithCrypto & APPLICATION_PGP)
991  && signatures[i]->type == TYPEAPPLICATION
992  && !ascii_strcasecmp (signatures[i]->subtype, "pgp-signature"))
993  {
994  if (crypt_pgp_verify_one (signatures[i], s, tempfile) != 0)
995  goodsig = 0;
996 
997  continue;
998  }
999 
1000  if ((WithCrypto & APPLICATION_SMIME)
1001  && signatures[i]->type == TYPEAPPLICATION
1002  && (!ascii_strcasecmp(signatures[i]->subtype, "x-pkcs7-signature")
1003  || !ascii_strcasecmp(signatures[i]->subtype, "pkcs7-signature")))
1004  {
1005  if (crypt_smime_verify_one (signatures[i], s, tempfile) != 0)
1006  goodsig = 0;
1007 
1008  continue;
1009  }
1010 
1011  state_printf (s, _("[-- Warning: "
1012  "We can't verify %s/%s signatures. --]\n\n"),
1013  TYPE(signatures[i]), signatures[i]->subtype);
1014  }
1015  }
1016 
1017  mutt_unlink (tempfile);
1018 
1019  b->goodsig = goodsig;
1020  b->badsig = !goodsig;
1021 
1022  /* Now display the signed body */
1023  state_attach_puts (_("[-- The following data is signed --]\n\n"), s);
1024 
1025 
1026  FREE (&signatures);
1027  }
1028  else
1029  state_attach_puts (_("[-- Warning: Can't find any signatures. --]\n\n"), s);
1030  }
1031 
1032  rc = mutt_body_handler (a, s);
1033 
1034  if (s->flags & MUTT_DISPLAY && sigcnt)
1035  state_attach_puts (_("\n[-- End of signed data --]\n"), s);
1036 
1037  return rc;
1038 }

1041 /* Obtain pointers to fingerprint or short or long key ID, if any.
1042  * See mutt_crypt.h for details.
1043  */
1044 const char* crypt_get_fingerprint_or_id (char *p, const char **pphint,
1045  const char **ppl, const char **pps)
1046 {
1047  const char *ps, *pl, *phint;
1048  char *pfcopy, *pf, *s1, *s2;
1049  char c;
1050  int isid;
1051  size_t hexdigits;
1052 
1053  /* User input may be partial name, fingerprint or short or long key ID,
1054  * independent of OPTPGPLONGIDS.
1055  * Fingerprint without spaces is 40 hex digits (SHA-1) or 32 hex digits (MD5).
1056  * Strip leading "0x" for key ID detection and prepare pl and ps to indicate
1057  * if an ID was found and to simplify logic in the key loop's inner
1058  * condition of the caller. */
1059 
1060  pf = mutt_skip_whitespace (p);
1061  if (!mutt_strncasecmp (pf, "0x", 2))
1062  pf += 2;
1063 
1064  /* Check if a fingerprint is given, must be hex digits only, blanks
1065  * separating groups of 4 hex digits are allowed. Also pre-check for ID. */
1066  isid = 2; /* unknown */
1067  hexdigits = 0;
1068  s1 = pf;
1069  do
1070  {
1071  c = *(s1++);
1072  if (('0' <= c && c <= '9') || ('A' <= c && c <= 'F') || ('a' <= c && c <= 'f'))
1073  {
1074  ++hexdigits;
1075  if (isid == 2)
1076  isid = 1; /* it is an ID so far */
1077  }
1078  else if (c)
1079  {
1080  isid = 0; /* not an ID */
1081  if (c == ' ' && ((hexdigits % 4) == 0))
1082  ; /* skip blank before or after 4 hex digits */
1083  else
1084  break; /* any other character or position */
1085  }
1086  } while (c);
1087 
1088  /* If at end of input, check for correct fingerprint length and copy if. */
1089  pfcopy = (!c && ((hexdigits == 40) || (hexdigits == 32)) ? safe_strdup (pf) : NULL);
1090 
1091  if (pfcopy)
1092  {
1093  /* Use pfcopy to strip all spaces from fingerprint and as hint. */
1094  s1 = s2 = pfcopy;
1095  do
1096  {
1097  *(s1++) = *(s2 = mutt_skip_whitespace (s2));
1098  } while (*(s2++));
1099 
1100  phint = pfcopy;
1101  ps = pl = NULL;
1102  }
1103  else
1104  {
1105  phint = p;
1106  ps = pl = NULL;
1107  if (isid == 1)
1108  {
1109  if (mutt_strlen (pf) == 16)
1110  pl = pf; /* long key ID */
1111  else if (mutt_strlen (pf) == 8)
1112  ps = pf; /* short key ID */
1113  }
1114  }
1115 
1116  *pphint = phint;
1117  *ppl = pl;
1118  *pps = ps;
1119  return pfcopy;
1120 }
