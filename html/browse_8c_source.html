39 /* imap_browse: IMAP hook into the folder browser, fills out browser_state,
40  * given a current folder to browse */
41 int imap_browse (char* path, struct browser_state* state)
42 {
43  IMAP_DATA* idata;
44  IMAP_LIST list;
45  char buf[LONG_STRING];
46  char mbox[LONG_STRING];
47  char munged_mbox[LONG_STRING];
48  char list_cmd[5];
49  int n;
50  char ctmp;
51  short showparents = 0;
52  int save_lsub;
53  IMAP_MBOX mx;
54 
55  if (imap_parse_path (path, &mx))
56  {
57  mutt_error (_("%s is an invalid IMAP path"), path);
58  return -1;
59  }
60 
61  save_lsub = option (OPTIMAPCHECKSUBSCRIBED);
62  unset_option (OPTIMAPCHECKSUBSCRIBED);
63  strfcpy (list_cmd, option (OPTIMAPLSUB) ? "LSUB" : "LIST", sizeof (list_cmd));
64 
65  if (!(idata = imap_conn_find (&(mx.account), 0)))
66  goto fail;
67 
68  mutt_message _("Getting folder list...");
69 
70  /* skip check for parents when at the root */
71  if (mx.mbox && mx.mbox[0] != '\0')
72  {
73  imap_fix_path (idata, mx.mbox, mbox, sizeof (mbox));
74  n = mutt_strlen (mbox);
75  }
76  else
77  {
78  mbox[0] = '\0';
79  n = 0;
80  }
81 
82  if (n)
83  {
84  int rc;
85  dprint (3, (debugfile, "imap_browse: mbox: %s\n", mbox));
86 
87  /* if our target exists and has inferiors, enter it if we
88  * aren't already going to */
89  imap_munge_mbox_name (idata, munged_mbox, sizeof (munged_mbox), mbox);
90  snprintf (buf, sizeof (buf), "%s \"\" %s", list_cmd, munged_mbox);
91  imap_cmd_start (idata, buf);
92  idata->cmdtype = IMAP_CT_LIST;
93  idata->cmddata = &list;
94  do
95  {
96  list.name = 0;
97  rc = imap_cmd_step (idata);
98  if (rc == IMAP_CMD_CONTINUE && list.name)
99  {
100  if (!list.noinferiors && list.name[0] &&
101  !imap_mxcmp (list.name, mbox) &&
102  n < sizeof (mbox) - 1)
103  {
104  mbox[n++] = list.delim;
105  mbox[n] = '\0';
106  }
107  }
108  }
109  while (rc == IMAP_CMD_CONTINUE);
110  idata->cmddata = NULL;
111 
112  /* if we're descending a folder, mark it as current in browser_state */
113  if (mbox[n-1] == list.delim)
114  {
115  showparents = 1;
116  imap_qualify_path (buf, sizeof (buf), &mx, mbox);
117  state->folder = safe_strdup (buf);
118  n--;
119  }
120 
121  /* Find superiors to list
122  * Note: UW-IMAP servers return folder + delimiter when asked to list
123  * folder + delimiter. Cyrus servers don't. So we ask for folder,
124  * and tack on delimiter ourselves.
125  * Further note: UW-IMAP servers return nothing when asked for
126  * NAMESPACES without delimiters at the end. Argh! */
127  for (n--; n >= 0 && mbox[n] != list.delim ; n--);
128  if (n > 0) /* "aaaa/bbbb/" -> "aaaa" */
129  {
130  /* forget the check, it is too delicate (see above). Have we ever
131  * had the parent not exist? */
132  ctmp = mbox[n];
133  mbox[n] = '\0';
134 
135  if (showparents)
136  {
137  dprint (3, (debugfile, "imap_init_browse: adding parent %s\n", mbox));
138  imap_add_folder (list.delim, mbox, 1, 0, state, 1);
139  }
140 
141  /* if our target isn't a folder, we are in our superior */
142  if (!state->folder)
143  {
144  /* store folder with delimiter */
145  mbox[n++] = ctmp;
146  ctmp = mbox[n];
147  mbox[n] = '\0';
148  imap_qualify_path (buf, sizeof (buf), &mx, mbox);
149  state->folder = safe_strdup (buf);
150  }
151  mbox[n] = ctmp;
152  }
153  /* "/bbbb/" -> add "/", "aaaa/" -> add "" */
154  else
155  {
156  char relpath[2];
157  /* folder may be "/" */
158  snprintf (relpath, sizeof (relpath), "%c" , n < 0 ? '\0' : idata->delim);
159  if (showparents)
160  imap_add_folder (idata->delim, relpath, 1, 0, state, 1);
161  if (!state->folder)
162  {
163  imap_qualify_path (buf, sizeof (buf), &mx, relpath);
164  state->folder = safe_strdup (buf);
165  }
166  }
167  }
168 
169  /* no namespace, no folder: set folder to host only */
170  if (!state->folder)
171  {
172  imap_qualify_path (buf, sizeof (buf), &mx, NULL);
173  state->folder = safe_strdup (buf);
174  }
175 
176  dprint (3, (debugfile, "imap_browse: Quoting mailbox scan: %s -> ", mbox));
177  snprintf (buf, sizeof (buf), "%s%%", mbox);
178  imap_munge_mbox_name (idata, munged_mbox, sizeof (munged_mbox), buf);
179  dprint (3, (debugfile, "%s\n", munged_mbox));
180  snprintf (buf, sizeof (buf), "%s \"\" %s", list_cmd, munged_mbox);
181  if (browse_add_list_result (idata, buf, state, 0))
182  goto fail;
183 
184  if (!state->entrylen)
185  {
186  mutt_error _("No such folder");
187  goto fail;
188  }
189 
190  mutt_clear_error ();
191 
192  if (save_lsub)
193  set_option (OPTIMAPCHECKSUBSCRIBED);
194 
195  FREE (&mx.mbox);
196  return 0;
197 
198  fail:
199  if (save_lsub)
200  set_option (OPTIMAPCHECKSUBSCRIBED);
201  FREE (&mx.mbox);
202  return -1;
203 }

206 int imap_mailbox_create (const char* folder)
207 {
208  IMAP_DATA* idata;
209  IMAP_MBOX mx;
210  char buf[LONG_STRING];
211  short n;
212 
213  if (imap_parse_path (folder, &mx) < 0)
214  {
215  dprint (1, (debugfile, "imap_mailbox_create: Bad starting path %s\n",
216  folder));
217  return -1;
218  }
219 
220  if (!(idata = imap_conn_find (&mx.account, MUTT_IMAP_CONN_NONEW)))
221  {
222  dprint (1, (debugfile, "imap_mailbox_create: Couldn't find open connection to %s", mx.account.host));
223  goto fail;
224  }
225 
226  strfcpy (buf, NONULL (mx.mbox), sizeof (buf));
227 
228  /* append a delimiter if necessary */
229  n = mutt_strlen (buf);
230  if (n && (n < sizeof (buf) - 1) && (buf[n-1] != idata->delim))
231  {
232  buf[n++] = idata->delim;
233  buf[n] = '\0';
234  }
235 
236  if (mutt_get_field (_("Create mailbox: "), buf, sizeof (buf), MUTT_FILE) < 0)
237  goto fail;
238 
239  if (!mutt_strlen (buf))
240  {
241  mutt_error (_("Mailbox must have a name."));
242  mutt_sleep(1);
243  goto fail;
244  }
245 
246  if (imap_create_mailbox (idata, buf) < 0)
247  goto fail;
248 
249  mutt_message _("Mailbox created.");
250  mutt_sleep (0);
251 
252  FREE (&mx.mbox);
253  return 0;
254 
255  fail:
256  FREE (&mx.mbox);

318 static int browse_add_list_result (IMAP_DATA* idata, const char* cmd,
319  struct browser_state* state, short isparent)
320 {
321  IMAP_LIST list;
322  IMAP_MBOX mx;
323  int rc;
324 
325  if (imap_parse_path (state->folder, &mx))
326  {
327  dprint (2, (debugfile,
328  "browse_add_list_result: current folder %s makes no sense\n", state->folder));
329  return -1;
330  }
331 
332  imap_cmd_start (idata, cmd);
333  idata->cmdtype = IMAP_CT_LIST;
334  idata->cmddata = &list;
335  do
336  {
337  list.name = NULL;
338  rc = imap_cmd_step (idata);
339 
340  if (rc == IMAP_CMD_CONTINUE && list.name)
341  {
342  /* Let a parent folder never be selectable for navigation */
343  if (isparent)
344  list.noselect = 1;
345  /* prune current folder from output */
346  if (isparent || mutt_strncmp (list.name, mx.mbox, strlen (list.name)))
347  imap_add_folder (list.delim, list.name, list.noselect, list.noinferiors,
348  state, isparent);
349  }
350  }
351  while (rc == IMAP_CMD_CONTINUE);
352  idata->cmddata = NULL;
353 
354  FREE (&mx.mbox);
355  return rc == IMAP_CMD_OK ? 0 : -1;
356 }

358 /* imap_add_folder:
359  * add a folder name to the browser list, formatting it as necessary.
360  *
361  * The folder parameter should already be 'unmunged' via
362  * imap_unmunge_mbox_name().
363  */
364 static void imap_add_folder (char delim, char *folder, int noselect,
365  int noinferiors, struct browser_state *state, short isparent)
366 {
367  char tmp[LONG_STRING];
368  char relpath[LONG_STRING];
369  IMAP_MBOX mx;
370  BUFFY *b;
371 
372  if (imap_parse_path (state->folder, &mx))
373  return;
374 
375  if (state->entrylen + 1 == state->entrymax)
376  {
377  safe_realloc (&state->entry,
378  sizeof (struct folder_file) * (state->entrymax += 256));
379  memset (state->entry + state->entrylen, 0,
380  (sizeof (struct folder_file) * (state->entrymax - state->entrylen)));
381  }
382 
383  /* render superiors as unix-standard ".." */
384  if (isparent)
385  strfcpy (relpath, "../", sizeof (relpath));
386  /* strip current folder from target, to render a relative path */
387  else if (!mutt_strncmp (mx.mbox, folder, mutt_strlen (mx.mbox)))
388  strfcpy (relpath, folder + mutt_strlen (mx.mbox), sizeof (relpath));
389  else
390  strfcpy (relpath, folder, sizeof (relpath));
391 
392  /* apply filemask filter. This should really be done at menu setup rather
393  * than at scan, since it's so expensive to scan. But that's big changes
394  * to browser.c */
395  if (!((regexec (Mask.rx, relpath, 0, NULL, 0) == 0) ^ Mask.not))
396  {
397  FREE (&mx.mbox);
398  return;
399  }
400 
401  imap_qualify_path (tmp, sizeof (tmp), &mx, folder);
402  (state->entry)[state->entrylen].name = safe_strdup (tmp);
403 
404  /* mark desc with delim in browser if it can have subfolders */
405  if (!isparent && !noinferiors && strlen (relpath) < sizeof (relpath) - 1)
406  {
407  relpath[strlen (relpath) + 1] = '\0';
408  relpath[strlen (relpath)] = delim;
409  }
410 
411  (state->entry)[state->entrylen].desc = safe_strdup (relpath);
412 
413  (state->entry)[state->entrylen].imap = 1;
414  /* delimiter at the root is useless. */
415  if (folder[0] == '\0')
416  delim = '\0';
417  (state->entry)[state->entrylen].delim = delim;
418  (state->entry)[state->entrylen].selectable = !noselect;
419  (state->entry)[state->entrylen].inferiors = !noinferiors;
420 
421  b = Incoming;
422  while (b && mutt_strcmp (tmp, b->path))
423  b = b->next;
424  if (b)
425  {
426  if (Context &&
427  !mutt_strcmp (b->realpath, Context->realpath))
428  {
429  b->msg_count = Context->msgcount;
430  b->msg_unread = Context->unread;
431  }
432  (state->entry)[state->entrylen].has_buffy = 1;
433  (state->entry)[state->entrylen].new = b->new;
434  (state->entry)[state->entrylen].msg_count = b->msg_count;
435  (state->entry)[state->entrylen].msg_unread = b->msg_unread;
436  }
437 
438  (state->entrylen)++;
439 
440  FREE (&mx.mbox);
441 }
