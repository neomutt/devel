101 {
102  struct folder_file *pa = (struct folder_file *) a;
103  struct folder_file *pb = (struct folder_file *) b;
104 
105  /* inbox should be sorted ahead of its siblings */
106  int same_path = mutt_same_path (pa->name, pb->name);

110 
111  return ((BrowserSort & SORT_REVERSE) ? -r : r);
112 }

176 /* Wild compare function that calls the others. It's useful
177  * because it provides a way to tell "../" is always on the
178  * top of the list, independently of the sort method.
179  */
180 static int browser_compare (const void *a, const void *b)
181 {
182  struct folder_file *pa = (struct folder_file *) a;
183  struct folder_file *pb = (struct folder_file *) b;
184 
185  if ((mutt_strcoll (pa->desc, "../") == 0) ||
186  (mutt_strcoll (pa->desc, "..") == 0))
187  return -1;
188  if ((mutt_strcoll (pb->desc, "../") == 0) ||
189  (mutt_strcoll (pb->desc, "..") == 0))
190  return 1;
191 
192  switch (BrowserSort & SORT_MASK)
193  {
194  case SORT_DATE:
195  return browser_compare_date(a, b);
196  case SORT_SIZE:
197  return browser_compare_size(a, b);
198  case SORT_DESC:
199  return browser_compare_desc(a, b);
200  case SORT_COUNT:
201  return browser_compare_count(a, b);
202  case SORT_COUNT_NEW:
203  return browser_compare_count_new(a, b);
204  case SORT_SUBJECT:
205  default:
206  return browser_compare_subject(a, b);
207  }
208 }

246 static const char *
247 folder_format_str (char *dest, size_t destlen, size_t col, int cols, char op, const char *src,
248  const char *fmt, const char *ifstring, const char *elsestring,
249  unsigned long data, format_flag flags)
250 {
251  char fn[SHORT_STRING], tmp[SHORT_STRING], permission[11];
252  char date[SHORT_STRING], *t_fmt;
253  time_t tnow;
254  FOLDER *folder = (FOLDER *) data;
255  struct passwd *pw;
256  struct group *gr;
257  int optional = (flags & MUTT_FORMAT_OPTIONAL);
258 
259  switch (op)
260  {
261  case 'C':
262  snprintf (tmp, sizeof (tmp), "%%%sd", fmt);
263  snprintf (dest, destlen, tmp, folder->num + 1);
264  break;
265 
266  case 'd':
267  case 'D':
268  if (folder->ff->local)
269  {
270  int do_locales = TRUE;
271 
272  if (op == 'D') {
273  t_fmt = NONULL(DateFmt);
274  if (*t_fmt == '!') {
275  ++t_fmt;
276  do_locales = FALSE;
277  }
278  } else {
279  tnow = time (NULL);
280  t_fmt = tnow - folder->ff->mtime < 31536000 ? "%b %d %H:%M" : "%b %d %Y";
281  }
282 
283  if (!do_locales)
284  setlocale (LC_TIME, "C");
285  strftime (date, sizeof (date), t_fmt, localtime (&folder->ff->mtime));
286  if (!do_locales)
287  setlocale (LC_TIME, "");
288 
289  mutt_format_s (dest, destlen, fmt, date);
290  }
291  else
292  mutt_format_s (dest, destlen, fmt, "");
293  break;
294 
295  case 'f':
296  {
297  char *s;
298 
299 #ifdef USE_NOTMUCH
300  if (mx_is_notmuch(folder->ff->name))
301  s = NONULL (folder->ff->desc);
302  else
303 #endif
304 #ifdef USE_IMAP
305  if (folder->ff->imap)
306  s = NONULL (folder->ff->desc);
307  else
308 #endif
309  s = NONULL (folder->ff->name);
310 
311  snprintf (fn, sizeof (fn), "%s%s", s,
312  folder->ff->local ? (S_ISLNK (folder->ff->mode) ? "@" :
313  (S_ISDIR (folder->ff->mode) ? "/" :
314  ((folder->ff->mode & S_IXUSR) != 0 ? "*" : ""))) : "");
315 
316  mutt_format_s (dest, destlen, fmt, fn);
317  break;
318  }
319  case 'F':
320  if (folder->ff->local)
321  {
322  snprintf (permission, sizeof (permission), "%c%c%c%c%c%c%c%c%c%c",
323  S_ISDIR(folder->ff->mode) ? 'd' : (S_ISLNK(folder->ff->mode) ? 'l' : '-'),
324  (folder->ff->mode & S_IRUSR) != 0 ? 'r': '-',
325  (folder->ff->mode & S_IWUSR) != 0 ? 'w' : '-',
326  (folder->ff->mode & S_ISUID) != 0 ? 's' : (folder->ff->mode & S_IXUSR) != 0 ? 'x': '-',
327  (folder->ff->mode & S_IRGRP) != 0 ? 'r' : '-',
328  (folder->ff->mode & S_IWGRP) != 0 ? 'w' : '-',
329  (folder->ff->mode & S_ISGID) != 0 ? 's' : (folder->ff->mode & S_IXGRP) != 0 ? 'x': '-',
330  (folder->ff->mode & S_IROTH) != 0 ? 'r' : '-',
331  (folder->ff->mode & S_IWOTH) != 0 ? 'w' : '-',
332  (folder->ff->mode & S_ISVTX) != 0 ? 't' : (folder->ff->mode & S_IXOTH) != 0 ? 'x': '-');
333  mutt_format_s (dest, destlen, fmt, permission);
334  }
335 #ifdef USE_IMAP
336  else if (folder->ff->imap)
337  {
338  /* mark folders with subfolders AND mail */
339  snprintf (permission, sizeof (permission), "IMAP %c",
340  (folder->ff->inferiors && folder->ff->selectable) ? '+' : ' ');
341  mutt_format_s (dest, destlen, fmt, permission);
342  }
343 #endif
344  else
345  mutt_format_s (dest, destlen, fmt, "");
346  break;
347 
348  case 'g':
349  if (folder->ff->local)
350  {
351  if ((gr = getgrgid (folder->ff->gid)))
352  mutt_format_s (dest, destlen, fmt, gr->gr_name);
353  else
354  {
355  snprintf (tmp, sizeof (tmp), "%%%sld", fmt);
356  snprintf (dest, destlen, tmp, folder->ff->gid);
357  }
358  }
359  else
360  mutt_format_s (dest, destlen, fmt, "");
361  break;
362 
363  case 'l':
364  if (folder->ff->local)
365  {
366  snprintf (tmp, sizeof (tmp), "%%%sd", fmt);
367  snprintf (dest, destlen, tmp, folder->ff->nlink);
368  }
369  else
370  mutt_format_s (dest, destlen, fmt, "");
371  break;
372 
373  case 'm':
374  if (!optional)
375  {
376  if (folder->ff->has_buffy)
377  {
378  snprintf (tmp, sizeof (tmp), "%%%sd", fmt);
379  snprintf (dest, destlen, tmp, folder->ff->msg_count);
380  }
381  else
382  mutt_format_s (dest, destlen, fmt, "");
383  }
384  else if (!folder->ff->msg_count)
385  optional = 0;
386  break;
387 
388  case 'N':
389  snprintf (tmp, sizeof (tmp), "%%%sc", fmt);
390  snprintf (dest, destlen, tmp, folder->ff->new ? 'N' : ' ');
391  break;
392 
393  case 'n':
394  if (!optional)
395  {
396  if (folder->ff->has_buffy)
397  {
398  snprintf (tmp, sizeof (tmp), "%%%sd", fmt);
399  snprintf (dest, destlen, tmp, folder->ff->msg_unread);
400  }
401  else
402  mutt_format_s (dest, destlen, fmt, "");
403  }
404  else if (!folder->ff->msg_unread)
405  optional = 0;
406  break;
407 
408  case 's':
409  if (folder->ff->local)
410  {
411  mutt_pretty_size(fn, sizeof(fn), folder->ff->size);
412  snprintf (tmp, sizeof (tmp), "%%%ss", fmt);
413  snprintf (dest, destlen, tmp, fn);
414  }
415  else
416  mutt_format_s (dest, destlen, fmt, "");
417  break;
418 
419  case 't':
420  snprintf (tmp, sizeof (tmp), "%%%sc", fmt);
421  snprintf (dest, destlen, tmp, folder->ff->tagged ? '*' : ' ');
422  break;
423 
424  case 'u':
425  if (folder->ff->local)
426  {
427  if ((pw = getpwuid (folder->ff->uid)))
428  mutt_format_s (dest, destlen, fmt, pw->pw_name);
429  else
430  {
431  snprintf (tmp, sizeof (tmp), "%%%sld", fmt);
432  snprintf (dest, destlen, tmp, folder->ff->uid);
433  }
434  }
435  else
436  mutt_format_s (dest, destlen, fmt, "");
437  break;
438 
439  default:
440  snprintf (tmp, sizeof (tmp), "%%%sc", fmt);
441  snprintf (dest, destlen, tmp, op);
442  break;
443  }
444 
445  if (optional)
446  mutt_FormatString (dest, destlen, col, cols, ifstring, folder_format_str, data, 0);
447  else if (flags & MUTT_FORMAT_OPTIONAL)
448  mutt_FormatString (dest, destlen, col, cols, elsestring, folder_format_str, data, 0);
449 
450  return (src);
451 }

617 /* get list of all files/newsgroups with mask */
618 static int examine_directory (MUTTMENU *menu, struct browser_state *state,
619  char *d, const char *prefix)
620 {
621 #ifdef USE_NNTP
622  if (option (OPTNEWS))
623  {
624  NNTP_SERVER *nserv = CurrentNewsSrv;
625  unsigned int i;
626 
627 /* mutt_buffy_check (0); */
628  init_state (state, menu);
629 
630  for (i = 0; i < nserv->groups_num; i++)
631  {
632  NNTP_DATA *nntp_data = nserv->groups_list[i];
633  if (!nntp_data)
634  continue;
635  if (prefix && *prefix &&
636  strncmp (prefix, nntp_data->group, strlen (prefix)))
637  continue;
638  if (!((regexec (Mask.rx, nntp_data->group, 0, NULL, 0) == 0) ^ Mask.not))
639  continue;
640  add_folder (menu, state, nntp_data->group, NULL, NULL, NULL, nntp_data);
641  }
642  }
643  else
644 #endif /* USE_NNTP */
645  {
646  struct stat s;
647  DIR *dp;
648  struct dirent *de;
649  char buffer[_POSIX_PATH_MAX + SHORT_STRING];
650  BUFFY *tmp;
651 
652  while (stat (d, &s) == -1)
653  {
654  if (errno == ENOENT)
655  {
656  /* The last used directory is deleted, try to use the parent dir. */
657  char *c = strrchr (d, '/');
658 
659  if (c && (c > d))
660  {
661  *c = 0;
662  continue;
663  }
664  }
665  mutt_perror (d);
666  return (-1);
667  }
668 
669  if (!S_ISDIR (s.st_mode))
