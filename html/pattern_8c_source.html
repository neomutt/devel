132 /* if no uppercase letters are given, do a case-insensitive search */
133 int mutt_which_case (const char *s)
134 {
135  wchar_t w;
136  mbstate_t mb;
137  size_t l;
138 
139  memset (&mb, 0, sizeof (mb));
140 
141  for (; (l = mbrtowc (&w, s, MB_CUR_MAX, &mb)) != 0; s += l)
142  {
143  if (l == (size_t) -2)
144  continue; /* shift sequences */
145  if (l == (size_t) -1)
146  return 0; /* error; assume case-sensitive */
147  if (iswalpha ((wint_t) w) && iswupper ((wint_t) w))
148  return 0; /* case-sensitive */
149  }
150 
151  return REG_ICASE; /* case-insensitive */
152 }

154 static int
155 msg_search (CONTEXT *ctx, pattern_t* pat, int msgno)
156 {
157  MESSAGE *msg = NULL;
158  STATE s;
159  FILE *fp = NULL;
160  long lng = 0;
161  int match = 0;
162  HEADER *h = ctx->hdrs[msgno];
163  char *buf;
164  size_t blen;
165 #ifdef USE_FMEMOPEN
166  char *temp;
167  size_t tempsize;
168 #else
169  char tempfile[_POSIX_PATH_MAX];
170  struct stat st;
171 #endif
172 
173  if ((msg = mx_open_message (ctx, msgno)) != NULL)
174  {
175  if (option (OPTTHOROUGHSRC))
176  {
177  /* decode the header / body */
178  memset (&s, 0, sizeof (s));
179  s.fpin = msg->fp;
180  s.flags = MUTT_CHARCONV;
181 #ifdef USE_FMEMOPEN
182  s.fpout = open_memstream (&temp, &tempsize);
183  if (!s.fpout) {
184  mutt_perror ("Error opening memstream");
185  return 0;
186  }
187 #else
188  mutt_mktemp (tempfile, sizeof (tempfile));
189  if ((s.fpout = safe_fopen (tempfile, "w+")) == NULL)
190  {
191  mutt_perror (tempfile);
192  return (0);
193  }
194 #endif
195 
196  if (pat->op != MUTT_BODY)
197  mutt_copy_header (msg->fp, h, s.fpout, CH_FROM | CH_DECODE, NULL);
198 
199  if (pat->op != MUTT_HEADER)
200  {
201  mutt_parse_mime_message (ctx, h);
202 
203  if (WithCrypto && (h->security & ENCRYPT)
204  && !crypt_valid_passphrase(h->security))
205  {
206  mx_close_message (ctx, &msg);
207  if (s.fpout)
208  {
209  safe_fclose (&s.fpout);
210 #ifdef USE_FMEMOPEN
211  FREE(&temp);
212 #else
213  unlink (tempfile);
214 #endif
215  }
216  return (0);
217  }
218 
219  fseeko (msg->fp, h->offset, 0);
220  mutt_body_handler (h->content, &s);
221  }
222 
223 #ifdef USE_FMEMOPEN
224  fclose (s.fpout);
225  lng = tempsize;
226 
227  if (tempsize) {
228  fp = fmemopen (temp, tempsize, "r");
229  if (!fp) {
230  mutt_perror ("Error re-opening memstream");
231  return 0;
232  }
233  } else { /* fmemopen cannot handle empty buffers */
234  fp = safe_fopen ("/dev/null", "r");
235  if (!fp) {
236  mutt_perror ("Error opening /dev/null");
237  return 0;
238  }
239  }
240 #else
241  fp = s.fpout;
242  fflush (fp);
243  fseek (fp, 0, 0);
244  fstat (fileno (fp), &st);
245  lng = (long) st.st_size;
246 #endif
247  }
248  else
249  {
250  /* raw header / body */
251  fp = msg->fp;
252  if (pat->op != MUTT_BODY)
253  {
254  fseeko (fp, h->offset, 0);
255  lng = h->content->offset - h->offset;
256  }
257  if (pat->op != MUTT_HEADER)
258  {
259  if (pat->op == MUTT_BODY)
260  fseeko (fp, h->content->offset, 0);
261  lng += h->content->length;
262  }
263  }
264 
265  blen = STRING;
266  buf = safe_malloc (blen);
267 
268  /* search the file "fp" */
269  while (lng > 0)
270  {
271  if (pat->op == MUTT_HEADER)
272  {
273  if (*(buf = mutt_read_rfc822_line (fp, buf, &blen)) == '\0')
274  break;
275  }
276  else if (fgets (buf, blen - 1, fp) == NULL)
277  break; /* don't loop forever */
278  if (patmatch (pat, buf) == 0)
279  {
280  match = 1;
281  break;
282  }
283  lng -= mutt_strlen (buf);
284  }
285 
286  FREE (&buf);
287 
288  mx_close_message (ctx, &msg);
289 
290  if (option (OPTTHOROUGHSRC))
291  {
292  safe_fclose (&fp);
293 #ifdef USE_FMEMOPEN
294  if (tempsize)
295  FREE(&temp);
296 #else
297  unlink (tempfile);
298 #endif
299  }
300  }
301 
302  return match;
303 }

305 static int eat_regexp (pattern_t *pat, BUFFER *s, BUFFER *err)
306 {
307  BUFFER buf;
308  char errmsg[STRING];
309  int r;
310  char *pexpr;
311 
312  mutt_buffer_init (&buf);
313  pexpr = s->dptr;
314  if (mutt_extract_token (&buf, s, MUTT_TOKEN_PATTERN | MUTT_TOKEN_COMMENT) != 0 ||
315  !buf.data)
316  {
317  snprintf (err->data, err->dsize, _("Error in expression: %s"), pexpr);
318  return (-1);
319  }
320  if (!*buf.data)
321  {
322  snprintf (err->data, err->dsize, _("Empty expression"));
323  return (-1);
324  }
325 
326 #if 0
327  /* If there are no RE metacharacters, use simple search anyway */
328  if (!pat->stringmatch && !strpbrk (buf.data, "|[{.*+?^$"))
329  pat->stringmatch = 1;
330 #endif
331 
332  if (pat->stringmatch)
333  {
334  pat->p.str = safe_strdup (buf.data);
335  pat->ign_case = mutt_which_case (buf.data) == REG_ICASE;
336  FREE (&buf.data);
337  }
338  else if (pat->groupmatch)
339  {
340  pat->p.g = mutt_pattern_group (buf.data);
341  FREE (&buf.data);
342  }
343  else
344  {
345  pat->p.rx = safe_malloc (sizeof (regex_t));
346  r = REGCOMP (pat->p.rx, buf.data, REG_NEWLINE | REG_NOSUB | mutt_which_case (buf.data));
347  if (r)
348  {
349  regerror (r, pat->p.rx, errmsg, sizeof (errmsg));
350  mutt_buffer_printf (err, "'%s': %s", buf.data, errmsg);
351  FREE (&buf.data);
352  FREE (&pat->p.rx);
353  return (-1);
354  }
355  FREE (&buf.data);
356  }
357 
358  return 0;
359 }

361 int eat_range (pattern_t *pat, BUFFER *s, BUFFER *err)
362 {
363  char *tmp;
364  int do_exclusive = 0;
365  int skip_quote = 0;
366 
367  /*
368  * If simple_search is set to "~m %s", the range will have double quotes
369  * around it...
370  */
371  if (*s->dptr == '"')
372  {
373  s->dptr++;
374  skip_quote = 1;
375  }
376  if (*s->dptr == '<')
377  do_exclusive = 1;
378  if ((*s->dptr != '-') && (*s->dptr != '<'))
379  {
380  /* range minimum */
381  if (*s->dptr == '>')
382  {
383  pat->max = MUTT_MAXRANGE;
384  pat->min = strtol (s->dptr + 1, &tmp, 0) + 1; /* exclusive range */
385  }
386  else
387  pat->min = strtol (s->dptr, &tmp, 0);
388  if (toupper ((unsigned char) *tmp) == 'K') /* is there a prefix? */
389  {
390  pat->min *= 1024;
391  tmp++;
392  }
393  else if (toupper ((unsigned char) *tmp) == 'M')
394  {
395  pat->min *= 1048576;
396  tmp++;
397  }
398  if (*s->dptr == '>')
399  {
400  s->dptr = tmp;
401  return 0;
402  }
403  if (*tmp != '-')
404  {
405  /* exact value */
406  pat->max = pat->min;
407  s->dptr = tmp;
408  return 0;
409  }
410  tmp++;
411  }
412  else
413  {
414  s->dptr++;
415  tmp = s->dptr;
416  }
417 
418  if (isdigit ((unsigned char) *tmp))
419  {
420  /* range maximum */
421  pat->max = strtol (tmp, &tmp, 0);
422  if (toupper ((unsigned char) *tmp) == 'K')
423  {
424  pat->max *= 1024;
425  tmp++;
426  }
427  else if (toupper ((unsigned char) *tmp) == 'M')
428  {
429  pat->max *= 1048576;
430  tmp++;
431  }
432  if (do_exclusive)
433  (pat->max)--;
434  }
435  else
436  pat->max = MUTT_MAXRANGE;
437 
438  if (skip_quote && *tmp == '"')
439  tmp++;
440 
441  SKIPWS (tmp);
442  s->dptr = tmp;
443  return 0;
444 }

446 static const char *getDate (const char *s, struct tm *t, BUFFER *err)
447 {
448  char *p;
449  time_t now = time (NULL);
450  struct tm *tm = localtime (&now);
451 
452  t->tm_mday = strtol (s, &p, 10);
453  if (t->tm_mday < 1 || t->tm_mday > 31)
454  {
455  snprintf (err->data, err->dsize, _("Invalid day of month: %s"), s);
456  return NULL;
457  }
458  if (*p != '/')
459  {
460  /* fill in today's month and year */
461  t->tm_mon = tm->tm_mon;
462  t->tm_year = tm->tm_year;
463  return p;
464  }
465  p++;
466  t->tm_mon = strtol (p, &p, 10) - 1;
467  if (t->tm_mon < 0 || t->tm_mon > 11)
468  {
469  snprintf (err->data, err->dsize, _("Invalid month: %s"), p);
470  return NULL;
471  }
472  if (*p != '/')
473  {
474  t->tm_year = tm->tm_year;
475  return p;
476  }
477  p++;
478  t->tm_year = strtol (p, &p, 10);
479  if (t->tm_year < 70) /* year 2000+ */
480  t->tm_year += 100;
481  else if (t->tm_year > 1900)
482  t->tm_year -= 1900;
483  return p;
484 }

490 static const char *get_offset (struct tm *tm, const char *s, int sign)
491 {
492  char *ps;
493  int offset = strtol (s, &ps, 0);
494  if ((sign < 0 && offset > 0) || (sign > 0 && offset < 0))
495  offset = -offset;
496 
497  switch (*ps)
498  {
499  case 'y':
500  tm->tm_year += offset;
501  break;
502  case 'm':
503  tm->tm_mon += offset;
504  break;
505  case 'w':
506  tm->tm_mday += 7 * offset;
507  break;
508  case 'd':
509  tm->tm_mday += offset;
510  break;
511  default:
512  return s;
513  }
514  mutt_normalize_time (tm);
515  return (ps + 1);
516 }

545 static const char * parse_date_range (const char* pc, struct tm *min,
546  struct tm *max, int haveMin, struct tm *baseMin, BUFFER *err)
547 {
548  int flag = MUTT_PDR_NONE;
549  while (*pc && ((flag & MUTT_PDR_DONE) == 0))
550  {
551  const char *pt;
552  char ch = *pc++;
553  SKIPWS (pc);
554  switch (ch)
555  {
556  case '-':
557  {
558  /* try a range of absolute date minus offset of Ndwmy */
559  pt = get_offset (min, pc, -1);
560  if (pc == pt)
561  {
562  if (flag == MUTT_PDR_NONE)
563  { /* nothing yet and no offset parsed => absolute date? */
564  if (!getDate (pc, max, err))
565  flag |= (MUTT_PDR_ABSOLUTE | MUTT_PDR_ERRORDONE); /* done bad */
566  else
567  {
568  /* reestablish initial base minimum if not specified */
569  if (!haveMin)
570  memcpy (min, baseMin, sizeof(struct tm));
571  flag |= (MUTT_PDR_ABSOLUTE | MUTT_PDR_DONE); /* done good */
572  }
573  }
574  else
575  flag |= MUTT_PDR_ERRORDONE;
576  }
577  else
578  {
579  pc = pt;
580  if (flag == MUTT_PDR_NONE && !haveMin)
581  { /* the very first "-3d" without a previous absolute date */
582  max->tm_year = min->tm_year;
583  max->tm_mon = min->tm_mon;
584  max->tm_mday = min->tm_mday;
585  }
586  flag |= MUTT_PDR_MINUS;
587  }
588  }
589  break;
590  case '+':
591  { /* enlarge plusRange */
592  pt = get_offset (max, pc, 1);
593  if (pc == pt)
594  flag |= MUTT_PDR_ERRORDONE;
595  else
596  {
597  pc = pt;
598  flag |= MUTT_PDR_PLUS;
599  }
600  }
601  break;
602  case '*':
603  { /* enlarge window in both directions */
604  pt = get_offset (min, pc, -1);
605  if (pc == pt)
606  flag |= MUTT_PDR_ERRORDONE;
607  else
608  {
609  pc = get_offset (max, pc, 1);
610  flag |= MUTT_PDR_WINDOW;
611  }
612  }
613  break;
614  default:
615  flag |= MUTT_PDR_ERRORDONE;
616  }
617  SKIPWS (pc);
618  }
619  if ((flag & MUTT_PDR_ERROR) && !(flag & MUTT_PDR_ABSOLUTE))
620  { /* getDate has its own error message, don't overwrite it here */
621  snprintf (err->data, err->dsize, _("Invalid relative date: %s"), pc-1);
622  }
623  return ((flag & MUTT_PDR_ERROR) ? NULL : pc);
624 }

626 static int eat_date (pattern_t *pat, BUFFER *s, BUFFER *err)
627 {
628  BUFFER buffer;
629  struct tm min, max;
630  char *pexpr;
631 
632  mutt_buffer_init (&buffer);
633  pexpr = s->dptr;
634  if (mutt_extract_token (&buffer, s, MUTT_TOKEN_COMMENT | MUTT_TOKEN_PATTERN) != 0
635  || !buffer.data)
636  {
637  snprintf (err->data, err->dsize, _("Error in expression: %s"), pexpr);
638  return (-1);
639  }
640  if (!*buffer.data)
641  {
642  snprintf (err->data, err->dsize, _("Empty expression"));
643  return (-1);
644  }
645 
646  memset (&min, 0, sizeof (min));
647  /* the `0' time is Jan 1, 1970 UTC, so in order to prevent a negative time
648  when doing timezone conversion, we use Jan 2, 1970 UTC as the base
649  here */
650  min.tm_mday = 2;
651  min.tm_year = 70;
652 
653  memset (&max, 0, sizeof (max));
654 
655  /* Arbitrary year in the future. Don't set this too high
656  or mutt_mktime() returns something larger than will
657  fit in a time_t on some systems */
658  max.tm_year = 130;
659  max.tm_mon = 11;
660  max.tm_mday = 31;
661  max.tm_hour = 23;
662  max.tm_min = 59;
663  max.tm_sec = 59;
664 
665  if (strchr ("<>=", buffer.data[0]))
666  {
667  /* offset from current time
668  <3d less than three days ago
669  >3d more than three days ago
670  =3d exactly three days ago */
671  time_t now = time (NULL);
672  struct tm *tm = localtime (&now);
673  int exact = 0;
674 
675  if (buffer.data[0] == '<')
676  {
677  memcpy (&min, tm, sizeof (min));
678  tm = &min;
679  }
680  else
681  {
682  memcpy (&max, tm, sizeof (max));
683  tm = &max;
684 
685  if (buffer.data[0] == '=')
686  exact++;
687  }
688  tm->tm_hour = 23;
689  tm->tm_min = tm->tm_sec = 59;
690 
691  /* force negative offset */
692  get_offset (tm, buffer.data + 1, -1);
693 
694  if (exact)
695  {
696  /* start at the beginning of the day in question */
697  memcpy (&min, &max, sizeof (max));
698  min.tm_hour = min.tm_sec = min.tm_min = 0;
699  }
700  }
701  else

828 pattern_t *mutt_pattern_comp (/* const */ char *s, int flags, BUFFER *err)
829 {
830  pattern_t *curlist = NULL;
831  pattern_t *tmp, *tmp2;
832  pattern_t *last = NULL;
833  int not = 0;
834  int alladdr = 0;
835  int or = 0;
836  int implicit = 1; /* used to detect logical AND operator */
837  int isalias = 0;
838  const struct pattern_flags *entry;
839  char *p;
840  char *buf;
841  BUFFER ps;
842 
843  mutt_buffer_init (&ps);
844  ps.dptr = s;
845  ps.dsize = mutt_strlen (s);
846 
847  while (*ps.dptr)
848  {
849  SKIPWS (ps.dptr);
850  switch (*ps.dptr)
851  {
852  case '^':
853  ps.dptr++;
854  alladdr = !alladdr;
855  break;
856  case '!':
857  ps.dptr++;
858  not = !not;
859  break;
860  case '@':
861  ps.dptr++;
862  isalias = !isalias;
863  break;
864  case '|':
865  if (!or)
866  {
867  if (!curlist)
868  {
869  snprintf (err->data, err->dsize, _("error in pattern at: %s"), ps.dptr);
870  return NULL;
871  }
872  if (curlist->next)
873  {
874  /* A & B | C == (A & B) | C */
875  tmp = new_pattern ();
876  tmp->op = MUTT_AND;
877  tmp->child = curlist;
878 
879  curlist = tmp;
880  last = curlist;
881  }
882 
883  or = 1;
884  }
885  ps.dptr++;
886  implicit = 0;
887  not = 0;
888  alladdr = 0;
889  isalias = 0;
890  break;
891  case '%':
892  case '=':
893  case '~':
894  if (!*(ps.dptr + 1))
895  {
896  snprintf (err->data, err->dsize, _("missing pattern: %s"), ps.dptr);
897  mutt_pattern_free (&curlist);
898  return NULL;
899  }
900  if (*(ps.dptr + 1) == '(')
901  {
902  ps.dptr ++; /* skip ~ */
903  p = find_matching_paren (ps.dptr + 1);
904  if (*p != ')')
905  {
906  snprintf (err->data, err->dsize, _("mismatched brackets: %s"), ps.dptr);
907  mutt_pattern_free (&curlist);
908  return NULL;
909  }
910  tmp = new_pattern ();
911  tmp->op = MUTT_THREAD;
912  if (last)
913  last->next = tmp;
914  else
915  curlist = tmp;
916  last = tmp;
917  tmp->not ^= not;
918  tmp->alladdr |= alladdr;
919  tmp->isalias |= isalias;
920  not = 0;
921  alladdr = 0;
922  isalias = 0;
923  /* compile the sub-expression */
924  buf = mutt_substrdup (ps.dptr + 1, p);
925  if ((tmp2 = mutt_pattern_comp (buf, flags, err)) == NULL)
926  {
927  FREE (&buf);
928  mutt_pattern_free (&curlist);
929  return NULL;
930  }
931  FREE (&buf);
932  tmp->child = tmp2;
933  ps.dptr = p + 1; /* restore location */
934  break;
935  }
936  if (implicit && or)
937  {
938  /* A | B & C == (A | B) & C */
939  tmp = new_pattern ();
940  tmp->op = MUTT_OR;
941  tmp->child = curlist;
942  curlist = tmp;
943  last = tmp;
944  or = 0;
945  }
946 
947  tmp = new_pattern ();
948  tmp->not = not;
949  tmp->alladdr = alladdr;
950  tmp->isalias = isalias;
951  tmp->stringmatch = (*ps.dptr == '=') ? 1 : 0;
952  tmp->groupmatch = (*ps.dptr == '%') ? 1 : 0;
953  not = 0;
954  alladdr = 0;
955  isalias = 0;
956 
957  if (last)
958  last->next = tmp;
959  else
960  curlist = tmp;
961  last = tmp;
962 
963  ps.dptr++; /* move past the ~ */
964  if ((entry = lookup_tag (*ps.dptr)) == NULL)
965  {
966  snprintf (err->data, err->dsize, _("%c: invalid pattern modifier"), *ps.dptr);
967  mutt_pattern_free (&curlist);
968  return NULL;
969  }
970  if (entry->class && (flags & entry->class) == 0)
971  {
972  snprintf (err->data, err->dsize, _("%c: not supported in this mode"), *ps.dptr);
973  mutt_pattern_free (&curlist);
974  return NULL;
975  }
976  tmp->op = entry->op;
977 
978  ps.dptr++; /* eat the operator and any optional whitespace */
979  SKIPWS (ps.dptr);
980 
981  if (entry->eat_arg)
982  {
983  if (!*ps.dptr)
984  {
985  snprintf (err->data, err->dsize, _("missing parameter"));
986  mutt_pattern_free (&curlist);
987  return NULL;
988  }
989  if (entry->eat_arg (tmp, &ps, err) == -1)
990  {
991  mutt_pattern_free (&curlist);
992  return NULL;
993  }
994  }
995  implicit = 1;
996  break;
997  case '(':
998  p = find_matching_paren (ps.dptr + 1);
999  if (*p != ')')
1000  {
1001  snprintf (err->data, err->dsize, _("mismatched parenthesis: %s"), ps.dptr);
1002  mutt_pattern_free (&curlist);
1003  return NULL;
1004  }
1005  /* compile the sub-expression */
1006  buf = mutt_substrdup (ps.dptr + 1, p);
1007  if ((tmp = mutt_pattern_comp (buf, flags, err)) == NULL)
1008  {
1009  FREE (&buf);
1010  mutt_pattern_free (&curlist);
1011  return NULL;
1012  }
1013  FREE (&buf);
1014  if (last)
1015  last->next = tmp;
1016  else
1017  curlist = tmp;
1018  last = tmp;
1019  tmp->not ^= not;
1020  tmp->alladdr |= alladdr;
1021  tmp->isalias |= isalias;
1022  not = 0;
1023  alladdr = 0;
1024  isalias = 0;
1025  ps.dptr = p + 1; /* restore location */
1026  break;
1027  default:
1028  snprintf (err->data, err->dsize, _("error in pattern at: %s"), ps.dptr);
1029  mutt_pattern_free (&curlist);
1030  return NULL;
1031  }
1032  }
1033  if (!curlist)
1034  {
1035  strfcpy (err->data, _("empty pattern"), err->dsize);
1036  return NULL;
1037  }
1038  if (curlist->next)
1039  {
1040  tmp = new_pattern ();
1041  tmp->op = or ? MUTT_OR : MUTT_AND;
1042  tmp->child = curlist;
1043  curlist = tmp;
1044  }
1045  return (curlist);
1046 }
