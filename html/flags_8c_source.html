28 void _mutt_set_flag (CONTEXT *ctx, HEADER *h, int flag, int bf, int upd_ctx)
29 {
30  if (!ctx || !h)
31  return;
32 
33  int changed = h->changed;
34  int deleted = ctx->deleted;
35  int tagged = ctx->tagged;
36  int flagged = ctx->flagged;
37  int update = 0;
38 
39  if (ctx->readonly && flag != MUTT_TAG)
40  return; /* don't modify anything if we are read-only */
41 
42  switch (flag)
43  {
44  case MUTT_DELETE:
45 
46  if (!mutt_bit_isset(ctx->rights,MUTT_ACL_DELETE))
47  return;
48 
49  if (bf)
50  {
51  if (!h->deleted && !ctx->readonly
52  && (!h->flagged || !option(OPTFLAGSAFE)))
53  {
54  h->deleted = 1;
55  update = 1;
56  if (upd_ctx) ctx->deleted++;
57 #ifdef USE_IMAP
58  /* deleted messages aren't treated as changed elsewhere so that the
59  * purge-on-sync option works correctly. This isn't applicable here */
60  if (ctx && ctx->magic == MUTT_IMAP)
61  {
62  h->changed = 1;
63  if (upd_ctx) ctx->changed = 1;
64  }
65 #endif
66  }
67  }
68  else if (h->deleted)
69  {
70  h->deleted = 0;
71  update = 1;
72  if (upd_ctx) ctx->deleted--;
73 #ifdef USE_IMAP
74  /* see my comment above */
75  if (ctx->magic == MUTT_IMAP)
76  {
77  h->changed = 1;
78  if (upd_ctx) ctx->changed = 1;
79  }
80 #endif
81  /*
82  * If the user undeletes a message which is marked as
83  * "trash" in the maildir folder on disk, the folder has
84  * been changed, and is marked accordingly. However, we do
85  * _not_ mark the message itself changed, because trashing
86  * is checked in specific code in the maildir folder
87  * driver.
88  */
89  if (ctx->magic == MUTT_MAILDIR && upd_ctx && h->trash)
90  ctx->changed = 1;
91  }
92  break;
93 
94  case MUTT_PURGE:
95 
96  if (!mutt_bit_isset(ctx->rights,MUTT_ACL_DELETE))
97  return;
98 
99  if (bf)
100  {
101  if (!h->purge && !ctx->readonly)
102  h->purge = 1;
103  }
104  else if (h->purge)
105  h->purge = 0;
106  break;
107 
108  case MUTT_NEW:
109 
110  if (!mutt_bit_isset(ctx->rights,MUTT_ACL_SEEN))
111  return;
112 
113  if (bf)
114  {
115  if (h->read || h->old)
116  {
117  update = 1;
118  h->old = 0;
119  if (upd_ctx) ctx->new++;
120  if (h->read)
121  {
122  h->read = 0;
123  if (upd_ctx) ctx->unread++;
124  }
125  h->changed = 1;
126  if (upd_ctx) ctx->changed = 1;
127  }
128  }
129  else if (!h->read)
130  {
131  update = 1;
132  if (!h->old)
133  if (upd_ctx) ctx->new--;
134  h->read = 1;
135  if (upd_ctx) ctx->unread--;
136  h->changed = 1;
137  if (upd_ctx) ctx->changed = 1;
138  }
139  break;
140 
141  case MUTT_OLD:
142 
143  if (!mutt_bit_isset(ctx->rights,MUTT_ACL_SEEN))
144  return;
145 
146  if (bf)
147  {
148  if (!h->old)
149  {
150  update = 1;
151  h->old = 1;
152  if (!h->read)
153  if (upd_ctx) ctx->new--;
154  h->changed = 1;
155  if (upd_ctx) ctx->changed = 1;
156  }
157  }
158  else if (h->old)
159  {
160  update = 1;
161  h->old = 0;
162  if (!h->read)
163  if (upd_ctx) ctx->new++;
164  h->changed = 1;
165  if (upd_ctx) ctx->changed = 1;
166  }
167  break;
168 
169  case MUTT_READ:
170 
171  if (!mutt_bit_isset(ctx->rights,MUTT_ACL_SEEN))
172  return;
173 
174  if (bf)
175  {
176  if (!h->read)
177  {
178  update = 1;
179  h->read = 1;
180  if (upd_ctx) ctx->unread--;
181  if (!h->old)
182  if (upd_ctx) ctx->new--;
183  h->changed = 1;
184  if (upd_ctx) ctx->changed = 1;
185  }
186  }
187  else if (h->read)
188  {
189  update = 1;
190  h->read = 0;
191  if (upd_ctx) ctx->unread++;
192  if (!h->old)
193  if (upd_ctx) ctx->new++;
194  h->changed = 1;
195  if (upd_ctx) ctx->changed = 1;
196  }
197  break;
198 
199  case MUTT_REPLIED:
200 
201  if (!mutt_bit_isset(ctx->rights,MUTT_ACL_WRITE))
202  return;
203 
204  if (bf)
205  {
206  if (!h->replied)
207  {
208  update = 1;
209  h->replied = 1;
210  if (!h->read)
211  {
212  h->read = 1;
213  if (upd_ctx) ctx->unread--;
214  if (!h->old)
215  if (upd_ctx) ctx->new--;
216  }
217  h->changed = 1;
218  if (upd_ctx) ctx->changed = 1;
219  }
220  }
221  else if (h->replied)
222  {
223  update = 1;
224  h->replied = 0;
225  h->changed = 1;
226  if (upd_ctx) ctx->changed = 1;
227  }
228  break;
229 
230  case MUTT_FLAG:
231 
232  if (!mutt_bit_isset(ctx->rights,MUTT_ACL_WRITE))
233  return;
234 
235  if (bf)
236  {
237  if (!h->flagged)
238  {
239  update = 1;
240  h->flagged = bf;
241  if (upd_ctx) ctx->flagged++;
242  h->changed = 1;
243  if (upd_ctx) ctx->changed = 1;
244  }
245  }
246  else if (h->flagged)
247  {
248  update = 1;
249  h->flagged = 0;
250  if (upd_ctx) ctx->flagged--;
251  h->changed = 1;
252  if (upd_ctx) ctx->changed = 1;
253  }
254  break;
255 
256  case MUTT_TAG:
257  if (bf)
258  {
259  if (!h->tagged)
260  {
261  update = 1;
262  h->tagged = 1;
263  if (upd_ctx) ctx->tagged++;
264  }
265  }
266  else if (h->tagged)
267  {
268  update = 1;
269  h->tagged = 0;
270  if (upd_ctx) ctx->tagged--;
271  }
272  break;
273  }
274 
275  if (update)
276  {
277  mutt_set_header_color(ctx, h);
278 #ifdef USE_SIDEBAR
279  SidebarNeedsRedraw = 1;
280 #endif
281  }
282 
283  /* if the message status has changed, we need to invalidate the cached
284  * search results so that any future search will match the current status
285  * of this message and not what it was at the time it was last searched.
286  */
287  if (h->searched && (changed != h->changed || deleted != ctx->deleted || tagged != ctx->tagged || flagged != ctx->flagged))
288  h->searched = 0;
289 }

299 int mutt_thread_set_flag (HEADER *hdr, int flag, int bf, int subthread)
300 {
301  THREAD *start, *cur = hdr->thread;
302 
303  if ((Sort & SORT_MASK) != SORT_THREADS)
304  {
305  mutt_error _("Threading is not enabled.");
306  return (-1);
307  }
308 
309  if (!subthread)
310  while (cur->parent)
311  cur = cur->parent;
312  start = cur;
313 
314  if (cur->message)
315  mutt_set_flag (Context, cur->message, flag, bf);
316 
317  if ((cur = cur->child) == NULL)
318  return (0);
319 
320  FOREVER
321  {
322  if (cur->message)
323  mutt_set_flag (Context, cur->message, flag, bf);
324 
325  if (cur->child)
326  cur = cur->child;
327  else if (cur->next)
328  cur = cur->next;
329  else
330  {
331  while (!cur->next)
332  {
333  cur = cur->parent;
334  if (cur == start)
335  return (0);
336  }
337  cur = cur->next;
338  }
339  }
340  /* not reached */
341 }

343 int mutt_change_flag (HEADER *h, int bf)
344 {
345  int i, flag;
346  event_t event;
347 
348  mutt_window_mvprintw (MuttMessageWindow, 0, 0,
349  "%s? (D/N/O/r/*/!): ", bf ? _("Set flag") : _("Clear flag"));
350  mutt_window_clrtoeol (MuttMessageWindow);
351 
352  event = mutt_getch();
353  i = event.ch;
354  if (i < 0)
355  {
356  mutt_window_clearline (MuttMessageWindow, 0);
357  return (-1);
358  }
359 
360  mutt_window_clearline (MuttMessageWindow, 0);
361 
362  switch (i)
363  {
364  case 'd':
365  case 'D':
366  if (!bf)
367  {
368  if (h)
369  mutt_set_flag (Context, h, MUTT_PURGE, bf);
370  else
371  mutt_tag_set_flag (MUTT_PURGE, bf);
372  }
373  flag = MUTT_DELETE;
374  break;
375 
376  case 'N':
377  case 'n':
378  flag = MUTT_NEW;
379  break;
380 
381  case 'o':
382  case 'O':
383  if (h)
384  mutt_set_flag (Context, h, MUTT_READ, !bf);
385  else
386  mutt_tag_set_flag (MUTT_READ, !bf);
387  flag = MUTT_OLD;
388  break;
389 
390  case 'r':
391  case 'R':
392  flag = MUTT_REPLIED;
393  break;
394 
395  case '*':
396  flag = MUTT_TAG;
397  break;
398 
399  case '!':
400  flag = MUTT_FLAG;
401  break;
402 
403  default:
404  BEEP ();
405  return (-1);
406  }
407 
408  if (h)
409  mutt_set_flag (Context, h, flag, bf);
410  else
411  mutt_tag_set_flag (flag, bf);
412 
413  return 0;
414 }
