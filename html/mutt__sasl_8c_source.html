36 static int getnameinfo_err(int ret)
37 {
38  int err;
39  dprint (1, (debugfile, "getnameinfo: "));
40  switch(ret)
41  {
42  case EAI_AGAIN:
43  dprint (1, (debugfile, "The name could not be resolved at this time. Future attempts may succeed.\n"));
44  err=SASL_TRYAGAIN;
45  break;
46  case EAI_BADFLAGS:
47  dprint (1, (debugfile, "The flags had an invalid value.\n"));
48  err=SASL_BADPARAM;
49  break;
50  case EAI_FAIL:
51  dprint (1, (debugfile, "A non-recoverable error occurred.\n"));
52  err=SASL_FAIL;
53  break;
54  case EAI_FAMILY:
55  dprint (1, (debugfile, "The address family was not recognized or the address length was invalid for the specified family.\n"));
56  err=SASL_BADPROT;
57  break;
58  case EAI_MEMORY:
59  dprint (1, (debugfile, "There was a memory allocation failure.\n"));
60  err=SASL_NOMEM;
61  break;
62  case EAI_NONAME:
63  dprint (1, (debugfile, "The name does not resolve for the supplied parameters. NI_NAMEREQD is set and the host's name cannot be located, or both nodename and servname were null.\n"));
64  err=SASL_FAIL; /* no real equivalent */
65  break;
66  case EAI_SYSTEM:
67  dprint (1, (debugfile, "A system error occurred. The error code can be found in errno(%d,%s)).\n",errno,strerror(errno)));
68  err=SASL_FAIL; /* no real equivalent */
69  break;
70  default:
71  dprint (1, (debugfile, "Unknown error %d\n",ret));
72  err=SASL_FAIL; /* no real equivalent */
73  break;
74  }
75  return err;
76 }

106 /* utility function, stolen from sasl2 sample code */
107 static int iptostring(const struct sockaddr *addr, socklen_t addrlen,
108  char *out, unsigned outlen)
109 {
110  char hbuf[NI_MAXHOST], pbuf[NI_MAXSERV];
111  int ret;
112 
113  if (!addr || !out)
114  return SASL_BADPARAM;
115 
116  ret = getnameinfo(addr, addrlen, hbuf, sizeof(hbuf), pbuf, sizeof(pbuf),
117  NI_NUMERICHOST |
118 #ifdef NI_WITHSCOPEID
119  NI_WITHSCOPEID |
120 #endif
121  NI_NUMERICSERV);
122  if (ret)
123  return getnameinfo_err(ret);
124 
125  if (outlen < strlen(hbuf) + strlen(pbuf) + 2)
126  return SASL_BUFOVER;
127 
128  snprintf(out, outlen, "%s;%s", hbuf, pbuf);
129 
130  return SASL_OK;
131 }

133 /* mutt_sasl_start: called before doing a SASL exchange - initialises library
134  * (if necessary). */
135 int mutt_sasl_start (void)
136 {
137  static unsigned char sasl_init = 0;
138 
139  static sasl_callback_t callbacks[2];
140  int rc;
141 
142  if (sasl_init)
143  return SASL_OK;
144 
145  /* set up default logging callback */
146  callbacks[0].id = SASL_CB_LOG;
147  callbacks[0].proc = (int (*)(void))mutt_sasl_cb_log;
148  callbacks[0].context = NULL;
149 
150  callbacks[1].id = SASL_CB_LIST_END;
151  callbacks[1].proc = NULL;
152  callbacks[1].context = NULL;
153 
154  rc = sasl_client_init (callbacks);
155 
156  if (rc != SASL_OK)
157  {
158  dprint (1, (debugfile, "mutt_sasl_start: libsasl initialisation failed.\n"));
159  return SASL_FAIL;
160  }
161 
162  sasl_init = 1;
163 
164  return SASL_OK;
165 }

167 /* mutt_sasl_client_new: wrapper for sasl_client_new which also sets various
168  * security properties. If this turns out to be fine for POP too we can
169  * probably stop exporting mutt_sasl_get_callbacks(). */
170 int mutt_sasl_client_new (CONNECTION* conn, sasl_conn_t** saslconn)
171 {
172  sasl_security_properties_t secprops;
173  struct sockaddr_storage local, remote;
174  socklen_t size;
175  char iplocalport[IP_PORT_BUFLEN], ipremoteport[IP_PORT_BUFLEN];
176  char* plp = NULL;
177  char* prp = NULL;
178  const char* service;
179  int rc;
180 
181  if (mutt_sasl_start () != SASL_OK)
182  return -1;
183 
184  switch (conn->account.type)
185  {
186  case MUTT_ACCT_TYPE_IMAP:
187  service = "imap";
188  break;
189  case MUTT_ACCT_TYPE_POP:
190  service = "pop";
191  break;
192  case MUTT_ACCT_TYPE_SMTP:
193  service = "smtp";
194  break;
195 #ifdef USE_NNTP
196  case MUTT_ACCT_TYPE_NNTP:
197  service = "nntp";
198  break;
199 #endif
200  default:
201  mutt_error (_("Unknown SASL profile"));
202  return -1;
203  }
204 
205  size = sizeof (local);
206  if (!getsockname (conn->fd, (struct sockaddr *)&local, &size)) {
207  if (iptostring((struct sockaddr *)&local, size, iplocalport,
208  IP_PORT_BUFLEN) == SASL_OK)
209  plp = iplocalport;
210  else
211  dprint (2, (debugfile, "SASL failed to parse local IP address\n"));
212  }
213  else
214  dprint (2, (debugfile, "SASL failed to get local IP address\n"));
215 
216  size = sizeof (remote);
217  if (!getpeername (conn->fd, (struct sockaddr *)&remote, &size)){
218  if (iptostring((struct sockaddr *)&remote, size, ipremoteport,
219  IP_PORT_BUFLEN) == SASL_OK)
220  prp = ipremoteport;
221  else
222  dprint (2, (debugfile, "SASL failed to parse remote IP address\n"));
223  }
224  else
225  dprint (2, (debugfile, "SASL failed to get remote IP address\n"));
226 
227  dprint (2, (debugfile, "SASL local ip: %s, remote ip:%s\n", NONULL(plp),
228  NONULL(prp)));
229 
230  rc = sasl_client_new (service, conn->account.host, plp, prp,
231  mutt_sasl_get_callbacks (&conn->account), 0, saslconn);
232 
233  if (rc != SASL_OK)
234  {
235  mutt_error (_("Error allocating SASL connection"));
236  mutt_sleep (2);
237  return -1;
238  }
239 
240  memset (&secprops, 0, sizeof (secprops));
241  /* Work around a casting bug in the SASL krb4 module */
242  secprops.max_ssf = 0x7fff;
243  secprops.maxbufsize = MUTT_SASL_MAXBUF;
244  if (sasl_setprop (*saslconn, SASL_SEC_PROPS, &secprops) != SASL_OK)
245  {
246  mutt_error (_("Error setting SASL security properties"));
247  return -1;
248  }
249 
250  if (conn->ssf)
251  {
252  /* I'm not sure this actually has an effect, at least with SASLv2 */
253  dprint (2, (debugfile, "External SSF: %d\n", conn->ssf));
254  if (sasl_setprop (*saslconn, SASL_SSF_EXTERNAL, &(conn->ssf)) != SASL_OK)
255  {
256  mutt_error (_("Error setting SASL external security strength"));
257  return -1;
258  }
259  }
260  if (conn->account.user[0])
261  {
262  dprint (2, (debugfile, "External authentication name: %s\n", conn->account.user));
263  if (sasl_setprop (*saslconn, SASL_AUTH_EXTERNAL, conn->account.user) != SASL_OK)
264  {
265  mutt_error (_("Error setting SASL external user name"));
266  return -1;
267  }
268  }
269 
270  return 0;
271 }

307 {
308  char prompt[SHORT_STRING];
309  char resp[SHORT_STRING];
310 
311  while (interaction->id != SASL_CB_LIST_END)
312  {
313  dprint (2, (debugfile, "mutt_sasl_interact: filling in SASL interaction %ld.\n", interaction->id));
314 
315  snprintf (prompt, sizeof (prompt), "%s: ", interaction->prompt);
316  resp[0] = '\0';
317  if (option (OPTNOCURSES) || mutt_get_field (prompt, resp, sizeof (resp), 0))
318  return SASL_FAIL;
319 
320  interaction->len = mutt_strlen (resp)+1;
321  interaction->result = safe_malloc (interaction->len);
322  memcpy ((char *)interaction->result, resp, interaction->len);
323 
324  interaction++;
325  }
326 
327  return SASL_OK;

330 /* SASL can stack a protection layer on top of an existing connection.
331  * To handle this, we store a saslconn_t in conn->sockdata, and write
332  * wrappers which en/decode the read/write stream, then replace sockdata
333  * with an embedded copy of the old sockdata and call the underlying
334  * functions (which we've also preserved). I thought about trying to make
335  * a general stackable connection system, but it seemed like overkill -
336  * something is wrong if we have 15 filters on top of a socket. Anyway,
337  * anything else which wishes to stack can use the same method. The only
338  * disadvantage is we have to write wrappers for all the socket methods,
339  * even if we only stack over read and write. Thinking about it, the
340  * abstraction problem is that there is more in CONNECTION than there
341  * needs to be. Ideally it would have only (void*)data and methods. */
342 
343 /* mutt_sasl_setup_conn: replace connection methods, sockdata with
344  * SASL wrappers, for protection layers. Also get ssf, as a fastpath
345  * for the read/write methods. */
346 void mutt_sasl_setup_conn (CONNECTION* conn, sasl_conn_t* saslconn)
347 {
348  SASL_DATA* sasldata = (SASL_DATA*) safe_malloc (sizeof (SASL_DATA));
349  /* work around sasl_getprop aliasing issues */
350  const void* tmp;
351 
352  sasldata->saslconn = saslconn;
353  /* get ssf so we know whether we have to (en|de)code read/write */
354  sasl_getprop (saslconn, SASL_SSF, &tmp);
355  sasldata->ssf = tmp;
356  dprint (3, (debugfile, "SASL protection strength: %u\n", *sasldata->ssf));
357  /* Add SASL SSF to transport SSF */
358  conn->ssf += *sasldata->ssf;
359  sasl_getprop (saslconn, SASL_MAXOUTBUF, &tmp);
360  sasldata->pbufsize = tmp;
361  dprint (3, (debugfile, "SASL protection buffer size: %u\n", *sasldata->pbufsize));
362 
363  /* clear input buffer */
364  sasldata->buf = NULL;
365  sasldata->bpos = 0;
366  sasldata->blen = 0;
367 
368  /* preserve old functions */
369  sasldata->sockdata = conn->sockdata;
370  sasldata->msasl_open = conn->conn_open;
371  sasldata->msasl_close = conn->conn_close;
372  sasldata->msasl_read = conn->conn_read;
373  sasldata->msasl_write = conn->conn_write;
374  sasldata->msasl_poll = conn->conn_poll;
375 
376  /* and set up new functions */
377  conn->sockdata = sasldata;
378  conn->conn_open = mutt_sasl_conn_open;
379  conn->conn_close = mutt_sasl_conn_close;
380  conn->conn_read = mutt_sasl_conn_read;
381  conn->conn_write = mutt_sasl_conn_write;
382  conn->conn_poll = mutt_sasl_conn_poll;
383 }

400  unsigned* len)
401 {

403 
404  if (!result)

411  if (!account)

413 
414  dprint (2, (debugfile, "mutt_sasl_cb_authname: getting %s for %s:%u\n",

417

419  {

426  if (mutt_account_getuser (account))
427  return SASL_FAIL;
428  *result = account->user;
429  }
430 
431  if (len)
432  *len = strlen (*result);
433 
434  return SASL_OK;
435 }

507 static int mutt_sasl_conn_read (CONNECTION* conn, char* buf, size_t len)
508 {
509  SASL_DATA* sasldata;
510  int rc;
511 
512  unsigned int olen;
513 
514  sasldata = (SASL_DATA*) conn->sockdata;
515 
516  /* if we still have data in our read buffer, copy it into buf */
517  if (sasldata->blen > sasldata->bpos)
518  {
519  olen = (sasldata->blen - sasldata->bpos > len) ? len :
520  sasldata->blen - sasldata->bpos;
521 
522  memcpy (buf, sasldata->buf+sasldata->bpos, olen);
523  sasldata->bpos += olen;
524 
525  return olen;
526  }
527 
528  conn->sockdata = sasldata->sockdata;
529 
530  sasldata->bpos = 0;
531  sasldata->blen = 0;
532 
533  /* and decode the result, if necessary */
534  if (*sasldata->ssf)
535  {
536  do
537  {
538  /* call the underlying read function to fill the buffer */
539  rc = (sasldata->msasl_read) (conn, buf, len);
540  if (rc <= 0)
541  goto out;
542 
543  rc = sasl_decode (sasldata->saslconn, buf, rc, &sasldata->buf,
544  &sasldata->blen);
545  if (rc != SASL_OK)
546  {
547  dprint (1, (debugfile, "SASL decode failed: %s\n",
548  sasl_errstring (rc, NULL, NULL)));
549  goto out;
550  }
551  }
552  while (!sasldata->blen);
553 
554  olen = (sasldata->blen - sasldata->bpos > len) ? len :
555  sasldata->blen - sasldata->bpos;
556 
557  memcpy (buf, sasldata->buf, olen);
558  sasldata->bpos += olen;
559 
560  rc = olen;
561  }
562  else
563  rc = (sasldata->msasl_read) (conn, buf, len);
564 
565  out:
566  conn->sockdata = sasldata;
567 
568  return rc;
569 }

571 static int mutt_sasl_conn_write (CONNECTION* conn, const char* buf,
572  size_t len)
573 {
574  SASL_DATA* sasldata;
575  int rc;
576 
577  const char *pbuf;
578  unsigned int olen, plen;
579 
580  sasldata = (SASL_DATA*) conn->sockdata;
581  conn->sockdata = sasldata->sockdata;
582 
583  /* encode data, if necessary */
584  if (*sasldata->ssf)
585  {
586  /* handle data larger than MAXOUTBUF */
587  do
588  {
589  olen = (len > *sasldata->pbufsize) ? *sasldata->pbufsize : len;
590 
591  rc = sasl_encode (sasldata->saslconn, buf, olen, &pbuf, &plen);
592  if (rc != SASL_OK)
593  {
594  dprint (1, (debugfile, "SASL encoding failed: %s\n",
595  sasl_errstring (rc, NULL, NULL)));
596  goto fail;
597  }
598 
599  rc = (sasldata->msasl_write) (conn, pbuf, plen);
600  if (rc != plen)
601  goto fail;
602 
603  len -= olen;
604  buf += olen;
605  }
606  while (len > *sasldata->pbufsize);
607  }
608  else
609  /* just write using the underlying socket function */
610  rc = (sasldata->msasl_write) (conn, buf, len);
611 
612  conn->sockdata = sasldata;
613 
614  return rc;
615 
616  fail:
617  conn->sockdata = sasldata;
618  return -1;
619 }
