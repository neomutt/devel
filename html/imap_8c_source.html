56  * TODO: ACL checks. Right now we assume if it exists we can
57  * mess with it. */
58 int imap_access (const char* path, int flags)
59 {
60  IMAP_DATA* idata;
61  IMAP_MBOX mx;
62  char buf[LONG_STRING];
63  char mailbox[LONG_STRING];
64  char mbox[LONG_STRING];
65  int rc;
66 
67  if (imap_parse_path (path, &mx))
68  return -1;
69 
70  if (!(idata = imap_conn_find (&mx.account,
71  option (OPTIMAPPASSIVE) ? MUTT_IMAP_CONN_NONEW : 0)))
72  {
73  FREE (&mx.mbox);
74  return -1;
75  }
76 
77  imap_fix_path (idata, mx.mbox, mailbox, sizeof (mailbox));
78  if (!*mailbox)
79  strfcpy (mailbox, "INBOX", sizeof (mailbox));
80 
81  /* we may already be in the folder we're checking */
82  if (!ascii_strcmp(idata->mailbox, mx.mbox))
83  {
84  FREE (&mx.mbox);
85  return 0;
86  }
87  FREE (&mx.mbox);
88 
89  if (imap_mboxcache_get (idata, mailbox, 0))
90  {
91  dprint (3, (debugfile, "imap_access: found %s in cache\n", mailbox));
92  return 0;
93  }
94 
95  imap_munge_mbox_name (idata, mbox, sizeof (mbox), mailbox);
96 
97  if (mutt_bit_isset (idata->capabilities, IMAP4REV1))
98  snprintf (buf, sizeof (buf), "STATUS %s (UIDVALIDITY)", mbox);
99  else if (mutt_bit_isset (idata->capabilities, STATUS))
100  snprintf (buf, sizeof (buf), "STATUS %s (UID-VALIDITY)", mbox);
101  else
102  {
103  dprint (2, (debugfile, "imap_access: STATUS not supported?\n"));
104  return -1;
105  }
106 
107  if ((rc = imap_exec (idata, buf, IMAP_CMD_FAIL_OK)) < 0)
108  {
109  dprint (1, (debugfile, "imap_access: Can't check STATUS of %s\n", mbox));
110  return rc;
111  }
112 
113  return 0;
114 }

318 /* imap_conn_find: Find an open IMAP connection matching account, or open
319  * a new one if none can be found. */
320 IMAP_DATA* imap_conn_find (const ACCOUNT* account, int flags)
321 {
322  CONNECTION* conn = NULL;
323  ACCOUNT* creds = NULL;
324  IMAP_DATA* idata = NULL;
325  int new = 0;
326 
327  while ((conn = mutt_conn_find (conn, account)))
328  {
329  if (!creds)
330  creds = &conn->account;
331  else
332  memcpy (&conn->account, creds, sizeof (ACCOUNT));
333 
334  idata = (IMAP_DATA*)conn->data;
335  if (flags & MUTT_IMAP_CONN_NONEW)
336  {
337  if (!idata)
338  {
339  /* This should only happen if we've come to the end of the list */
340  mutt_socket_free (conn);
341  return NULL;
342  }
343  else if (idata->state < IMAP_AUTHENTICATED)
344  continue;
345  }
346  if (flags & MUTT_IMAP_CONN_NOSELECT && idata && idata->state >= IMAP_SELECTED)
347  continue;
348  if (idata && idata->status == IMAP_FATAL)
349  continue;
350  break;
351  }
352  if (!conn)
353  return NULL; /* this happens when the initial connection fails */
354 
355  if (!idata)
356  {
357  /* The current connection is a new connection */
358  if (! (idata = imap_new_idata ()))
359  {
360  mutt_socket_free (conn);
361  return NULL;
362  }
363 
364  conn->data = idata;
365  idata->conn = conn;
366  new = 1;
367  }
368 
369  if (idata->state == IMAP_DISCONNECTED)
370  imap_open_connection (idata);
371  if (idata->state == IMAP_CONNECTED)
372  {
373  if (!imap_authenticate (idata))
374  {
375  idata->state = IMAP_AUTHENTICATED;
376  FREE (&idata->capstr);
377  new = 1;
378  if (idata->conn->ssf)
379  dprint (2, (debugfile, "Communication encrypted at %d bits\n",
380  idata->conn->ssf));
381  }
382  else
383  mutt_account_unsetpass (&idata->conn->account);
384  }
385  if (new && idata->state == IMAP_AUTHENTICATED)
386  {
387  /* capabilities may have changed */
388  imap_exec (idata, "CAPABILITY", IMAP_CMD_QUEUE);
389  /* enable RFC6855, if the server supports that */
390  if (mutt_bit_isset (idata->capabilities, ENABLE))
391  imap_exec (idata, "ENABLE UTF8=ACCEPT", IMAP_CMD_QUEUE);
392  /* get root delimiter, '/' as default */
393  idata->delim = '/';
394  imap_exec (idata, "LIST \"\" \"\"", IMAP_CMD_QUEUE);
395  if (option (OPTIMAPCHECKSUBSCRIBED))
396  imap_exec (idata, "LSUB \"\" \"*\"", IMAP_CMD_QUEUE);
397  /* we may need the root delimiter before we open a mailbox */
398  imap_exec (idata, NULL, IMAP_CMD_FAIL_OK);
399  }
400 
401  return idata;
402 }

404 int imap_open_connection (IMAP_DATA* idata)
405 {
406  char buf[LONG_STRING];
407 
408  if (mutt_socket_open (idata->conn) < 0)
409  return -1;
410 
411  idata->state = IMAP_CONNECTED;
412 
413  if (imap_cmd_step (idata) != IMAP_CMD_CONTINUE)
414  {
415  imap_close_connection (idata);
416  return -1;
417  }
418 
419  if (ascii_strncasecmp ("* OK", idata->buf, 4) == 0)
420  {
421  if (ascii_strncasecmp ("* OK [CAPABILITY", idata->buf, 16)
422  && imap_check_capabilities (idata))
423  goto bail;
424 #if defined(USE_SSL)
425  /* Attempt STARTTLS if available and desired. */
426  if (!idata->conn->ssf && (option(OPTSSLFORCETLS) ||
427  mutt_bit_isset (idata->capabilities, STARTTLS)))
428  {
429  int rc;
430 
431  if (option(OPTSSLFORCETLS))
432  rc = MUTT_YES;
433  else if ((rc = query_quadoption (OPT_SSLSTARTTLS,
434  _("Secure connection with TLS?"))) == MUTT_ABORT)
435  goto err_close_conn;
436  if (rc == MUTT_YES) {
437  if ((rc = imap_exec (idata, "STARTTLS", IMAP_CMD_FAIL_OK)) == -1)
438  goto bail;
439  if (rc != -2)
440  {
441  if (mutt_ssl_starttls (idata->conn))
442  {
443  mutt_error (_("Could not negotiate TLS connection"));
444  mutt_sleep (1);
445  goto err_close_conn;
446  }
447  else
448  {
449  /* RFC 2595 demands we recheck CAPABILITY after TLS completes. */
450  if (imap_exec (idata, "CAPABILITY", 0))
451  goto bail;
452  }
453  }
454  }
455  }
456 
457  if (option(OPTSSLFORCETLS) && ! idata->conn->ssf)
458  {
459  mutt_error _("Encrypted connection unavailable");
460  mutt_sleep (1);
461  goto err_close_conn;
462  }
463 #endif
464  }
465  else if (ascii_strncasecmp ("* PREAUTH", idata->buf, 9) == 0)
466  {
467  idata->state = IMAP_AUTHENTICATED;
468  if (imap_check_capabilities (idata) != 0)
469  goto bail;
470  FREE (&idata->capstr);
471  }
472  else
473  {
474  imap_error ("imap_open_connection()", buf);
475  goto bail;
476  }
477 
478  return 0;
479 
480 #if defined(USE_SSL)
481  err_close_conn:
482  imap_close_connection (idata);
483 #endif
484  bail:
485  FREE (&idata->capstr);
486  return -1;
487 }

500 /* imap_get_flags: Make a simple list out of a FLAGS response.
501  * return stream following FLAGS response */
502 static char* imap_get_flags (LIST** hflags, char* s)
503 {
504  LIST* flags;
505  char* flag_word;
506  char ctmp;
507 
508  /* sanity-check string */
509  if (ascii_strncasecmp ("FLAGS", s, 5) != 0)
510  {
511  dprint (1, (debugfile, "imap_get_flags: not a FLAGS response: %s\n",
512  s));
513  return NULL;
514  }
515  s += 5;
516  SKIPWS(s);
517  if (*s != '(')
518  {
519  dprint (1, (debugfile, "imap_get_flags: bogus FLAGS response: %s\n",
520  s));
521  return NULL;
522  }
523 
524  /* create list, update caller's flags handle */
525  flags = mutt_new_list();
526  *hflags = flags;
527 
528  while (*s && *s != ')')
529  {
530  s++;
531  SKIPWS(s);
532  flag_word = s;
533  while (*s && (*s != ')') && !ISSPACE (*s))
534  s++;
535  ctmp = *s;
536  *s = '\0';
537  if (*flag_word)
538  mutt_add_list (flags, flag_word);
539  *s = ctmp;
540  }
541 
542  /* note bad flags response */
543  if (*s != ')')
544  {
545  dprint (1, (debugfile,
546  "imap_get_flags: Unterminated FLAGS response: %s\n", s));
547  mutt_free_list (hflags);
548 
549  return NULL;
550  }
551 
552  s++;
553 
554  return s;
555 }

557 static int imap_open_mailbox (CONTEXT* ctx)
558 {
559  IMAP_DATA *idata;
560  IMAP_STATUS* status;
561  char buf[LONG_STRING];
562  char bufout[LONG_STRING];
563  int count = 0;
564  IMAP_MBOX mx, pmx;
565  int rc;
566 
567  if (imap_parse_path (ctx->path, &mx))
568  {
569  mutt_error (_("%s is an invalid IMAP path"), ctx->path);
570  return -1;
571  }
572 
573  /* we require a connection which isn't currently in IMAP_SELECTED state */
574  if (!(idata = imap_conn_find (&(mx.account), MUTT_IMAP_CONN_NOSELECT)))
575  goto fail_noidata;
576  if (idata->state < IMAP_AUTHENTICATED)
577  goto fail;
578 
579  /* once again the context is new */
580  ctx->data = idata;
581 
582  /* Clean up path and replace the one in the ctx */
583  imap_fix_path (idata, mx.mbox, buf, sizeof (buf));
584  if (!*buf)
585  strfcpy (buf, "INBOX", sizeof (buf));
586  FREE(&(idata->mailbox));
587  idata->mailbox = safe_strdup (buf);
588  imap_qualify_path (buf, sizeof (buf), &mx, idata->mailbox);
589 
590  FREE (&(ctx->path));
591  FREE (&(ctx->realpath));
592  ctx->path = safe_strdup (buf);
593  ctx->realpath = safe_strdup (ctx->path);
594 
595  idata->ctx = ctx;
596 
597  /* clear mailbox status */
598  idata->status = 0;
599  memset (idata->ctx->rights, 0, sizeof (idata->ctx->rights));
600  idata->newMailCount = 0;
601 
602  mutt_message (_("Selecting %s..."), idata->mailbox);
603  imap_munge_mbox_name (idata, buf, sizeof(buf), idata->mailbox);
604 
605  /* pipeline ACL test */
606  if (mutt_bit_isset (idata->capabilities, ACL))
607  {
608  snprintf (bufout, sizeof (bufout), "MYRIGHTS %s", buf);
609  imap_exec (idata, bufout, IMAP_CMD_QUEUE);
610  }
611  /* assume we have all rights if ACL is unavailable */
612  else
613  {
614  mutt_bit_set (idata->ctx->rights, MUTT_ACL_LOOKUP);
615  mutt_bit_set (idata->ctx->rights, MUTT_ACL_READ);
616  mutt_bit_set (idata->ctx->rights, MUTT_ACL_SEEN);
617  mutt_bit_set (idata->ctx->rights, MUTT_ACL_WRITE);
618  mutt_bit_set (idata->ctx->rights, MUTT_ACL_INSERT);
619  mutt_bit_set (idata->ctx->rights, MUTT_ACL_POST);
620  mutt_bit_set (idata->ctx->rights, MUTT_ACL_CREATE);
621  mutt_bit_set (idata->ctx->rights, MUTT_ACL_DELETE);
622  }
623  /* pipeline the postponed count if possible */
624  pmx.mbox = NULL;
625  if (mx_is_imap (Postponed) && !imap_parse_path (Postponed, &pmx)
626  && mutt_account_match (&pmx.account, &mx.account))
627  imap_status (Postponed, 1);
628  FREE (&pmx.mbox);
629 
630  snprintf (bufout, sizeof (bufout), "%s %s",
631  ctx->readonly ? "EXAMINE" : "SELECT", buf);
632 
633  idata->state = IMAP_SELECTED;
634 
635  imap_cmd_start (idata, bufout);
636 
637  status = imap_mboxcache_get (idata, idata->mailbox, 1);
638 
639  do
640  {
641  char *pc;
642 
643  if ((rc = imap_cmd_step (idata)) != IMAP_CMD_CONTINUE)
644  break;
645 
646  pc = idata->buf + 2;
647 
648  /* Obtain list of available flags here, may be overridden by a
649  * PERMANENTFLAGS tag in the OK response */
650  if (ascii_strncasecmp ("FLAGS", pc, 5) == 0)
651  {
652  /* don't override PERMANENTFLAGS */
653  if (!idata->flags)
654  {
655  dprint (3, (debugfile, "Getting mailbox FLAGS\n"));
656  if ((pc = imap_get_flags (&(idata->flags), pc)) == NULL)
657  goto fail;
658  }
659  }
660  /* PERMANENTFLAGS are massaged to look like FLAGS, then override FLAGS */
661  else if (ascii_strncasecmp ("OK [PERMANENTFLAGS", pc, 18) == 0)
662  {
663  dprint (3, (debugfile, "Getting mailbox PERMANENTFLAGS\n"));
664  /* safe to call on NULL */
665  mutt_free_list (&(idata->flags));
666  /* skip "OK [PERMANENT" so syntax is the same as FLAGS */
667  pc += 13;
668  if ((pc = imap_get_flags (&(idata->flags), pc)) == NULL)
669  goto fail;
670  }
671  /* save UIDVALIDITY for the header cache */
672  else if (ascii_strncasecmp ("OK [UIDVALIDITY", pc, 14) == 0)
673  {
674  dprint (3, (debugfile, "Getting mailbox UIDVALIDITY\n"));
675  pc += 3;
676  pc = imap_next_word (pc);
677  idata->uid_validity = strtol (pc, NULL, 10);

878 /* Note: headers must be in SORT_ORDER. See imap_exec_msgset for args.
879  * Pos is an opaque pointer a la strtok. It should be 0 at first call. */
880 static int imap_make_msg_set (IMAP_DATA* idata, BUFFER* buf, int flag,
881  int changed, int invert, int* pos)
882 {
883  HEADER** hdrs = idata->ctx->hdrs;
884  int count = 0; /* number of messages in message set */
885  int match = 0; /* whether current message matches flag condition */
886  unsigned int setstart = 0; /* start of current message range */
887  int n;
888  int started = 0;
889 
890  hdrs = idata->ctx->hdrs;
891 
892  for (n = *pos;
893  n < idata->ctx->msgcount && buf->dptr - buf->data < IMAP_MAX_CMDLEN;
894  n++)
895  {
896  match = 0;
897  /* don't include pending expunged messages */
898  if (hdrs[n]->active)
899  switch (flag)
900  {
901  case MUTT_DELETED:
902  if (hdrs[n]->deleted != HEADER_DATA(hdrs[n])->deleted)
903  match = invert ^ hdrs[n]->deleted;
904  break;
905  case MUTT_FLAG:
906  if (hdrs[n]->flagged != HEADER_DATA(hdrs[n])->flagged)
907  match = invert ^ hdrs[n]->flagged;
908  break;
909  case MUTT_OLD:
910  if (hdrs[n]->old != HEADER_DATA(hdrs[n])->old)
911  match = invert ^ hdrs[n]->old;
912  break;
913  case MUTT_READ:
914  if (hdrs[n]->read != HEADER_DATA(hdrs[n])->read)
915  match = invert ^ hdrs[n]->read;
916  break;
917  case MUTT_REPLIED:
918  if (hdrs[n]->replied != HEADER_DATA(hdrs[n])->replied)
919  match = invert ^ hdrs[n]->replied;
920  break;
921 
922  case MUTT_TAG:
923  if (hdrs[n]->tagged)
924  match = 1;
925  break;
926  case MUTT_TRASH:
927  if (hdrs[n]->deleted && !hdrs[n]->purge)
928  match = 1;
929  break;
930  }
931 
932  if (match && (!changed || hdrs[n]->changed))
933  {
934  count++;
935  if (setstart == 0)
936  {
937  setstart = HEADER_DATA (hdrs[n])->uid;
938  if (started == 0)
939  {
940  mutt_buffer_printf (buf, "%u", HEADER_DATA (hdrs[n])->uid);
941  started = 1;
942  }
943  else
944  mutt_buffer_printf (buf, ",%u", HEADER_DATA (hdrs[n])->uid);
945  }
946  /* tie up if the last message also matches */
947  else if (n == idata->ctx->msgcount-1)
948  mutt_buffer_printf (buf, ":%u", HEADER_DATA (hdrs[n])->uid);
949  }
950  /* End current set if message doesn't match or we've reached the end
951  * of the mailbox via inactive messages following the last match. */
952  else if (setstart && (hdrs[n]->active || n == idata->ctx->msgcount-1))
953  {
954  if (HEADER_DATA (hdrs[n-1])->uid > setstart)
955  mutt_buffer_printf (buf, ":%u", HEADER_DATA (hdrs[n-1])->uid);
956  setstart = 0;
957  }
958  }
959 
960  *pos = n;
961 
962  return count;
963 }

965 /* Prepares commands for all messages matching conditions (must be flushed
966  * with imap_exec)
967  * Params:
968  * idata: IMAP_DATA containing context containing header set
969  * pre, post: commands are of the form "%s %s %s %s", tag,
970  * pre, message set, post
971  * flag: enum of flag type on which to filter
972  * changed: include only changed messages in message set
973  * invert: invert sense of flag, eg MUTT_READ matches unread messages
974  * Returns: number of matched messages, or -1 on failure */
975 int imap_exec_msgset (IMAP_DATA* idata, const char* pre, const char* post,
976  int flag, int changed, int invert)
977 {
978  HEADER** hdrs = NULL;
979  short oldsort;
980  BUFFER* cmd;
981  int pos;
982  int rc;
983  int count = 0;
984 
985  if (! (cmd = mutt_buffer_new ()))
986  {
987  dprint (1, (debugfile, "imap_exec_msgset: unable to allocate buffer\n"));
988  return -1;
989  }
990 
991  /* We make a copy of the headers just in case resorting doesn't give
992  exactly the original order (duplicate messages?), because other parts of
993  the ctx are tied to the header order. This may be overkill. */
994  oldsort = Sort;
995  if (Sort != SORT_ORDER)
996  {
997  hdrs = idata->ctx->hdrs;
998  idata->ctx->hdrs = safe_malloc (idata->ctx->msgcount * sizeof (HEADER*));
999  memcpy (idata->ctx->hdrs, hdrs, idata->ctx->msgcount * sizeof (HEADER*));
1000 
1001  Sort = SORT_ORDER;
1002  qsort (idata->ctx->hdrs, idata->ctx->msgcount, sizeof (HEADER*),
1003  mutt_get_sort_func (SORT_ORDER));
1004  }
1005 
1006  pos = 0;
1007 
1008  do
1009  {
1010  cmd->dptr = cmd->data;
1011  mutt_buffer_printf (cmd, "%s ", pre);
1012  rc = imap_make_msg_set (idata, cmd, flag, changed, invert, &pos);
1013  if (rc > 0)
1014  {
1015  mutt_buffer_printf (cmd, " %s", post);
1016  if (imap_exec (idata, cmd->data, IMAP_CMD_QUEUE))
1017  {
1018  rc = -1;
1019  goto out;
1020  }
1021  count += rc;
1022  }
1023  }
1024  while (rc > 0);
1025 
1026  rc = count;
1027 
1028 out:
1029  mutt_buffer_free (&cmd);
1030  if (oldsort != Sort)
1031  {
1032  Sort = oldsort;
1033  FREE (&idata->ctx->hdrs);
1034  idata->ctx->hdrs = hdrs;
1035  }
1036 
1037  return rc;
1038 }

1181

1185  return -1;
1186  }

1189  * to be changed. */
1190  imap_allow_reopen (ctx);
1191 
1192  if ((rc = imap_check_mailbox (ctx, index_hint, 0)) != 0)

1207  {

1211  if (ctx->hdrs[n]->deleted && ctx->hdrs[n]->changed)

1219 #endif

1229 #if USE_HCACHE
1230  imap_hcache_del (idata, HEADER_DATA(h)->uid);

1243  h->attach_del || h->label_changed)
