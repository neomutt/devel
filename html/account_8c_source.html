29 /* mutt_account_match: compare account info (host/port/user) */
30 int mutt_account_match (const ACCOUNT* a1, const ACCOUNT* a2)
31 {
32  const char* user = NONULL (Username);
33 
34  if (a1->type != a2->type)
35  return 0;
36  if (ascii_strcasecmp (a1->host, a2->host))
37  return 0;
38  if (a1->port != a2->port)
39  return 0;
40 
41 #ifdef USE_IMAP
42  if (a1->type == MUTT_ACCT_TYPE_IMAP)
43  {
44  if (ImapUser)
45  user = ImapUser;
46  }
47 #endif
48 
49 #ifdef USE_POP
50  if (a1->type == MUTT_ACCT_TYPE_POP && PopUser)
51  user = PopUser;
52 #endif
53 
54 #ifdef USE_NNTP
55  if (a1->type == MUTT_ACCT_TYPE_NNTP && NntpUser)
56  user = NntpUser;
57 #endif
58 
59  if (a1->flags & a2->flags & MUTT_ACCT_USER)
60  return (!strcmp (a1->user, a2->user));
61 #ifdef USE_NNTP
62  if (a1->type == MUTT_ACCT_TYPE_NNTP)
63  return a1->flags & MUTT_ACCT_USER && a1->user[0] ? 0 : 1;
64 #endif
65  if (a1->flags & MUTT_ACCT_USER)
66  return (!strcmp (a1->user, user));
67  if (a2->flags & MUTT_ACCT_USER)
68  return (!strcmp (a2->user, user));
69 
70  return 1;
71 }

73 /* mutt_account_fromurl: fill account with information from url. */
74 int mutt_account_fromurl (ACCOUNT* account, ciss_url_t* url)
75 {
76  /* must be present */
77  if (url->host)
78  strfcpy (account->host, url->host, sizeof (account->host));
79  else
80  return -1;
81 
82  if (url->user)
83  {
84  strfcpy (account->user, url->user, sizeof (account->user));
85  account->flags |= MUTT_ACCT_USER;
86  }
87  if (url->pass)
88  {
89  strfcpy (account->pass, url->pass, sizeof (account->pass));
90  account->flags |= MUTT_ACCT_PASS;
91  }
92  if (url->port)
93  {
94  account->port = url->port;
95  account->flags |= MUTT_ACCT_PORT;
96  }
97 
98  return 0;
99 }

161 /* mutt_account_getuser: retrieve username into ACCOUNT, if necessary */
162 int mutt_account_getuser (ACCOUNT* account)
163 {
164  char prompt[SHORT_STRING];
165 
166  /* already set */
167  if (account->flags & MUTT_ACCT_USER)
168  return 0;
169 #ifdef USE_IMAP
170  else if ((account->type == MUTT_ACCT_TYPE_IMAP) && ImapUser)
171  strfcpy (account->user, ImapUser, sizeof (account->user));
172 #endif
173 #ifdef USE_POP
174  else if ((account->type == MUTT_ACCT_TYPE_POP) && PopUser)
175  strfcpy (account->user, PopUser, sizeof (account->user));
176 #endif
177 #ifdef USE_NNTP
178  else if ((account->type == MUTT_ACCT_TYPE_NNTP) && NntpUser)
179  strfcpy (account->user, NntpUser, sizeof (account->user));
180 #endif
181  else if (option (OPTNOCURSES))
182  return -1;
183  /* prompt (defaults to unix username), copy into account->user */
184  else
185  {
186  /* L10N: Example: Username at myhost.com */
187  snprintf (prompt, sizeof (prompt), _("Username at %s: "), account->host);
188  strfcpy (account->user, NONULL (Username), sizeof (account->user));
189  if (mutt_get_field_unbuffered (prompt, account->user, sizeof (account->user), 0))
190  return -1;
191  }
192 
193  account->flags |= MUTT_ACCT_USER;
194 
195  return 0;
196 }

256  if (mutt_get_password (prompt, account->pass, sizeof (account->pass)))
257  return -1;
258  }
259 
260  account->flags |= MUTT_ACCT_PASS;
261 
262  return 0;
