128 static const char *cb_format_str(char *dest, size_t destlen, size_t col, int cols, char op,
129  const char *src, const char *prefix, const char *ifstring,
130  const char *elsestring, unsigned long data, format_flag flags)
131 {
132  SBENTRY *sbe = (SBENTRY *) data;
133  unsigned int optional;
134  char fmt[STRING];
135 
136  if (!sbe || !dest)
137  return src;
138 
139  dest[0] = 0; /* Just in case there's nothing to do */
140 
141  BUFFY *b = sbe->buffy;
142  if (!b)
143  return src;
144 
145  int c = Context && (mutt_strcmp (Context->realpath, b->realpath) == 0);
146 
147  optional = flags & MUTT_FORMAT_OPTIONAL;
148 
149  switch (op)
150  {
151  case 'B':
152  mutt_format_s (dest, destlen, prefix, sbe->box);
153  break;
154 
155  case 'd':
156  if (!optional)
157  {
158  snprintf (fmt, sizeof (fmt), "%%%sd", prefix);
159  snprintf (dest, destlen, fmt, c ? Context->deleted : 0);
160  }
161  else if ((c && Context->deleted == 0) || !c)
162  optional = 0;
163  break;
164 
165  case 'F':
166  if (!optional)
167  {
168  snprintf (fmt, sizeof (fmt), "%%%sd", prefix);
169  snprintf (dest, destlen, fmt, b->msg_flagged);
170  }
171  else if (b->msg_flagged == 0)
172  optional = 0;
173  break;
174 
175  case 'L':
176  if (!optional)
177  {
178  snprintf (fmt, sizeof (fmt), "%%%sd", prefix);
179  snprintf (dest, destlen, fmt, c ? Context->vcount : b->msg_count);
180  }
181  else if ((c && Context->vcount == b->msg_count) || !c)
182  optional = 0;
183  break;
184 
185  case 'N':
186  if (!optional)
187  {
188  snprintf (fmt, sizeof (fmt), "%%%sd", prefix);
189  snprintf (dest, destlen, fmt, b->msg_unread);
190  }
191  else if (b->msg_unread == 0)
192  optional = 0;
193  break;
194 
195  case 'n':
196  if (!optional)
197  {
198  snprintf (fmt, sizeof (fmt), "%%%sc", prefix);
199  snprintf (dest, destlen, fmt, b->new ? 'N' : ' ');
200  }
201  else if (b->new == 0)
202  optional = 0;
203  break;
204 
205  case 'S':
206  if (!optional)
207  {
208  snprintf (fmt, sizeof (fmt), "%%%sd", prefix);
209  snprintf (dest, destlen, fmt, b->msg_count);
210  }
211  else if (b->msg_count == 0)
212  optional = 0;
213  break;
214 
215  case 't':
216  if (!optional)
217  {
218  snprintf (fmt, sizeof (fmt), "%%%sd", prefix);
219  snprintf (dest, destlen, fmt, c ? Context->tagged : 0);
220  }
221  else if ((c && Context->tagged == 0) || !c)
222  optional = 0;
223  break;
224 
225  case '!':
226  if (b->msg_flagged == 0)
227  mutt_format_s (dest, destlen, prefix, "");
228  else if (b->msg_flagged == 1)
229  mutt_format_s (dest, destlen, prefix, "!");
230  else if (b->msg_flagged == 2)
231  mutt_format_s (dest, destlen, prefix, "!!");
232  else
233  {
234  snprintf (fmt, sizeof (fmt), "%d!", b->msg_flagged);
235  mutt_format_s (dest, destlen, prefix, fmt);
236  }
237  break;
238  }
239 
240  if (optional)
241  mutt_FormatString (dest, destlen, col, SidebarWidth, ifstring, cb_format_str, (unsigned long) sbe, flags);
242  else if (flags & MUTT_FORMAT_OPTIONAL)
243  mutt_FormatString (dest, destlen, col, SidebarWidth, elsestring, cb_format_str, (unsigned long) sbe, flags);
244 
245  /* We return the format string, unchanged */
246  return src;
247 }

261 static void make_sidebar_entry (char *buf, unsigned int buflen, int width, char *box,
262  SBENTRY *sbe)
263 {
264  if (!buf || !box || !sbe)
265  return;
266 
267  strfcpy (sbe->box, box, sizeof (sbe->box));
268 
269  mutt_FormatString (buf, buflen, 0, width, NONULL(SidebarFormat), cb_format_str, (unsigned long) sbe, 0);
270 
271  /* Force string to be exactly the right width */
272  int w = mutt_strwidth (buf);
273  int s = mutt_strlen (buf);
274  width = MIN(buflen, width);
275  if (w < width)
276  {
277  /* Pad with spaces */
278  memset (buf + s, ' ', width - w);
279  buf[s + width - w] = 0;
280  }
281  else if (w > width)
282  {
283  /* Truncate to fit */
284  int len = mutt_wstr_trunc (buf, buflen, width, NULL);
285  buf[len] = 0;
286  }
287 }

299 static int cb_qsort_sbe (const void *a, const void *b)
300 {
301  const SBENTRY *sbe1 = *(const SBENTRY **) a;
302  const SBENTRY *sbe2 = *(const SBENTRY **) b;
303  BUFFY *b1 = sbe1->buffy;
304  BUFFY *b2 = sbe2->buffy;
305 
306  int result = 0;
307 
308  switch ((SidebarSortMethod & SORT_MASK))
309  {
310  case SORT_COUNT:
311  result = (b2->msg_count - b1->msg_count);
312  break;
313  case SORT_COUNT_NEW:
314  result = (b2->msg_unread - b1->msg_unread);
315  break;
316  case SORT_DESC:
317  result = mutt_strcmp (b1->desc, b2->desc);
318  break;
319  case SORT_FLAGGED:
320  result = (b2->msg_flagged - b1->msg_flagged);
321  break;
322  case SORT_PATH:
323  {
324  int same_path = mutt_same_path (b1->path, b2->path);
325  result = (same_path && mutt_is_inbox (b1->path)) ? -1 :
326  (same_path && mutt_is_inbox (b2->path)) ? 1 :
327  mutt_strcoll (b1->path, b2->path);
328  break;
329  }
330  }
331 
332  if (SidebarSortMethod & SORT_REVERSE)
333  result = -result;
334 
335  return result;
336 }

349 static void update_entries_visibility (void)
350 {
351  short new_only = option (OPTSIDEBARNEWMAILONLY);
352  SBENTRY *sbe;
353  int i;
354 
355  for (i = 0; i < EntryCount; i++)
356  {
357  sbe = Entries[i];
358 
359  sbe->is_hidden = 0;
360 
361  if (!new_only)
362  continue;
363 
364  if ((i == OpnIndex) || (sbe->buffy->msg_unread > 0) || sbe->buffy->new ||
365  (sbe->buffy->msg_flagged > 0))
366  continue;
367 
368  if (Context && (mutt_strcmp (sbe->buffy->realpath, Context->realpath) == 0))
369  /* Spool directory */
370  continue;
371 
372  if (mutt_find_list (SidebarWhitelist, sbe->buffy->path) ||
373  mutt_find_list (SidebarWhitelist, sbe->buffy->desc))
374  /* Explicitly asked to be visible */
375  continue;
376 
377  sbe->is_hidden = 1;
378  }
379 }

384 static void unsort_entries (void)
385 {
386  BUFFY *cur = get_incoming();
387  int i = 0, j;
388  SBENTRY *tmp;
389 
390  while (cur && (i < EntryCount))
391  {
392  j = i;
393  while ((j < EntryCount) &&
394  (Entries[j]->buffy != cur))
395  j++;
396  if (j < EntryCount)
397  {
398  if (j != i)
399  {
400  tmp = Entries[i];
401  Entries[i] = Entries[j];
402  Entries[j] = tmp;
403  }
404  i++;
405  }
406  cur = cur->next;
407  }
408 }

419 static void sort_entries (void)
420 {
421  short ssm = (SidebarSortMethod & SORT_MASK);
422 
423  /* These are the only sort methods we understand */
424  if ((ssm == SORT_COUNT) ||
425  (ssm == SORT_COUNT_NEW) ||
426  (ssm == SORT_FLAGGED) ||
427  (ssm == SORT_PATH))
428  qsort (Entries, EntryCount, sizeof (*Entries), cb_qsort_sbe);
429  else if ((ssm == SORT_ORDER) &&
430  (SidebarSortMethod != PreviousSort))
431  unsort_entries ();
432 }

448 static int prepare_sidebar (int page_size)
449 {
450  int i;
451  SBENTRY *opn_entry = NULL, *hil_entry = NULL;
452  int page_entries;
453 
454  if (!EntryCount || (page_size <= 0))
455  return 0;
456 
457  if (OpnIndex >= 0)
458  opn_entry = Entries[OpnIndex];
459  if (HilIndex >= 0)
460  hil_entry = Entries[HilIndex];
461 
462  update_entries_visibility ();
463  sort_entries ();
464 
465  for (i = 0; i < EntryCount; i++)
466  {
467  if (opn_entry == Entries[i])
468  OpnIndex = i;
469  if (hil_entry == Entries[i])
470  HilIndex = i;
471  }
472 
473  if ((HilIndex < 0) || Entries[HilIndex]->is_hidden ||
474  (SidebarSortMethod != PreviousSort))
475  {
476  if (OpnIndex >= 0)
477  HilIndex = OpnIndex;
478  else
479  {
480  HilIndex = 0;
481  if (Entries[HilIndex]->is_hidden)
482  select_next ();
483  }
484  }
485 
486  /* Set the Top and Bottom to frame the HilIndex in groups of page_size */
487 
488  /* If OPTSIDEBARNEMAILONLY is set, some entries may be hidden so we
489  * need to scan for the framing interval */
490  if (option (OPTSIDEBARNEWMAILONLY))
491  {
492  TopIndex = BotIndex = -1;
493  while (BotIndex < HilIndex)
494  {
495  TopIndex = BotIndex + 1;
496  page_entries = 0;
497  while (page_entries < page_size)
498  {
499  BotIndex++;
500  if (BotIndex >= EntryCount)
501  break;
502  if (! Entries[BotIndex]->is_hidden)
503  page_entries++;
504  }
505  }
506  }
507  /* Otherwise we can just calculate the interval */
508  else
509  {
510  TopIndex = (HilIndex / page_size) * page_size;
511  BotIndex = TopIndex + page_size - 1;
512  }
513 
514  if (BotIndex > (EntryCount - 1))
515  BotIndex = EntryCount - 1;
516 
517  PreviousSort = SidebarSortMethod;
518  return 1;
519 }

537 static int draw_divider (int num_rows, int num_cols)
538 {
539  if ((num_rows < 1) || (num_cols < 1))
540  return 0;
541 
542  int i;
543  int delim_len;
544  enum div_type altchar = SB_DIV_UTF8;
545 
546  /* Calculate the width of the delimiter in screen cells */
547  delim_len = mutt_strwidth (SidebarDividerChar);
548  if (delim_len < 0)
549  {
550  delim_len = 1; /* Bad character */
551  }
552  else if (delim_len == 0)
553  {
554  if (SidebarDividerChar)
555  return 0; /* User has set empty string */
556 
557  delim_len = 1; /* Unset variable */
558  }
559  else
560  {
561  altchar = SB_DIV_USER; /* User config */
562  }
563 
564  if (option (OPTASCIICHARS) && (altchar != SB_DIV_ASCII))
565  {
566  /* $ascii_chars overrides Unicode divider chars */
567  if (altchar == SB_DIV_UTF8)
568  {
569  altchar = SB_DIV_ASCII;
570  }
571  else if (SidebarDividerChar)
572  {
573  for (i = 0; i < delim_len; i++)
574  {
575  if (SidebarDividerChar[i] & ~0x7F) /* high-bit is set */
576  {
577  altchar = SB_DIV_ASCII;
578  delim_len = 1;
579  break;
580  }
581  }
582  }
583  }
584 
585  if (delim_len > num_cols)
586  return 0;
587 
588  SETCOLOR(MT_COLOR_DIVIDER);
589 
590  int col = option (OPTSIDEBARONRIGHT) ? 0 : (SidebarWidth - delim_len);
591 
592  for (i = 0; i < num_rows; i++)
593  {
594  mutt_window_move (MuttSidebarWindow, i, col);
595 
596  switch (altchar)
597  {
598  case SB_DIV_USER:
599  addstr (NONULL(SidebarDividerChar));
600  break;
601  case SB_DIV_ASCII:
602  addch ('|');
603  break;
604  case SB_DIV_UTF8:
605  addch (ACS_VLINE);
606  break;
607  }
608  }
609 
610  return delim_len;
611 }

662 static void draw_sidebar (int num_rows, int num_cols, int div_width)
663 {
664  int entryidx;
665  SBENTRY *entry;
666  BUFFY *b;
667  if (TopIndex < 0)
668  return;
669 
670  int w = MIN(num_cols, (SidebarWidth - div_width));
671  int row = 0;
672  for (entryidx = TopIndex; (entryidx < EntryCount) && (row < num_rows); entryidx++)
673  {
674  entry = Entries[entryidx];
675  if (entry->is_hidden)
676  continue;
677  b = entry->buffy;
678 
679  if (entryidx == OpnIndex)
680  {
681  if ((ColorDefs[MT_COLOR_SB_INDICATOR] != 0))
682  SETCOLOR(MT_COLOR_SB_INDICATOR);
683  else
684  SETCOLOR(MT_COLOR_INDICATOR);
685  }
686  else if (entryidx == HilIndex)
687  SETCOLOR(MT_COLOR_HIGHLIGHT);
688  else if ((b->msg_unread > 0) || (b->new))
689  SETCOLOR(MT_COLOR_NEW);
690  else if (b->msg_flagged > 0)
691  SETCOLOR(MT_COLOR_FLAGGED);
692  else if ((ColorDefs[MT_COLOR_SB_SPOOLFILE] != 0) &&
693  (mutt_strcmp (b->path, Spoolfile) == 0))
694  SETCOLOR(MT_COLOR_SB_SPOOLFILE);
695  else
696  {
697  if (ColorDefs[MT_COLOR_ORDINARY] != 0)
698  SETCOLOR(MT_COLOR_ORDINARY);
699  else
700  SETCOLOR(MT_COLOR_NORMAL);
701  }
702 
703  int col = 0;
704  if (option (OPTSIDEBARONRIGHT))
705  col = div_width;
706 
707  mutt_window_move (MuttSidebarWindow, row, col);
708  if (Context && Context->realpath &&
709  !mutt_strcmp (b->realpath, Context->realpath))
710  {
711 #ifdef USE_NOTMUCH
712  if (b->magic == MUTT_NOTMUCH)
713  nm_nonctx_get_count(b->realpath, &b->msg_count, &b->msg_unread);
714  else
715 #endif
716  {
717  b->msg_unread = Context->unread;
718  b->msg_count = Context->msgcount;
719  }
720  b->msg_flagged = Context->flagged;
721  }
722 
723  /* compute length of Maildir without trailing separator */
724  size_t maildirlen = mutt_strlen (Maildir);
725  if (maildirlen &&
726  SidebarDelimChars &&
727  strchr (SidebarDelimChars, Maildir[maildirlen - 1]))
728  maildirlen--;
729 
730  /* check whether Maildir is a prefix of the current folder's path */
731  short maildir_is_prefix = 0;
732  if ((mutt_strlen (b->path) > maildirlen) &&
733  (mutt_strncmp (Maildir, b->path, maildirlen) == 0) &&
734  SidebarDelimChars &&
735  strchr (SidebarDelimChars, b->path[maildirlen]))
736  maildir_is_prefix = 1;
737 
738  /* calculate depth of current folder and generate its display name with indented spaces */
739  int sidebar_folder_depth = 0;
740  char *sidebar_folder_name;
741  int i;
742  if (option (OPTSIDEBARSHORTPATH))
743  {
744  /* disregard a trailing separator, so strlen() - 2 */
745  sidebar_folder_name = b->path;
746  for (i = mutt_strlen (sidebar_folder_name) - 2; i >= 0; i--)
747  {
748  if (SidebarDelimChars &&
749  strchr (SidebarDelimChars, sidebar_folder_name[i]))
750  {
751  sidebar_folder_name += (i + 1);
752  break;
753  }
754  }
755  }
756  else
757  sidebar_folder_name = b->path + maildir_is_prefix * (maildirlen + 1);
758 
759  if (maildir_is_prefix && option (OPTSIDEBARFOLDERINDENT))
760  {
761  const char *tmp_folder_name;
762  int lastsep = 0;
763  tmp_folder_name = b->path + maildirlen + 1;
764  int tmplen = (int) mutt_strlen (tmp_folder_name) - 1;
765  for (i = 0; i < tmplen; i++)
766  {
767  if (SidebarDelimChars && strchr (SidebarDelimChars, tmp_folder_name[i]))
768  {
769  sidebar_folder_depth++;
770  lastsep = i + 1;
771  }
772  }
773  if (sidebar_folder_depth > 0)
774  {
775  if (option (OPTSIDEBARSHORTPATH))
776  tmp_folder_name += lastsep; /* basename */
777  int sfn_len = mutt_strlen (tmp_folder_name) +
778  sidebar_folder_depth*mutt_strlen (SidebarIndentString) + 1;
779  sidebar_folder_name = safe_malloc (sfn_len);
780  sidebar_folder_name[0]=0;
781  for (i=0; i < sidebar_folder_depth; i++)
782  safe_strcat (sidebar_folder_name, sfn_len, NONULL(SidebarIndentString));
783  safe_strcat (sidebar_folder_name, sfn_len, tmp_folder_name);
784  }
785  }
786 #ifdef USE_NOTMUCH
787  else if (b->magic == MUTT_NOTMUCH)
788  {
789  sidebar_folder_name = b->desc;
790  }
791 #endif
792  char str[STRING];
793  make_sidebar_entry (str, sizeof (str), w, sidebar_folder_name, entry);
794  printw ("%s", str);
795  if (sidebar_folder_depth > 0)
796  FREE (&sidebar_folder_name);
797  row++;
798  }
799 
800  fill_empty_space (row, num_rows - row, div_width, w);
801 }

810 void mutt_sb_draw (void)
811 {
812  if (!option (OPTSIDEBAR))
813  return;
814 
815 #ifdef USE_SLANG_CURSES
816  int x = SLsmg_get_column();
817  int y = SLsmg_get_row();
818 #else
819  int x = getcurx (stdscr);
820  int y = getcury (stdscr);
821 #endif
822 
823  int num_rows = MuttSidebarWindow->rows;
824  int num_cols = MuttSidebarWindow->cols;
825 
826  int div_width = draw_divider (num_rows, num_cols);
827 
828  BUFFY *b;
829  if (Entries == NULL)
830  for (b = get_incoming(); b; b = b->next)
831  mutt_sb_notify_mailbox (b, 1);
832 
833  if (!get_incoming())
834  {
835  fill_empty_space (0, num_rows, div_width, num_cols - div_width);
836  return;
837  }
838 
839  if (!prepare_sidebar (num_rows))
840  return;
841 
842  draw_sidebar (num_rows, num_cols, div_width);
843  move (y, x);
844 }

854 {

857  if (!EntryCount || HilIndex < 0)

880 static int select_next_new (void)
881 {
882  int entry = HilIndex;
883 
884  if (!EntryCount || HilIndex < 0)
885  return 0;
886 
887  do
888  {
889  entry++;
890  if (entry == EntryCount)
891  {
892  if (option (OPTSIDEBARNEXTNEWWRAP))
893  entry = 0;
894  else
895  return 0;
