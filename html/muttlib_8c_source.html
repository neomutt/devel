85 {
86  char prefix[_POSIX_PATH_MAX];

88  struct stat sb;
89 
90  if (s[0] == '\0')
91  {
92  mutt_mktemp (s, l);

94  else
95  {
96  strfcpy (prefix, s, sizeof (prefix));
97  mutt_sanitize_filename (prefix, 1);
98  snprintf (s, l, "%s/%s", NONULL (Tempdir), prefix);
99  if (lstat (s, &sb) == -1 && errno == ENOENT)
100  return;
101 
102  if ((suffix = strrchr (prefix, '.')) != NULL)
103  {
104  *suffix = 0;
105  ++suffix;
106  }
107  mutt_mktemp_pfx_sfx (s, l, prefix, suffix);
108  }

113 int mutt_copy_body (FILE *fp, BODY **tgt, BODY *src)
114 {
115  char tmp[_POSIX_PATH_MAX];
116  BODY *b;
117 
118  PARAMETER *par, **ppar;
119 
120  short use_disp;
121 
122  if (src->filename)
123  {
124  use_disp = 1;
125  strfcpy (tmp, src->filename, sizeof (tmp));
126  }
127  else
128  {
129  use_disp = 0;
130  tmp[0] = '\0';
131  }
132 
133  mutt_adv_mktemp (tmp, sizeof (tmp));
134  if (mutt_save_attachment (fp, src, tmp, 0, NULL) == -1)
135  return -1;
136 
137  *tgt = mutt_new_body ();
138  b = *tgt;
139 
140  memcpy (b, src, sizeof (BODY));
141  b->parts = NULL;
142  b->next = NULL;
143 
144  b->filename = safe_strdup (tmp);
145  b->use_disp = use_disp;
146  b->unlink = 1;
147 
148  if (mutt_is_text_part (b))
149  b->noconv = 1;
150 
151  b->xtype = safe_strdup (b->xtype);
152  b->subtype = safe_strdup (b->subtype);
153  b->form_name = safe_strdup (b->form_name);
154  b->d_filename = safe_strdup (b->d_filename);
155  /* mutt_adv_mktemp() will mangle the filename in tmp,
156  * so preserve it in d_filename */
157  if (!b->d_filename && use_disp)
158  b->d_filename = safe_strdup (src->filename);
159  b->description = safe_strdup (b->description);
160 
161  /*
162  * we don't seem to need the HEADER structure currently.
163  * XXX - this may change in the future
164  */
165 
166  if (b->hdr) b->hdr = NULL;
167 
168  /* copy parameters */
169  for (par = b->parameter, ppar = &b->parameter; par; ppar = &(*ppar)->next, par = par->next)
170  {
171  *ppar = mutt_new_parameter ();
172  (*ppar)->attribute = safe_strdup (par->attribute);
173  (*ppar)->value = safe_strdup (par->value);
174  }
175 
176  mutt_stamp_attachment (b);
177 
178  return 0;
179 }

431 char *_mutt_expand_path (char *s, size_t slen, int rx)
432 {
433  char p[_POSIX_PATH_MAX] = "";
434  char q[_POSIX_PATH_MAX] = "";
435  char tmp[_POSIX_PATH_MAX];
436  char *t;
437 
438  char *tail = "";
439 
440  int recurse = 0;
441 
442  do
443  {
444  recurse = 0;
445 
446  switch (*s)
447  {
448  case '~':
449  {
450  if (*(s + 1) == '/' || *(s + 1) == 0)
451  {
452  strfcpy (p, NONULL(Homedir), sizeof (p));
453  tail = s + 1;
454  }
455  else
456  {
457  struct passwd *pw;
458  if ((t = strchr (s + 1, '/')))
459  *t = 0;
460 
461  if ((pw = getpwnam (s + 1)))
462  {
463  strfcpy (p, pw->pw_dir, sizeof (p));
464  if (t)
465  {
466  *t = '/';
467  tail = t;
468  }
469  else
470  tail = "";
471  }
472  else
473  {
474  /* user not found! */
475  if (t)
476  *t = '/';
477  *p = '\0';
478  tail = s;
479  }
480  }
481  }
482  break;
483 
484  case '=':
485  case '+':
486  {
487 #ifdef USE_IMAP
488  /* if folder = {host} or imap[s]://host/: don't append slash */
489  if (mx_is_imap (NONULL (Maildir)) &&
490  (Maildir[strlen (Maildir) - 1] == '}' ||
491  Maildir[strlen (Maildir) - 1] == '/'))
492  strfcpy (p, NONULL (Maildir), sizeof (p));
493  else
494 #endif
495 #ifdef USE_NOTMUCH
496  if (mx_is_notmuch (NONULL (Maildir)))
497  strfcpy (p, NONULL (Maildir), sizeof (p));
498  else
499 #endif
500  if (Maildir && *Maildir && Maildir[strlen (Maildir) - 1] == '/')
501  strfcpy (p, NONULL (Maildir), sizeof (p));
502  else
503  snprintf (p, sizeof (p), "%s/", NONULL (Maildir));
504 
505  tail = s + 1;
506  }
507  break;
508 
509  /* elm compatibility, @ expands alias to user name */
510 
511  case '@':
512  {
513  HEADER *h;
514  ADDRESS *alias;
515 
516  if ((alias = mutt_lookup_alias (s + 1)))
517  {
518  h = mutt_new_header();
519  h->env = mutt_new_envelope();
520  h->env->from = h->env->to = alias;
521  mutt_default_save (p, sizeof (p), h);
522  h->env->from = h->env->to = NULL;
523  mutt_free_header (&h);
524  /* Avoid infinite recursion if the resulting folder starts with '@' */
525  if (*p != '@')
526  recurse = 1;
527 
528  tail = "";
529  }
530  }
531  break;
532 
533  case '>':
534  {
535  strfcpy (p, NONULL(Inbox), sizeof (p));
536  tail = s + 1;
537  }
538  break;
539 
540  case '<':
541  {
542  strfcpy (p, NONULL(Outbox), sizeof (p));
543  tail = s + 1;
544  }
545  break;
546 
547  case '!':
548  {
549  if (*(s+1) == '!')
550  {
551  strfcpy (p, NONULL(LastFolder), sizeof (p));
552  tail = s + 2;
553  }
554  else
555  {
556  strfcpy (p, NONULL(Spoolfile), sizeof (p));
557  tail = s + 1;
558  }
559  }
560  break;
561 
562  case '-':
563  {
564  strfcpy (p, NONULL(LastFolder), sizeof (p));
565  tail = s + 1;
566  }
567  break;
568 
569  case '^':
570  {
571  strfcpy (p, NONULL(CurrentFolder), sizeof (p));
572  tail = s + 1;
573  }
574  break;
575 
576  default:
577  {
578  *p = '\0';
579  tail = s;
580  }
581  }
582 
583  if (rx && *p && !recurse)
584  {
585  mutt_rx_sanitize_string (q, sizeof (q), p);
586  snprintf (tmp, sizeof (tmp), "%s%s", q, tail);
587  }
588  else
589  snprintf (tmp, sizeof (tmp), "%s%s", p, tail);
590 
591  strfcpy (s, tmp, slen);
592  }
593  while (recurse);
594 
595 #ifdef USE_IMAP
596  /* Rewrite IMAP path in canonical form - aids in string comparisons of
597  * folders. May possibly fail, in which case s should be the same. */
598  if (mx_is_imap (s))
599  imap_expand_path (s, slen);
600 #endif
601 
602  return (s);
603 }

605 /* Extract the real name from /etc/passwd's GECOS field.
606  * When set, honor the regular expression in GecosMask,
607  * otherwise assume that the GECOS field is a
608  * comma-separated list.
609  * Replace "&" by a capitalized version of the user's login
610  * name.
611  */
612 
613 char *mutt_gecos_name (char *dest, size_t destlen, struct passwd *pw)
614 {
615  regmatch_t pat_match[1];
616  size_t pwnl;
617  int idx;
618  char *p;
619 
620  if (!pw || !pw->pw_gecos)
621  return NULL;
622 
623  memset (dest, 0, destlen);
624 
625  if (GecosMask.rx)
626  {
627  if (regexec (GecosMask.rx, pw->pw_gecos, 1, pat_match, 0) == 0)
628  strfcpy (dest, pw->pw_gecos + pat_match[0].rm_so,
629  MIN (pat_match[0].rm_eo - pat_match[0].rm_so + 1, destlen));
630  }
631  else if ((p = strchr (pw->pw_gecos, ',')))
632  strfcpy (dest, pw->pw_gecos, MIN (destlen, p - pw->pw_gecos + 1));
633  else
634  strfcpy (dest, pw->pw_gecos, destlen);
635 
636  pwnl = strlen (pw->pw_name);
637 
638  for (idx = 0; dest[idx]; idx++)
639  {
640  if (dest[idx] == '&')
641  {
642  memmove (&dest[idx + pwnl], &dest[idx + 1],
643  MAX((ssize_t)(destlen - idx - pwnl - 1), 0));
644  memcpy (&dest[idx], pw->pw_name, MIN(destlen - idx - 1, pwnl));
645  dest[idx] = toupper ((unsigned char) dest[idx]);
646  }
647  }
648 
649  return dest;
650 }
