130 static int tls_socket_read (CONNECTION* conn, char* buf, size_t len)
131 {
132  tlssockdata *data = conn->sockdata;
133  int ret;
134 
135  if (!data)
136  {
137  mutt_error (_("Error: no TLS socket open"));
138  mutt_sleep (2);
139  return -1;
140  }
141 
142  do {
143  ret = gnutls_record_recv (data->state, buf, len);
144  if ((ret < 0 &&
145  gnutls_error_is_fatal(ret) == 1) ||
146  ret == GNUTLS_E_INTERRUPTED
147  )
148  {
149  mutt_error ("tls_socket_read (%s)", gnutls_strerror (ret));
150  mutt_sleep (2);
151  return -1;
152  }
153  } while (ret == GNUTLS_E_AGAIN);
154 
155  return ret;
156 }

158 static int tls_socket_write (CONNECTION* conn, const char* buf, size_t len)
159 {
160  tlssockdata *data = conn->sockdata;
161  int ret;
162  size_t sent = 0;
163 
164  if (!data)
165  {
166  mutt_error (_("Error: no TLS socket open"));
167  mutt_sleep (2);
168  return -1;
169  }
170 
171  do
172  {
173  ret = gnutls_record_send (data->state, buf + sent, len - sent);
174  if (ret < 0)
175  {
176  if (gnutls_error_is_fatal(ret) == 1 || ret == GNUTLS_E_INTERRUPTED)
177  {
178  mutt_error ("tls_socket_write (%s)", gnutls_strerror (ret));
179  mutt_sleep (4);
180  return -1;
181  }
182  return ret;
183  }
184  sent += ret;
185  } while (sent < len);
186 
187  return sent;
188 }

219 static void tls_get_client_cert (CONNECTION* conn)
220 {
221  tlssockdata *data = conn->sockdata;
222  const gnutls_datum_t* crtdata;
223  gnutls_x509_crt_t clientcrt;
224  char* dn;
225  char* cn;
226  char* cnend;
227  size_t dnlen;
228 
229  /* get our cert CN if we have one */
230  if (!(crtdata = gnutls_certificate_get_ours (data->state)))
231  return;
232 
233  if (gnutls_x509_crt_init (&clientcrt) < 0)
234  {
235  dprint (1, (debugfile, "Failed to init gnutls crt\n"));
236  return;
237  }
238  if (gnutls_x509_crt_import (clientcrt, crtdata, GNUTLS_X509_FMT_DER) < 0)
239  {
240  dprint (1, (debugfile, "Failed to import gnutls client crt\n"));
241  goto err_crt;
242  }
243  /* get length of DN */
244  dnlen = 0;
245  gnutls_x509_crt_get_dn (clientcrt, NULL, &dnlen);
246  if (!(dn = calloc (1, dnlen)))
247  {
248  dprint (1, (debugfile, "could not allocate DN\n"));
249  goto err_crt;
250  }
251  gnutls_x509_crt_get_dn (clientcrt, dn, &dnlen);
252  dprint (2, (debugfile, "client certificate DN: %s\n", dn));
253 
254  /* extract CN to use as external user name */
255  if (!(cn = strstr (dn, "CN=")))
256  {
257  dprint (1, (debugfile, "no CN found in DN\n"));
258  goto err_dn;
259  }
260  cn += 3;
261 
262  if ((cnend = strstr (dn, ",EMAIL=")))
263  *cnend = '\0';
264 
265  /* if we are using a client cert, SASL may expect an external auth name */
266  mutt_account_getuser (&conn->account);
267 
268 err_dn:
269  FREE (&dn);
270 err_crt:
271  gnutls_x509_crt_deinit (clientcrt);
272 }

371 /* tls_negotiate: After TLS state has been initialized, attempt to negotiate
372  * TLS over the wire, including certificate checks. */
373 static int tls_negotiate (CONNECTION * conn)
374 {
375  tlssockdata *data;
376  int err;
377 
378  data = (tlssockdata *) safe_calloc (1, sizeof (tlssockdata));
379  conn->sockdata = data;
380  err = gnutls_certificate_allocate_credentials (&data->xcred);
381  if (err < 0)
382  {
383  FREE(&conn->sockdata);
384  mutt_error ("gnutls_certificate_allocate_credentials: %s", gnutls_strerror(err));
385  mutt_sleep (2);
386  return -1;
387  }
388 
389  gnutls_certificate_set_x509_trust_file (data->xcred, SslCertFile,
390  GNUTLS_X509_FMT_PEM);
391  /* ignore errors, maybe file doesn't exist yet */
392 
393  if (SslCACertFile)
394  {
395  gnutls_certificate_set_x509_trust_file (data->xcred, SslCACertFile,
396  GNUTLS_X509_FMT_PEM);
397  }
398 
399  if (SslClientCert)
400  {
401  dprint (2, (debugfile, "Using client certificate %s\n", SslClientCert));
402  gnutls_certificate_set_x509_key_file (data->xcred, SslClientCert,
403  SslClientCert, GNUTLS_X509_FMT_PEM);
404  }
405 
406 #if HAVE_DECL_GNUTLS_VERIFY_DISABLE_TIME_CHECKS
407  /* disable checking certificate activation/expiration times
408  in gnutls, we do the checks ourselves */
409  gnutls_certificate_set_verify_flags(data->xcred, GNUTLS_VERIFY_DISABLE_TIME_CHECKS);
410 #endif
411 
412  if ((err = gnutls_init(&data->state, GNUTLS_CLIENT)))
413  {
414  mutt_error ("gnutls_handshake: %s", gnutls_strerror(err));
415  mutt_sleep (2);
416  goto fail;
417  }
418 
419  /* set socket */
420  gnutls_transport_set_ptr (data->state, (gnutls_transport_ptr_t)(long)conn->fd);
421 
422  if (tls_set_priority(data) < 0) {
423  goto fail;
424  }
425 
426  if (SslDHPrimeBits > 0)
427  {
428  gnutls_dh_set_prime_bits (data->state, SslDHPrimeBits);
429  }
430 
431 /*
432  gnutls_set_cred (data->state, GNUTLS_ANON, NULL);
433 */
434 
435  gnutls_credentials_set (data->state, GNUTLS_CRD_CERTIFICATE, data->xcred);
436 
437  err = gnutls_handshake(data->state);
438 
439  while (err == GNUTLS_E_AGAIN)
440  {
441  err = gnutls_handshake(data->state);
442  }
443  if (err < 0) {
444  if (err == GNUTLS_E_FATAL_ALERT_RECEIVED)
445  {
446  mutt_error("gnutls_handshake: %s(%s)", gnutls_strerror(err),
447  gnutls_alert_get_name(gnutls_alert_get(data->state)));
448  }
449  else
450  {
451  mutt_error("gnutls_handshake: %s", gnutls_strerror(err));
452  }
453  mutt_sleep (2);
454  goto fail;
455  }
456 
457  if (!tls_check_certificate(conn))
458  goto fail;
459 
460  /* set Security Strength Factor (SSF) for SASL */
461  /* NB: gnutls_cipher_get_key_size() returns key length in bytes */
462  conn->ssf = gnutls_cipher_get_key_size (gnutls_cipher_get (data->state)) * 8;
463 
464  tls_get_client_cert (conn);
465 
466  if (!option(OPTNOCURSES)) {
467  mutt_message (_("SSL/TLS connection using %s (%s/%s/%s)"),
468  gnutls_protocol_get_name (gnutls_protocol_get_version (data->state)),
469  gnutls_kx_get_name (gnutls_kx_get (data->state)),
470  gnutls_cipher_get_name (gnutls_cipher_get (data->state)),
471  gnutls_mac_get_name (gnutls_mac_get (data->state)));
472  mutt_sleep (0);
473  }
474 
475  return 0;
476 
477  fail:
478  gnutls_certificate_free_credentials (data->xcred);
479  gnutls_deinit (data->state);
480  FREE(&conn->sockdata);
481  return -1;
482 }

484 static int tls_socket_close (CONNECTION* conn)
485 {
486  tlssockdata *data = conn->sockdata;
487  if (data)
488  {
489  /* shut down only the write half to avoid hanging waiting for the remote to respond.
490  *
491  * RFC5246 7.2.1. "Closure Alerts"
492  *
493  * It is not required for the initiator of the close to wait for the
494  * responding close_notify alert before closing the read side of the
495  * connection.
496  */
497  gnutls_bye (data->state, GNUTLS_SHUT_WR);
498 
499  gnutls_certificate_free_credentials (data->xcred);
500  gnutls_deinit (data->state);
501  FREE (&conn->sockdata);
502  }
503 
504  return raw_socket_close (conn);
505 }

521 /* this bit is based on read_ca_file() in gnutls */
522 static int tls_compare_certificates (const gnutls_datum_t *peercert)
523 {
524  gnutls_datum_t cert;
525  unsigned char *ptr;
526  FILE *fd1;
527  int ret;
528  gnutls_datum_t b64_data;
529  unsigned char *b64_data_data;
530  struct stat filestat;
531 
532  if (stat(SslCertFile, &filestat) == -1)
533  return 0;
534 
535  b64_data.size = filestat.st_size+1;
536  b64_data_data = (unsigned char *) safe_calloc (1, b64_data.size);
537  b64_data_data[b64_data.size-1] = '\0';
538  b64_data.data = b64_data_data;
539 
540  fd1 = fopen(SslCertFile, "r");
541  if (fd1 == NULL) {
542  return 0;
543  }
544 
545  b64_data.size = fread(b64_data.data, 1, b64_data.size, fd1);
546  safe_fclose (&fd1);
547 
548  do {
549  ret = gnutls_pem_base64_decode_alloc(NULL, &b64_data, &cert);
550  if (ret != 0)
551  {
552  FREE (&b64_data_data);
553  return 0;
554  }
555 
556  /* find start of cert, skipping junk */
557  ptr = (unsigned char *)strstr((char*)b64_data.data, CERT_SEP);
558  if (!ptr)
559  {
560  gnutls_free(cert.data);
561  FREE (&b64_data_data);
562  return 0;
563  }
564  /* find start of next cert */
565  ptr = (unsigned char *)strstr((char*)ptr + 1, CERT_SEP);
566 
567  b64_data.size = b64_data.size - (ptr - b64_data.data);
568  b64_data.data = ptr;
569 
570  if (cert.size == peercert->size)
571  {
572  if (memcmp (cert.data, peercert->data, cert.size) == 0)
573  {
574  /* match found */
575  gnutls_free(cert.data);
576  FREE (&b64_data_data);
577  return 1;
578  }
579  }
580 
581  gnutls_free(cert.data);
582  } while (ptr != NULL);
583 
584  /* no match found */
585  FREE (&b64_data_data);
586  return 0;
587 }

589 static void tls_fingerprint (gnutls_digest_algorithm_t algo,
590  char* s, int l, const gnutls_datum_t* data)
591 {
592  unsigned char md[36];
593  size_t n;
594  int j;
595 
596  n = 36;
597 
598  if (gnutls_fingerprint (algo, data, (char *)md, &n) < 0)
599  {
600  snprintf (s, l, _("[unable to calculate]"));
601  }
602  else
603  {
604  for (j = 0; j < (int) n; j++)
605  {
606  char ch[8];
607  snprintf (ch, 8, "%02X%s", md[j], (j % 2 ? " " : ""));
608  safe_strcat (s, l, ch);
609  }
610  s[2*n+n/2-1] = '\0'; /* don't want trailing space */
611  }
612 }

628 static int tls_check_stored_hostname (const gnutls_datum_t *cert,
629  const char *hostname)
630 {
631  char buf[80];
632  FILE *fp;
633  char *linestr = NULL;
634  size_t linestrsize;
635  int linenum = 0;
636  regex_t preg;
637  regmatch_t pmatch[3];
638 
639  /* try checking against names stored in stored certs file */
640  if ((fp = fopen (SslCertFile, "r")))
641  {
642  if (REGCOMP(&preg, "^#H ([a-zA-Z0-9_\\.-]+) ([0-9A-F]{4}( [0-9A-F]{4}){7})[ \t]*$",
643  REG_ICASE) != 0)
644  {
645  return 0;
646  }
647 
648  buf[0] = '\0';
649  tls_fingerprint (GNUTLS_DIG_MD5, buf, sizeof (buf), cert);
650  while ((linestr = mutt_read_line(linestr, &linestrsize, fp, &linenum, 0)) != NULL)
651  {
652  if(linestr[0] == '#' && linestr[1] == 'H')
653  {
654  if (regexec(&preg, linestr, 3, pmatch, 0) == 0)
655  {
656  linestr[pmatch[1].rm_eo] = '\0';
657  linestr[pmatch[2].rm_eo] = '\0';
658  if (strcmp(linestr + pmatch[1].rm_so, hostname) == 0 &&
659  strcmp(linestr + pmatch[2].rm_so, buf) == 0)
660  {
661  regfree(&preg);
662  FREE(&linestr);
663  safe_fclose (&fp);
664  return 1;
665  }
666  }
667  }
668  }
669 
670  regfree(&preg);
671  safe_fclose (&fp);
672  }
673 
674  /* not found a matching name */
675  return 0;
676 }

678 static int tls_check_preauth (const gnutls_datum_t *certdata,
679  gnutls_certificate_status_t certstat,
680  const char *hostname, int chainidx, int* certerr,

1104 static int tls_check_certificate (CONNECTION* conn)
1105 {
1106  tlssockdata *data = conn->sockdata;
1107  gnutls_session_t state = data->state;
1108  const gnutls_datum_t *cert_list;
1109  unsigned int cert_list_size = 0;
1110  gnutls_certificate_status_t certstat;
1111  int certerr, i, preauthrc, savedcert, rc = 0;
1112  int rcpeer = -1; /* the result of tls_check_preauth() on the peer's EE cert */
1113 
1114  if (gnutls_auth_get_type (state) != GNUTLS_CRD_CERTIFICATE)
1115  {
1116  mutt_error (_("Unable to get certificate from peer"));
1117  mutt_sleep (2);
1118  return 0;
1119  }
1120 
1121  certstat = tls_verify_peers (state);
1122 
1123  cert_list = gnutls_certificate_get_peers (state, &cert_list_size);
1124  if (!cert_list)
1125  {
1126  mutt_error (_("Unable to get certificate from peer"));
1127  mutt_sleep (2);
1128  return 0;
1129  }
1130 
1131  /* tls_verify_peers doesn't check hostname or expiration, so walk
1132  * from most specific to least checking these. If we see a saved certificate,
1133  * its status short-circuits the remaining checks. */
1134  preauthrc = 0;
1135  for (i = 0; i < cert_list_size; i++) {
1136  rc = tls_check_preauth(&cert_list[i], certstat, conn->account.host, i,
1137  &certerr, &savedcert);
1138  preauthrc += rc;
1139  if (i == 0)
1140  {
1141  /* This is the peer's end-entity X.509 certificate. Stash the result
1142  * to check later in this function.
1143  */
1144  rcpeer = rc;
1145  }
1146 
1147  if (savedcert)
1148  {
1149  if (!preauthrc)
1150  return 1;
1151  else
1152  break;
1153  }
1154  }
1155 
1156  /* then check interactively, starting from chain root */
1157  for (i = cert_list_size - 1; i >= 0; i--)
1158  {
1159  rc = tls_check_one_certificate (&cert_list[i], certstat, conn->account.host,
1160  i, cert_list_size);
1161 
1162  /* add signers to trust set, then reverify */
1163  if (i && rc) {
1164  rc = gnutls_certificate_set_x509_trust_mem (data->xcred, &cert_list[i],
1165  GNUTLS_X509_FMT_DER);
1166  if (rc != 1)
1167  dprint (1, (debugfile, "error trusting certificate %d: %d\n", i, rc));
1168 
1169  certstat = tls_verify_peers (state);
1170  /* If the cert chain now verifies, and the peer's cert was otherwise
1171  * valid (rcpeer==0), we are done.
1172  */
1173  if (!certstat && !rcpeer)
1174  return 1;
1175  }
1176  }
1177 
1178  return rc;
1179 }
