65 size_t mutt_to_base64 (char *out, const char *cin, size_t len, size_t olen)
66 {
67  unsigned char *begin = (unsigned char *)out;
68  const unsigned char *in = (const unsigned char *)cin;
69  while (len >= 3 && olen > 10)
70  {
71  *out++ = B64Chars[in[0] >> 2];
72  *out++ = B64Chars[((in[0] << 4) & 0x30) | (in[1] >> 4)];
73  *out++ = B64Chars[((in[1] << 2) & 0x3c) | (in[2] >> 6)];
74  *out++ = B64Chars[in[2] & 0x3f];
75  olen -= 4;
76  len -= 3;
77  in += 3;
78  }
79 
80  /* clean up remainder */
81  if (len > 0 && olen > 4)
82  {
83  unsigned char fragment;
84 
85  *out++ = B64Chars[in[0] >> 2];
86  fragment = (in[0] << 4) & 0x30;
87  if (len > 1)
88  fragment |= in[1] >> 4;
89  *out++ = B64Chars[fragment];
90  *out++ = (len < 2) ? '=' : B64Chars[(in[1] << 2) & 0x3c];
91  *out++ = '=';
92  }
93  *out = '\0';
94  return out - (char *)begin;
95 }

108 int mutt_from_base64 (char *out, const char *in)
109 {
110  int len = 0;
111  register unsigned char digit1, digit2, digit3, digit4;
112 
113  do
114  {
115  digit1 = in[0];
116  if (digit1 > 127 || base64val (digit1) == BAD)
117  return -1;
118  digit2 = in[1];
119  if (digit2 > 127 || base64val (digit2) == BAD)
120  return -1;
121  digit3 = in[2];
122  if (digit3 > 127 || ((digit3 != '=') && (base64val (digit3) == BAD)))
123  return -1;
124  digit4 = in[3];
125  if (digit4 > 127 || ((digit4 != '=') && (base64val (digit4) == BAD)))
126  return -1;
127  in += 4;
128 
129  /* digits are already sanity-checked */
130  *out++ = (base64val(digit1) << 2) | (base64val(digit2) >> 4);
131  len++;
132  if (digit3 != '=')
133  {
134  *out++ = ((base64val(digit2) << 4) & 0xf0) | (base64val(digit3) >> 2);
135  len++;
136  if (digit4 != '=')
137  {
138  *out++ = ((base64val(digit3) << 6) & 0xc0) | base64val(digit4);
139  len++;
140  }
141  }
142  }
143  while (*in && digit4 != '=');
144 
145  return len;
146 }
