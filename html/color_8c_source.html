233 int mutt_alloc_color (int fg, int bg)
234 {
235  COLOR_LIST *p = ColorList;
236  int i;
237 
238 #if defined (USE_SLANG_CURSES)
239  char fgc[SHORT_STRING], bgc[SHORT_STRING];
240 #endif
241 
242  /* check to see if this color is already allocated to save space */
243  while (p)
244  {
245  if (p->fg == fg && p->bg == bg)
246  {
247  (p->count)++;
248  return (COLOR_PAIR (p->index));
249  }
250  p = p->next;
251  }
252 
253  /* check to see if there are colors left */
254  if (++UserColors > COLOR_PAIRS) return (A_NORMAL);
255 
256  /* find the smallest available index (object) */
257  i = 1;
258  FOREVER
259  {
260  p = ColorList;
261  while (p)
262  {
263  if (p->index == i) break;
264  p = p->next;
265  }
266  if (p == NULL) break;
267  i++;
268  }
269 
270  p = (COLOR_LIST *) safe_malloc (sizeof (COLOR_LIST));
271  p->next = ColorList;
272  ColorList = p;
273 
274  p->index = i;
275  p->count = 1;
276  p->bg = bg;
277  p->fg = fg;
278 
279 #if defined (USE_SLANG_CURSES)
280  if (fg == COLOR_DEFAULT || bg == COLOR_DEFAULT)
281  SLtt_set_color (i, NULL, get_color_name (fgc, sizeof (fgc), fg), get_color_name (bgc, sizeof (bgc), bg));
282  else
283 #elif defined (HAVE_USE_DEFAULT_COLORS)
284  if (fg == COLOR_DEFAULT)
285  fg = -1;
286  if (bg == COLOR_DEFAULT)
287  bg = -1;
288 #endif
289 
290  init_pair(i, fg, bg);
291 
292  dprint (3, (debugfile,"mutt_alloc_color(): Color pairs used so far: %d\n",
293  UserColors));
294 
295  return (COLOR_PAIR (p->index));
296 }

298 void mutt_free_color (int fg, int bg)
299 {
300  COLOR_LIST *p, *q;
301 
302  p = ColorList;
303  while (p)
304  {
305  if (p->fg == fg && p->bg == bg)
306  {
307  (p->count)--;
308  if (p->count > 0) return;
309 
310  UserColors--;
311  dprint(1,(debugfile,"mutt_free_color(): Color pairs used so far: %d\n",
312  UserColors));
313 
314  if (p == ColorList)
315  {
316  ColorList = ColorList->next;
317  FREE (&p);
318  return;
319  }
320  q = ColorList;
321  while (q)
322  {
323  if (q->next == p)
324  {
325  q->next = p->next;
326  FREE (&p);
327  return;
328  }
329  q = q->next;
330  }
331  /* can't get here */
332  }
333  p = p->next;
334  }
335 }

342 static int
343 parse_color_name (const char *s, int *col, int *attr, int is_fg, BUFFER *err)
344 {
345  char *eptr;
346  int is_bright = 0;
347 
348  if (ascii_strncasecmp (s, "bright", 6) == 0)
349  {
350  is_bright = 1;
351  s += 6;
352  }
353 
354  /* allow aliases for xterm color resources */
355  if (ascii_strncasecmp (s, "color", 5) == 0)
356  {
357  s += 5;
358  *col = strtol (s, &eptr, 10);
359  if (!*s || *eptr || *col < 0 ||
360  (*col >= COLORS && !option(OPTNOCURSES) && has_colors()))
361  {
362  snprintf (err->data, err->dsize, _("%s: color not supported by term"), s);
363  return (-1);
364  }
365  }
366  else if ((*col = mutt_getvaluebyname (s, Colors)) == -1)
367  {
368  snprintf (err->data, err->dsize, _("%s: no such color"), s);
369  return (-1);
370  }
371 
372  if (is_bright)
373  {
374  if (is_fg)
375  {
376  *attr |= A_BOLD;
377  }
378  else if (COLORS < 16)
379  {
380  /* A_BLINK turns the background color brite on some terms */
381  *attr |= A_BLINK;
382  }
383  else
384  {
385  /* Advance the color by 8 to get the bright version */
386  *col += 8;
387  }
388  }
389 
390  return 0;
391 }

421 static void
422 mutt_do_uncolor (BUFFER *buf, BUFFER *s, COLOR_LINE **ColorList,
423  int *do_cache, int parse_uncolor)
424 {
425  COLOR_LINE *tmp, *last = NULL;
426 
427  do
428  {
429  mutt_extract_token (buf, s, 0);
430  if (mutt_strcmp ("*", buf->data) == 0)
431  {
432  for (tmp = *ColorList; tmp; )
433  {
434  if (!*do_cache)
435  {
436  *do_cache = 1;
437  }
438  last = tmp;
439  tmp = tmp->next;
440  mutt_free_color_line (&last, parse_uncolor);
441  }
442  *ColorList = NULL;
443  }
444  else
445  {
446  for (last = NULL, tmp = *ColorList; tmp; last = tmp, tmp = tmp->next)
447  {
448  if (mutt_strcmp (buf->data, tmp->pattern) == 0)
449  {
450  if (!*do_cache)
451  {
452  *do_cache = 1;
453  }
454  dprint (1, (debugfile,"Freeing pattern \"%s\" from ColorList\n",
455  tmp->pattern));
456  if (last)
457  {
458  last->next = tmp->next;
459  }
460  else
461  {
462  *ColorList = tmp->next;
463  }
464  mutt_free_color_line (&tmp, parse_uncolor);
465  break;
466  }
467  }
468  }
469  } while (MoreArgs (s));
470 }

472 static int _mutt_parse_uncolor (BUFFER *buf, BUFFER *s, unsigned long data,
473  BUFFER *err, short parse_uncolor)
474 {
475  int object = 0, do_cache = 0;
476 
477  mutt_extract_token (buf, s, 0);
478 
479  if ((object = mutt_getvaluebyname (buf->data, Fields)) == -1)
480  {
481  snprintf (err->data, err->dsize, _("%s: no such object"), buf->data);
482  return (-1);
483  }
484 
485  if (object > MT_COLOR_INDEX_SUBJECT) { /* uncolor index column */
486  ColorDefs[object] = 0;
487  set_option (OPTFORCEREDRAWINDEX);
488  return 0;
489  }
490 
491  if ((mutt_strncmp (buf->data, "body", 4) != 0) &&
492  (mutt_strncmp (buf->data, "header", 6) != 0) &&
493  (mutt_strncmp (buf->data, "index", 5) != 0))
494  {
495  snprintf (err->data, err->dsize,
496  _("%s: command valid only for index, body, header objects"),
497  parse_uncolor ? "uncolor" : "unmono");
498  return (-1);
499  }
500 
501  if (!MoreArgs (s))
502  {
503  snprintf (err->data, err->dsize,
504  _("%s: too few arguments"), parse_uncolor ? "uncolor" : "unmono");
505  return (-1);
506  }
507 
508  if(
509 #ifdef HAVE_COLOR
510  /* we're running without curses */
511  option (OPTNOCURSES)
512  || /* we're parsing an uncolor command, and have no colors */
513  (parse_uncolor && !has_colors())
514  /* we're parsing an unmono command, and have colors */
515  || (!parse_uncolor && has_colors())
516 #else
517  /* We don't even have colors compiled in */
518  parse_uncolor
519 #endif
520  )
521  {
522  /* just eat the command, but don't do anything real about it */
523  do
524  mutt_extract_token (buf, s, 0);
525  while (MoreArgs (s));
526 
527  return 0;
528  }
529 
530  if (object == MT_COLOR_BODY)
531  mutt_do_uncolor (buf, s, &ColorBodyList, &do_cache, parse_uncolor);
532  else if (object == MT_COLOR_HEADER)
533  mutt_do_uncolor (buf, s, &ColorHdrList, &do_cache, parse_uncolor);
534  else if (object == MT_COLOR_ATTACH_HEADERS)
535  mutt_do_uncolor (buf, s, &ColorAttachList, &do_cache, parse_uncolor);
536  else if (object == MT_COLOR_INDEX)
537  mutt_do_uncolor (buf, s, &ColorIndexList, &do_cache, parse_uncolor);
538  else if (object == MT_COLOR_INDEX_AUTHOR)
539  mutt_do_uncolor (buf, s, &ColorIndexAuthorList, &do_cache, parse_uncolor);
540  else if (object == MT_COLOR_INDEX_FLAGS)
541  mutt_do_uncolor (buf, s, &ColorIndexFlagsList, &do_cache, parse_uncolor);
542  else if (object == MT_COLOR_INDEX_SUBJECT)
543  mutt_do_uncolor (buf, s, &ColorIndexSubjectList, &do_cache, parse_uncolor);
544 #ifdef USE_NOTMUCH
545  else if (object == MT_COLOR_INDEX_TAG)
546  mutt_do_uncolor(buf, s, &ColorIndexTagList, &do_cache, parse_uncolor);
547 #endif
548 
549  if (do_cache && !option (OPTNOCURSES))
550  {
551  int i;
552  set_option (OPTFORCEREDRAWINDEX);
553  /* force re-caching of index colors */
554  for (i = 0; Context && i < Context->msgcount; i++)
555  Context->hdrs[i]->pair = 0;
556  }
557  return (0);
558 }

561 static int
562 add_pattern (COLOR_LINE **top, const char *s, int sensitive,
563  int fg, int bg, int attr, BUFFER *err,
564  int is_index, int match)
565 {
566 
567  /* is_index used to store compiled pattern
568  * only for `index' color object
569  * when called from mutt_parse_color() */
570 
571  COLOR_LINE *tmp = *top;
572 
573  while (tmp)
574  {
575  if (sensitive)
576  {
577  if (mutt_strcmp (s, tmp->pattern) == 0)
578  break;
579  }
580  else
581  {
582  if (mutt_strcasecmp (s, tmp->pattern) == 0)
583  break;
584  }
585  tmp = tmp->next;
586  }
587 
588  if (tmp)
589  {
590 #ifdef HAVE_COLOR
591  if (fg != -1 && bg != -1)
592  {
593  if (tmp->fg != fg || tmp->bg != bg)
594  {
595  mutt_free_color (tmp->fg, tmp->bg);
596  tmp->fg = fg;
597  tmp->bg = bg;
598  attr |= mutt_alloc_color (fg, bg);
599  }
600  else
601  attr |= (tmp->pair & ~A_BOLD);
602  }
603 #endif /* HAVE_COLOR */
604  tmp->pair = attr;
605  }
606  else
607  {
608  int r;
609  char buf[LONG_STRING];
610 
611  tmp = mutt_new_color_line ();
612  if (is_index)
613  {
614  int i;
615 
616  strfcpy(buf, NONULL(s), sizeof(buf));
617  mutt_check_simple (buf, sizeof (buf), NONULL(SimpleSearch));
618  if((tmp->color_pattern = mutt_pattern_comp (buf, MUTT_FULL_MSG, err)) == NULL)
619  {
620  mutt_free_color_line(&tmp, 1);
621  return -1;
622  }
623  /* force re-caching of index colors */
624  for (i = 0; Context && i < Context->msgcount; i++)
625  Context->hdrs[i]->pair = 0;
626  }
627  else if ((r = REGCOMP (&tmp->rx, s, (sensitive ? mutt_which_case (s) : REG_ICASE))) != 0)
628  {
629  regerror (r, &tmp->rx, err->data, err->dsize);
630  mutt_free_color_line(&tmp, 1);
631  return (-1);
632  }
633  tmp->next = *top;
634  tmp->pattern = safe_strdup (s);
635  tmp->match = match;
636 #ifdef HAVE_COLOR
637  if(fg != -1 && bg != -1)
638  {
639  tmp->fg = fg;
640  tmp->bg = bg;
641  attr |= mutt_alloc_color (fg, bg);
642  }
643 #endif
644  tmp->pair = attr;
645  *top = tmp;
646  }
647 
648  return 0;
649 }

773 static int
774 _mutt_parse_color (BUFFER *buf, BUFFER *s, BUFFER *err,
775  parser_callback_t callback, short dry_run)
776 {
777  int object = 0, attr = 0, fg = 0, bg = 0, q_level = 0;
778  int r = 0, match = 0;
779 
780  if(parse_object(buf, s, &object, &q_level, err) == -1)
781  return -1;
782 
783  if(callback(buf, s, &fg, &bg, &attr, err) == -1)
784  return -1;
785 
786  /* extract a regular expression if needed */
787 
788  if ((object == MT_COLOR_BODY) ||
789  (object == MT_COLOR_HEADER) ||
790  (object == MT_COLOR_ATTACH_HEADERS) ||
791  (object == MT_COLOR_INDEX) ||
792  (object == MT_COLOR_INDEX_AUTHOR) ||
793  (object == MT_COLOR_INDEX_FLAGS) ||
794 #ifdef USE_NOTMUCH
795  (object == MT_COLOR_INDEX_TAG) ||
796 #endif
797  (object == MT_COLOR_INDEX_SUBJECT)) {
798  if (!MoreArgs (s))
799  {
800  strfcpy (err->data, _("too few arguments"), err->dsize);
801  return -1;
802  }
803 
804  mutt_extract_token (buf, s, 0);
805  }
806 
807  if (MoreArgs (s) && (object != MT_COLOR_STATUS))
808  {
809  strfcpy (err->data, _("too many arguments"), err->dsize);
810  return (-1);
811  }
812 
813  /* dry run? */
814 
815  if(dry_run)
816  {
817  *s->dptr = '\0'; /* fake that we're done parsing */
818  return 0;
819  }
820 
821 
822 #ifdef HAVE_COLOR
823 # ifdef HAVE_USE_DEFAULT_COLORS
824  if (!option (OPTNOCURSES) && has_colors()
825  /* delay use_default_colors() until needed, since it initializes things */
826  && (fg == COLOR_DEFAULT || bg == COLOR_DEFAULT)
827  && use_default_colors () != OK)
828  {
829  strfcpy (err->data, _("default colors not supported"), err->dsize);
830  return (-1);
831  }
832 # endif /* HAVE_USE_DEFAULT_COLORS */
833 #endif
834 
835  if (object == MT_COLOR_HEADER)
836  r = add_pattern (&ColorHdrList, buf->data, 0, fg, bg, attr, err, 0, match);
837  else if (object == MT_COLOR_BODY)
838  r = add_pattern (&ColorBodyList, buf->data, 1, fg, bg, attr, err, 0, match);
839  else if (object == MT_COLOR_ATTACH_HEADERS)
840  r = add_pattern (&ColorAttachList, buf->data, 1, fg, bg, attr, err, 0, match);
841  else if ((object == MT_COLOR_STATUS) && MoreArgs (s)) {
842  /* 'color status fg bg' can have up to 2 arguments:
843  * 0 arguments: sets the default status color (handled below by else part)
844  * 1 argument : colorize pattern on match
845  * 2 arguments: colorize nth submatch of pattern
846  */
847  mutt_extract_token (buf, s, 0);
848 
849  if (MoreArgs (s)) {
850  BUFFER temporary;
851  memset (&temporary, 0, sizeof (BUFFER));
852  mutt_extract_token (&temporary, s, 0);
853  match = atoi (temporary.data);
854  FREE(&temporary.data);
855  }
856 
857  if (MoreArgs (s)) {
858  strfcpy (err->data, _("too many arguments"), err->dsize);
859  return -1;
860  }
861 
862  r = add_pattern (&ColorStatusList, buf->data, 1,
863  fg, bg, attr, err, 0, match);
864  }
865  else if (object == MT_COLOR_INDEX)
866  {
867  r = add_pattern (&ColorIndexList, buf->data, 1,
868  fg, bg, attr, err, 1, match);
869  set_option (OPTFORCEREDRAWINDEX);
870  }
871  else if (object == MT_COLOR_INDEX_AUTHOR)
872  {
873  r = add_pattern (&ColorIndexAuthorList, buf->data, 1,
874  fg, bg, attr, err, 1, match);
875  set_option (OPTFORCEREDRAWINDEX);
876  }
877  else if (object == MT_COLOR_INDEX_FLAGS)
878  {
879  r = add_pattern (&ColorIndexFlagsList, buf->data, 1,
880  fg, bg, attr, err, 1, match);
881  set_option (OPTFORCEREDRAWINDEX);
882  }
883  else if (object == MT_COLOR_INDEX_SUBJECT)
884  {
885  r = add_pattern (&ColorIndexSubjectList, buf->data, 1,
886  fg, bg, attr, err, 1, match);
887  set_option (OPTFORCEREDRAWINDEX);
888  }
889 #ifdef USE_NOTMUCH
890  else if (object == MT_COLOR_INDEX_TAG)
891  {
892  r = add_pattern (&ColorIndexTagList, buf->data, 1,
893  fg, bg, attr, err, 1, match);
894  set_option (OPTFORCEREDRAWINDEX);
895  }
896 #endif
897  else if (object == MT_COLOR_QUOTED)
898  {
899  if (q_level >= ColorQuoteSize)
900  {
901  safe_realloc (&ColorQuote, (ColorQuoteSize += 2) * sizeof (int));
902  ColorQuote[ColorQuoteSize-2] = ColorDefs[MT_COLOR_QUOTED];
903  ColorQuote[ColorQuoteSize-1] = ColorDefs[MT_COLOR_QUOTED];
904  }
905  if (q_level >= ColorQuoteUsed)
906  ColorQuoteUsed = q_level + 1;
907  if (q_level == 0)
908  {
909  ColorDefs[MT_COLOR_QUOTED] = fgbgattr_to_color(fg, bg, attr);
910 
911  ColorQuote[0] = ColorDefs[MT_COLOR_QUOTED];
912  for (q_level = 1; q_level < ColorQuoteUsed; q_level++)
913  {
914  if (ColorQuote[q_level] == A_NORMAL)
915  ColorQuote[q_level] = ColorDefs[MT_COLOR_QUOTED];
916  }
917  }
918  else
919  ColorQuote[q_level] = fgbgattr_to_color(fg, bg, attr);
920  }
921  else
922  {
923  ColorDefs[object] = fgbgattr_to_color(fg, bg, attr);
924  if (object > MT_COLOR_INDEX_AUTHOR)
925  set_option (OPTFORCEREDRAWINDEX);
926  }
927 
928  return (r);
929 }
