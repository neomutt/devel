28 static void *
29 hcache_gdbm_open(const char *path)
30 {
31  int pagesize;
32 
33  if (mutt_atoi (HeaderCachePageSize, &pagesize) < 0 || pagesize <= 0)
34  pagesize = 16384;
35 
36  GDBM_FILE db = gdbm_open((char *) path, pagesize, GDBM_WRCREAT, 00600, NULL);
37  if (db)
38  return db;
39 
40  /* if rw failed try ro */
41  return gdbm_open((char *) path, pagesize, GDBM_READER, 00600, NULL);
42 }
