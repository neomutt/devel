142 static int parse_fkey(char *s)
143 {
144  char *t;
145  int n = 0;
146 
147  if(s[0] != '<' || ascii_tolower(s[1]) != 'f')
148  return -1;
149 
150  for(t = s + 2; *t && isdigit((unsigned char) *t); t++)
151  {
152  n *= 10;
153  n += *t - '0';
154  }
155 
156  if(*t != '>')
157  return -1;
158  else
159  return n;
160 }

162 /*
163  * This function parses the string <NNN> and uses the octal value as the key
164  * to bind.
165  */
166 static int parse_keycode (const char *s)
167 {
168  char *endChar;
169  long int result = strtol(s+1, &endChar, 8);
170  /* allow trailing whitespace, eg. < 1001 > */
171  while (ISSPACE(*endChar))
172  ++endChar;
173  /* negative keycodes don't make sense, also detect overflow */
174  if (*endChar != '>' || result < 0 || result == LONG_MAX) {
175  return -1;
176  }
177 
178  return result;
179 }

181 static int parsekeys (const char *str, keycode_t *d, int max)
182 {
183  int n, len = max;
184  char buff[SHORT_STRING];
185  char c;
186  char *s, *t;
187 
188  strfcpy(buff, str, sizeof(buff));
189  s = buff;
190 
191  while (*s && len)
192  {
193  *d = '\0';
194  if(*s == '<' && (t = strchr(s, '>')))
195  {
196  t++; c = *t; *t = '\0';
197 
198  if ((n = mutt_getvaluebyname (s, KeyNames)) != -1)
199  {
200  s = t;
201  *d = n;
202  }
203  else if ((n = parse_fkey(s)) > 0)
204  {
205  s = t;
206  *d = KEY_F (n);
207  }
208  else if ((n = parse_keycode(s)) > 0)
209  {
210  s = t;
211  *d = n;
212  }
213 
214  *t = c;
215  }
216 
217  if(!*d)
218  {
219  *d = (unsigned char)*s;
220  s++;
221  }
222  d++;
223  len--;
224  }
225 
226  return (max - len);
227 }

229 /* insert a key sequence into the specified map. the map is sorted by ASCII
230  * value (lowest to highest)
231  */
232 void km_bind (char *s, int menu, int op, char *macro, char *descr)
233 {
234  struct keymap_t *map, *tmp, *last = NULL, *next;
235  keycode_t buf[MAX_SEQ];
236  int len, pos = 0, lastpos = 0;
237 
238  len = parsekeys (s, buf, MAX_SEQ);
239 
240  map = allocKeys (len, buf);
241  map->op = op;
242  map->macro = safe_strdup (macro);
243  map->descr = safe_strdup (descr);
244 
245  tmp = Keymaps[menu];
246 
247  while (tmp)
248  {
249  if (pos >= len || pos >= tmp->len)
250  {
251  /* map and tmp match, but have different lengths, so overwrite */
252  do
253  {
254  len = tmp->eq;
255  next = tmp->next;
256  FREE (&tmp->macro);
257  FREE (&tmp->keys);
258  FREE (&tmp->descr);
259  FREE (&tmp);
260  tmp = next;
261  }
262  while (tmp && len >= pos);
263  map->eq = len;
264  break;
265  }
266  else if (buf[pos] == tmp->keys[pos])
267  pos++;
268  else if (buf[pos] < tmp->keys[pos])
269  {
270  /* found location to insert between last and tmp */
271  map->eq = pos;
272  break;
273  }
274  else /* buf[pos] > tmp->keys[pos] */
275  {
276  last = tmp;
277  lastpos = pos;
278  if (pos > tmp->eq)
279  pos = tmp->eq;
280  tmp = tmp->next;
281  }
282  }
283 
284  map->next = tmp;
285  if (last)
286  {
287  last->next = map;
288  last->eq = lastpos;
289  }
290  else
291  Keymaps[menu] = map;
292 }

300 {

304  {

326 /* Parses s for <function> syntax and adds the whole sequence to
327  * either the macro or unget buffer. This function is invoked by the next
328  * two defines below.
329  */
330 static void generic_tokenize_push_string (char *s, void (*generic_push) (int, int))
331 {
332  char *pp, *p = s + mutt_strlen (s) - 1;
333  size_t l;
334  int i, op = OP_NULL;
335 
336  while (p >= s)
337  {
338  /* if we see something like "<PageUp>", look to see if it is a real
339  function name and return the corresponding value */
340  if (*p == '>')
341  {
342  for (pp = p - 1; pp >= s && *pp != '<'; pp--)
343  ;
344  if (pp >= s)
345  {
346  if ((i = parse_fkey (pp)) > 0)
347  {
348  generic_push (KEY_F (i), 0);
349  p = pp - 1;
350  continue;
351  }
352 
353  l = p - pp + 1;
354  for (i = 0; KeyNames[i].name; i++)
355  {
356  if (!ascii_strncasecmp (pp, KeyNames[i].name, l))
357  break;
358  }
359  if (KeyNames[i].name)
360  {
361  /* found a match */
362  generic_push (KeyNames[i].value, 0);
363  p = pp - 1;
364  continue;
365  }
366 
367  /* See if it is a valid command
368  * skip the '<' and the '>' when comparing */
369  for (i = 0; Menus[i].name; i++)
370  {
371  const struct binding_t *binding = km_get_table (Menus[i].value);
372  if (binding)
373  {
374  op = get_op (binding, pp + 1, l - 2);
375  if (op != OP_NULL)
376  break;
377  }
378  }
379 
380  if (op != OP_NULL)
381  {
382  generic_push (0, op);
383  p = pp - 1;
384  continue;
385  }
386  }
387  }
388  generic_push ((unsigned char)*p--, 0); /* independent 8 bits chars */
389  }
390 }

397 static int retry_generic (int menu, keycode_t *keys, int keyslen, int lastkey)
398 {
399  if (menu != MENU_EDITOR && menu != MENU_GENERIC && menu != MENU_PAGER)
400  {
401  if (lastkey)
402  mutt_unget_event (lastkey, 0);
403  for (; keyslen; keyslen--)
404  mutt_unget_event (keys[keyslen - 1], 0);
405  return (km_dokey (MENU_GENERIC));
406  }
407  if (menu != MENU_EDITOR)
408  {
409  /* probably a good idea to flush input here so we can abort macros */
410  mutt_flushinp ();
411  }
412  return OP_NULL;
413 }

415 /* return values:
416  * >0 function to execute
417  * OP_NULL no function bound to key sequence
418  * -1 error occurred while reading input
419  */
420 int km_dokey (int menu)
421 {
422  event_t tmp;
423  struct keymap_t *map = Keymaps[menu];
424  int pos = 0;
425  int n = 0;
426  int i;
427 
428  if (!map)
429  return (retry_generic (menu, NULL, 0, 0));
430 
431  FOREVER
432  {
433  i = Timeout > 0 ? Timeout : 60;
434 #ifdef USE_IMAP
435  /* keepalive may need to run more frequently than Timeout allows */
436  if (ImapKeepalive)
437  {
438  if (ImapKeepalive >= i)
439  imap_keepalive ();
440  else
441  while (ImapKeepalive && ImapKeepalive < i)
442  {
443  timeout (ImapKeepalive * 1000);
444  tmp = mutt_getch ();
445  timeout (-1);
446  /* If a timeout was not received, or the window was resized, exit the
447  * loop now. Otherwise, continue to loop until reaching a total of
448  * $timeout seconds.
449  */
450  if (tmp.ch != -2 || SigWinch)
451  goto gotkey;
452  i -= ImapKeepalive;
453  imap_keepalive ();
454  }
455  }
456 #endif
457 
458  timeout (i * 1000);
459  tmp = mutt_getch();
460  timeout (-1);
461 
462 #ifdef USE_IMAP
463  gotkey:
464 #endif
465  /* hide timeouts and window resizes from line editor. */
466  if (menu == MENU_EDITOR && tmp.ch == -2)
467  continue;
468 
469  LastKey = tmp.ch;
470  if (LastKey < 0)
471  return -1;
472 
473  /* do we have an op already? */
474  if (tmp.op)
475  {
476  char *func = NULL;
477  const struct binding_t *bindings;
478 
479  /* is this a valid op for this menu? */
480  if ((bindings = km_get_table (menu)) &&
481  (func = get_func (bindings, tmp.op)))
482  return tmp.op;
483 
484  if (menu == MENU_EDITOR && get_func (OpEditor, tmp.op))
485  return tmp.op;
486 
487  if (menu != MENU_EDITOR && menu != MENU_PAGER)
488  {
489  /* check generic menu */
490  bindings = OpGeneric;
491  if ((func = get_func (bindings, tmp.op)))
492  return tmp.op;
493  }
494 
495  /* Sigh. Valid function but not in this context.
496  * Find the literal string and push it back */
497  for (i = 0; Menus[i].name; i++)
498  {
499  bindings = km_get_table (Menus[i].value);
500  if (bindings)
501  {
502  func = get_func (bindings, tmp.op);
503  if (func)
504  {
505  mutt_unget_event ('>', 0);
506  mutt_unget_string (func);
507  mutt_unget_event ('<', 0);
508  break;
509  }
510  }
511  }
512  /* continue to chew */
513  if (func)
514  continue;
515  }
516 
517  /* Nope. Business as usual */
518  while (LastKey > map->keys[pos])
519  {
520  if (pos > map->eq || !map->next)
521  return (retry_generic (menu, map->keys, pos, LastKey));
522  map = map->next;
523  }
524 
525  if (LastKey != map->keys[pos])
526  return (retry_generic (menu, map->keys, pos, LastKey));
527 
528  if (++pos == map->len)
529  {
530 
531  if (map->op != OP_MACRO)
532  return map->op;
533 
534  if (option (OPTIGNOREMACROEVENTS))
535  {
536  mutt_error _("Macros are currently disabled.");
537  return -1;
538  }
539 
540  if (n++ == 10)
541  {
542  mutt_flushinp ();
543  mutt_error _("Macro loop detected.");
544  return -1;
545  }
546 
547  tokenize_push_macro_string (map->macro);
548  map = Keymaps[menu];
549  pos = 0;
550  }
551  }
552 
553  /* not reached */
554 }

565 static const char *km_keyname (int c)
566 {
567  static char buf[10];
568  const char *p;
569 
570  if ((p = mutt_getnamebyvalue (c, KeyNames)))
571  return p;
572 
573  if (c < 256 && c > -128 && iscntrl ((unsigned char) c))
574  {
575  if (c < 0)
576  c += 256;
577 
578  if (c < 128)
579  {
580  buf[0] = '^';
581  buf[1] = (c + '@') & 0x7f;
582  buf[2] = 0;
583  }
584  else
585  snprintf (buf, sizeof (buf), "\\%d%d%d", c >> 6, (c >> 3) & 7, c & 7);
586  }
587  else if (c >= KEY_F0 && c < KEY_F(256)) /* this maximum is just a guess */
588  sprintf (buf, "<F%d>", c - KEY_F0);
589  else if (IsPrint (c))
590  snprintf (buf, sizeof (buf), "%c", (unsigned char) c);
591  else
592  snprintf (buf, sizeof (buf), "\\x%hx", (unsigned short) c);
593  return (buf);
594 }

684 /* Determine the keycodes for ncurses extended keys and fill in the KeyNames array.
685  *
686  * This function must be called *after* initscr(), or tigetstr() returns -1. This
687  * creates a bit of a chicken-and-egg problem because km_init() is called prior to
688  * start_curses(). This means that the default keybindings can't include any of the
689  * extended keys because they won't be defined until later.
690  */
691 void init_extended_keys(void)
692 {
693 #ifdef NCURSES_VERSION
694  int j;
695 
696  use_extended_names(TRUE);
697 
698  for (j = 0; KeyNames[j].name; ++j)
699  {
700  if (KeyNames[j].value == -1)
701  {
702  const char *keyname = find_ext_name(KeyNames[j].name);
703 
704  if (keyname)
705  {
706  char *s = tigetstr((char *)keyname);
707  if (s && (long)(s) != -1)
708  {
709  int code = key_defined(s);
710  if (code > 0)
711  KeyNames[j].value = code;
712  }
713  }
714  }
715  }
716 #endif
717 }

880 /* expects to see: <menu-string>,<menu-string>,... <key-string> */
881 static char *parse_keymap (int *menu, BUFFER *s, int maxmenus, int *nummenus, BUFFER *err)
882 {
883  BUFFER buf;
884  int i=0;
885  char *p, *q;
886 
887  mutt_buffer_init (&buf);
888 
889  /* menu name */
890  mutt_extract_token (&buf, s, 0);
891  p = buf.data;
892  if (MoreArgs (s))
893  {
894  while (i < maxmenus)
895  {
896  q = strchr(p,',');
897  if (q)
898  *q = '\0';
899 
900  if ((menu[i] = mutt_check_menu (p)) == -1)
901  {
902  snprintf (err->data, err->dsize, _("%s: no such menu"), p);
903  goto error;
904  }
905  ++i;
906  if (q)
907  p = q+1;
908  else
909  break;
910  }
911  *nummenus=i;
912  /* key sequence */
913  mutt_extract_token (&buf, s, 0);
914 
915  if (!*buf.data)
916  {
917  strfcpy (err->data, _("null key sequence"), err->dsize);
918  }
919  else if (MoreArgs (s))
920  return (buf.data);
921  }
922  else
923  {
924  strfcpy (err->data, _("too few arguments"), err->dsize);
925  }
926 error:
927  FREE (&buf.data);
928  return (NULL);
929 }

990 int mutt_parse_bind (BUFFER *buf, BUFFER *s, unsigned long data, BUFFER *err)
991 {

995

997  &nummenus, err)) == NULL)

999

1001  mutt_extract_token (buf, s, 0);
1002  if (MoreArgs (s))
1003  {

1005  r = -1;

1008  {

1010  {

1017  {

1020  menu[i] == MENU_GENERIC ||

1028  r = -1;
1029  }
1030  }
1031  }
1032  }
1033  FREE (&key);
