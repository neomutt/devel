65 {

67  int yr;
68

71

73  return 0;

76  if (!*s)

80

88  {

90  {
91  if (*++p == '\0')
92  return 0;

94  else if (*p == '"')
95  {
96  q = !q;
97  }
98  }
99

103  if (ascii_strncasecmp(p, " at ", 4) == 0)
104  {
105  p = strchr(p + 4, ' ');
106  if (!p)
107  {
108  dprint (1, (debugfile, "is_from(): error parsing what appears to be a pipermail-style obscured return_path: %s\n", s));
109  return 0;
110  }
111  }
112 
113  if (path)
114  {
115  len = (size_t) (p - s);
116  if (len + 1 > pathlen)
117  len = pathlen - 1;
118  memcpy (path, s, len);

121  }
122

125  if (!*s)
126  return 0;
127 
128  if (!is_day_name (s))
129  {

141  */

150

174  if (!*s) return 0;
175 
176  /*

178  * From xxxx@yyyyyyy.fr Wed Aug 2 00:39:12 MET DST 1995
179  */
180  if (isalpha ((unsigned char) *s))
181  {
182  s = next_word (s);
183  if (!*s) return 0;

185  }
