56 static void *
57 hcache_bdb_open(const char *path)
58 {
59  struct stat sb;
60  int ret;
61  u_int32_t createflags = DB_CREATE;
62  int pagesize;
63 
64  hcache_db_ctx_t *ctx = safe_malloc(sizeof(hcache_db_ctx_t));
65 
66  if (mutt_atoi (HeaderCachePageSize, &pagesize) < 0 || pagesize <= 0)
67  pagesize = 16384;
68 
69  snprintf (ctx->lockfile, _POSIX_PATH_MAX, "%s-lock-hack", path);
70 
71  ctx->fd = open (ctx->lockfile, O_WRONLY | O_CREAT, S_IRUSR | S_IWUSR);
72  if (ctx->fd < 0)
73  {
74  FREE(&ctx);
75  return NULL;
76  }
77 
78  if (mx_lock_file (ctx->lockfile, ctx->fd, 1, 0, 5))
79  goto fail_close;
80 
81  ret = db_env_create (&ctx->env, 0);
82  if (ret)
83  goto fail_unlock;
84 
85  ret = (*ctx->env->open)(ctx->env, NULL, DB_INIT_MPOOL | DB_CREATE | DB_PRIVATE, 0600);
86  if (ret)
87  goto fail_env;
88 
89  ctx->db = NULL;
90  ret = db_create (&ctx->db, ctx->env, 0);
91  if (ret)
92  goto fail_env;
93 
94  if (stat(path, &sb) != 0 && errno == ENOENT)
95  {
96  createflags |= DB_EXCL;
97  ctx->db->set_pagesize(ctx->db, pagesize);
98  }
99 
100  ret = (*ctx->db->open)(ctx->db, NULL, path, NULL, DB_BTREE, createflags,
101  0600);
102  if (ret)
103  goto fail_db;
104 
105  return ctx;
106 
107 fail_db:
108  ctx->db->close (ctx->db, 0);
109 fail_env:
110  ctx->env->close (ctx->env, 0);
111 fail_unlock:
112  mx_unlock_file (ctx->lockfile, ctx->fd, 0);
113 fail_close:
114  close (ctx->fd);
115  unlink (ctx->lockfile);
116  FREE(&ctx);
117 
118  return NULL;
119 }

123 {
