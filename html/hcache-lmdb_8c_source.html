48 static int
49 mdb_get_r_txn(hcache_lmdb_ctx_t *ctx)
50 {
51  int rc;
52 
53  if (ctx->txn && (ctx->txn_mode & (txn_read | txn_write)) > 0)
54  return MDB_SUCCESS;
55 
56  if (ctx->txn)
57  rc = mdb_txn_renew(ctx->txn);
58  else
59  rc = mdb_txn_begin(ctx->env, NULL, MDB_RDONLY, &ctx->txn);
60 
61  if (rc == MDB_SUCCESS)
62  ctx->txn_mode = txn_read;
63 
64  return rc;
65 }

67 static int
68 mdb_get_w_txn(hcache_lmdb_ctx_t *ctx)
69 {
70  int rc;
71 
72  if (ctx->txn && (ctx->txn_mode == txn_write))
73  return MDB_SUCCESS;
74 
75  if (ctx->txn)
76  {
77  if (ctx->txn_mode == txn_read)
78  mdb_txn_reset(ctx->txn);
79  ctx->txn = NULL;
80  }
81 
82  rc = mdb_txn_begin(ctx->env, NULL, 0, &ctx->txn);
83  if (rc == MDB_SUCCESS)
84  ctx->txn_mode = txn_write;
85 
86  return rc;
87 }

89 static void *
90 hcache_lmdb_open(const char *path)
91 {
92  int rc;
93 
94  hcache_lmdb_ctx_t *ctx = safe_malloc(sizeof(hcache_lmdb_ctx_t));
95  ctx->txn = NULL;
96  ctx->db = 0;
97 
98  rc = mdb_env_create(&ctx->env);
99  if (rc != MDB_SUCCESS)
100  {
101  fprintf(stderr, "hcache_open_lmdb: mdb_env_create: %s", mdb_strerror(rc));
102  FREE(&ctx);
103  return NULL;
104  }
105 
106  mdb_env_set_mapsize(ctx->env, LMDB_DB_SIZE);
107 
108  rc = mdb_env_open(ctx->env, path, MDB_NOSUBDIR, 0644);
109  if (rc != MDB_SUCCESS)
110  {
111  fprintf(stderr, "hcache_open_lmdb: mdb_env_open: %s", mdb_strerror(rc));
112  goto fail_env;
113  }
114 
115  rc = mdb_get_r_txn(ctx);
116  if (rc != MDB_SUCCESS)
117  {
118  fprintf(stderr, "hcache_open_lmdb: mdb_txn_begin: %s", mdb_strerror(rc));
119  goto fail_env;
120  }
121 
122  rc = mdb_dbi_open(ctx->txn, NULL, MDB_CREATE, &ctx->db);
123  if (rc != MDB_SUCCESS)
124  {
125  fprintf(stderr, "hcache_open_lmdb: mdb_dbi_open: %s", mdb_strerror(rc));
126  goto fail_dbi;
127  }
128 
129  mdb_txn_reset(ctx->txn);
130  ctx->txn_mode = txn_uninitialized;
131  return ctx;
132 
133 fail_dbi:
134  mdb_txn_abort(ctx->txn);
135  ctx->txn_mode = txn_uninitialized;
136  ctx->txn = NULL;
137 
138 fail_env:
139  mdb_env_close(ctx->env);
140  FREE(&ctx);
141  return NULL;
142 }

144 static void *
145 hcache_lmdb_fetch(void *vctx, const char *key, size_t keylen)
146 {
147  MDB_val dkey;
148  MDB_val data;
149  int rc;
150 
151  if (!vctx)
152  return NULL;
153 
154  hcache_lmdb_ctx_t *ctx = vctx;
155 
156  dkey.mv_data = (void *)key;
157  dkey.mv_size = keylen;
158  data.mv_data = NULL;
159  data.mv_size = 0;
160  rc = mdb_get_r_txn(ctx);
161  if (rc != MDB_SUCCESS)
162  {
163  ctx->txn = NULL;
164  fprintf(stderr, "txn_renew: %s\n", mdb_strerror(rc));
165  return NULL;
166  }
167  rc = mdb_get(ctx->txn, ctx->db, &dkey, &data);
168  if (rc == MDB_NOTFOUND)
169  {
170  return NULL;
171  }
172  if (rc != MDB_SUCCESS)
173  {
174  fprintf(stderr, "mdb_get: %s\n", mdb_strerror(rc));
175  return NULL;
176  }
177 
178  return data.mv_data;
179 }

187 static int
188 hcache_lmdb_store(void *vctx, const char *key, size_t keylen, void *data, size_t dlen)
189 {
190  MDB_val dkey;
191  MDB_val databuf;
192  int rc;
193 
194  if (!vctx)
195  return -1;
196 
197  hcache_lmdb_ctx_t *ctx = vctx;
198 
199  dkey.mv_data = (void *)key;
200  dkey.mv_size = keylen;
201  databuf.mv_data = data;
202  databuf.mv_size = dlen;
203  rc = mdb_get_w_txn(ctx);
204  if (rc != MDB_SUCCESS)
205  {
206  fprintf(stderr, "txn_begin: %s\n", mdb_strerror(rc));
207  return rc;
208  }
209  rc = mdb_put(ctx->txn, ctx->db, &dkey, &databuf, 0);
210  if (rc != MDB_SUCCESS)
211  {
212  fprintf(stderr, "mdb_put: %s\n", mdb_strerror(rc));
213  mdb_txn_abort(ctx->txn);
214  ctx->txn_mode = txn_uninitialized;
215  ctx->txn = NULL;
216  return rc;
217  }
218  return rc;
219 }

256 static void
257 hcache_lmdb_close(void **vctx)
258 {
259  if (!vctx || !*vctx)
260  return;
261 
262  hcache_lmdb_ctx_t *ctx = *vctx;
263 
264  if (ctx->txn && ctx->txn_mode == txn_write)
265  {
266  mdb_txn_commit(ctx->txn);
267  ctx->txn_mode = txn_uninitialized;
268  ctx->txn = NULL;
269  }
270 
271  mdb_env_close(ctx->env);
272  FREE(vctx); /* __FREE_CHECKED__ */
273 }
