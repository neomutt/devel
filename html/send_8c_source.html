98 /* removes addresses from "b" which are contained in "a" */
99 ADDRESS *mutt_remove_xrefs (ADDRESS *a, ADDRESS *b)
100 {
101  ADDRESS *top, *p, *prev = NULL;
102 
103  top = b;
104  while (b)
105  {
106  for (p = a; p; p = p->next)
107  {
108  if (mutt_addrcmp (p, b))
109  break;
110  }
111  if (p)
112  {
113  if (prev)
114  {
115  prev->next = b->next;
116  b->next = NULL;
117  rfc822_free_address (&b);
118  b = prev;
119  }
120  else
121  {
122  top = top->next;
123  b->next = NULL;
124  rfc822_free_address (&b);
125  b = top;
126  }
127  }
128  else
129  {
130  prev = b;
131  b = b->next;
132  }
133  }
134  return top;
135 }

137 /* remove any address which matches the current user. if `leave_only' is
138  * nonzero, don't remove the user's address if it is the only one in the list
139  */
140 static ADDRESS *remove_user (ADDRESS *a, int leave_only)
141 {
142  ADDRESS *top = NULL, *last = NULL;
143 
144  while (a)
145  {
146  if (!mutt_addr_is_user (a))
147  {
148  if (top)
149  {
150  last->next = a;
151  last = last->next;
152  }
153  else
154  last = top = a;
155  a = a->next;
156  last->next = NULL;
157  }
158  else
159  {
160  ADDRESS *tmp = a;
161 
162  a = a->next;
163  if (!leave_only || a || last)
164  {
165  tmp->next = NULL;
166  rfc822_free_address (&tmp);
167  }
168  else
169  last = top = tmp;
170  }
171  }
172  return top;
173 }

198 static int edit_address (ADDRESS **a, /* const */ char *field)
199 {
200  char buf[HUGE_STRING];
201  char *err = NULL;
202  int idna_ok = 0;
203 
204  do
205  {
206  buf[0] = 0;
207  mutt_addrlist_to_local (*a);
208  rfc822_write_address (buf, sizeof (buf), *a, 0);
209  if (mutt_get_field (field, buf, sizeof (buf), MUTT_ALIAS) != 0)
210  return (-1);
211  rfc822_free_address (a);
212  *a = mutt_expand_aliases (mutt_parse_adrlist (NULL, buf));
213  if ((idna_ok = mutt_addrlist_to_intl (*a, &err)) != 0)
214  {
215  mutt_error (_("Error: '%s' is a bad IDN."), err);
216  mutt_refresh ();
217  mutt_sleep (2);
218  FREE (&err);
219  }
220  }
221  while (idna_ok != 0);
222  return 0;
223 }

243  strfcpy (buf, en->followup_to, sizeof (buf));
244  else
245  buf[0] = 0;

247  return (-1);

256  return (-1);

259  }

268  return (-1);

270  (flags & (SENDREPLY|SENDLISTREPLY|SENDGROUPREPLY)) &&
271  (edit_address (&en->from, "From: ") == -1))
272  return (-1);
273  }
274 
275  if (en->subject)
276  {
277  if (option (OPTFASTREPLY))
278  return (0);
279  else
280  strfcpy (buf, en->subject, sizeof (buf));
281  }
282  else
283  {
284  const char *p;
285 
286  buf[0] = 0;
287  for (; uh; uh = uh->next)
288  {
289  if (ascii_strncasecmp ("subject:", uh->data, 8) == 0)
290  {
291  p = skip_email_wsp(uh->data + 8);
292  strncpy (buf, p, sizeof (buf));
293  }
294  }
295  }
296 
297  if (mutt_get_field ("Subject: ", buf, sizeof (buf), 0) != 0 ||
298  (!buf[0] && query_quadoption (OPT_SUBJECT, _("No subject, abort?")) != MUTT_NO))
299  {
300  mutt_message _("No subject, aborting.");
301  return (-1);
302  }
303  mutt_str_replace (&en->subject, buf);
304 
305  return 0;
306 }

339 static void process_user_header (ENVELOPE *env)
340 {
341  LIST *uh = UserHeader;
342  LIST *last = env->userhdrs;
343 
344  if (last)
345  while (last->next)
346  last = last->next;
347 
348  for (; uh; uh = uh->next)
349  {
350  if (ascii_strncasecmp ("from:", uh->data, 5) == 0)
351  {
352  /* User has specified a default From: address. Remove default address */
353  rfc822_free_address (&env->from);
354  env->from = rfc822_parse_adrlist (env->from, uh->data + 5);
355  }
356  else if (ascii_strncasecmp ("reply-to:", uh->data, 9) == 0)
357  {
358  rfc822_free_address (&env->reply_to);
359  env->reply_to = rfc822_parse_adrlist (env->reply_to, uh->data + 9);
360  }
361  else if (ascii_strncasecmp ("message-id:", uh->data, 11) == 0)
362  {
363  char *tmp = mutt_extract_message_id (uh->data + 11, NULL);
364  if (rfc822_valid_msgid (tmp) >= 0)
365  {
366  FREE(&env->message_id);
367  env->message_id = tmp;
368  } else
369  FREE(&tmp);
370  }
371  else if (ascii_strncasecmp ("to:", uh->data, 3) != 0 &&
372  ascii_strncasecmp ("cc:", uh->data, 3) != 0 &&
373  ascii_strncasecmp ("bcc:", uh->data, 4) != 0 &&
374 #ifdef USE_NNTP
375  ascii_strncasecmp ("newsgroups:", uh->data, 11) != 0 &&
376  ascii_strncasecmp ("followup-to:", uh->data, 12) != 0 &&
377  ascii_strncasecmp ("x-comment-to:", uh->data, 13) != 0 &&
378 #endif
379  ascii_strncasecmp ("supersedes:", uh->data, 11) != 0 &&
380  ascii_strncasecmp ("subject:", uh->data, 8) != 0 &&
381  ascii_strncasecmp ("return-path:", uh->data, 12) != 0)
382  {
383  if (last)
384  {
385  last->next = mutt_new_list ();
386  last = last->next;
387  }
388  else
389  last = env->userhdrs = mutt_new_list ();
390  last->data = safe_strdup (uh->data);
391  }
392  }

503 static int default_to (ADDRESS **to, ENVELOPE *env, int flags, int hmfupto)
504 {
505  char prompt[STRING];
506 
507  if (flags && env->mail_followup_to && hmfupto == MUTT_YES)
508  {
509  rfc822_append (to, env->mail_followup_to, 1);
510  return 0;
511  }
512 
513  /* Exit now if we're setting up the default Cc list for list-reply
514  * (only set if Mail-Followup-To is present and honoured).
515  */
516  if (flags & SENDLISTREPLY)
517  return 0;
518 
519  if (!option(OPTREPLYSELF) && mutt_addr_is_user (env->from))
520  {
521  /* mail is from the user, assume replying to recipients */
522  rfc822_append (to, env->to, 1);
523  }
524  else if (env->reply_to)
525  {
526  if ((mutt_addrcmp (env->from, env->reply_to) && !env->reply_to->next) ||
527  (option (OPTIGNORELISTREPLYTO) &&
528  mutt_is_mail_list (env->reply_to) &&
529  (mutt_addrsrc (env->reply_to, env->to) ||
530  mutt_addrsrc (env->reply_to, env->cc))))
531  {
532  /* If the Reply-To: address is a mailing list, assume that it was
533  * put there by the mailing list, and use the From: address
534  *
535  * We also take the from header if our correspondent has a reply-to
536  * header which is identical to the electronic mail address given
537  * in his From header.
538  *
539  */
540  rfc822_append (to, env->from, 0);
541  }
542  else if (!(mutt_addrcmp (env->from, env->reply_to) &&
543  !env->reply_to->next) &&
544  quadoption (OPT_REPLYTO) != MUTT_YES)
545  {
546  /* There are quite a few mailing lists which set the Reply-To:
547  * header field to the list address, which makes it quite impossible
548  * to send a message to only the sender of the message. This
549  * provides a way to do that.
550  */
551  /* L10N:
552  Asks whether the user respects the reply-to header.
553  If she says no, mutt will reply to the from header's address instead. */
554  snprintf (prompt, sizeof (prompt), _("Reply to %s%s?"),
555  env->reply_to->mailbox,
556  env->reply_to->next?",...":"");
557  switch (query_quadoption (OPT_REPLYTO, prompt))
558  {
559  case MUTT_YES:
560  rfc822_append (to, env->reply_to, 0);
561  break;
562 
563  case MUTT_NO:
564  rfc822_append (to, env->from, 0);
565  break;
566 
567  default:
568  return (-1); /* abort */
569  }
570  }
571  else
572  rfc822_append (to, env->reply_to, 0);
573  }
574  else
575  rfc822_append (to, env->from, 0);
576 
577  return (0);
578 }

580 int mutt_fetch_recips (ENVELOPE *out, ENVELOPE *in, int flags)
581 {
582  char prompt[STRING];
583  ADDRESS *tmp;
584  int hmfupto = -1;
585 
586  if ((flags & (SENDLISTREPLY|SENDGROUPREPLY)) && in->mail_followup_to)
587  {
588  snprintf (prompt, sizeof (prompt), _("Follow-up to %s%s?"),
589  in->mail_followup_to->mailbox,
590  in->mail_followup_to->next ? ",..." : "");
591 
592  if ((hmfupto = query_quadoption (OPT_MFUPTO, prompt)) == MUTT_ABORT)
593  return -1;
594  }
595 
596  if (flags & SENDLISTREPLY)
597  {
598  tmp = find_mailing_lists (in->to, in->cc);
599  rfc822_append (&out->to, tmp, 0);
600  rfc822_free_address (&tmp);
601 
602  if (in->mail_followup_to && hmfupto == MUTT_YES &&
603  default_to (&out->cc, in, flags & SENDLISTREPLY, hmfupto) == MUTT_ABORT)
604  return (-1); /* abort */
605  }
606  else
607  {
608  if (default_to (&out->to, in, flags & SENDGROUPREPLY, hmfupto) == MUTT_ABORT)
609  return (-1); /* abort */
610 
611  if ((flags & SENDGROUPREPLY) && (!in->mail_followup_to || hmfupto != MUTT_YES))
612  {
613  /* if(!mutt_addr_is_user(in->to)) */
614  rfc822_append (&out->cc, in->to, 1);
615  rfc822_append (&out->cc, in->cc, 1);
616  }
617  }
618  return 0;
