76 /* Search for a mailing list in the list of addresses pointed to by adr.
77  * If one is found, print pfx and the name of the list into buf, then
78  * return 1. Otherwise, simply return 0.

81 check_for_mailing_list (ADDRESS *adr, const char *pfx, char *buf, int buflen)
82 {
83  for (; adr; adr = adr->next)
84  {
85  if (mutt_is_subscribed_list (adr))
86  {
87  if (pfx && buf && buflen)
88  snprintf (buf, buflen, "%s%s", pfx, mutt_get_name (adr));
89  return 1;
90  }
91  }
92  return 0;

133 static size_t
134 add_index_color (char *buf, size_t buflen, format_flag flags, char color)
135 {
136  int len;
137 
138  /* only add color markers if we are operating on main index entries. */
139  if (!(flags & MUTT_FORMAT_INDEX))
140  return 0;
141 
142  if (color == MT_COLOR_INDEX) { /* buf might be uninitialized other cases */
143  len = mutt_strlen (buf);
144  buf += len;
145  buflen -= len;
146  }
147 
148  if (buflen < 2)
149  return 0;
150 
151  buf[0] = MUTT_SPECIAL_INDEX;
152  buf[1] = color;
153  buf[2] = '\0';
154 
155  return 2;
156 }

175 static const char *make_from_prefix(enum FieldType disp)

206 static void make_from (ENVELOPE *env, char *buf, size_t len, int do_lists)
207 {
208  if (!env || !buf)
209  return;
210 
211  int me;
212  enum FieldType disp;
213  ADDRESS *name;
214 
215  me = mutt_addr_is_user (env->from);
216 
217  if (do_lists || me)
218  {
219  if (check_for_mailing_list (env->to, make_from_prefix(DISP_TO), buf, len))
220  return;
221  if (check_for_mailing_list (env->cc, make_from_prefix(DISP_CC), buf, len))
222  return;
223  }
224 
225  if (me && env->to)
226  {
227  disp = DISP_TO;
228  name = env->to;
229  }
230  else if (me && env->cc)
231  {
232  disp = DISP_CC;
233  name = env->cc;
234  }
235  else if (me && env->bcc)
236  {
237  disp = DISP_BCC;
238  name = env->bcc;
239  }
240  else if (env->from)
241  {
242  disp = DISP_FROM;
243  name = env->from;
244  }
245  else
246  {
247  *buf = '\0';
248  return;
249  }
250 
251  snprintf (buf, len, "%s%s", make_from_prefix(disp), mutt_get_name (name));
252 }

254 static void make_from_addr (ENVELOPE *hdr, char *buf, size_t len, int do_lists)
255 {
256  if (!hdr || !buf)
257  return;
258 
259  int me = mutt_addr_is_user (hdr->from);
260 
261  if (do_lists || me)
262  {
263  if (check_for_mailing_list_addr (hdr->to, buf, len))
264  return;
265  if (check_for_mailing_list_addr (hdr->cc, buf, len))
266  return;
267  }
268 
269  if (me && hdr->to)
270  snprintf (buf, len, "%s", hdr->to->mailbox);
271  else if (me && hdr->cc)
272  snprintf (buf, len, "%s", hdr->cc->mailbox);
273  else if (hdr->from)
274  strfcpy (buf, hdr->from->mailbox, len);
275  else
276  *buf = 0;
277 }

287 /* Return values:
288  * 0: user is not in list
289  * 1: user is unique recipient
290  * 2: user is in the TO list
291  * 3: user is in the CC list
292  * 4: user is originator
293  * 5: sent to a subscribed mailinglist
294  */
295 int mutt_user_is_recipient (HEADER *h)
296 {
297  if (!h || !h->env)
298  return 0;
299 
300  ENVELOPE *env = h->env;
301 
302  if(!h->recip_valid)
303  {
304  h->recip_valid = 1;
305 
306  if (mutt_addr_is_user (env->from))
307  h->recipient = 4;
308  else if (user_in_addr (env->to))
309  {
310  if (env->to->next || env->cc)
311  h->recipient = 2; /* non-unique recipient */
312  else
313  h->recipient = 1; /* unique recipient */
314  }
315  else if (user_in_addr (env->cc))
316  h->recipient = 3;
317  else if (check_for_mailing_list (env->to, NULL, NULL, 0))
318  h->recipient = 5;
319  else if (check_for_mailing_list (env->cc, NULL, NULL, 0))
320  h->recipient = 5;
321  else
322  h->recipient = 0;
323  }
324 
325  return h->recipient;
326 }

339 static int get_initials(const char *name, char *buf, int buflen)
340 {
341  if (!name || !buf)
342  return 0;
343 
344  while (*name)
345  {
346  /* Char's length in bytes */
347  int clen = mutt_charlen(name, NULL);
348  if (clen < 1)
349  return 0;
350 
351  /* Ignore punctuation at the beginning of a word */
352  if ((clen == 1) && ispunct (*name))
353  {
354  name++;
355  continue;
356  }
357 
358  if (clen >= buflen)
359  return 0;
360 
361  /* Copy one multibyte character */
362  buflen -= clen;
363  while (clen--)
364  *buf++ = *name++;
365 
366  /* Skip to end-of-word */
367  for (; *name; name += clen)
368  {
369  clen = mutt_charlen(name, NULL);
370  if (clen < 1)
371  return 0;
372  else if ((clen == 1) && (isspace(*name) || (*name == '-')))
373  break;
374  }
375 
376  /* Skip any whitespace, or hyphens */
377  while (*name && (isspace(*name) || (*name == '-')))
378  name++;
379  }
380 
381  *buf = 0;
382  return 1;
383 }

406 /* %a = address of author
407  * %A = reply-to address (if present; otherwise: address of author
408  * %b = filename of the originating folder
409  * %B = the list to which the letter was sent
410  * %c = size of message in bytes
411  * %C = current message number
412  * %d = date and time of message using $date_format and sender's timezone
413  * %D = date and time of message using $date_format and local timezone
414  * %e = current message number in thread
415  * %E = number of messages in current thread
416  * %f = entire from line
417  * %F = like %n, unless from self
418  * %g = newsgroup name (if compiled with NNTP support)
419  * %g = message labels (e.g. notmuch tags)
420  * %i = message-id
421  * %I = initials of author
422  * %l = number of lines in the message
423  * %L = like %F, except `lists' are displayed first
424  * %m = number of messages in the mailbox
425  * %n = name of author
426  * %N = score
427  * %O = like %L, except using address instead of name
428  * %P = progress indicator for builtin pager
429  * %r = comma separated list of To: recipients
430  * %R = comma separated list of Cc: recipients
431  * %s = subject
432  * %S = short message status (e.g., N/O/D/!/r/-)
433  * %t = `to:' field (recipients)
434  * %T = $to_chars
435  * %u = user (login) name of author
436  * %v = first name of author, unless from self
437  * %W = where user is (organization)
438  * %x = `x-comment-to:' field (if present and compiled with NNTP support)
439  * %X = number of MIME attachments
440  * %y = `x-label:' field (if present)
441  * %Y = `x-label:' field (if present, tree unfolded, and != parent's x-label)
442  * %Z = status flags */

444 static const char *
445 hdr_format_str (char *dest,
446  size_t destlen,
447  size_t col,
448  int cols,
449  char op,
450  const char *src,
451  const char *prefix,
452  const char *ifstring,
453  const char *elsestring,
454  unsigned long data,
455  format_flag flags)
456 {
457  struct hdr_format_info *hfi = (struct hdr_format_info *) data;
458  HEADER *hdr, *htmp;
459  CONTEXT *ctx;
460  char fmt[SHORT_STRING], buf2[LONG_STRING], *p;
461  char *wch;
462  int do_locales, i;
463  int optional = (flags & MUTT_FORMAT_OPTIONAL);
464  int threads = ((Sort & SORT_MASK) == SORT_THREADS);
465  int is_index = (flags & MUTT_FORMAT_INDEX);
466 #define THREAD_NEW (threads && hdr->collapsed && hdr->num_hidden > 1 && mutt_thread_contains_unread (ctx, hdr) == 1)
467 #define THREAD_OLD (threads && hdr->collapsed && hdr->num_hidden > 1 && mutt_thread_contains_unread (ctx, hdr) == 2)
468  size_t len;
469  size_t colorlen;
470 
471  hdr = hfi->hdr;
472  ctx = hfi->ctx;
473 
474  if (!hdr || !hdr->env)
475  return src;
476  dest[0] = 0;
477  switch (op)
478  {
479  case 'A':
480  case 'I':
481  if (op == 'A')
482  {
483  if(hdr->env->reply_to && hdr->env->reply_to->mailbox)
484  {
485  colorlen = add_index_color (dest, destlen, flags, MT_COLOR_INDEX_AUTHOR);
486  mutt_format_s (dest + colorlen, destlen - colorlen, prefix, mutt_addr_for_display (hdr->env->reply_to));
487  add_index_color (dest + colorlen, destlen - colorlen, flags, MT_COLOR_INDEX);
488  break;
489  }
490  }
491  else
492  {
493  if (get_initials(mutt_get_name (hdr->env->from), buf2, sizeof (buf2)))
494  {
495  colorlen = add_index_color (dest, destlen, flags, MT_COLOR_INDEX_AUTHOR);
496  mutt_format_s (dest + colorlen, destlen - colorlen, prefix, buf2);
497  add_index_color (dest + colorlen, destlen - colorlen, flags, MT_COLOR_INDEX);
498  break;
499  }
500  }
501  /* fall through on failure */
502 
503  case 'a':
504  colorlen = add_index_color (dest, destlen, flags, MT_COLOR_INDEX_AUTHOR);
505  if(hdr->env->from && hdr->env->from->mailbox)
506  mutt_format_s (dest + colorlen, destlen - colorlen, prefix, mutt_addr_for_display (hdr->env->from));
507  else
508  mutt_format_s (dest + colorlen, destlen - colorlen, prefix, "");
509  add_index_color (dest + colorlen, destlen - colorlen, flags, MT_COLOR_INDEX);
510  break;
511 
512  case 'B':
513  if (!first_mailing_list (dest, destlen, hdr->env->to) &&
514  !first_mailing_list (dest, destlen, hdr->env->cc))
515  dest[0] = 0;
516  if (dest[0])
517  {
518  strfcpy (buf2, dest, sizeof(buf2));
519  mutt_format_s (dest, destlen, prefix, buf2);
520  break;
521  }
522  /* fall through if 'B' returns nothing */
523 
524  case 'b':
525  if(ctx)
526  {
527  if ((p = strrchr (ctx->path, '/')))
528  strfcpy (dest, p + 1, destlen);
529  else
530  strfcpy (dest, ctx->path, destlen);
531  }
532  else
533  strfcpy(dest, "(null)", destlen);
534  strfcpy (buf2, dest, sizeof(buf2));
535  mutt_format_s (dest, destlen, prefix, buf2);
536  break;
537 
538  case 'c':
539  colorlen = add_index_color (dest, destlen, flags, MT_COLOR_INDEX_SIZE);
540  mutt_pretty_size (buf2, sizeof (buf2), (long) hdr->content->length);
541  mutt_format_s (dest + colorlen, destlen - colorlen, prefix, buf2);
542  add_index_color (dest + colorlen, destlen - colorlen, flags, MT_COLOR_INDEX);
543  break;
544 
545  case 'C':
546  colorlen = add_index_color (fmt, sizeof (fmt), flags, MT_COLOR_INDEX_NUMBER);
547  snprintf (fmt + colorlen, sizeof (fmt) - colorlen, "%%%sd", prefix);
548  add_index_color (fmt + colorlen, sizeof (fmt) - colorlen, flags, MT_COLOR_INDEX);
549  snprintf (dest, destlen, fmt, hdr->msgno + 1);
550  break;
551 
552  case 'd':
553  case 'D':
554  case '{':
555  case '[':
556  case '(':
557  case '<':
558 
559  /* preprocess $date_format to handle %Z */
560  {
561  const char *cp;
562  struct tm *tm;
563  time_t T;
564  int i = 0, invert = 0;
565 
566  if (optional && ((op == '[') || (op == '('))) {
567  char *is;
568  T = time (NULL);
569  tm = localtime (&T);
570  T -= (op == '(') ? hdr->received : hdr->date_sent;
571 
572  is = (char *) prefix;
573  if (*is == '>') {
574  invert = 1;
575  is++;
576  }
577 
578  while (*is && (*is != '?')) {
579  int t = strtol (is, &is, 10);
580  /* semi-broken (assuming 30 days in all months) */
581  switch (*(is++)) {
582  case 'y':
583  if (t > 1) {
584  t--;
585  t *= (60 * 60 * 24 * 365);
586  }
587  t += ((tm->tm_mon * 60 * 60 * 24 * 30) +
588  (tm->tm_mday * 60 * 60 * 24) +
589  (tm->tm_hour * 60 * 60) +
590  (tm->tm_min * 60) +
591  tm->tm_sec);
592  break;
593 
594  case 'm':
595  if (t > 1) {
596  t--;
597  t *= (60 * 60 * 24 * 30);
598  }
599  t += ((tm->tm_mday * 60 * 60 * 24) +
600  (tm->tm_hour * 60 * 60) +
601  (tm->tm_min * 60) +
602  tm->tm_sec);
603  break;
604 
605  case 'w':
606  if (t > 1) {
607  t--;
608  t *= (60 * 60 * 24 * 7);
609  }
610  t += ((tm->tm_wday * 60 * 60 * 24) +
611  (tm->tm_hour * 60 * 60) +
612  (tm->tm_min * 60) +
613  tm->tm_sec);
614  break;
615 
616  case 'd':
617  if (t > 1) {
618  t--;
619  t *= (60 * 60 * 24);
620  }
621  t += ((tm->tm_hour * 60 * 60) +
622  (tm->tm_min * 60) +
623  tm->tm_sec);
624  break;
625 
626  case 'H':
627  if (t > 1) {
628  t--;
629  t *= (60 * 60);
630  }
631  t += ((tm->tm_min * 60) +
632  tm->tm_sec);
633  break;
634 
635  case 'M':
636  if (t > 1) {
637  t--;
638  t *= (60);
639  }
640  t += (tm->tm_sec);
641  break;
642 
643  default:
644  break;
645  }
646  i += t;
647  }
648 
649  if (i < 0)
650  i *= -1;
651 
652  if (((T > i) || (T < (-1*i))) ^ invert)
653  optional = 0;
654  break;
655  }
656 
657  p = dest;
658 
659  cp = (op == 'd' || op == 'D') ? (NONULL (DateFmt)) : src;
660  if (*cp == '!')
661  {
662  do_locales = 0;
663  cp++;
664  }
665  else
666  do_locales = 1;
667 
668  len = destlen - 1;
669  while (len > 0 && (((op == 'd' || op == 'D') && *cp) ||
670  (op == '{' && *cp != '}') ||
671  (op == '[' && *cp != ']') ||
672  (op == '(' && *cp != ')') ||
673  (op == '<' && *cp != '>')))
674  {
675  if (*cp == '%')
676  {
677  cp++;
678  if ((*cp == 'Z' || *cp == 'z') && (op == 'd' || op == '{'))
679  {
680  if (len >= 5)
681  {
682  sprintf (p, "%c%02u%02u", hdr->zoccident ? '-' : '+',
683  hdr->zhours, hdr->zminutes);
684  p += 5;
685  len -= 5;
686  }
687  else
688  break; /* not enough space left */
689  }
690  else
691  {
692  if (len >= 2)
693  {
694  *p++ = '%';
695  *p++ = *cp;
696  len -= 2;
697  }
698  else
699  break; /* not enough space */
700  }
701  cp++;
702  }
703  else
704  {
705  *p++ = *cp++;
706  len--;
707  }
708  }
709  *p = 0;
710 
711  if (op == '[' || op == 'D')
712  tm = localtime (&hdr->date_sent);
713  else if (op == '(')
714  tm = localtime (&hdr->received);
715  else if (op == '<')
716  {
717  T = time (NULL);
718  tm = localtime (&T);
719  }
720  else
721  {
722  /* restore sender's time zone */
723  T = hdr->date_sent;
724  if (hdr->zoccident)
725  T -= (hdr->zhours * 3600 + hdr->zminutes * 60);
726  else
727  T += (hdr->zhours * 3600 + hdr->zminutes * 60);
728  tm = gmtime (&T);
729  }
730 
731  if (!do_locales)
732  setlocale (LC_TIME, "C");
733  strftime (buf2, sizeof (buf2), dest, tm);
734  if (!do_locales)
735  setlocale (LC_TIME, "");
736 
737  colorlen = add_index_color (dest, destlen, flags, MT_COLOR_INDEX_DATE);
738  mutt_format_s (dest + colorlen, destlen - colorlen, prefix, buf2);
739  add_index_color (dest + colorlen, destlen - colorlen, flags, MT_COLOR_INDEX);
740 
741  if (len > 0 && op != 'd' && op != 'D') /* Skip ending op */
742  src = cp + 1;
743  }
744  break;
745 
746  case 'e':
747  snprintf (fmt, sizeof (fmt), "%%%sd", prefix);
748  snprintf (dest, destlen, fmt, mutt_messages_in_thread(ctx, hdr, 1));
749  break;
750 
751  case 'E':
752  if (!optional)
753  {
754  snprintf (fmt, sizeof (fmt), "%%%sd", prefix);
755  snprintf (dest, destlen, fmt, mutt_messages_in_thread(ctx, hdr, 0));
756  }
757  else if (mutt_messages_in_thread(ctx, hdr, 0) <= 1)
758  optional = 0;
759  break;
760 
761  case 'f':
762  buf2[0] = 0;
763  rfc822_write_address (buf2, sizeof (buf2), hdr->env->from, 1);
764  mutt_format_s (dest, destlen, prefix, buf2);
765  break;
766 
767  case 'F':
768  if (!optional)
769  {
