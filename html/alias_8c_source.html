42 static ADDRESS *mutt_expand_aliases_r (ADDRESS *a, LIST **expn)
43 {
44  ADDRESS *head = NULL, *last = NULL, *t, *w;
45  LIST *u;
46  char i;
47  const char *fqdn;
48 
49  while (a)
50  {
51  if (!a->group && !a->personal && a->mailbox && strchr (a->mailbox, '@') == NULL)
52  {
53  t = mutt_lookup_alias (a->mailbox);
54 
55  if (t)
56  {
57  i = 0;
58  for (u = *expn; u; u = u->next)
59  {
60  if (mutt_strcmp (a->mailbox, u->data) == 0) /* alias already found */
61  {
62  dprint (1, (debugfile, "mutt_expand_aliases_r(): loop in alias found for '%s'\n", a->mailbox));
63  i = 1;
64  break;
65  }
66  }
67 
68  if (!i)
69  {
70  u = safe_malloc (sizeof (LIST));
71  u->data = safe_strdup (a->mailbox);
72  u->next = *expn;
73  *expn = u;
74  w = rfc822_cpy_adr (t, 0);
75  w = mutt_expand_aliases_r (w, expn);
76  if (head)
77  last->next = w;
78  else
79  head = last = w;
80  while (last && last->next)
81  last = last->next;
82  }
83  t = a;
84  a = a->next;
85  t->next = NULL;
86  rfc822_free_address (&t);
87  continue;
88  }
89  else
90  {
91  struct passwd *pw = getpwnam (a->mailbox);
92 
93  if (pw)
94  {
95  char namebuf[STRING];
96 
97  mutt_gecos_name (namebuf, sizeof (namebuf), pw);
98  mutt_str_replace (&a->personal, namebuf);
99 
100 #ifdef EXACT_ADDRESS
101  FREE (&a->val);
102 #endif
103  }
104  }
105  }
106 
107  if (head)
108  {
109  last->next = a;
110  last = last->next;
111  }
112  else
113  head = last = a;
114  a = a->next;
115  last->next = NULL;
116  }
117 
118  if (option (OPTUSEDOMAIN) && (fqdn = mutt_fqdn(1)))
119  {
120  /* now qualify all local addresses */
121  rfc822_qualify (head, fqdn);
122  }
123 
124  return (head);
125 }

148 /*
149  * if someone has an address like
150  * From: Michael `/bin/rm -f ~` Elkins <me@mutt.org>
151  * and the user creates an alias for this, Mutt could wind up executing
152  * the backticks because it writes aliases like
153  * alias me Michael `/bin/rm -f ~` Elkins <me@mutt.org>
154  * To avoid this problem, use a backslash (\) to quote any backticks. We also
155  * need to quote backslashes as well, since you could defeat the above by
156  * doing
157  * From: Michael \`/bin/rm -f ~\` Elkins <me@mutt.org>
158  * since that would get aliased as
159  * alias me Michael \\`/bin/rm -f ~\\` Elkins <me@mutt.org>
160  * which still gets evaluated because the double backslash is not a quote.
161  *
162  * Additionally, we need to quote ' and " characters - otherwise, mutt will
163  * interpret them on the wrong parsing step.
164  *
165  * $ wants to be quoted since it may indicate the start of an environment
166  * variable.
167  */
168 
169 static void write_safe_address (FILE *fp, char *s)
170 {
171  while (*s)
172  {
173  if (*s == '\\' || *s == '`' || *s == '\'' || *s == '"'
174  || *s == '$')
175  fputc ('\\', fp);
176  fputc (*s, fp);
177  s++;
178  }
179 }

207  pfx = "From";
208  }
209 
210  if (pfxp) *pfxp = pfx;
211 
212  return adr;

229 void mutt_create_alias (ENVELOPE *cur, ADDRESS *iadr)
230 {
231  ALIAS *new, *t;
232  char buf[LONG_STRING], tmp[LONG_STRING], prompt[SHORT_STRING], *pc;
233  char *err = NULL;
234  char fixed[LONG_STRING];
235  FILE *rc;
236  ADDRESS *adr = NULL;
237 
238  if (cur)
239  {
240  adr = mutt_get_address (cur, NULL);
241  }
242  else if (iadr)
243  {
244  adr = iadr;
245  }
246 
247  if (adr && adr->mailbox)
248  {
249  strfcpy (tmp, adr->mailbox, sizeof (tmp));
250  if ((pc = strchr (tmp, '@')))
251  *pc = 0;
252  }
253  else
254  tmp[0] = '\0';
255 
256  /* Don't suggest a bad alias name in the event of a strange local part. */
257  mutt_check_alias_name (tmp, buf, sizeof (buf));
258 
259 retry_name:
260  /* L10N: prompt to add a new alias */
261  if (mutt_get_field (_("Alias as: "), buf, sizeof (buf), 0) != 0 || !buf[0])
262  return;
263 
264  /* check to see if the user already has an alias defined */
265  if (mutt_lookup_alias (buf))
266  {
267  mutt_error _("You already have an alias defined with that name!");
268  return;
269  }
270 
271  if (mutt_check_alias_name (buf, fixed, sizeof (fixed)))
272  {
273  switch (mutt_yesorno (_("Warning: This alias name may not work. Fix it?"), MUTT_YES))
274  {
275  case MUTT_YES:
276  strfcpy (buf, fixed, sizeof (buf));
277  goto retry_name;
278  case MUTT_ABORT:
279  return;
280  }
281  }
282 
283  new = safe_calloc (1, sizeof (ALIAS));
284  new->self = new;
285  new->name = safe_strdup (buf);
286 
287  mutt_addrlist_to_local (adr);
288 
289  if (adr)
290  strfcpy (buf, adr->mailbox, sizeof (buf));
291  else
292  buf[0] = 0;
293 
294  mutt_addrlist_to_intl (adr, NULL);
295 
296  do
297  {
298  if (mutt_get_field (_("Address: "), buf, sizeof (buf), 0) != 0 || !buf[0])
299  {
300  mutt_free_alias (&new);
301  return;
302  }
303 
304  if((new->addr = rfc822_parse_adrlist (new->addr, buf)) == NULL)
305  BEEP ();
306  if (mutt_addrlist_to_intl (new->addr, &err))
307  {
308  mutt_error (_("Error: '%s' is a bad IDN."), err);
309  mutt_sleep (2);
310  continue;
311  }
312  }
313  while(new->addr == NULL);
314 
315  if (adr && adr->personal && !mutt_is_mail_list (adr))
316  strfcpy (buf, adr->personal, sizeof (buf));
317  else
318  buf[0] = 0;
319 
320  if (mutt_get_field (_("Personal name: "), buf, sizeof (buf), 0) != 0)
321  {
322  mutt_free_alias (&new);
323  return;
324  }
325  new->addr->personal = safe_strdup (buf);
326 
327  buf[0] = 0;
328  rfc822_write_address (buf, sizeof (buf), new->addr, 1);
329  snprintf (prompt, sizeof (prompt), _("[%s = %s] Accept?"), new->name, buf);
330  if (mutt_yesorno (prompt, MUTT_YES) != MUTT_YES)
331  {
332  mutt_free_alias (&new);
333  return;
334  }
335 
336  mutt_alias_add_reverse (new);
337 
338  if ((t = Aliases))
339  {
340  while (t->next)
341  t = t->next;
342  t->next = new;
343  }
344  else
345  Aliases = new;
346 
347  strfcpy (buf, NONULL (AliasFile), sizeof (buf));
348  if (mutt_get_field (_("Save to file: "), buf, sizeof (buf), MUTT_FILE) != 0)
349  return;
350  mutt_expand_path (buf, sizeof (buf));
351  if ((rc = fopen (buf, "a+")))
352  {
353  /* terminate existing file with \n if necessary */
354  if (fseek (rc, 0, SEEK_END))
355  goto fseek_err;
356  if (ftell(rc) > 0)
357  {
358  if (fseek (rc, -1, SEEK_CUR) < 0)
359  goto fseek_err;
360  if (fread(buf, 1, 1, rc) != 1)
361  {
362  mutt_perror (_("Error reading alias file"));
363  safe_fclose (&rc);
364  return;
365  }
366  if (fseek (rc, 0, SEEK_END) < 0)
367  goto fseek_err;
368  if (buf[0] != '\n')
369  fputc ('\n', rc);
370  }
371 
372  if (mutt_check_alias_name (new->name, NULL, 0))
373  mutt_quote_filename (buf, sizeof (buf), new->name);
374  else
375  strfcpy (buf, new->name, sizeof (buf));
376  recode_buf (buf, sizeof (buf));
377  fprintf (rc, "alias %s ", buf);
378  buf[0] = 0;
379  rfc822_write_address (buf, sizeof (buf), new->addr, 0);
380  recode_buf (buf, sizeof (buf));
381  write_safe_address (rc, buf);
382  fputc ('\n', rc);
383  if (safe_fsync_close(&rc) != 0)
384  mutt_message ("Trouble adding alias: %s.", strerror(errno));
385  else
386  mutt_message (_("Alias added."));
387  }
388  else
389  mutt_perror (buf);
390 
391  return;
392 
393  fseek_err:
394  mutt_perror (_("Error seeking in alias file"));
395  safe_fclose (&rc);
396  return;
397 }

404 int mutt_check_alias_name (const char *s, char *dest, size_t destlen)
405 {
406  wchar_t wc;
407  mbstate_t mb;
408  size_t l;
409  int rv = 0, bad = 0, dry = !dest || !destlen;
410 
411  memset (&mb, 0, sizeof (mbstate_t));
412 
413  if (!dry)
414  destlen--;
415  for (; s && *s && (dry || destlen) &&
416  (l = mbrtowc (&wc, s, MB_CUR_MAX, &mb)) != 0;
417  s += l, destlen -= l)
418  {
419  bad = l == (size_t)(-1) || l == (size_t)(-2); /* conversion error */
420  bad = bad || (!dry && l > destlen); /* too few room for mb char */
421  if (l == 1)
422  bad = bad || (strchr ("-_+=.", *s) == NULL && !iswalnum (wc));
423  else
424  bad = bad || !iswalnum (wc);
425  if (bad)
426  {
427  if (dry)
428  return -1;
429  if (l == (size_t)(-1))
430  memset (&mb, 0, sizeof (mbstate_t));
431  *dest++ = '_';
432  rv = -1;
433  }
434  else if (!dry)
435  {
436  memcpy (dest, s, l);
437  dest += l;
438  }
439  }
440  if (!dry)
441  *dest = 0;
442  return rv;
443 }

483 /* alias_complete() -- alias completion routine
484  *
485  * given a partial alias, this routine attempts to fill in the alias
486  * from the alias list as much as possible. if given empty search string
487  * or found nothing, present all aliases
488  */
489 int mutt_alias_complete (char *s, size_t buflen)
490 {
491  ALIAS *a = Aliases;
492  ALIAS *a_list = NULL, *a_cur = NULL;
493  char bestname[HUGE_STRING];
494  int i;
495 
496 #ifndef min
497 #define min(a,b) ((a<b)?a:b)
498 #endif
499 
500  if (s[0] != 0) /* avoid empty string as strstr argument */
501  {
502  memset (bestname, 0, sizeof (bestname));
503 
504  while (a)
505  {
506  if (a->name && strstr (a->name, s) == a->name)
507  {
508  if (!bestname[0]) /* init */
509  strfcpy (bestname, a->name,
510  min (mutt_strlen (a->name) + 1, sizeof (bestname)));
511  else
512  {
513  for (i = 0 ; a->name[i] && a->name[i] == bestname[i] ; i++)
514  ;
515  bestname[i] = 0;
516  }
517  }
518  a = a->next;
519  }
520 
521  if (bestname[0] != 0)
522  {
523  if (mutt_strcmp (bestname, s) != 0)
524  {
525  /* we are adding something to the completion */
526  strfcpy (s, bestname, mutt_strlen (bestname) + 1);
527  return 1;
528  }
529 
530  /* build alias list and show it */
531 
532  a = Aliases;
533  while (a)
534  {
535  if (a->name && (strstr (a->name, s) == a->name))
536  {
537  if (!a_list) /* init */
538  a_cur = a_list = (ALIAS *) safe_malloc (sizeof (ALIAS));
539  else
540  {
541  a_cur->next = (ALIAS *) safe_malloc (sizeof (ALIAS));
542  a_cur = a_cur->next;
543  }
544  memcpy (a_cur, a, sizeof (ALIAS));
545  a_cur->next = NULL;
546  }
547  a = a->next;
548  }
549  }
550  }
551 
552  bestname[0] = 0;
553  mutt_alias_menu (bestname, sizeof(bestname), a_list ? a_list : Aliases);
554  if (bestname[0] != 0)
555  strfcpy (s, bestname, buflen);
556 
557  /* free the alias list */
558  while (a_list)
559  {
560  a_cur = a_list;
561  a_list = a_list->next;
562  FREE (&a_cur);
563  }
564 
565  /* remove any aliases marked for deletion */
566  a_list = NULL;
567  for (a_cur = Aliases; a_cur;)
568  {
569  if (a_cur->del)
570  {
571  if (a_list)
572  a_list->next = a_cur->next;
573  else
574  Aliases = a_cur->next;
575 
576  a_cur->next = NULL;
577  mutt_free_alias (&a_cur);
578 
579  if (a_list)
580  a_cur = a_list;
581  else
582  a_cur = Aliases;
583  }
584  else
585  {
586  a_list = a_cur;
587  a_cur = a_cur->next;
588  }
589  }
590 
591  return 0;
592 }

606 int mutt_addr_is_user (ADDRESS *addr)
607 {

609 
610  /* NULL address is assumed to be the user. */
611  if (!addr)
612  {
613  dprint (5, (debugfile, "mutt_addr_is_user: yes, NULL address\n"));
614  return 1;
615  }
616  if (!addr->mailbox)
617  {
618  dprint (5, (debugfile, "mutt_addr_is_user: no, no mailbox\n"));
619  return 0;
620  }
621 
622  if (ascii_strcasecmp (addr->mailbox, Username) == 0)
623  {
624  dprint (5, (debugfile, "mutt_addr_is_user: yes, %s = %s\n", addr->mailbox, Username));
625  return 1;
626  }
627  if (string_is_address(addr->mailbox, Username, Hostname))
628  {
629  dprint (5, (debugfile, "mutt_addr_is_user: yes, %s = %s @ %s \n", addr->mailbox, Username, Hostname));
630  return 1;
631  }
632  fqdn = mutt_fqdn (0);
633  if (string_is_address(addr->mailbox, Username, fqdn))
634  {
635  dprint (5, (debugfile, "mutt_addr_is_user: yes, %s = %s @ %s \n", addr->mailbox, Username, NONULL(fqdn)));
636  return 1;
637  }
638  fqdn = mutt_fqdn (1);
639  if (string_is_address(addr->mailbox, Username, fqdn))
640  {
641  dprint (5, (debugfile, "mutt_addr_is_user: yes, %s = %s @ %s \n", addr->mailbox, Username, NONULL(fqdn)));
642  return 1;
643  }
644 
645  if (From && !ascii_strcasecmp (From->mailbox, addr->mailbox))
646  {
647  dprint (5, (debugfile, "mutt_addr_is_user: yes, %s = %s\n", addr->mailbox, From->mailbox));
648  return 1;
649  }
650 
651  if (mutt_match_rx_list (addr->mailbox, Alternates))
652  {
653  dprint (5, (debugfile, "mutt_addr_is_user: yes, %s matched by alternates.\n", addr->mailbox));

655  dprint (5, (debugfile, "mutt_addr_is_user: but, %s matched by unalternates.\n", addr->mailbox));
656  else
657  return 1;

661  return 0;
662 }
