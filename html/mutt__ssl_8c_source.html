89 /* mutt_ssl_starttls: Negotiate TLS over an already opened connection.
90  * TODO: Merge this code better with ssl_socket_open. */
91 int mutt_ssl_starttls (CONNECTION* conn)
92 {
93  sslsockdata* ssldata;
94  int maxbits;
95  long ssl_options = 0;
96 
97  if (ssl_init())
98  goto bail;
99 
100  ssldata = (sslsockdata*) safe_calloc (1, sizeof (sslsockdata));
101  /* the ssl_use_xxx protocol options don't apply. We must use TLS in TLS.
102  *
103  * However, we need to be able to negotiate amongst various TLS versions,
104  * which at present can only be done with the SSLv23_client_method;
105  * TLSv1_client_method gives us explicitly TLSv1.0, not 1.1 or 1.2 (True as
106  * of OpenSSL 1.0.1c)
107  */
108  if (! (ssldata->ctx = SSL_CTX_new (SSLv23_client_method())))
109  {
110  dprint (1, (debugfile, "mutt_ssl_starttls: Error allocating SSL_CTX\n"));
111  goto bail_ssldata;
112  }
113 #ifdef SSL_OP_NO_TLSv1_2
114  if (!option(OPTTLSV1_2))
115  ssl_options |= SSL_OP_NO_TLSv1_2;
116 #endif
117 #ifdef SSL_OP_NO_TLSv1_1
118  if (!option(OPTTLSV1_1))
119  ssl_options |= SSL_OP_NO_TLSv1_1;
120 #endif
121 #ifdef SSL_OP_NO_TLSv1
122  if (!option(OPTTLSV1))
123  ssl_options |= SSL_OP_NO_TLSv1;
124 #endif
125  /* these are always set */
126 #ifdef SSL_OP_NO_SSLv3
127  ssl_options |= SSL_OP_NO_SSLv3;
128 #endif
129 #ifdef SSL_OP_NO_SSLv2
130  ssl_options |= SSL_OP_NO_SSLv2;
131 #endif
132  if (! SSL_CTX_set_options(ssldata->ctx, ssl_options))
133  {
134  dprint(1, (debugfile, "mutt_ssl_starttls: Error setting options to %ld\n", ssl_options));
135  goto bail_ctx;
136  }
137 
138  if (option (OPTSSLSYSTEMCERTS))
139  {
140  if (! SSL_CTX_set_default_verify_paths (ssldata->ctx))
141  {
142  dprint (1, (debugfile, "mutt_ssl_starttls: Error setting default verify paths\n"));
143  goto bail_ctx;
144  }
145  }
146 
147  if (SslCertFile && ! SSL_CTX_load_verify_locations (ssldata->ctx, SslCertFile, NULL))
148  dprint (1, (debugfile, "mutt_ssl_starttls: Error loading trusted certificates\n"));
149 
150  ssl_get_client_cert(ssldata, conn);
151 
152  if (SslCiphers) {
153  if (!SSL_CTX_set_cipher_list (ssldata->ctx, SslCiphers)) {
154  dprint (1, (debugfile, "mutt_ssl_starttls: Could not select preferred ciphers\n"));
155  goto bail_ctx;
156  }
157  }
158 
159  if (! (ssldata->ssl = SSL_new (ssldata->ctx)))
160  {
161  dprint (1, (debugfile, "mutt_ssl_starttls: Error allocating SSL\n"));
162  goto bail_ctx;
163  }
164 
165  if (SSL_set_fd (ssldata->ssl, conn->fd) != 1)
166  {
167  dprint (1, (debugfile, "mutt_ssl_starttls: Error setting fd\n"));
168  goto bail_ssl;
169  }
170 
171  if (ssl_negotiate (conn, ssldata))
172  goto bail_ssl;
173 
174  ssldata->isopen = 1;
175 
176  /* hmm. watch out if we're starting TLS over any method other than raw. */
177  conn->sockdata = ssldata;
178  conn->conn_read = ssl_socket_read;
179  conn->conn_write = ssl_socket_write;
180  conn->conn_close = tls_close;
181 
182  conn->ssf = SSL_CIPHER_get_bits (SSL_get_current_cipher (ssldata->ssl),
183  &maxbits);
184 
185  return 0;
186 
187  bail_ssl:
188  FREE (&ssldata->ssl);
189  bail_ctx:
190  FREE (&ssldata->ctx);
191  bail_ssldata:
192  FREE (&ssldata);
193  bail:
194  return -1;
195 }

197 /*
198  * OpenSSL library needs to be fed with sufficient entropy. On systems
199  * with /dev/urandom, this is done transparently by the library itself,
200  * on other systems we need to fill the entropy pool ourselves.
201  *
202  * Even though only OpenSSL 0.9.5 and later will complain about the
203  * lack of entropy, we try to our best and fill the pool with older
204  * versions also. (That's the reason for the ugly #ifdefs and macros,
205  * otherwise I could have simply #ifdef'd the whole ssl_init funcion)
206  */
207 static int ssl_init (void)
208 {
209  char path[_POSIX_PATH_MAX];
210  static unsigned char init_complete = 0;
211 
212  if (init_complete)
213  return 0;
214 
215  if (! HAVE_ENTROPY())
216  {
217  /* load entropy from files */
218  add_entropy (SslEntropyFile);
219  add_entropy (RAND_file_name (path, sizeof (path)));
220 
221  /* load entropy from egd sockets */
222 #ifdef HAVE_RAND_EGD
223  add_entropy (getenv ("EGDSOCKET"));
224  snprintf (path, sizeof(path), "%s/.entropy", NONULL(Homedir));
225  add_entropy (path);
226  add_entropy ("/tmp/entropy");
227 #endif
228 
229  /* shuffle $RANDFILE (or ~/.rnd if unset) */
230  RAND_write_file (RAND_file_name (path, sizeof (path)));
231  mutt_clear_error ();
232  if (! HAVE_ENTROPY())
233  {
234  mutt_error (_("Failed to find enough entropy on your system"));
235  mutt_sleep (2);
236  return -1;
237  }
238  }
239 
240  /* I don't think you can do this just before reading the error. The call
241  * itself might clobber the last SSL error. */
242  SSL_load_error_strings();
243  SSL_library_init();
244  init_complete = 1;
245  return 0;
246 }

248 static int add_entropy (const char *file)
249 {
250  struct stat st;
251  int n = -1;
252 
253  if (!file) return 0;
254 
255  if (stat (file, &st) == -1)
256  return errno == ENOENT ? 0 : -1;
257 
258  mutt_message (_("Filling entropy pool: %s...\n"),
259  file);
260 
261  /* check that the file permissions are secure */
262  if (st.st_uid != getuid () ||
263  ((st.st_mode & (S_IWGRP | S_IRGRP)) != 0) ||
264  ((st.st_mode & (S_IWOTH | S_IROTH)) != 0))
265  {
266  mutt_error (_("%s has insecure permissions!"), file);
267  mutt_sleep (2);
268  return -1;
269  }
270 
271 #ifdef HAVE_RAND_EGD
272  n = RAND_egd (file);
273 #endif
274  if (n <= 0)
275  n = RAND_load_file (file, -1);
276 
277 #ifndef HAVE_RAND_STATUS
278  if (n > 0) entropy_byte_count += n;
279 #endif
280  return n;
281 }

309 {
310  sslsockdata *data = conn->sockdata;

312

323

344 static int ssl_socket_open (CONNECTION * conn)
345 {
346  sslsockdata *data;
347  int maxbits;
348 
349  if (raw_socket_open (conn) < 0)
350  return -1;
351 
352  data = (sslsockdata *) safe_calloc (1, sizeof (sslsockdata));
353  conn->sockdata = data;
354 
355  if (! (data->ctx = SSL_CTX_new (SSLv23_client_method ())))
356  {
357  /* L10N: an SSL context is a data structure returned by the OpenSSL
358  function SSL_CTX_new(). In this case it returned NULL: an
359  error condition. */
360  mutt_error (_("Unable to create SSL context"));
361  ssl_dprint_err_stack ();
362  mutt_socket_close (conn);
363  return -1;
364  }
365 
366  /* disable SSL protocols as needed */
367  if (!option(OPTTLSV1))
368  {
369  SSL_CTX_set_options(data->ctx, SSL_OP_NO_TLSv1);
370  }
371  /* TLSv1.1/1.2 support was added in OpenSSL 1.0.1, but some OS distros such
372  * as Fedora 17 are on OpenSSL 1.0.0.
373  */
374 #ifdef SSL_OP_NO_TLSv1_1
375  if (!option(OPTTLSV1_1))
376  {
377  SSL_CTX_set_options(data->ctx, SSL_OP_NO_TLSv1_1);
378  }
379 #endif
380 #ifdef SSL_OP_NO_TLSv1_2
381  if (!option(OPTTLSV1_2))
382  {
383  SSL_CTX_set_options(data->ctx, SSL_OP_NO_TLSv1_2);
384  }
385 #endif
386  if (!option(OPTSSLV2))
387  {
388  SSL_CTX_set_options(data->ctx, SSL_OP_NO_SSLv2);
389  }
390  if (!option(OPTSSLV3))
391  {
392  SSL_CTX_set_options(data->ctx, SSL_OP_NO_SSLv3);
393  }
394 
395  if (option (OPTSSLSYSTEMCERTS))
396  {
397  if (! SSL_CTX_set_default_verify_paths (data->ctx))
398  {
399  dprint (1, (debugfile, "ssl_socket_open: Error setting default verify paths\n"));
400  mutt_socket_close (conn);
401  return -1;
402  }
403  }
404 
405  if (SslCertFile && ! SSL_CTX_load_verify_locations (data->ctx, SslCertFile, NULL))
406  dprint (1, (debugfile, "ssl_socket_open: Error loading trusted certificates\n"));
407 
408  ssl_get_client_cert(data, conn);
409 
410  if (SslCiphers) {
411  SSL_CTX_set_cipher_list (data->ctx, SslCiphers);
412  }
413 
414  data->ssl = SSL_new (data->ctx);
415  SSL_set_fd (data->ssl, conn->fd);
416 
417  if (ssl_negotiate(conn, data))
418  {
419  mutt_socket_close (conn);
420  return -1;
421  }
422 
423  data->isopen = 1;
424 
425  conn->ssf = SSL_CIPHER_get_bits (SSL_get_current_cipher (data->ssl),
426  &maxbits);
427 
428  return 0;
429 }

431 /* ssl_negotiate: After SSL state has been initialized, attempt to negotiate
432  * SSL over the wire, including certificate checks. */
433 static int ssl_negotiate (CONNECTION *conn, sslsockdata* ssldata)
434 {
435  int err;
436  const char* errmsg;
437 
438  if ((HostExDataIndex = SSL_get_ex_new_index (0, "host", NULL, NULL, NULL)) == -1)
439  {
440  dprint (1, (debugfile, "failed to get index for application specific data\n"));
441  return -1;
442  }
443 
444  if (! SSL_set_ex_data (ssldata->ssl, HostExDataIndex, conn->account.host))
445  {
446  dprint (1, (debugfile, "failed to save hostname in SSL structure\n"));
447  return -1;
448  }
449 
450  SSL_set_verify (ssldata->ssl, SSL_VERIFY_PEER, ssl_verify_callback);
451  SSL_set_mode (ssldata->ssl, SSL_MODE_AUTO_RETRY);
452 
453 #if (OPENSSL_VERSION_NUMBER >= 0x0090806fL) && !defined(OPENSSL_NO_TLSEXT)
454  /* TLS Virtual-hosting requires that the server present the correct
455  * certificate; to do this, the ServerNameIndication TLS extension is used.
456  * If TLS is negotiated, and OpenSSL is recent enough that it might have
457  * support, and support was enabled when OpenSSL was built, mutt supports
458  * sending the hostname we think we're connecting to, so a server can send
459  * back the correct certificate.
460  * This has been tested over SMTP against Exim 4.80.
461  * Not yet found an IMAP server which supports this. */
462  SSL_set_tlsext_host_name (ssldata->ssl, conn->account.host);
463 #endif
464 
465  if ((err = SSL_connect (ssldata->ssl)) != 1)
466  {
467  switch (SSL_get_error (ssldata->ssl, err))
468  {
469  case SSL_ERROR_SYSCALL:
470  errmsg = _("I/O error");
471  break;
472  case SSL_ERROR_SSL:
473  errmsg = ERR_error_string (ERR_get_error (), NULL);
474  break;
475  default:
476  errmsg = _("unknown error");
477  }
478 
479  mutt_error (_("SSL failed: %s"), errmsg);
480  mutt_sleep (1);
481 
482  return -1;
483  }
484 
485  return 0;
486 }

521 static void ssl_err (sslsockdata *data, int err)
522 {
523  const char* errmsg;
524  unsigned long sslerr;
525 
526  switch (SSL_get_error (data->ssl, err))
527  {
528  case SSL_ERROR_NONE:
529  return;
530  case SSL_ERROR_ZERO_RETURN:
531  errmsg = "SSL connection closed";
532  data->isopen = 0;
533  break;
534  case SSL_ERROR_WANT_READ:
535  errmsg = "retry read";
536  break;
537  case SSL_ERROR_WANT_WRITE:
538  errmsg = "retry write";
539  break;
540  case SSL_ERROR_WANT_CONNECT:
541  errmsg = "retry connect";
542  break;
543  case SSL_ERROR_WANT_ACCEPT:
544  errmsg = "retry accept";
545  break;
546  case SSL_ERROR_WANT_X509_LOOKUP:
547  errmsg = "retry x509 lookup";
548  break;
549  case SSL_ERROR_SYSCALL:
550  errmsg = "I/O error";
551  data->isopen = 0;
552  break;
553  case SSL_ERROR_SSL:
554  sslerr = ERR_get_error ();
555  switch (sslerr)
556  {
557  case 0:
558  switch (err)
559  {
560  case 0:
561  errmsg = "EOF";
562  break;
563  default:
564  errmsg = strerror(errno);
565  }
566  break;
567  default:
568  errmsg = ERR_error_string (sslerr, NULL);
569  }
570  break;
571  default:
572  errmsg = "unknown error";
573  }
574 
575  dprint (1, (debugfile, "SSL error: %s\n", errmsg));
576 }

613 static void x509_fingerprint (char *s, int l, X509 * cert)
614 {
615  unsigned char md[EVP_MAX_MD_SIZE];
616  unsigned int n;
617  int j;
618 
619  if (!X509_digest (cert, EVP_md5 (), md, &n))
620  {
621  snprintf (s, l, _("[unable to calculate]"));
622  }
623  else
624  {
625  for (j = 0; j < (int) n; j++)
626  {
627  char ch[8];
628  snprintf (ch, 8, "%02X%s", md[j], (j % 2 ? " " : ""));
629  safe_strcat (s, l, ch);
630  }
631  }

652 static int compare_certificates (X509 *cert, X509 *peercert,
653  unsigned char *peermd, unsigned int peermdlen)
654 {
655  unsigned char md[EVP_MAX_MD_SIZE];
656  unsigned int mdlen;
657 
658  /* Avoid CPU-intensive digest calculation if the certificates are
659  * not even remotely equal.
660  */
661  if (X509_subject_name_cmp (cert, peercert) != 0 ||
662  X509_issuer_name_cmp (cert, peercert) != 0)
663  return -1;
664 
665  if (!X509_digest (cert, EVP_sha1(), md, &mdlen) || peermdlen != mdlen)
666  return -1;
667 
668  if (memcmp(peermd, md, mdlen) != 0)
669  return -1;
670 
671  return 0;
672 }

674 static int check_certificate_cache (X509 *peercert)
675 {
676  unsigned char peermd[EVP_MAX_MD_SIZE];
677  unsigned int peermdlen;
678  X509 *cert;
679  int i;
680 
681  if (!X509_digest (peercert, EVP_sha1(), peermd, &peermdlen)
682  || !SslSessionCerts)
683  {
684  return 0;
685  }
686 
687  for (i = sk_X509_num (SslSessionCerts); i-- > 0;)
688  {
689  cert = sk_X509_value (SslSessionCerts, i);
690  if (!compare_certificates (cert, peercert, peermd, peermdlen))
691  {
692  return 1;
693  }
694  }
695 
696  return 0;
697 }

699 static int check_certificate_by_digest (X509 *peercert)

786 static int check_host (X509 *x509cert, const char *hostname, char *err, size_t errlen)
787 {
788  int i, rc = 0;
789  /* hostname in ASCII format: */
790  char *hostname_ascii = NULL;
791  /* needed to get the common name: */
792  X509_NAME *x509_subject;
793  char *buf = NULL;
794  int bufsize;
795  /* needed to get the DNS subjectAltNames: */
796  STACK_OF(GENERAL_NAME) *subj_alt_names;
797  int subj_alt_names_count;
798  GENERAL_NAME *subj_alt_name;
799  /* did we find a name matching hostname? */
800  int match_found;
801 
802  /* Check if 'hostname' matches the one of the subjectAltName extensions of
803  * type DNS or the Common Name (CN). */
804 
805 #ifdef HAVE_LIBIDN
806  if (idna_to_ascii_lz(hostname, &hostname_ascii, 0) != IDNA_SUCCESS)
807  {
808  hostname_ascii = safe_strdup(hostname);
809  }
810 #else
811  hostname_ascii = safe_strdup(hostname);
812 #endif
813 
814  /* Try the DNS subjectAltNames. */
815  match_found = 0;
816  if ((subj_alt_names = X509_get_ext_d2i(x509cert, NID_subject_alt_name,
817  NULL, NULL)))
818  {
819  subj_alt_names_count = sk_GENERAL_NAME_num(subj_alt_names);
820  for (i = 0; i < subj_alt_names_count; i++)
821  {
822  subj_alt_name = sk_GENERAL_NAME_value(subj_alt_names, i);
823  if (subj_alt_name->type == GEN_DNS)
824  {
825  if (subj_alt_name->d.ia5->length >= 0 &&
826  mutt_strlen((char *)subj_alt_name->d.ia5->data) == (size_t)subj_alt_name->d.ia5->length &&
827  (match_found = hostname_match(hostname_ascii,
828  (char *)(subj_alt_name->d.ia5->data))))
829  {
830  break;
831  }
832  }
833  }
834  }
835 
836  if (!match_found)
837  {
838  /* Try the common name */
839  if (!(x509_subject = X509_get_subject_name(x509cert)))
840  {
841  if (err && errlen)
842  strfcpy (err, _("cannot get certificate subject"), errlen);
843  goto out;
844  }
845 
846  /* first get the space requirements */
847  bufsize = X509_NAME_get_text_by_NID(x509_subject, NID_commonName,
848  NULL, 0);
849  if (bufsize == -1)
850  {
851  if (err && errlen)
852  strfcpy (err, _("cannot get certificate common name"), errlen);
853  goto out;
854  }
855  bufsize++; /* space for the terminal nul char */
856  buf = safe_malloc((size_t)bufsize);
857  if (X509_NAME_get_text_by_NID(x509_subject, NID_commonName,
858  buf, bufsize) == -1)
859  {
860  if (err && errlen)
861  strfcpy (err, _("cannot get certificate common name"), errlen);
862  goto out;
863  }
864  /* cast is safe since bufsize is incremented above, so bufsize-1 is always
865  * zero or greater.
866  */
867  if (mutt_strlen(buf) == (size_t)bufsize - 1) {
868  match_found = hostname_match(hostname_ascii, buf);
869  }
870  }
871 
872  if (!match_found)
873  {
874  if (err && errlen)
875  snprintf (err, errlen, _("certificate owner does not match hostname %s"),
876  hostname);
877  goto out;
878  }
879 
880  rc = 1;
881 
882 out:
883  FREE(&buf);
884  FREE(&hostname_ascii);
885 
886  return rc;
887 }

898  * sent by the peer, starting from the root; returning 1 means that the given

900 static int ssl_verify_callback (int preverify_ok, X509_STORE_CTX *ctx)
901 {

904  int len, pos;
905  X509 *cert;
906  SSL *ssl;
907

911  return 0;

914  {

921  len = sk_X509_num (X509_STORE_CTX_get_chain (ctx));
922 
923  dprint (1, (debugfile, "ssl_verify_callback: checking cert chain entry %s (preverify: %d)\n",
924  X509_NAME_oneline (X509_get_subject_name (cert),
925  buf, sizeof (buf)), preverify_ok));
926 
927  /* check session cache first */
928  if (check_certificate_cache (cert))
929  {
930  dprint (2, (debugfile, "ssl_verify_callback: using cached certificate\n"));
931  return 1;
932  }

935  buf[0] = 0;
936  if (pos == 0 && option (OPTSSLVERIFYHOST) != MUTT_NO)
937  {
938  if (!check_host (cert, host, buf, sizeof (buf)))
939  {
940  mutt_error (_("Certificate host check failed: %s"), buf);
941  mutt_sleep (2);
942  return interactive_check_cert (cert, pos, len);
943  }
944  dprint (2, (debugfile, "ssl_verify_callback: hostname check passed\n"));
945  }

950  if (SslCertFile && check_certificate_by_digest (cert))

963  }
964 #endif
965

968  }
