44 static void print_gss_error(OM_uint32 err_maj, OM_uint32 err_min)
45 {
46  OM_uint32 maj_stat, min_stat;
47  OM_uint32 msg_ctx = 0;
48  gss_buffer_desc status_string;
49  char buf_maj[512];
50  char buf_min[512];
51 
52  do
53  {
54  maj_stat = gss_display_status (&min_stat,
55  err_maj,
56  GSS_C_GSS_CODE,
57  GSS_C_NO_OID,
58  &msg_ctx,
59  &status_string);
60  if (GSS_ERROR(maj_stat))
61  break;
62  strncpy(buf_maj, (char*) status_string.value, sizeof(buf_maj));
63  gss_release_buffer(&min_stat, &status_string);
64 
65  maj_stat = gss_display_status (&min_stat,
66  err_min,
67  GSS_C_MECH_CODE,
68  GSS_C_NULL_OID,
69  &msg_ctx,
70  &status_string);
71  if (!GSS_ERROR(maj_stat))
72  {
73  strncpy(buf_min, (char*) status_string.value, sizeof(buf_min));
74  gss_release_buffer(&min_stat, &status_string);
75  }
76  } while (!GSS_ERROR(maj_stat) && msg_ctx != 0);
77 
78  dprint (2, (debugfile, "((%s:%d )(%s:%d))", buf_maj, err_maj, buf_min, err_min));
79 }

82 imap_auth_res_t imap_auth_gss (IMAP_DATA* idata, const char* method)
83 {
84  gss_buffer_desc request_buf, send_token;
85  gss_buffer_t sec_token;
86  gss_name_t target_name;
87  gss_ctx_id_t context;
88 #ifdef DEBUG
89  gss_OID mech_name;
90  char server_conf_flags;
91 #endif
92  gss_qop_t quality;
93  int cflags;
94  OM_uint32 maj_stat, min_stat;
95  char buf1[GSS_BUFSIZE], buf2[GSS_BUFSIZE];
96  unsigned long buf_size;
97  int rc;
98 
99  if (!mutt_bit_isset (idata->capabilities, AGSSAPI))
100  return IMAP_AUTH_UNAVAIL;
101 
102  if (mutt_account_getuser (&idata->conn->account))
103  return IMAP_AUTH_FAILURE;
104 
105  /* get an IMAP service ticket for the server */
106  snprintf (buf1, sizeof (buf1), "imap@%s", idata->conn->account.host);
107  request_buf.value = buf1;
108  request_buf.length = strlen (buf1) + 1;
109  maj_stat = gss_import_name (&min_stat, &request_buf, gss_nt_service_name,
110  &target_name);
111  if (maj_stat != GSS_S_COMPLETE)
112  {
113  dprint (2, (debugfile, "Couldn't get service name for [%s]\n", buf1));
114  return IMAP_AUTH_UNAVAIL;
115  }
116 #ifdef DEBUG
117  else if (debuglevel >= 2)
118  {
119  maj_stat = gss_display_name (&min_stat, target_name, &request_buf,
120  &mech_name);
121  dprint (2, (debugfile, "Using service name [%s]\n",
122  (char*) request_buf.value));
123  maj_stat = gss_release_buffer (&min_stat, &request_buf);
124  }
125 #endif
126  /* Acquire initial credentials - without a TGT GSSAPI is UNAVAIL */
127  sec_token = GSS_C_NO_BUFFER;
128  context = GSS_C_NO_CONTEXT;
129 
130  /* build token */
131  maj_stat = gss_init_sec_context (&min_stat, GSS_C_NO_CREDENTIAL, &context,
132  target_name, GSS_C_NO_OID, GSS_C_MUTUAL_FLAG | GSS_C_SEQUENCE_FLAG, 0,
133  GSS_C_NO_CHANNEL_BINDINGS, sec_token, NULL, &send_token,
134  (unsigned int*) &cflags, NULL);
135  if (maj_stat != GSS_S_COMPLETE && maj_stat != GSS_S_CONTINUE_NEEDED)
136  {
137  print_gss_error(maj_stat, min_stat);
138  dprint (1, (debugfile, "Error acquiring credentials - no TGT?\n"));
139  gss_release_name (&min_stat, &target_name);
140 
141  return IMAP_AUTH_UNAVAIL;
142  }
143 
144  /* now begin login */
145  mutt_message _("Authenticating (GSSAPI)...");
146 
147  imap_cmd_start (idata, "AUTHENTICATE GSSAPI");
148 
149  /* expect a null continuation response ("+") */
150  do
151  rc = imap_cmd_step (idata);
152  while (rc == IMAP_CMD_CONTINUE);
153 
154  if (rc != IMAP_CMD_RESPOND)
155  {
156  dprint (2, (debugfile, "Invalid response from server: %s\n", buf1));
157  gss_release_name (&min_stat, &target_name);
158  goto bail;
159  }
160 
161  /* now start the security context initialisation loop... */
162  dprint (2, (debugfile, "Sending credentials\n"));
163  mutt_to_base64 (buf1, send_token.value, send_token.length,
164  sizeof (buf1) - 2);
165  gss_release_buffer (&min_stat, &send_token);
166  safe_strcat (buf1, sizeof (buf1), "\r\n");
167  mutt_socket_write (idata->conn, buf1);
168 
169  while (maj_stat == GSS_S_CONTINUE_NEEDED)
170  {
171  /* Read server data */
172  do
173  rc = imap_cmd_step (idata);
174  while (rc == IMAP_CMD_CONTINUE);
175 
176  if (rc != IMAP_CMD_RESPOND)
177  {
178  dprint (1, (debugfile, "Error receiving server response.\n"));
179  gss_release_name (&min_stat, &target_name);
180  goto bail;
181  }
182 
183  request_buf.length = mutt_from_base64 (buf2, idata->buf + 2);
184  request_buf.value = buf2;
185  sec_token = &request_buf;
186 
187  /* Write client data */
188  maj_stat = gss_init_sec_context (&min_stat, GSS_C_NO_CREDENTIAL, &context,
189  target_name, GSS_C_NO_OID, GSS_C_MUTUAL_FLAG | GSS_C_SEQUENCE_FLAG, 0,
190  GSS_C_NO_CHANNEL_BINDINGS, sec_token, NULL, &send_token,
191  (unsigned int*) &cflags, NULL);
192  if (maj_stat != GSS_S_COMPLETE && maj_stat != GSS_S_CONTINUE_NEEDED)
193  {
194  print_gss_error(maj_stat, min_stat);
195  dprint (1, (debugfile, "Error exchanging credentials\n"));
196  gss_release_name (&min_stat, &target_name);
197 
198  goto err_abort_cmd;
199  }
200  mutt_to_base64 (buf1, send_token.value, send_token.length,
201  sizeof (buf1) - 2);
202  gss_release_buffer (&min_stat, &send_token);
203  safe_strcat (buf1, sizeof (buf1), "\r\n");
204  mutt_socket_write (idata->conn, buf1);
205  }
206 
207  gss_release_name (&min_stat, &target_name);
208 
209  /* get security flags and buffer size */
210  do
211  rc = imap_cmd_step (idata);
212  while (rc == IMAP_CMD_CONTINUE);
213 
214  if (rc != IMAP_CMD_RESPOND)
215  {
216  dprint (1, (debugfile, "Error receiving server response.\n"));
217  goto bail;
218  }
219  request_buf.length = mutt_from_base64 (buf2, idata->buf + 2);
220  request_buf.value = buf2;
221 
222  maj_stat = gss_unwrap (&min_stat, context, &request_buf, &send_token,
223  &cflags, &quality);
224  if (maj_stat != GSS_S_COMPLETE)
225  {
226  print_gss_error(maj_stat, min_stat);
227  dprint (2, (debugfile, "Couldn't unwrap security level data\n"));
228  gss_release_buffer (&min_stat, &send_token);
229  goto err_abort_cmd;
230  }
231  dprint (2, (debugfile, "Credential exchange complete\n"));
232 
233  /* first octet is security levels supported. We want NONE */
234 #ifdef DEBUG
235  server_conf_flags = ((char*) send_token.value)[0];
236 #endif
237  if ( !(((char*) send_token.value)[0] & GSS_AUTH_P_NONE) )
238  {
239  dprint (2, (debugfile, "Server requires integrity or privacy\n"));
240  gss_release_buffer (&min_stat, &send_token);
241  goto err_abort_cmd;
242  }
243 
244  /* we don't care about buffer size if we don't wrap content. But here it is */
245  ((char*) send_token.value)[0] = 0;
246  buf_size = ntohl (*((long *) send_token.value));
247  gss_release_buffer (&min_stat, &send_token);
248  dprint (2, (debugfile, "Unwrapped security level flags: %c%c%c\n",
249  server_conf_flags & GSS_AUTH_P_NONE ? 'N' : '-',
250  server_conf_flags & GSS_AUTH_P_INTEGRITY ? 'I' : '-',
251  server_conf_flags & GSS_AUTH_P_PRIVACY ? 'P' : '-'));
252  dprint (2, (debugfile, "Maximum GSS token size is %ld\n", buf_size));
253 
254  /* agree to terms (hack!) */
255  buf_size = htonl (buf_size); /* not relevant without integrity/privacy */
256  memcpy (buf1, &buf_size, 4);
257  buf1[0] = GSS_AUTH_P_NONE;
258  /* server decides if principal can log in as user */
259  strncpy (buf1 + 4, idata->conn->account.user, sizeof (buf1) - 4);
260  request_buf.value = buf1;
261  request_buf.length = 4 + strlen (idata->conn->account.user) + 1;
262  maj_stat = gss_wrap (&min_stat, context, 0, GSS_C_QOP_DEFAULT, &request_buf,
263  &cflags, &send_token);
264  if (maj_stat != GSS_S_COMPLETE)
265  {
266  dprint (2, (debugfile, "Error creating login request\n"));
267  goto err_abort_cmd;
268  }
269 
270  mutt_to_base64 (buf1, send_token.value, send_token.length,
271  sizeof (buf1) - 2);
272  dprint (2, (debugfile, "Requesting authorisation as %s\n",
273  idata->conn->account.user));
274  safe_strcat (buf1, sizeof (buf1), "\r\n");
275  mutt_socket_write (idata->conn, buf1);
276 
277  /* Joy of victory or agony of defeat? */
278  do
279  rc = imap_cmd_step (idata);
280  while (rc == IMAP_CMD_CONTINUE);
281  if (rc == IMAP_CMD_RESPOND)
282  {
283  dprint (1, (debugfile, "Unexpected server continuation request.\n"));
284  goto err_abort_cmd;
285  }
286  if (imap_code (idata->buf))
287  {
288  /* flush the security context */
289  dprint (2, (debugfile, "Releasing GSS credentials\n"));
290  maj_stat = gss_delete_sec_context (&min_stat, &context, &send_token);
291  if (maj_stat != GSS_S_COMPLETE)
292  dprint (1, (debugfile, "Error releasing credentials\n"));
293 
294  /* send_token may contain a notification to the server to flush
295  * credentials. RFC 1731 doesn't specify what to do, and since this
296  * support is only for authentication, we'll assume the server knows
297  * enough to flush its own credentials */
298  gss_release_buffer (&min_stat, &send_token);
299 
300  return IMAP_AUTH_SUCCESS;
301  }
302  else
303  goto bail;
304 
305  err_abort_cmd:
306  mutt_socket_write (idata->conn, "*\r\n");
307  do
308  rc = imap_cmd_step (idata);
309  while (rc == IMAP_CMD_CONTINUE);
310 
311  bail:
312  mutt_error _("GSSAPI authentication failed.");
313  mutt_sleep (2);
314  return IMAP_AUTH_FAILURE;
315 }
