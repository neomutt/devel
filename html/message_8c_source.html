55 /* imap_read_headers:
56  * Changed to read many headers instead of just one. It will return the
57  * msgno of the last message read. It will return a value other than
58  * msgend if mail comes in while downloading headers (in theory).
59  */
60 int imap_read_headers (IMAP_DATA* idata, int msgbegin, int msgend)
61 {
62  CONTEXT* ctx;
63  char *hdrreq = NULL;
64  FILE *fp;
65  char tempfile[_POSIX_PATH_MAX];
66  int msgno, idx = msgbegin - 1;
67  IMAP_HEADER h;
68  IMAP_STATUS* status;
69  int rc, mfhrc, oldmsgcount;
70  int fetchlast = 0;
71  int maxuid = 0;
72  static const char * const want_headers = "DATE FROM SUBJECT TO CC MESSAGE-ID REFERENCES CONTENT-TYPE CONTENT-DESCRIPTION IN-REPLY-TO REPLY-TO LINES LIST-POST X-LABEL X-KEYWORDS X-MOZILLA-KEYS KEYWORDS X-ORIGINAL-TO";
73  progress_t progress;
74  int retval = -1;
75 
76 #if USE_HCACHE
77  char buf[LONG_STRING];
78  void *uid_validity = NULL;
79  void *puidnext = NULL;
80  unsigned int uidnext = 0;
81  int evalhc = 0;
82 #endif /* USE_HCACHE */
83 
84  ctx = idata->ctx;
85 
86  if (mutt_bit_isset (idata->capabilities,IMAP4REV1))
87  {
88  safe_asprintf (&hdrreq, "BODY.PEEK[HEADER.FIELDS (%s%s%s)]",
89  want_headers, ImapHeaders ? " " : "", NONULL (ImapHeaders));
90  }
91  else if (mutt_bit_isset (idata->capabilities,IMAP4))
92  {
93  safe_asprintf (&hdrreq, "RFC822.HEADER.LINES (%s%s%s)",
94  want_headers, ImapHeaders ? " " : "", NONULL (ImapHeaders));
95  }
96  else
97  { /* Unable to fetch headers for lower versions */
98  mutt_error _("Unable to fetch headers from this IMAP server version.");
99  mutt_sleep (2); /* pause a moment to let the user see the error */
100  goto error_out_0;
101  }
102 
103  /* instead of downloading all headers and then parsing them, we parse them
104  * as they come in. */
105  mutt_mktemp (tempfile, sizeof (tempfile));
106  if (!(fp = safe_fopen (tempfile, "w+")))
107  {
108  mutt_error (_("Could not create temporary file %s"), tempfile);
109  mutt_sleep (2);
110  goto error_out_0;
111  }
112  unlink (tempfile);
113 
114  /* make sure context has room to hold the mailbox */
115  while ((msgend) >= idata->ctx->hdrmax)
116  mx_alloc_memory (idata->ctx);
117 
118  oldmsgcount = ctx->msgcount;
119  idata->reopen &= ~(IMAP_REOPEN_ALLOW|IMAP_NEWMAIL_PENDING);
120  idata->newMailCount = 0;
121 
122 #if USE_HCACHE
123  idata->hcache = imap_hcache_open (idata, NULL);
124 
125  if (idata->hcache && !msgbegin)
126  {
127  uid_validity = mutt_hcache_fetch_raw (idata->hcache, "/UIDVALIDITY", 12);
128  puidnext = mutt_hcache_fetch_raw (idata->hcache, "/UIDNEXT", 8);
129  if (puidnext)
130  {
131  uidnext = *(unsigned int *)puidnext;
132  mutt_hcache_free (idata->hcache, &puidnext);
133  }
134  if (uid_validity && uidnext &&
135  *(unsigned int *)uid_validity == idata->uid_validity)
136  evalhc = 1;
137  mutt_hcache_free (idata->hcache, &uid_validity);
138  }
139  if (evalhc)
140  {
141  /* L10N:
142  Comparing the cached data with the IMAP server's data */
143  mutt_progress_init (&progress, _("Evaluating cache..."),
144  MUTT_PROGRESS_MSG, ReadInc, msgend + 1);
145 
146  snprintf (buf, sizeof (buf),
147  "UID FETCH 1:%u (UID FLAGS)", uidnext - 1);
148 
149  imap_cmd_start (idata, buf);
150 
151  rc = IMAP_CMD_CONTINUE;
152  for (msgno = msgbegin; rc == IMAP_CMD_CONTINUE; msgno++)
153  {
154  mutt_progress_update (&progress, msgno + 1, -1);
155 
156  memset (&h, 0, sizeof (h));
157  h.data = safe_calloc (1, sizeof (IMAP_HEADER_DATA));
158  do
159  {
160  mfhrc = 0;
161 
162  rc = imap_cmd_step (idata);
163  if (rc != IMAP_CMD_CONTINUE)
164  {
165  imap_free_header_data (&h.data);
166  break;
167  }
168 
169  /* hole in the header cache */
170  if (!evalhc)
171  continue;
172 
173  if ((mfhrc = msg_fetch_header (ctx, &h, idata->buf, NULL)) == -1)
174  continue;
175  else if (mfhrc < 0)
176  {
177  imap_free_header_data (&h.data);
178  break;
179  }
180 
181  if (!h.data->uid)
182  {
183  dprint (2, (debugfile, "imap_read_headers: skipping hcache FETCH "
184  "response for unknown message number %d\n", h.sid));
185  mfhrc = -1;
186  continue;
187  }
188 
189  idx++;
190  ctx->hdrs[idx] = imap_hcache_get (idata, h.data->uid);
191  if (ctx->hdrs[idx])
192  {
193  ctx->hdrs[idx]->index = idx;
194  /* messages which have not been expunged are ACTIVE (borrowed from mh
195  * folders) */
196  ctx->hdrs[idx]->active = 1;
197  ctx->hdrs[idx]->read = h.data->read;
198  ctx->hdrs[idx]->old = h.data->old;
199  ctx->hdrs[idx]->deleted = h.data->deleted;
200  ctx->hdrs[idx]->flagged = h.data->flagged;
201  ctx->hdrs[idx]->replied = h.data->replied;
202  ctx->hdrs[idx]->changed = h.data->changed;
203  /* ctx->hdrs[msgno]->received is restored from mutt_hcache_restore */
204  ctx->hdrs[idx]->data = (void *) (h.data);
205 
206  ctx->msgcount++;
207  ctx->size += ctx->hdrs[idx]->content->length;
208  }
209  else
210  {
211  /* bad header in the cache, we'll have to refetch. */
212  dprint (3, (debugfile, "bad cache entry at %d, giving up\n", h.sid - 1));
213  imap_free_header_data(&h.data);
214  evalhc = 0;
215  idx--;
216  }
217  }
218  while (rc != IMAP_CMD_OK && mfhrc == -1);
219  if (rc == IMAP_CMD_OK)
220  break;
221  if ((mfhrc < -1) || ((rc != IMAP_CMD_CONTINUE) && (rc != IMAP_CMD_OK)))
222  {
223  imap_free_header_data (&h.data);
224  imap_hcache_close (idata);
225  goto error_out_1;
226  }
227  }
228  /* could also look for first null header in case hcache is holey */
229  msgbegin = ctx->msgcount;
230  }
231 #endif /* USE_HCACHE */
232 
233  mutt_progress_init (&progress, _("Fetching message headers..."),
234  MUTT_PROGRESS_MSG, ReadInc, msgend + 1);
235 
236  for (msgno = msgbegin; msgno <= msgend ; msgno++)
237  {
238  mutt_progress_update (&progress, msgno + 1, -1);
239 
240  /* we may get notification of new mail while fetching headers */
241  if (msgno + 1 > fetchlast)
242  {
243  char *cmd;
244 
245  fetchlast = msgend + 1;
246  safe_asprintf (&cmd, "FETCH %d:%d (UID FLAGS INTERNALDATE RFC822.SIZE %s)",
247  msgno + 1, fetchlast, hdrreq);
248  imap_cmd_start (idata, cmd);
249  FREE (&cmd);
250  }
251 
252  rewind (fp);
253  memset (&h, 0, sizeof (h));
254  h.data = safe_calloc (1, sizeof (IMAP_HEADER_DATA));
255 
256  /* this DO loop does two things:
257  * 1. handles untagged messages, so we can try again on the same msg
258  * 2. fetches the tagged response at the end of the last message.
259  */
260  do
261  {
262  mfhrc = 0;
263 
264  rc = imap_cmd_step (idata);
265  if (rc != IMAP_CMD_CONTINUE)
266  break;
267 
268  if ((mfhrc = msg_fetch_header (ctx, &h, idata->buf, fp)) == -1)
269  continue;
270  else if (mfhrc < 0)
271  break;
272 
273  if (!ftello (fp))
274  {
275  dprint (2, (debugfile, "msg_fetch_header: ignoring fetch response with no body\n"));
276  mfhrc = -1;
277  msgend--;
278  continue;
279  }
280 
281  /* make sure we don't get remnants from older larger message headers */
282  fputs ("\n\n", fp);
283 
284  idx++;
285  if (idx > msgend)
286  {
287  dprint (1, (debugfile, "imap_read_headers: skipping FETCH response for "
288  "unknown message number %d\n", h.sid));
289  mfhrc = -1;
290  idx--;
291  continue;
292  }
293  /* May receive FLAGS updates in a separate untagged response (#2935) */
294  if (idx < ctx->msgcount)
295  {
296  dprint (2, (debugfile, "imap_read_headers: message %d is not new\n",
297  h.sid));
298  idx--;
299  continue;
300  }
301 
302  ctx->hdrs[idx] = mutt_new_header ();
303 
304  ctx->hdrs[idx]->index = h.sid - 1;
305  /* messages which have not been expunged are ACTIVE (borrowed from mh
306  * folders) */
307  ctx->hdrs[idx]->active = 1;
308  ctx->hdrs[idx]->read = h.data->read;
309  ctx->hdrs[idx]->old = h.data->old;
310  ctx->hdrs[idx]->deleted = h.data->deleted;
311  ctx->hdrs[idx]->flagged = h.data->flagged;
312  ctx->hdrs[idx]->replied = h.data->replied;
313  ctx->hdrs[idx]->changed = h.data->changed;
314  ctx->hdrs[idx]->received = h.received;
315  ctx->hdrs[idx]->data = (void *) (h.data);
316 
317  if (maxuid < h.data->uid)
318  maxuid = h.data->uid;
319 
320  rewind (fp);
321  /* NOTE: if Date: header is missing, mutt_read_rfc822_header depends
322  * on h.received being set */
323  ctx->hdrs[idx]->env = mutt_read_rfc822_header (fp, ctx->hdrs[idx],
324  0, 0);
325  /* content built as a side-effect of mutt_read_rfc822_header */
326  ctx->hdrs[idx]->content->length = h.content_length;
327  ctx->size += h.content_length;
328 
329 #if USE_HCACHE
330  imap_hcache_put (idata, ctx->hdrs[idx]);
331 #endif /* USE_HCACHE */
332 
333  ctx->msgcount++;
334  }
335  while ((rc != IMAP_CMD_OK) && ((mfhrc == -1) ||
336  ((msgno + 1) >= fetchlast)));
337 
338  if ((mfhrc < -1) || ((rc != IMAP_CMD_CONTINUE) && (rc != IMAP_CMD_OK)))
339  {
340  imap_free_header_data (&h.data);
341 #if USE_HCACHE
342  imap_hcache_close (idata);
343 #endif
344  goto error_out_1;
345  }
346 
347  /* in case we get new mail while fetching the headers */
348  if (idata->reopen & IMAP_NEWMAIL_PENDING)
349  {
350  msgend = idata->newMailCount - 1;
351  while ((msgend) >= ctx->hdrmax)
352  mx_alloc_memory (ctx);
353  idata->reopen &= ~IMAP_NEWMAIL_PENDING;
354  idata->newMailCount = 0;
355  }
356  }
357 
358  if (maxuid && (status = imap_mboxcache_get (idata, idata->mailbox, 0)) &&
359  (status->uidnext < maxuid + 1))
360  status->uidnext = maxuid + 1;
361 
362 #if USE_HCACHE
363  mutt_hcache_store_raw (idata->hcache, "/UIDVALIDITY", 12,
364  &idata->uid_validity, sizeof (idata->uid_validity));
365  if (maxuid && idata->uidnext < maxuid + 1)
366  {
367  dprint (2, (debugfile, "Overriding UIDNEXT: %u -> %u\n", idata->uidnext, maxuid + 1));
368  idata->uidnext = maxuid + 1;
369  }
370  if (idata->uidnext > 1)
371  mutt_hcache_store_raw (idata->hcache, "/UIDNEXT", 8,
372  &idata->uidnext, sizeof (idata->uidnext));
373 
374  imap_hcache_close (idata);
375 #endif /* USE_HCACHE */
376 
377  if (ctx->msgcount > oldmsgcount)
378  {
379  mx_alloc_memory(ctx);
380  mx_update_context (ctx, ctx->msgcount - oldmsgcount);
381  }
382 
383  idata->reopen |= IMAP_REOPEN_ALLOW;
384 
385  retval = msgend;
386 
387 error_out_1:
388  safe_fclose (&fp);
389 
390 error_out_0:
391  FREE (&hdrreq);
392 
393  return retval;
394 }

396 int imap_fetch_message (CONTEXT *ctx, MESSAGE *msg, int msgno)
397 {
398  IMAP_DATA* idata;
399  HEADER* h;
400  ENVELOPE* newenv;
401  char buf[LONG_STRING];
402  char path[_POSIX_PATH_MAX];
403  char *pc;
404  long bytes;
405  progress_t progressbar;
406  int uid;
407  int cacheno;
408  IMAP_CACHE *cache;
409  int read;
410  int rc;
411 
412  /* Sam's weird courier server returns an OK response even when FETCH
413  * fails. Thanks Sam. */
414  short fetched = 0;
415 
416  idata = (IMAP_DATA*) ctx->data;
417  h = ctx->hdrs[msgno];
418 
419  if ((msg->fp = msg_cache_get (idata, h)))
420  {
421  if (HEADER_DATA(h)->parsed)
422  return 0;
423  else
424  goto parsemsg;
425  }
426 
427  /* we still do some caching even if imap_cachedir is unset */
428  /* see if we already have the message in our cache */
429  cacheno = HEADER_DATA(h)->uid % IMAP_CACHE_LEN;
430  cache = &idata->cache[cacheno];
431 
432  if (cache->path)
433  {
434  /* don't treat cache errors as fatal, just fall back. */
435  if (cache->uid == HEADER_DATA(h)->uid &&
436  (msg->fp = fopen (cache->path, "r")))
437  return 0;
438  else
439  {
440  unlink (cache->path);
441  FREE (&cache->path);
442  }
443  }
444 
445  if (!isendwin())
446  mutt_message _("Fetching message...");
447 
448  if (!(msg->fp = msg_cache_put (idata, h)))
449  {
450  cache->uid = HEADER_DATA(h)->uid;
451  mutt_mktemp (path, sizeof (path));
452  cache->path = safe_strdup (path);
453  if (!(msg->fp = safe_fopen (path, "w+")))
454  {
455  FREE (&cache->path);
456  return -1;
457  }
458  }
459 
460  /* mark this header as currently inactive so the command handler won't
461  * also try to update it. HACK until all this code can be moved into the
462  * command handler */
463  h->active = 0;
464 
465  snprintf (buf, sizeof (buf), "UID FETCH %u %s", HEADER_DATA(h)->uid,
466  (mutt_bit_isset (idata->capabilities, IMAP4REV1) ?
467  (option (OPTIMAPPEEK) ? "BODY.PEEK[]" : "BODY[]") :
468  "RFC822"));
469 
470  imap_cmd_start (idata, buf);
471  do
472  {
473  if ((rc = imap_cmd_step (idata)) != IMAP_CMD_CONTINUE)
474  break;
475 
476  pc = idata->buf;
477  pc = imap_next_word (pc);
478  pc = imap_next_word (pc);
479 
480  if (!ascii_strncasecmp ("FETCH", pc, 5))
481  {
482  while (*pc)
483  {
484  pc = imap_next_word (pc);
485  if (pc[0] == '(')
486  pc++;
487  if (ascii_strncasecmp ("UID", pc, 3) == 0)
488  {
489  pc = imap_next_word (pc);
490  uid = atoi (pc);
491  if (uid != HEADER_DATA(h)->uid)
492  mutt_error (_("The message index is incorrect. Try reopening the mailbox."));
493  }
494  else if ((ascii_strncasecmp ("RFC822", pc, 6) == 0) ||
495  (ascii_strncasecmp ("BODY[]", pc, 6) == 0))
496  {
497  pc = imap_next_word (pc);
498  if (imap_get_literal_count(pc, &bytes) < 0)
499  {
500  imap_error ("imap_fetch_message()", buf);
501  goto bail;
502  }
503  mutt_progress_init (&progressbar, _("Fetching message..."),
504  MUTT_PROGRESS_SIZE, NetInc, bytes);
505  if (imap_read_literal (msg->fp, idata, bytes, &progressbar) < 0)
506  goto bail;
507  /* pick up trailing line */
508  if ((rc = imap_cmd_step (idata)) != IMAP_CMD_CONTINUE)
509  goto bail;
510  pc = idata->buf;
511 
512  fetched = 1;
513  }
514  /* UW-IMAP will provide a FLAGS update here if the FETCH causes a
515  * change (eg from \Unseen to \Seen).
516  * Uncommitted changes in mutt take precedence. If we decide to
517  * incrementally update flags later, this won't stop us syncing */
518  else if ((ascii_strncasecmp ("FLAGS", pc, 5) == 0) && !h->changed)
519  {
520  if ((pc = imap_set_flags (idata, h, pc)) == NULL)
521  goto bail;
522  }
523  }
524  }
525  }
526  while (rc == IMAP_CMD_CONTINUE);
527 
528  /* see comment before command start. */
529  h->active = 1;
530 
531  fflush (msg->fp);
532  if (ferror (msg->fp))
533  {
534  mutt_perror (cache->path);
535  goto bail;
536  }
537 
538  if (rc != IMAP_CMD_OK)
539  goto bail;
540 
541  if (!fetched || !imap_code (idata->buf))
542  goto bail;
543 
544  msg_cache_commit (idata, h);
545 
546  parsemsg:
547  /* Update the header information. Previously, we only downloaded a
548  * portion of the headers, those required for the main display.
549  */
550  rewind (msg->fp);
551  /* It may be that the Status header indicates a message is read, but the
552  * IMAP server doesn't know the message has been \Seen. So we capture
553  * the server's notion of 'read' and if it differs from the message info
554  * picked up in mutt_read_rfc822_header, we mark the message (and context
555  * changed). Another possibility: ignore Status on IMAP?*/
556  read = h->read;
557  newenv = mutt_read_rfc822_header (msg->fp, h, 0, 0);
558  mutt_merge_envelopes(h->env, &newenv);
559 
560  /* see above. We want the new status in h->read, so we unset it manually
561  * and let mutt_set_flag set it correctly, updating context. */
562  if (read != h->read)
563  {
564  h->read = read;
565  mutt_set_flag (ctx, h, MUTT_NEW, read);
566  }
567 
568  h->lines = 0;
569  fgets (buf, sizeof (buf), msg->fp);
570  while (!feof (msg->fp))
571  {
572  h->lines++;
573  fgets (buf, sizeof (buf), msg->fp);
574  }
575 
576  h->content->length = ftell (msg->fp) - h->content->offset;
577 
578  /* This needs to be done in case this is a multipart message */
579 #if defined(HAVE_PGP) || defined(HAVE_SMIME)
580  h->security = crypt_query (h->content);
581 #endif
582 
583  mutt_clear_error();
584  rewind (msg->fp);
585  HEADER_DATA(h)->parsed = 1;
586 
587  return 0;
588 
589 bail:
590  safe_fclose (&msg->fp);
591  imap_cache_del (idata, h);
592  if (cache->path)
593  {
594  unlink (cache->path);
595  FREE (&cache->path);
596  }
597 
598  return -1;
599 }

616 int imap_append_message (CONTEXT *ctx, MESSAGE *msg)
617 {
618  IMAP_DATA* idata;
619  FILE *fp;
620  char buf[LONG_STRING];
621  char mbox[LONG_STRING];
622  char mailbox[LONG_STRING];
623  char internaldate[IMAP_DATELEN];
624  char imap_flags[SHORT_STRING];
625  size_t len;
626  progress_t progressbar;
627  size_t sent;
628  int c, last;
629  IMAP_MBOX mx;
630  int rc;
631 
632  idata = (IMAP_DATA*) ctx->data;
633 
634  if (imap_parse_path (ctx->path, &mx))
635  return -1;
636 
637  imap_fix_path (idata, mx.mbox, mailbox, sizeof (mailbox));
638  if (!*mailbox)
639  strfcpy (mailbox, "INBOX", sizeof (mailbox));
640 
641  if ((fp = fopen (msg->path, "r")) == NULL)
642  {
643  mutt_perror (msg->path);
644  goto fail;
645  }
646 
647  /* currently we set the \Seen flag on all messages, but probably we
648  * should scan the message Status header for flag info. Since we're
649  * already rereading the whole file for length it isn't any more
650  * expensive (it'd be nice if we had the file size passed in already
651  * by the code that writes the file, but that's a lot of changes.
652  * Ideally we'd have a HEADER structure with flag info here... */
653  for (last = EOF, len = 0; (c = fgetc(fp)) != EOF; last = c)
654  {
655  if(c == '\n' && last != '\r')
656  len++;
657 
658  len++;
659  }
660  rewind (fp);
661 
662  mutt_progress_init (&progressbar, _("Uploading message..."),
