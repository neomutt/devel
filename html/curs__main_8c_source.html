141 static void collapse_all(MUTTMENU *menu, int toggle)
142 {
143  HEADER *h, *base;
144  THREAD *thread, *top;
145  int final;
146 
147  if (!Context || (Context->msgcount == 0))
148  return;
149 
150  /* Figure out what the current message would be after folding / unfolding,
151  * so that we can restore the cursor in a sane way afterwards. */
152  if (CURHDR->collapsed && toggle)
153  final = mutt_uncollapse_thread (Context, CURHDR);
154  else if (option (OPTCOLLAPSEUNREAD) || !UNREAD (CURHDR))
155  final = mutt_collapse_thread (Context, CURHDR);
156  else
157  final = CURHDR->virtual;
158 
159  base = Context->hdrs[Context->v2r[final]];
160 
161  /* Iterate all threads, perform collapse/uncollapse as needed */
162  top = Context->tree;
163  Context->collapsed = toggle ? !Context->collapsed : 1;
164  while ((thread = top) != NULL)
165  {
166  while (!thread->message)
167  thread = thread->child;
168  h = thread->message;
169 
170  if (h->collapsed != Context->collapsed)
171  {
172  if (h->collapsed)
173  mutt_uncollapse_thread (Context, h);
174  else if (option (OPTCOLLAPSEUNREAD) || !UNREAD (h))
175  mutt_collapse_thread (Context, h);
176  }
177  top = top->next;
178  }
179 
180  /* Restore the cursor */
181  mutt_set_virtual (Context);
182  int j;
183  for (j = 0; j < Context->vcount; j++)
184  {
185  if (Context->hdrs[Context->v2r[j]]->index == base->index)
186  {
187  menu->current = j;
188  break;
189  }
190  }
191 
192  menu->redraw = REDRAW_INDEX | REDRAW_STATUS;
193 }

271 {

273  return;

276  if (!h)

299 
300  for (tmp = h->thread->parent; tmp; tmp = tmp->parent)
301  {
302  if (!tmp->message)
303  continue;
304 
305  /* if no ancestor is visible on current screen, provisionally force
306  * subject... */
307  if (reverse ? tmp->message->msgno > edgemsgno : tmp->message->msgno < edgemsgno)
308  {
309  flag |= MUTT_FORMAT_FORCESUBJ;
310  break;

313  break;
314  }
315  if (flag & MUTT_FORMAT_FORCESUBJ)
316  {
317  for (tmp = h->thread->prev; tmp; tmp = tmp->prev)
318  {
319  if (!tmp->message)
320  continue;
321 
322  /* ...but if a previous sibling is available, don't force it */
323  if (reverse ? tmp->message->msgno > edgemsgno : tmp->message->msgno < edgemsgno)
324  break;
325  else if (tmp->message->virtual >= 0)
326  {
327  flag &= ~MUTT_FORMAT_FORCESUBJ;
328  break;
329  }
330  }
331  }
332  }
333  }
334 
335  _mutt_make_string (s, l, NONULL (HdrFmt), Context, h, flag);

376  * unread message.

378 static int ci_first_message (void)
379 {
380  int old = -1, i;
381 
382  if (Context && Context->msgcount)
383  {
384  for (i=0; i < Context->vcount; i++)
385  {
386  if (! Context->hdrs[Context->v2r[i]]->read &&
387  ! Context->hdrs[Context->v2r[i]]->deleted)
388  {
389  if (! Context->hdrs[Context->v2r[i]]->old)
390  return (i);
391  else if (old == -1)
392  old = i;
393  }
394  }
395  if (old != -1)
396  return (old);
397 
398  /* If Sort is reverse and not threaded, the latest message is first.
399  * If Sort is threaded, the latest message is first iff exactly one
400  * of Sort and SortAux are reverse.
401  */
402  if (((Sort & SORT_REVERSE) && (Sort & SORT_MASK) != SORT_THREADS) ||
403  ((Sort & SORT_MASK) == SORT_THREADS &&
404  ((Sort ^ SortAux) & SORT_REVERSE)))
405  return 0;
406  else
407  return (Context->vcount ? Context->vcount - 1 : 0);
408  }
409  return 0;
410 }

438 void update_index (MUTTMENU *menu, CONTEXT *ctx, int check,
439  int oldcount, int index_hint)
440 {
441  /* store pointers to the newly added messages */
442  HEADER **save_new = NULL;
443  int j;
444 
445  if (!menu || !ctx)
446  return;
447 
448  /* take note of the current message */
449  if (oldcount)
450  {
451  if (menu->current < ctx->vcount)
452  menu->oldcurrent = index_hint;
453  else
454  oldcount = 0; /* invalid message number! */
455  }
456 
457  /* We are in a limited view. Check if the new message(s) satisfy
458  * the limit criteria. If they do, set their virtual msgno so that
459  * they will be visible in the limited view */
460  if (ctx->pattern)
461  {
462 #define THIS_BODY ctx->hdrs[j]->content
463  for (j = (check == MUTT_REOPENED) ? 0 : oldcount; j < ctx->msgcount; j++)
464  {
465  if (!j)
466  ctx->vcount = 0;
467 
468  if (mutt_pattern_exec (ctx->limit_pattern,
469  MUTT_MATCH_FULL_ADDRESS,
470  ctx, ctx->hdrs[j]))
471  {
472  assert (ctx->vcount < ctx->msgcount);
473  ctx->hdrs[j]->virtual = ctx->vcount;
474  ctx->v2r[ctx->vcount] = j;
475  ctx->hdrs[j]->limited = 1;
476  ctx->vcount++;
477  ctx->vsize += THIS_BODY->length + THIS_BODY->offset - THIS_BODY->hdr_offset;
478  }
479  }
480 #undef THIS_BODY
481  }
482 
483  /* save the list of new messages */
484  if (option(OPTUNCOLLAPSENEW) && oldcount && check != MUTT_REOPENED
485  && ((Sort & SORT_MASK) == SORT_THREADS))
486  {
487  save_new = (HEADER **) safe_malloc (sizeof (HEADER *) * (ctx->msgcount - oldcount));
488  for (j = oldcount; j < ctx->msgcount; j++)
489  save_new[j-oldcount] = ctx->hdrs[j];
490  }
491 
492  /* if the mailbox was reopened, need to rethread from scratch */
493  mutt_sort_headers (ctx, (check == MUTT_REOPENED));
494 
495  /* uncollapse threads with new mail */
496  if (option(OPTUNCOLLAPSENEW) && ((Sort & SORT_MASK) == SORT_THREADS))
497  {
498  if (check == MUTT_REOPENED)
499  {
500  THREAD *h, *j;
501 
502  ctx->collapsed = 0;
503 
504  for (h = ctx->tree; h; h = h->next)
505  {
506  for (j = h; !j->message; j = j->child)
507  ;
508  mutt_uncollapse_thread (ctx, j->message);
509  }
510  mutt_set_virtual (ctx);
511  }
512  else if (oldcount)
513  {
514  for (j = 0; j < ctx->msgcount - oldcount; j++)
515  {
516  int k;
517 
518  for (k = 0; k < ctx->msgcount; k++)
519  {
520  HEADER *h = ctx->hdrs[k];
521  if (h == save_new[j] && (!ctx->pattern || h->limited))
522  mutt_uncollapse_thread (ctx, h);
523  }
524  }
525  FREE (&save_new);
526  mutt_set_virtual (ctx);
527  }
528  }
529 
530  menu->current = -1;
531  if (oldcount)
532  {
533  /* restore the current message to the message it was pointing to */
534  for (j = 0; j < ctx->vcount; j++)
535  {
536  if (ctx->hdrs[ctx->v2r[j]]->index == menu->oldcurrent)
537  {
538  menu->current = j;
539  break;
540  }
541  }
542  }
543 
544  if (menu->current < 0)
545  menu->current = ci_first_message ();
546 
547 }

587 void
588 mutt_draw_statusline (int cols, const char *buf, int buflen)
589 {
590  int i = 0;
591  int offset = 0;
592  int found = 0;
593  int chunks = 0;
594  int len = 0;
595 
596  struct syntax_t
597  {
598  int color;
599  int first;
600  int last;
601  } *syntax = NULL;
602 
603  if (!buf)
604  return;
605 
606  do
607  {
608  COLOR_LINE *cl;
609  found = 0;
610 
611  if (!buf[offset])
612  break;
613 
614  /* loop through each "color status regex" */
615  for (cl = ColorStatusList; cl; cl = cl->next)
616  {
617  regmatch_t pmatch[cl->match + 1];
618 
619  if (regexec (&cl->rx, buf + offset, cl->match + 1, pmatch, 0) != 0)
620  continue; /* regex doesn't match the status bar */
621 
622  int first = pmatch[cl->match].rm_so + offset;
623  int last = pmatch[cl->match].rm_eo + offset;
624 
625  if (first == last)
626  continue; /* ignore an empty regex */
627 
628  if (!found)
629  {
630  chunks++;
631  safe_realloc (&syntax, chunks * sizeof (struct syntax_t));
632  }
633 
634  i = chunks - 1;
635  if (!found || (first < syntax[i].first) || ((first == syntax[i].first) && (last > syntax[i].last)))
636  {
637  syntax[i].color = cl->pair;
638  syntax[i].first = first;
639  syntax[i].last = last;
640  }
641  found = 1;
642  }
643 
644  if (syntax)
645  {
646  offset = syntax[i].last;
647  }
648  } while (found);
649 
650  /* Only 'len' bytes will fit into 'cols' screen columns */
651  len = mutt_wstr_trunc (buf, buflen, cols, NULL);
652 
653  offset = 0;
654 
655  if ((chunks > 0) && (syntax[0].first > 0))
656  {
657  /* Text before the first highlight */
658  addnstr (buf, MIN(len, syntax[0].first));
659  attrset (ColorDefs[MT_COLOR_STATUS]);
660  if (len <= syntax[0].first)
661  goto dsl_finish; /* no more room */
662 
663  offset = syntax[0].first;
664  }
665 
666  for (i = 0; i < chunks; i++)
667  {
668  /* Highlighted text */
669  attrset (syntax[i].color);
670  addnstr (buf + offset, MIN(len, syntax[i].last) - offset);
671  if (len <= syntax[i].last)
672  goto dsl_finish; /* no more room */
673 
674  int next;
675  if ((i + 1) == chunks)
676  {
677  next = len;
678  }
679  else
680  {
681  next = MIN (len, syntax[i+1].first);
682  }
683 
684  attrset (ColorDefs[MT_COLOR_STATUS]);
685  offset = syntax[i].last;
686  addnstr (buf + offset, next - offset);
687 
688  offset = next;
689  if (offset >= len)
690  goto dsl_finish; /* no more room */
691  }
692 
693  attrset (ColorDefs[MT_COLOR_STATUS]);
694  if (offset < len)
695  {
696  /* Text after the last highlight */
697  addnstr (buf + offset, len - offset);
698  }
699 
700  int width = mutt_strwidth (buf);
701  if (width < cols)
702  {
703  /* Pad the rest of the line with whitespace */
704  mutt_paddstr (cols - width, "");
705  }
706 dsl_finish:
707  FREE(&syntax);
708 }
