99  * key key to hash on
100  * data data to associate with `key'
101  * allow_dup if nonzero, duplicate keys are allowed in the table

103 int hash_insert (HASH * table, const char *key, void *data, int allow_dup)
104 {
105  struct hash_elem *ptr;
106  unsigned int h;
107 
108  ptr = (struct hash_elem *) safe_malloc (sizeof (struct hash_elem));
109  h = table->hash_string ((unsigned char *) key, table->nelem);
110  ptr->key = key;
111  ptr->data = data;
112 
113  if (allow_dup)
114  {
115  ptr->next = table->table[h];
116  table->table[h] = ptr;
117  table->curnelem++;
118  }
119  else
120  {
121  struct hash_elem *tmp, *last;
122  int r;
123 
124  for (tmp = table->table[h], last = NULL; tmp; last = tmp, tmp = tmp->next)
125  {
126  r = table->cmp_string (tmp->key, key);
127  if (r == 0)
128  {
129  FREE (&ptr);
130  return (-1);
131  }
132  if (r > 0)
133  break;
134  }
135  if (last)
136  last->next = ptr;
137  else
138  table->table[h] = ptr;
139  ptr->next = tmp;
140  table->curnelem++;
141  }
142  return h;
143 }

221 struct hash_elem *hash_walk(const HASH *table, struct hash_walk_state *state)
222 {
223  if (state->last && state->last->next)
224  {
225  state->last = state->last->next;
226  return state->last;
227  }
228 
229  if (state->last)
230  state->index++;
231 
232  while (state->index < table->nelem)
233  {
234  if (table->table[state->index])
235  {
236  state->last = table->table[state->index];
237  return state->last;
238  }
239  state->index++;
240  }
241 
242  state->index = 0;
243  state->last = NULL;
244  return NULL;
