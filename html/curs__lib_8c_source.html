81 void mutt_refresh (void)
82 {
83  /* don't refresh when we are waiting for a child. */
84  if (option (OPTKEEPQUIET))
85  return;
86 
87  /* don't refresh in the middle of macros unless necessary */
88  if (UngetCount && !option (OPTFORCEREFRESH))
89  return;
90 
91  /* else */
92  refresh ();
93 }

106 event_t mutt_getch (void)
107 {
108  int ch;
109  event_t err = {-1, OP_NULL }, ret;
110  event_t timeout = {-2, OP_NULL};
111 
112  if (UngetCount)
113  return (UngetKeyEvents[--UngetCount]);
114 
115  if (!option(OPTIGNOREMACROEVENTS) && MacroBufferCount)
116  return (MacroEvents[--MacroBufferCount]);
117 
118  SigInt = 0;
119 
120  mutt_allow_interrupt (1);
121 #ifdef KEY_RESIZE
122  /* ncurses 4.2 sends this when the screen is resized */
123  ch = KEY_RESIZE;
124  while (ch == KEY_RESIZE)
125 #endif /* KEY_RESIZE */
126  ch = getch ();
127  mutt_allow_interrupt (0);
128 
129  if (SigInt)
130  {
131  mutt_query_exit ();
132  return err;
133  }
134 
135  /* either timeout, a sigwinch (if timeout is set), or the terminal
136  * has been lost */
137  if (ch == ERR)
138  {
139  if (!isatty (0))
140  {
141  endwin ();
142  exit (1);
143  }
144  return timeout;
145  }
146 
147  if ((ch & 0x80) && option (OPTMETAKEY))
148  {
149  /* send ALT-x as ESC-x */
150  ch &= ~0x80;
151  mutt_unget_event (ch, 0);
152  ret.ch = '\033';
153  ret.op = 0;
154  return ret;
155  }
156 
157  ret.ch = ch;
158  ret.op = 0;
159  return (ch == ctrl ('G') ? err : ret);
160 }

163 {

223 int mutt_yesorno (const char *msg, int def)
224 {
225  event_t ch;
226  char *yes = _("yes");
227  char *no = _("no");
228  char *answer_string;
229  size_t answer_string_len;
230  size_t msglen;
231 
232 #ifdef HAVE_LANGINFO_YESEXPR
233  char *expr;
234  regex_t reyes;
235  regex_t reno;
236  int reyes_ok;
237  int reno_ok;
238  char answer[2];
239 
240  answer[1] = 0;
241 
242  reyes_ok = (expr = nl_langinfo (YESEXPR)) && expr[0] == '^' &&
243  !REGCOMP (&reyes, expr, REG_NOSUB);
244  reno_ok = (expr = nl_langinfo (NOEXPR)) && expr[0] == '^' &&
245  !REGCOMP (&reno, expr, REG_NOSUB);
246 #endif
247 
248  mutt_window_clearline (MuttMessageWindow, 0);
249 
250  /*
251  * In order to prevent the default answer to the question to wrapped
252  * around the screen in the even the question is wider than the screen,
253  * ensure there is enough room for the answer and truncate the question
254  * to fit.
255  */
256  safe_asprintf (&answer_string, " ([%s]/%s): ", def == MUTT_YES ? yes : no, def == MUTT_YES ? no : yes);
257  answer_string_len = mutt_strwidth (answer_string);
258  /* maxlen here is sort of arbitrary, so pick a reasonable upper bound */
259  msglen = mutt_wstr_trunc (msg, 4*MuttMessageWindow->cols, MuttMessageWindow->cols - answer_string_len, NULL);
260  SETCOLOR (MT_COLOR_PROMPT);
261  addnstr (msg, msglen);
262  addstr (answer_string);
263  NORMAL_COLOR;
264  FREE (&answer_string);
265 
266  FOREVER
267  {
268  mutt_refresh ();
269  ch = mutt_getch ();
270  if (CI_is_return (ch.ch))
271  break;
272  if (ch.ch < 0)
273  {
274  def = MUTT_ABORT;
275  break;
276  }
277 
278 #ifdef HAVE_LANGINFO_YESEXPR
279  answer[0] = ch.ch;
280  if (reyes_ok ?
281  (regexec (& reyes, answer, 0, 0, 0) == 0) :
282 #else
283  if (
284 #endif
285  (tolower (ch.ch) == 'y'))
286  {
287  def = MUTT_YES;
288  break;
289  }
290  else if (
291 #ifdef HAVE_LANGINFO_YESEXPR
292  reno_ok ?
293  (regexec (& reno, answer, 0, 0, 0) == 0) :
294 #endif
295  (tolower (ch.ch) == 'n'))
296  {
297  def = MUTT_NO;
298  break;
299  }
300  else
301  {
302  BEEP();
303  }
304  }
305 
306 #ifdef HAVE_LANGINFO_YESEXPR
307  if (reyes_ok)
308  regfree (& reyes);
309  if (reno_ok)
310  regfree (& reno);
311 #endif
312 
313  if (def != MUTT_ABORT)
314  {
315  addstr ((char *) (def == MUTT_YES ? yes : no));
316  mutt_refresh ();
317  }
318  else
319  {
320  /* when the users cancels with ^G, clear the message stored with
321  * mutt_message() so it isn't displayed when the screen is refreshed. */
322  mutt_clear_error();
323  }
324  return (def);
325 }

328 void mutt_query_exit (void)
329 {
330  mutt_flushinp ();
331  curs_set (1);
332  if (Timeout)
333  timeout (-1); /* restore blocking operation */
334  if (mutt_yesorno (_("Exit Mutt?"), MUTT_YES) == MUTT_YES)

341  SigInt = 0;
342 }

389 void mutt_progress_init (progress_t* progress, const char *msg,
390  unsigned short flags, unsigned short inc,
391  long size)
392 {
393  struct timeval tv = { 0, 0 };
394 
395  if (!progress)
396  return;
397  if (option(OPTNOCURSES))
398  return;
399 
400  memset (progress, 0, sizeof (progress_t));
401  progress->inc = inc;
402  progress->flags = flags;
403  progress->msg = msg;
404  progress->size = size;
405  if (progress->size) {
406  if (progress->flags & MUTT_PROGRESS_SIZE)
407  mutt_pretty_size (progress->sizestr, sizeof (progress->sizestr),
408  progress->size);
409  else
410  snprintf (progress->sizestr, sizeof (progress->sizestr), "%ld",
411  progress->size);
412  }
413  if (!inc)
414  {
415  if (size)
416  mutt_message ("%s (%s)", msg, progress->sizestr);
417  else
418  mutt_message (msg);
419  return;
420  }
421  if (gettimeofday (&tv, NULL) < 0)
422  dprint (1, (debugfile, "gettimeofday failed: %d\n", errno));
423  /* if timestamp is 0 no time-based suppression is done */
424  if (TimeInc)
425  progress->timestamp = ((unsigned int) tv.tv_sec * 1000)
426  + (unsigned int) (tv.tv_usec / 1000);
427  mutt_progress_update (progress, 0, 0);
428 }

436 static void
437 message_bar (int percent, const char *fmt, ...)
438 {
439  va_list ap;
440  char buf[STRING], buf2[STRING];
441  int w = percent * COLS / 100;
442  size_t l;
443 
444  va_start (ap, fmt);
445  vsnprintf (buf, sizeof (buf), fmt, ap);
446  l = mutt_strwidth (buf);
447  va_end (ap);
448 
449  mutt_format_string (buf2, sizeof (buf2), 0, COLS-2, FMT_LEFT, 0, buf, sizeof (buf), 0);
450 
451  move (LINES - 1, 0);
452 
453  if (ColorDefs[MT_COLOR_PROGRESS] == 0)
454  {
455  addstr (buf2);
456  }
457  else
458  {
459  if (l < w)
460  {
461  /* The string fits within the colour bar */
462  SETCOLOR(MT_COLOR_PROGRESS);
463  addstr (buf2);
464  w -= l;
465  while (w--)
466  {
467  addch (' ');
468  }
469  SETCOLOR(MT_COLOR_NORMAL);
470  }
471  else
472  {
473  /* The string is too long for the colour bar */
474  char ch;
475  int off = mutt_wstr_trunc (buf2, sizeof (buf2), w, NULL);
476 
477  ch = buf2[off];
478  buf2[off] = 0;
479  SETCOLOR(MT_COLOR_PROGRESS);
480  addstr (buf2);
481  buf2[off] = ch;
482  SETCOLOR(MT_COLOR_NORMAL);
483  addstr (&buf2[off]);
484  }
485  }
486 
487  clrtoeol();
488  mutt_refresh();
489 }

491 void mutt_progress_update (progress_t* progress, long pos, int percent)
492 {
493  char posstr[SHORT_STRING];
494  short update = 0;
495  struct timeval tv = { 0, 0 };
496  unsigned int now = 0;
497 
498  if (option(OPTNOCURSES))
499  return;
500 
501  if (!progress->inc)
502  goto out;
503 
504  /* refresh if size > inc */
505  if (progress->flags & MUTT_PROGRESS_SIZE &&
506  (pos >= progress->pos + (progress->inc << 10)))
507  update = 1;
508  else if (pos >= progress->pos + progress->inc)
509  update = 1;
510 
511  /* skip refresh if not enough time has passed */
512  if (update && progress->timestamp && !gettimeofday (&tv, NULL)) {
513  now = ((unsigned int) tv.tv_sec * 1000)
514  + (unsigned int) (tv.tv_usec / 1000);
515  if (now && now - progress->timestamp < TimeInc)
516  update = 0;
517  }
518 
519  /* always show the first update */
520  if (!pos)
521  update = 1;
522 
523  if (update)
524  {
525  if (progress->flags & MUTT_PROGRESS_SIZE)
526  {
527  pos = pos / (progress->inc << 10) * (progress->inc << 10);
528  mutt_pretty_size (posstr, sizeof (posstr), pos);
529  }
530  else
531  snprintf (posstr, sizeof (posstr), "%ld", pos);
532 
533  dprint (5, (debugfile, "updating progress: %s\n", posstr));
534 
535  progress->pos = pos;
536  if (now)
537  progress->timestamp = now;
538 
539  if (progress->size > 0)
540  {
541  message_bar ((percent > 0) ? percent : (int) (100.0 * (double) progress->pos / progress->size),
542  "%s %s/%s (%d%%)", progress->msg, posstr, progress->sizestr,
543  (percent > 0) ? percent : (int) (100.0 * (double) progress->pos / progress->size));
544  }
545  else
546  {
547  if (percent > 0)
548  message_bar (percent, "%s %s (%d%%)", progress->msg, posstr, percent);
549  else
550  mutt_message ("%s %s", progress->msg, posstr);
551  }
552  }
553 
554 out:
555  if (pos >= progress->size)
556  mutt_clear_error ();
557 }

1005 void mutt_format_string (char *dest, size_t destlen,
1006  int min_width, int max_width,
1007  int justify, char m_pad_char,
1008  const char *s, size_t n,
1009  int arboreal)
1010 {
1011  char *p;
1012  wchar_t wc;
1013  int w;
1014  size_t k, k2;
1015  char scratch[MB_LEN_MAX];
1016  mbstate_t mbstate1, mbstate2;
1017  int escaped = 0;
1018 
1019  memset(&mbstate1, 0, sizeof (mbstate1));
1020  memset(&mbstate2, 0, sizeof (mbstate2));
1021  --destlen;
1022  p = dest;
1023  for (; n && (k = mbrtowc (&wc, s, n, &mbstate1)); s += k, n -= k)
1024  {
1025  if (k == (size_t)(-1) || k == (size_t)(-2))
1026  {
1027  if (k == (size_t)(-1) && errno == EILSEQ)
1028  memset (&mbstate1, 0, sizeof (mbstate1));
1029 
1030  k = (k == (size_t)(-1)) ? 1 : n;
1031  wc = replacement_char ();
1032  }
1033  if (escaped) {
1034  escaped = 0;
1035  w = 0;
1036  } else if (arboreal && wc == MUTT_SPECIAL_INDEX) {
1037  escaped = 1;
1038  w = 0;
1039  } else if (arboreal && wc < MUTT_TREE_MAX) {
1040  w = 1; /* hack */
1041  }
1042  else
1043  {
1044 #ifdef HAVE_ISWBLANK
1045  if (iswblank (wc))
1046  wc = ' ';
1047  else
1048 #endif
1049  if (!IsWPrint (wc))
1050  wc = '?';
1051  w = wcwidth (wc);
1052  }
1053  if (w >= 0)
1054  {
1055  if (w > max_width || (k2 = wcrtomb (scratch, wc, &mbstate2)) > destlen)
1056  break;
1057  min_width -= w;
1058  max_width -= w;
1059  strncpy (p, scratch, k2);
1060  p += k2;
1061  destlen -= k2;
1062  }
1063  }
1064  w = (int)destlen < min_width ? destlen : min_width;
1065  if (w <= 0)
1066  *p = '\0';
1067  else if (justify == FMT_RIGHT) /* right justify */
1068  {
1069  p[w] = '\0';
1070  while (--p >= dest)
1071  p[w] = *p;
1072  while (--w >= 0)
1073  dest[w] = m_pad_char;
1074  }
1075  else if (justify == FMT_CENTER) /* center */
1076  {
1077  char *savedp = p;
1078  int half = (w+1) / 2; /* half of cushion space */
1079 
1080  p[w] = '\0';
1081 
1082  /* move str to center of buffer */
1083  while (--p >= dest)
1084  p[half] = *p;
1085 
1086  /* fill rhs */
1087  p = savedp + half;
1088  while (--w >= half)
1089  *p++ = m_pad_char;
1090 
1091  /* fill lhs */
1092  while (half--)
1093  dest[half] = m_pad_char;
1094  }
1095  else /* left justify */
1096  {
1097  while (--w >= 0)
1098  *p++ = m_pad_char;
1099  *p = '\0';
1100  }
1101 }
