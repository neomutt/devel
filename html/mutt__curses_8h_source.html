208 static inline int mutt_window_wrap_cols(mutt_window_t *win, short wrap)
209 {
210  if (wrap < 0)
211  return win->cols > -wrap ? win->cols + wrap : win->cols;
212  else if (wrap)
213  return wrap < win->cols ? wrap : win->cols;
214  else
215  return win->cols;
216 }
