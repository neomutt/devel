28 static void *
29 hcache_kyotocabinet_open(const char *path)
30 {
31  char kcdbpath[_POSIX_PATH_MAX];
32  int printfresult;
33 
34  printfresult = snprintf(kcdbpath, sizeof(kcdbpath),
35  "%s#type=kct#opts=%s#rcomp=lex",
36  path, option(OPTHCACHECOMPRESS) ? "lc" : "l");
37  if ((printfresult < 0) || (printfresult >= sizeof(kcdbpath)))
38  {
39  return NULL;
40  }
41 
42  KCDB *db = kcdbnew();
43  if (!db)
44  return NULL;
45 
46  if (kcdbopen(db, kcdbpath, KCOWRITER | KCOCREATE))
47  return db;
48  else
49  {
50 #ifdef DEBUG
51  int ecode = kcdbecode (db);
52  dprint (2, (debugfile, "kcdbopen failed for %s: %s (ecode %d)\n", kcdbpath, kcdbemsg (db), ecode));
53 #endif
54  kcdbdel(db);
55  return NULL;
56  }
57 }
