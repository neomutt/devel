56 int mbox_lock_mailbox (CONTEXT *ctx, int excl, int retry)
57 {
58  int r;
59 
60  if ((r = mx_lock_file (ctx->path, fileno (ctx->fp), excl, 1, retry)) == 0)
61  ctx->locked = 1;
62  else if (retry && !excl)
63  {
64  ctx->readonly = 1;
65  return 0;
66  }
67 
68  return (r);

82 int mmdf_parse_mailbox (CONTEXT *ctx)
83 {
84  char buf[HUGE_STRING];
85  char return_path[LONG_STRING];
86  int count = 0, oldmsgcount = ctx->msgcount;
87  int lines;
88  time_t t;
89  LOFF_T loc, tmploc;
90  HEADER *hdr;
91  struct stat sb;
92 #ifdef NFS_ATTRIBUTE_HACK
93  struct utimbuf newtime;
94 #endif
95  progress_t progress;
96  char msgbuf[STRING];
97 
98  if (stat (ctx->path, &sb) == -1)
99  {
100  mutt_perror (ctx->path);
101  return (-1);
102  }
103  ctx->atime = sb.st_atime;
104  ctx->mtime = sb.st_mtime;
105  ctx->size = sb.st_size;
106 
107 #ifdef NFS_ATTRIBUTE_HACK
108  if (sb.st_mtime > sb.st_atime)
109  {
110  newtime.modtime = sb.st_mtime;
111  newtime.actime = time (NULL);
112  utime (ctx->path, &newtime);
113  }
114 #endif
115 
116  buf[sizeof (buf) - 1] = 0;
117 
118  if (!ctx->quiet)
119  {
120  snprintf (msgbuf, sizeof (msgbuf), _("Reading %s..."), ctx->path);
121  mutt_progress_init (&progress, msgbuf, MUTT_PROGRESS_MSG, ReadInc, 0);
122  }
123 
124  FOREVER
125  {
126  if (fgets (buf, sizeof (buf) - 1, ctx->fp) == NULL)
127  break;
128 
129  if (mutt_strcmp (buf, MMDF_SEP) == 0)
130  {
131  loc = ftello (ctx->fp);
132 
133  count++;
134  if (!ctx->quiet)
135  mutt_progress_update (&progress, count,
136  (int) (loc / (ctx->size / 100 + 1)));
137 
138  if (ctx->msgcount == ctx->hdrmax)
139  mx_alloc_memory (ctx);
140  ctx->hdrs[ctx->msgcount] = hdr = mutt_new_header ();
141  hdr->offset = loc;
142  hdr->index = ctx->msgcount;
143 
144  if (fgets (buf, sizeof (buf) - 1, ctx->fp) == NULL)
145  {
146  /* TODO: memory leak??? */
147  dprint (1, (debugfile, "mmdf_parse_mailbox: unexpected EOF\n"));
148  break;
149  }
150 
151  return_path[0] = 0;
152 
153  if (!is_from (buf, return_path, sizeof (return_path), &t))
154  {
155  if (fseeko (ctx->fp, loc, SEEK_SET) != 0)
156  {
157  dprint (1, (debugfile, "mmdf_parse_mailbox: fseek() failed\n"));
158  mutt_error _("Mailbox is corrupt!");
159  return (-1);
160  }
161  }
162  else
163  hdr->received = t - mutt_local_tz (t);
164 
165  hdr->env = mutt_read_rfc822_header (ctx->fp, hdr, 0, 0);
166 
167  loc = ftello (ctx->fp);
168 
169  if (hdr->content->length > 0 && hdr->lines > 0)
170  {
171  tmploc = loc + hdr->content->length;
172 
173  if (0 < tmploc && tmploc < ctx->size)
174  {
175  if (fseeko (ctx->fp, tmploc, SEEK_SET) != 0 ||
176  fgets (buf, sizeof (buf) - 1, ctx->fp) == NULL ||
177  mutt_strcmp (MMDF_SEP, buf) != 0)
178  {
179  if (fseeko (ctx->fp, loc, SEEK_SET) != 0)
180  dprint (1, (debugfile, "mmdf_parse_mailbox: fseek() failed\n"));
181  hdr->content->length = -1;
182  }
183  }
184  else
185  hdr->content->length = -1;
186  }
187  else
188  hdr->content->length = -1;
189 
190  if (hdr->content->length < 0)
191  {
192  lines = -1;
193  do {
194  loc = ftello (ctx->fp);
195  if (fgets (buf, sizeof (buf) - 1, ctx->fp) == NULL)
196  break;
197  lines++;
198  } while (mutt_strcmp (buf, MMDF_SEP) != 0);
199 
200  hdr->lines = lines;
201  hdr->content->length = loc - hdr->content->offset;
202  }
203 
204  if (!hdr->env->return_path && return_path[0])
205  hdr->env->return_path = rfc822_parse_adrlist (hdr->env->return_path, return_path);
206 
207  if (!hdr->env->from)
208  hdr->env->from = rfc822_cpy_adr (hdr->env->return_path, 0);
209 
210  ctx->msgcount++;
211  }
212  else
213  {
214  dprint (1, (debugfile, "mmdf_parse_mailbox: corrupt mailbox!\n"));
215  mutt_error _("Mailbox is corrupt!");
216  return (-1);
217  }
218  }
219 
220  if (ctx->msgcount > oldmsgcount)
221  mx_update_context (ctx, ctx->msgcount - oldmsgcount);
222 
223  return (0);
224 }

226 /* Note that this function is also called when new mail is appended to the
227  * currently open folder, and NOT just when the mailbox is initially read.
228  *
229  * NOTE: it is assumed that the mailbox being read has been locked before
230  * this routine gets called. Strange things could happen if it's not!
231  */
232 int mbox_parse_mailbox (CONTEXT *ctx)
233 {
234  struct stat sb;
235  char buf[HUGE_STRING], return_path[STRING];
236  HEADER *curhdr;
237  time_t t;
238  int count = 0, lines = 0;
239  LOFF_T loc;
240 #ifdef NFS_ATTRIBUTE_HACK
241  struct utimbuf newtime;
242 #endif
243  progress_t progress;
244  char msgbuf[STRING];
245 
246  /* Save information about the folder at the time we opened it. */
247  if (stat (ctx->path, &sb) == -1)
248  {
249  mutt_perror (ctx->path);
250  return (-1);
251  }
252 
253  ctx->size = sb.st_size;
254  ctx->mtime = sb.st_mtime;
255  ctx->atime = sb.st_atime;
256 
257 #ifdef NFS_ATTRIBUTE_HACK
258  if (sb.st_mtime > sb.st_atime)
259  {
260  newtime.modtime = sb.st_mtime;
261  newtime.actime = time (NULL);
262  utime (ctx->path, &newtime);
263  }
264 #endif
265 
266  if (!ctx->readonly)
267  ctx->readonly = access (ctx->path, W_OK) ? 1 : 0;
268 
269  if (!ctx->quiet)
270  {
271  snprintf (msgbuf, sizeof (msgbuf), _("Reading %s..."), ctx->path);
272  mutt_progress_init (&progress, msgbuf, MUTT_PROGRESS_MSG, ReadInc, 0);
273  }
274 
275  loc = ftello (ctx->fp);
276  while (fgets (buf, sizeof (buf), ctx->fp) != NULL)
277  {
278  if (is_from (buf, return_path, sizeof (return_path), &t))
279  {
280  /* Save the Content-Length of the previous message */
281  if (count > 0)
282  {
283 #define PREV ctx->hdrs[ctx->msgcount-1]
284 
285  if (PREV->content->length < 0)
286  {
287  PREV->content->length = loc - PREV->content->offset - 1;
288  if (PREV->content->length < 0)
289  PREV->content->length = 0;
290  }
291  if (!PREV->lines)
292  PREV->lines = lines ? lines - 1 : 0;
293  }
294 
295  count++;
296 
297  if (!ctx->quiet)
298  mutt_progress_update (&progress, count,
299  (int)(ftello (ctx->fp) / (ctx->size / 100 + 1)));
300 
301  if (ctx->msgcount == ctx->hdrmax)
302  mx_alloc_memory (ctx);
303 
304  curhdr = ctx->hdrs[ctx->msgcount] = mutt_new_header ();
305  curhdr->received = t - mutt_local_tz (t);
306  curhdr->offset = loc;
307  curhdr->index = ctx->msgcount;
308 
309  curhdr->env = mutt_read_rfc822_header (ctx->fp, curhdr, 0, 0);
310 
311  /* if we know how long this message is, either just skip over the body,
312  * or if we don't know how many lines there are, count them now (this will
313  * save time by not having to search for the next message marker).
314  */
315  if (curhdr->content->length > 0)
316  {
317  LOFF_T tmploc;
318 
319  loc = ftello (ctx->fp);
320  tmploc = loc + curhdr->content->length + 1;
321 
322  if (0 < tmploc && tmploc < ctx->size)
323  {
324  /*
325  * check to see if the content-length looks valid. we expect to
326  * to see a valid message separator at this point in the stream
327  */
328  if (fseeko (ctx->fp, tmploc, SEEK_SET) != 0 ||
329  fgets (buf, sizeof (buf), ctx->fp) == NULL ||
330  mutt_strncmp ("From ", buf, 5) != 0)
331  {
332  dprint (1, (debugfile, "mbox_parse_mailbox: bad content-length in message %d (cl=" OFF_T_FMT ")\n", curhdr->index, curhdr->content->length));
333  dprint (1, (debugfile, "\tLINE: %s", buf));
334  if (fseeko (ctx->fp, loc, SEEK_SET) != 0) /* nope, return the previous position */
335  {
336  dprint (1, (debugfile, "mbox_parse_mailbox: fseek() failed\n"));
337  }
338  curhdr->content->length = -1;
339  }
340  }
341  else if (tmploc != ctx->size)
342  {
343  /* content-length would put us past the end of the file, so it
344  * must be wrong
345  */
346  curhdr->content->length = -1;
347  }
348 
349  if (curhdr->content->length != -1)
350  {
351  /* good content-length. check to see if we know how many lines
352  * are in this message.
353  */
354  if (curhdr->lines == 0)
355  {
356  int cl = curhdr->content->length;
357 
358  /* count the number of lines in this message */
359  if (fseeko (ctx->fp, loc, SEEK_SET) != 0)
360  dprint (1, (debugfile, "mbox_parse_mailbox: fseek() failed\n"));
361  while (cl-- > 0)
362  {
363  if (fgetc (ctx->fp) == '\n')
364  curhdr->lines++;
365  }
366  }
367 
368  /* return to the offset of the next message separator */
369  if (fseeko (ctx->fp, tmploc, SEEK_SET) != 0)
370  dprint (1, (debugfile, "mbox_parse_mailbox: fseek() failed\n"));
371  }
372  }
373 
374  ctx->msgcount++;
375 
376  if (!curhdr->env->return_path && return_path[0])
377  curhdr->env->return_path = rfc822_parse_adrlist (curhdr->env->return_path, return_path);
378 
379  if (!curhdr->env->from)
380  curhdr->env->from = rfc822_cpy_adr (curhdr->env->return_path, 0);
381 
382  lines = 0;
383  }
384  else
385  lines++;
386 
387  loc = ftello (ctx->fp);
388  }
389 
390  /*
391  * Only set the content-length of the previous message if we have read more
392  * than one message during _this_ invocation. If this routine is called
393  * when new mail is received, we need to make sure not to clobber what
394  * previously was the last message since the headers may be sorted.
395  */
396  if (count > 0)
397  {
398  if (PREV->content->length < 0)
399  {
400  PREV->content->length = ftello (ctx->fp) - PREV->content->offset - 1;
401  if (PREV->content->length < 0)
402  PREV->content->length = 0;
403  }
404 
405  if (!PREV->lines)
406  PREV->lines = lines ? lines - 1 : 0;
407 
408  mx_update_context (ctx, count);
409  }
410 
411  return (0);
412 }

416 /* open a mbox or mmdf style mailbox */
417 static int mbox_open_mailbox (CONTEXT *ctx)
418 {
419  int rc;
420 
421  if ((ctx->fp = fopen (ctx->path, "r")) == NULL)
422  {
423  mutt_perror (ctx->path);
424  return (-1);
425  }
426  mutt_block_signals ();
427  if (mbox_lock_mailbox (ctx, 0, 1) == -1)
428  {
429  mutt_unblock_signals ();
430  return (-1);
431  }
432 
433  if (ctx->magic == MUTT_MBOX)
434  rc = mbox_parse_mailbox (ctx);
435  else if (ctx->magic == MUTT_MMDF)
436  rc = mmdf_parse_mailbox (ctx);
437  else
438  rc = -1;
439  mutt_touch_atime (fileno (ctx->fp));
440 
441  mbox_unlock_mailbox (ctx);
442  mutt_unblock_signals ();
443  return (rc);
444 }

532 {

534  {
535  if (mutt_strcmp (a->mailbox, b->mailbox) ||

537  return (0);
538 
539  a = a->next;

545  return (1);

564 static int strict_cmp_envelopes (const ENVELOPE *e1, const ENVELOPE *e2)
565 {
566  if (e1 && e2)
567  {
568  if (mutt_strcmp (e1->message_id, e2->message_id) ||
569  mutt_strcmp (e1->subject, e2->subject) ||
570  !strict_cmp_lists (e1->references, e2->references) ||
571  !strict_addrcmp (e1->from, e2->from) ||
572  !strict_addrcmp (e1->sender, e2->sender) ||
573  !strict_addrcmp (e1->reply_to, e2->reply_to) ||
574  !strict_addrcmp (e1->to, e2->to) ||
575  !strict_addrcmp (e1->cc, e2->cc) ||
576  !strict_addrcmp (e1->return_path, e2->return_path))
577  return (0);
578  else
579  return (1);
580  }
581  else
582  {
583  if (e1 == NULL && e2 == NULL)
584  return (1);
585  else
586  return (0);
587  }
588 }

647 /* check to see if the mailbox has changed on disk.
648  *
649  * return values:
650  * MUTT_REOPENED mailbox has been reopened
651  * MUTT_NEW_MAIL new mail has arrived!
652  * MUTT_LOCKED couldn't lock the file
653  * 0 no change
654  * -1 error
655  */
656 static int mbox_check_mailbox (CONTEXT *ctx, int *index_hint)
657 {
658  struct stat st;
659  char buffer[LONG_STRING];
660  int unlock = 0;
661  int modified = 0;
662 
663  if (stat (ctx->path, &st) == 0)
664  {
665  if (st.st_mtime == ctx->mtime && st.st_size == ctx->size)
666  return (0);
667 
668  if (st.st_size == ctx->size)
669  {
670  /* the file was touched, but it is still the same length, so just exit */
671  ctx->mtime = st.st_mtime;
672  return (0);
673  }
674 
675  if (st.st_size > ctx->size)
676  {
677  /* lock the file if it isn't already */
678  if (!ctx->locked)
679  {
680  mutt_block_signals ();
681  if (mbox_lock_mailbox (ctx, 0, 0) == -1)
682  {
683  mutt_unblock_signals ();
684  /* we couldn't lock the mailbox, but nothing serious happened:
685  * probably the new mail arrived: no reason to wait till we can
686  * parse it: we'll get it on the next pass
687  */
688  return (MUTT_LOCKED);
689  }
690  unlock = 1;
691  }
692 
693  /*
694  * Check to make sure that the only change to the mailbox is that
695  * message(s) were appended to this file. My heuristic is that we should
696  * see the message separator at *exactly* what used to be the end of the
697  * folder.
698  */
699  if (fseeko (ctx->fp, ctx->size, SEEK_SET) != 0)
700  dprint (1, (debugfile, "mbox_check_mailbox: fseek() failed\n"));
701  if (fgets (buffer, sizeof (buffer), ctx->fp) != NULL)
702  {
703  if ((ctx->magic == MUTT_MBOX && mutt_strncmp ("From ", buffer, 5) == 0) ||
704  (ctx->magic == MUTT_MMDF && mutt_strcmp (MMDF_SEP, buffer) == 0))
705  {
706  if (fseeko (ctx->fp, ctx->size, SEEK_SET) != 0)
707  dprint (1, (debugfile, "mbox_check_mailbox: fseek() failed\n"));
708  if (ctx->magic == MUTT_MBOX)
709  mbox_parse_mailbox (ctx);
710  else
711  mmdf_parse_mailbox (ctx);
712 
713  /* Only unlock the folder if it was locked inside of this routine.
714  * It may have been locked elsewhere, like in
715  * mutt_checkpoint_mailbox().
716  */
717 
718  if (unlock)
719  {
720  mbox_unlock_mailbox (ctx);
721  mutt_unblock_signals ();
722  }
723 
724  return (MUTT_NEW_MAIL); /* signal that new mail arrived */
725  }
726  else
727  modified = 1;
728  }
729  else
730  {
731  dprint (1, (debugfile, "mbox_check_mailbox: fgets returned NULL.\n"));
732  modified = 1;
733  }
734  }
735  else
736  modified = 1;
737  }
738 
739  if (modified)
740  {
741  if (mutt_reopen_mailbox (ctx, index_hint) != -1)
742  {
743  if (unlock)
744  {
745  mbox_unlock_mailbox (ctx);
746  mutt_unblock_signals ();
747  }
748  return (MUTT_REOPENED);
749  }
750  }
751 
752  /* fatal error */
753 
754  mbox_unlock_mailbox (ctx);
755  mx_fastclose_mailbox (ctx);
756  mutt_unblock_signals ();
757  mutt_error _("Mailbox was corrupted!");
758  return (-1);
759 }

761 /*
762  * Returns 1 if the mailbox has at least 1 new messages (not old)
763  * otherwise returns 0.
764  */
765 static int mbox_has_new(CONTEXT *ctx)
766 {
767  int i;
768 
769  for (i = 0; i < ctx->msgcount; i++)
770  if (!ctx->hdrs[i]->deleted && !ctx->hdrs[i]->read && !ctx->hdrs[i]->old)
771  return 1;
772  return 0;
773 }

775 /* if mailbox has at least 1 new message, sets mtime > atime of mailbox
776  * so buffy check reports new mail */
777 void mbox_reset_atime (CONTEXT *ctx, struct stat *st)
778 {
779  struct utimbuf utimebuf;
780  struct stat _st;
781 
782  if (!st)
783  {
784  if (stat (ctx->path, &_st) < 0)
785  return;
786  st = &_st;
787  }
788 
789  utimebuf.actime = st->st_atime;
790  utimebuf.modtime = st->st_mtime;
791 
792  /*
793  * When $mbox_check_recent is set, existing new mail is ignored, so do not
794  * reset the atime to mtime-1 to signal new mail.
795  */
796  if (!option(OPTMAILCHECKRECENT) && utimebuf.actime >= utimebuf.modtime && mbox_has_new(ctx))
797  utimebuf.actime = utimebuf.modtime - 1;
798 
799  utime (ctx->path, &utimebuf);

802 /* return values:
803  * 0 success
804  * -1 failure
805  */
806 static int mbox_sync_mailbox (CONTEXT *ctx, int *index_hint)
807 {
808  char tempfile[_POSIX_PATH_MAX];
809  char buf[32];
810  int i, j, save_sort = SORT_ORDER;
811  int rc = -1;
812  int need_sort = 0; /* flag to resort mailbox if new mail arrives */
813  int first = -1; /* first message to be written */
814  LOFF_T offset; /* location in mailbox to write changed messages */
815  struct stat statbuf;
816  struct m_update_t *newOffset = NULL;
817  struct m_update_t *oldOffset = NULL;
818  FILE *fp = NULL;
819  progress_t progress;
820  char msgbuf[STRING];
821  BUFFY *tmp = NULL;
822 
823  /* sort message by their position in the mailbox on disk */
824  if (Sort != SORT_ORDER)
825  {
826  save_sort = Sort;
827  Sort = SORT_ORDER;
828  mutt_sort_headers (ctx, 0);
829  Sort = save_sort;
830  need_sort = 1;
831  }
832 
833  /* need to open the file for writing in such a way that it does not truncate
834  * the file, so use read-write mode.
835  */
836  if ((ctx->fp = freopen (ctx->path, "r+", ctx->fp)) == NULL)
837  {
838  mx_fastclose_mailbox (ctx);
839  mutt_error _("Fatal error! Could not reopen mailbox!");
840  return (-1);
841  }
842 
843  mutt_block_signals ();
844 
845  if (mbox_lock_mailbox (ctx, 1, 1) == -1)
846  {
847  mutt_unblock_signals ();
848  mutt_error _("Unable to lock mailbox!");
849  goto bail;
850  }
851 
852  /* Check to make sure that the file hasn't changed on disk */
853  if ((i = mbox_check_mailbox (ctx, index_hint)) == MUTT_NEW_MAIL || i == MUTT_REOPENED)
854  {
855  /* new mail arrived, or mailbox reopened */
856  need_sort = i;
857  rc = i;
858  goto bail;
859  }
860  else if (i < 0)
861  /* fatal error */
862  return (-1);
863 
864  /* Create a temporary file to write the new version of the mailbox in. */
865  mutt_mktemp (tempfile, sizeof (tempfile));
866  if ((i = open (tempfile, O_WRONLY | O_EXCL | O_CREAT, 0600)) == -1 ||
867  (fp = fdopen (i, "w")) == NULL)
868  {
869  if (-1 != i)
870  {
871  close (i);
872  unlink (tempfile);
873  }
874  mutt_error _("Could not create temporary file!");
875  mutt_sleep (5);
876  goto bail;
877  }
878 
879  /* find the first deleted/changed message. we save a lot of time by only
880  * rewriting the mailbox from the point where it has actually changed.
881  */
882  for (i = 0 ; i < ctx->msgcount && !ctx->hdrs[i]->deleted &&
883  !ctx->hdrs[i]->changed && !ctx->hdrs[i]->attach_del; i++)
884  ;
885  if (i == ctx->msgcount)
886  {
887  /* this means ctx->changed or ctx->deleted was set, but no
888  * messages were found to be changed or deleted. This should
889  * never happen, is we presume it is a bug in mutt.
890  */
891  mutt_error _("sync: mbox modified, but no modified messages! (report this bug)");
892  mutt_sleep(5); /* the mutt_error /will/ get cleared! */
893  dprint(1, (debugfile, "mbox_sync_mailbox(): no modified messages.\n"));
894  unlink (tempfile);
895  goto bail;
896  }
897 
898  /* save the index of the first changed/deleted message */
899  first = i;
900  /* where to start overwriting */
901  offset = ctx->hdrs[i]->offset;
902 
903  /* the offset stored in the header does not include the MMDF_SEP, so make
904  * sure we seek to the correct location
905  */
906  if (ctx->magic == MUTT_MMDF)
907  offset -= (sizeof MMDF_SEP - 1);
908 
909  /* allocate space for the new offsets */
910  newOffset = safe_calloc (ctx->msgcount - first, sizeof (struct m_update_t));
911  oldOffset = safe_calloc (ctx->msgcount - first, sizeof (struct m_update_t));
912 
913  if (!ctx->quiet)
914  {
915  snprintf (msgbuf, sizeof (msgbuf), _("Writing %s..."), ctx->path);
916  mutt_progress_init (&progress, msgbuf, MUTT_PROGRESS_MSG, WriteInc, ctx->msgcount);
917  }
918 
919  for (i = first, j = 0; i < ctx->msgcount; i++)
920  {
921  if (!ctx->quiet)
922  mutt_progress_update (&progress, i, (int)(ftello (ctx->fp) / (ctx->size / 100 + 1)));
923  /*
924  * back up some information which is needed to restore offsets when
925  * something fails.
926  */
927 
928  oldOffset[i-first].valid = 1;
929  oldOffset[i-first].hdr = ctx->hdrs[i]->offset;
930  oldOffset[i-first].body = ctx->hdrs[i]->content->offset;
931  oldOffset[i-first].lines = ctx->hdrs[i]->lines;
932  oldOffset[i-first].length = ctx->hdrs[i]->content->length;
933 
934  if (! ctx->hdrs[i]->deleted)
935  {
936  j++;
937 
938  if (ctx->magic == MUTT_MMDF)
939  {
940  if (fputs (MMDF_SEP, fp) == EOF)
941  {
942  mutt_perror (tempfile);
943  mutt_sleep (5);
944  unlink (tempfile);
945  goto bail;
946  }
947 
948  }
949 
950  /* save the new offset for this message. we add `offset' because the
951  * temporary file only contains saved message which are located after
952  * `offset' in the real mailbox
953  */
954  newOffset[i - first].hdr = ftello (fp) + offset;
955 
956  if (mutt_copy_message (fp, ctx, ctx->hdrs[i], MUTT_CM_UPDATE,
957  CH_FROM | CH_UPDATE | CH_UPDATE_LEN) != 0)
958  {
959  mutt_perror (tempfile);
960  mutt_sleep (5);
961  unlink (tempfile);
962  goto bail;
963  }
964 
965  /* Since messages could have been deleted, the offsets stored in memory
966  * will be wrong, so update what we can, which is the offset of this
967  * message, and the offset of the body. If this is a multipart message,
968  * we just flush the in memory cache so that the message will be reparsed
969  * if the user accesses it later.
970  */
971  newOffset[i - first].body = ftello (fp) - ctx->hdrs[i]->content->length + offset;
972  mutt_free_body (&ctx->hdrs[i]->content->parts);
973 
974  switch(ctx->magic)
975  {
976  case MUTT_MMDF:
977  if(fputs(MMDF_SEP, fp) == EOF)
978  {
979  mutt_perror (tempfile);
980  mutt_sleep (5);
981  unlink (tempfile);
982  goto bail;
983  }
984  break;
985  default:
986  if(fputs("\n", fp) == EOF)
987  {
988  mutt_perror (tempfile);
989  mutt_sleep (5);
990  unlink (tempfile);
991  goto bail;
992  }
993  }
994  }
995  }
996 
997  if (fclose (fp) != 0)
998  {
999  fp = NULL;
1000  dprint(1, (debugfile, "mbox_sync_mailbox: safe_fclose (&) returned non-zero.\n"));
1001  unlink (tempfile);
1002  mutt_perror (tempfile);
1003  mutt_sleep (5);
1004  goto bail;
1005  }
1006  fp = NULL;
1007 
1008  /* Save the state of this folder. */
1009  if (stat (ctx->path, &statbuf) == -1)
1010  {
1011  mutt_perror (ctx->path);
1012  mutt_sleep (5);
1013  unlink (tempfile);
1014  goto bail;
1015  }
1016 
1017  if ((fp = fopen (tempfile, "r")) == NULL)
1018  {
1019  mutt_unblock_signals ();
1020  mx_fastclose_mailbox (ctx);
1021  dprint (1, (debugfile, "mbox_sync_mailbox: unable to reopen temp copy of mailbox!\n"));
1022  mutt_perror (tempfile);
1023  mutt_sleep (5);
1024  return (-1);
1025  }
1026 
1027  if (fseeko (ctx->fp, offset, SEEK_SET) != 0 || /* seek the append location */
1028  /* do a sanity check to make sure the mailbox looks ok */
1029  fgets (buf, sizeof (buf), ctx->fp) == NULL ||
1030  (ctx->magic == MUTT_MBOX && mutt_strncmp ("From ", buf, 5) != 0) ||
1031  (ctx->magic == MUTT_MMDF && mutt_strcmp (MMDF_SEP, buf) != 0))
1032  {
1033  dprint (1, (debugfile, "mbox_sync_mailbox: message not in expected position."));
1034  dprint (1, (debugfile, "\tLINE: %s\n", buf));
1035  i = -1;
1036  }
1037  else
1038  {
1039  if (fseeko (ctx->fp, offset, SEEK_SET) != 0) /* return to proper offset */
1040  {
1041  i = -1;
1042  dprint (1, (debugfile, "mbox_sync_mailbox: fseek() failed\n"));
1043  }
1044  else
1045  {
1046  /* copy the temp mailbox back into place starting at the first
1047  * change/deleted message
1048  */
1049  if (!ctx->quiet)
1050  mutt_message _("Committing changes...");
1051  i = mutt_copy_stream (fp, ctx->fp);
1052 
1053  if (ferror (ctx->fp))
1054  i = -1;
1055  }
1056  if (i == 0)
1057  {
1058  ctx->size = ftello (ctx->fp); /* update the size of the mailbox */
1059  if (ftruncate (fileno (ctx->fp), ctx->size) != 0)
1060  {
1061  i = -1;
1062  dprint (1, (debugfile, "mbox_sync_mailbox: ftruncate() failed\n"));
1063  }
1064  }
1065  }
1066 
1067  safe_fclose (&fp);
1068  fp = NULL;
1069  mbox_unlock_mailbox (ctx);
1070 
1071  if (safe_fclose (&ctx->fp) != 0 || i == -1)
1072  {
1073  /* error occurred while writing the mailbox back, so keep the temp copy
1074  * around
1075  */
1076 
1077  char savefile[_POSIX_PATH_MAX];
1078 
1079  snprintf (savefile, sizeof (savefile), "%s/mutt.%s-%s-%u",
1080  NONULL (Tempdir), NONULL(Username), NONULL(Hostname), (unsigned int)getpid ());
1081  rename (tempfile, savefile);
1082  mutt_unblock_signals ();
1083  mx_fastclose_mailbox (ctx);
1084  mutt_pretty_mailbox (savefile, sizeof (savefile));
1085  mutt_error (_("Write failed! Saved partial mailbox to %s"), savefile);
1086  mutt_sleep (5);
1087  return (-1);
1088  }
1089 
1090  /* Restore the previous access/modification times */
1091  mbox_reset_atime (ctx, &statbuf);
1092 
1093  /* reopen the mailbox in read-only mode */
1094  if ((ctx->fp = fopen (ctx->path, "r")) == NULL)
1095  {
1096  unlink (tempfile);
1097  mutt_unblock_signals ();
1098  mx_fastclose_mailbox (ctx);
1099  mutt_error _("Fatal error! Could not reopen mailbox!");
1100  return (-1);
1101  }
1102 
1103  /* update the offsets of the rewritten messages */
1104  for (i = first, j = first; i < ctx->msgcount; i++)
1105  {
1106  if (!ctx->hdrs[i]->deleted)
1107  {
1108  ctx->hdrs[i]->offset = newOffset[i - first].hdr;
1109  ctx->hdrs[i]->content->hdr_offset = newOffset[i - first].hdr;
1110  ctx->hdrs[i]->content->offset = newOffset[i - first].body;
1111  ctx->hdrs[i]->index = j++;
1112  }
1113  }
1114  FREE (&newOffset);
1115  FREE (&oldOffset);
1116  unlink (tempfile); /* remove partial copy of the mailbox */
1117  mutt_unblock_signals ();
1118 
1119  if (option(OPTCHECKMBOXSIZE))
1120  {
1121  tmp = mutt_find_mailbox (ctx->path);
1122  if (tmp && tmp->new == 0)
1123  mutt_update_mailbox (tmp);
1124  }
1125 
1126  return (0); /* signal success */
1127 
1128 bail: /* Come here in case of disaster */
1129 
1130  safe_fclose (&fp);
1131 
1132  /* restore offsets, as far as they are valid */
1133  if (first >= 0 && oldOffset)
1134  {
1135  for (i = first; i < ctx->msgcount && oldOffset[i-first].valid; i++)
1136  {
1137  ctx->hdrs[i]->offset = oldOffset[i-first].hdr;
1138  ctx->hdrs[i]->content->hdr_offset = oldOffset[i-first].hdr;
1139  ctx->hdrs[i]->content->offset = oldOffset[i-first].body;
1140  ctx->hdrs[i]->lines = oldOffset[i-first].lines;
1141  ctx->hdrs[i]->content->length = oldOffset[i-first].length;
1142  }
1143  }
1144 
1145  /* this is ok to call even if we haven't locked anything */
1146  mbox_unlock_mailbox (ctx);
1147 
1148  mutt_unblock_signals ();
1149  FREE (&newOffset);
1150  FREE (&oldOffset);
1151 
1152  if ((ctx->fp = freopen (ctx->path, "r", ctx->fp)) == NULL)
1153  {
1154  mutt_error _("Could not reopen mailbox!");
1155  mx_fastclose_mailbox (ctx);
1156  return (-1);
1157  }
1158 
1159  if (need_sort)
1160  /* if the mailbox was reopened, the thread tree will be invalid so make
1161  * sure to start threading from scratch. */
1162  mutt_sort_headers (ctx, (need_sort == MUTT_REOPENED));
1163 
1164  return rc;
1165 }

1167 int mutt_reopen_mailbox (CONTEXT *ctx, int *index_hint)
1168 {
1169  int (*cmp_headers) (const HEADER *, const HEADER *) = NULL;
1170  HEADER **old_hdrs;
1171  int old_msgcount;
1172  int msg_mod = 0;
1173  int index_hint_set;
1174  int i, j;
1175  int rc = -1;
1176 
1177  /* silent operations */
1178  ctx->quiet = 1;
1179 
1180  if (!ctx->quiet)
1181  mutt_message _("Reopening mailbox...");
1182 
1183  /* our heuristics require the old mailbox to be unsorted */
1184  if (Sort != SORT_ORDER)
1185  {
1186  short old_sort;
1187 
1188  old_sort = Sort;
1189  Sort = SORT_ORDER;
1190  mutt_sort_headers (ctx, 1);
1191  Sort = old_sort;
1192  }
1193 
1194  old_hdrs = NULL;
1195  old_msgcount = 0;
1196 
1197  /* simulate a close */
1198  if (ctx->id_hash)
1199  hash_destroy (&ctx->id_hash, NULL);
1200  if (ctx->subj_hash)
1201  hash_destroy (&ctx->subj_hash, NULL);
1202  mutt_clear_threads (ctx);
1203  FREE (&ctx->v2r);
1204  if (ctx->readonly)
1205  {
1206  for (i = 0; i < ctx->msgcount; i++)
1207  mutt_free_header (&(ctx->hdrs[i])); /* nothing to do! */
1208  FREE (&ctx->hdrs);
1209  }
1210  else
1211  {
1212  /* save the old headers */
1213  old_msgcount = ctx->msgcount;
1214  old_hdrs = ctx->hdrs;
1215  ctx->hdrs = NULL;
1216  }
1217 
1218  ctx->hdrmax = 0; /* force allocation of new headers */
1219  ctx->msgcount = 0;
1220  ctx->vcount = 0;
1221  ctx->tagged = 0;
1222  ctx->deleted = 0;
1223  ctx->new = 0;
1224  ctx->unread = 0;
1225  ctx->flagged = 0;
1226  ctx->changed = 0;
1227  ctx->id_hash = NULL;
1228  ctx->subj_hash = NULL;
1229 
1230  switch (ctx->magic)
1231  {
1232  case MUTT_MBOX:
1233  case MUTT_MMDF:
1234  cmp_headers = mbox_strict_cmp_headers;
1235  safe_fclose (&ctx->fp);
1236  if (!(ctx->fp = safe_fopen (ctx->path, "r")))
1237  rc = -1;
1238  else
1239  rc = ((ctx->magic == MUTT_MBOX) ? mbox_parse_mailbox
1240  : mmdf_parse_mailbox) (ctx);
1241  break;
1242 
1243  default:
1244  rc = -1;
1245  break;
1246  }
1247 
1248  if (rc == -1)
1249  {
1250  /* free the old headers */
1251  for (j = 0; j < old_msgcount; j++)
1252  mutt_free_header (&(old_hdrs[j]));
1253  FREE (&old_hdrs);
1254 
1255  ctx->quiet = 0;
1256  return (-1);
1257  }
1258 
1259  mutt_touch_atime (fileno (ctx->fp));
1260 
1261  /* now try to recover the old flags */
1262 
1263  index_hint_set = (index_hint == NULL);
1264 
1265  if (!ctx->readonly)
1266  {
1267  for (i = 0; i < ctx->msgcount; i++)
1268  {
1269  int found = 0;
1270 
1271  /* some messages have been deleted, and new messages have been
1272  * appended at the end; the heuristic is that old messages have then
1273  * "advanced" towards the beginning of the folder, so we begin the
1274  * search at index "i"
1275  */
1276  for (j = i; j < old_msgcount; j++)
1277  {
1278  if (old_hdrs[j] == NULL)
1279  continue;
1280  if (cmp_headers (ctx->hdrs[i], old_hdrs[j]))
1281  {
1282  found = 1;
1283  break;
1284  }
1285  }
1286  if (!found)
1287  {
1288  for (j = 0; j < i && j < old_msgcount; j++)
1289  {
1290  if (old_hdrs[j] == NULL)
1291  continue;
1292  if (cmp_headers (ctx->hdrs[i], old_hdrs[j]))
1293  {
1294  found = 1;
1295  break;
1296  }
1297  }
1298  }
1299 
1300  if (found)
1301  {
1302  /* this is best done here */
1303  if (!index_hint_set && *index_hint == j)
1304  *index_hint = i;
1305 
1306  if (old_hdrs[j]->changed)
1307  {
1308  /* Only update the flags if the old header was changed;
1309  * otherwise, the header may have been modified externally,
1310  * and we don't want to lose _those_ changes
1311  */
1312  mutt_set_flag (ctx, ctx->hdrs[i], MUTT_FLAG, old_hdrs[j]->flagged);
1313  mutt_set_flag (ctx, ctx->hdrs[i], MUTT_REPLIED, old_hdrs[j]->replied);
1314  mutt_set_flag (ctx, ctx->hdrs[i], MUTT_OLD, old_hdrs[j]->old);
1315  mutt_set_flag (ctx, ctx->hdrs[i], MUTT_READ, old_hdrs[j]->read);
1316  }
1317  mutt_set_flag (ctx, ctx->hdrs[i], MUTT_DELETE, old_hdrs[j]->deleted);
1318  mutt_set_flag (ctx, ctx->hdrs[i], MUTT_PURGE, old_hdrs[j]->purge);
1319  mutt_set_flag (ctx, ctx->hdrs[i], MUTT_TAG, old_hdrs[j]->tagged);
1320 
1321  /* we don't need this header any more */
1322  mutt_free_header (&(old_hdrs[j]));
1323  }
1324  }
1325 
1326  /* free the remaining old headers */
1327  for (j = 0; j < old_msgcount; j++)
1328  {
1329  if (old_hdrs[j])
1330  {
1331  mutt_free_header (&(old_hdrs[j]));
1332  msg_mod = 1;
1333  }
1334  }
1335  FREE (&old_hdrs);
1336  }
1337 
1338  ctx->quiet = 0;
1339 
1340  return ((ctx->changed || msg_mod) ? MUTT_REOPENED : MUTT_NEW_MAIL);
1341 }
