28 static void *
29 hcache_tokyocabinet_open(const char *path)
30 {
31  TCBDB *db = tcbdbnew();
32  if (!db)
33  return NULL;
34  if (option(OPTHCACHECOMPRESS))
35  tcbdbtune(db, 0, 0, 0, -1, -1, BDBTDEFLATE);
36  if (tcbdbopen(db, path, BDBOWRITER | BDBOCREAT))
37  return db;
38  else
39  {
40 #ifdef DEBUG
41  int ecode = tcbdbecode (db);
42  dprint (2, (debugfile, "tcbdbopen failed for %s: %s (ecode %d)\n", path, tcbdberrmsg (ecode), ecode));
43 #endif
44  tcbdbdel(db);
45  return NULL;
46  }
47 }
