346 static void
347 print_compile_options (void)
348 {
349  int i;
350  char c;
351  int len;
352  int used = 0;
353 
354  for (i = 0; comp_opts[i].name; i++)
355  {
356  len = strlen (comp_opts[i].name) + 2; /* +/- and a space */
357  if ((used + len) > SCREEN_WIDTH)
358  {
359  used = 0;
360  puts ("");
361  }
362  used += len;
363  c = comp_opts[i].enabled ? '+' : '-';
364  printf ("%c%s ", c, comp_opts[i].name);
365  }
366  puts ("");
367 }

377 static char *
378 rstrip_in_place (char *s)
379 {
380  if (!s)
381  return NULL;
382 
383  char *p;
384 
385  p = &s[strlen (s)];
386  if (p == s)
387  return s;
388  p--;
389  while ((p >= s) && ((*p == '\n') || (*p == '\r')))
390  *p-- = '\0';
391  return s;
392 }
