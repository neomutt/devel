331 void mutt_unlink (const char *s)
332 {
333  int fd;
334  int flags;
335  FILE *f;
336  struct stat sb, sb2;
337  char buf[2048];
338 
339  /* Defend against symlink attacks */
340 
341 #ifdef O_NOFOLLOW
342  flags = O_RDWR | O_NOFOLLOW;
343 #else
344  flags = O_RDWR;
345 #endif
346 
347  if (lstat (s, &sb) == 0 && S_ISREG(sb.st_mode))
348  {
349  if ((fd = open (s, flags)) < 0)
350  return;
351 
352  if ((fstat (fd, &sb2) != 0) || !S_ISREG (sb2.st_mode)
353  || (sb.st_dev != sb2.st_dev) || (sb.st_ino != sb2.st_ino))
354  {
355  close (fd);
356  return;
357  }
358 
359  if ((f = fdopen (fd, "r+")))
360  {
361  unlink (s);
362  memset (buf, 0, sizeof (buf));
363  while (sb.st_size > 0)
364  {
365  fwrite (buf, 1, MIN (sizeof (buf), sb.st_size), f);
366  sb.st_size -= MIN (sizeof (buf), sb.st_size);
367  }
368  safe_fclose (&f);
369  }
370  }
371 }

374 {
375  char buf[2048];
376  size_t chunk;
377 
378  while (size > 0)
379  {
380  chunk = (size > sizeof (buf)) ? sizeof (buf) : size;
381  if ((chunk = fread (buf, 1, chunk, in)) < 1)
382  break;
383  if (fwrite (buf, 1, chunk, out) != chunk)
384  {
385  /* dprint (1, (debugfile, "mutt_copy_bytes(): fwrite() returned short byte count\n")); */
386  return (-1);
387  }
388  size -= chunk;
389  }
390 
391  if (fflush(out) != 0) return -1;
392  return 0;

422 int safe_symlink(const char *oldpath, const char *newpath)
423 {
424  struct stat osb, nsb;
425 
426  if(!oldpath || !newpath)
427  return -1;
428 
429  if(unlink(newpath) == -1 && errno != ENOENT)
430  return -1;
431 
432  if (oldpath[0] == '/')
433  {
434  if (symlink (oldpath, newpath) == -1)
435  return -1;
436  }
437  else
438  {
439  char abs_oldpath[_POSIX_PATH_MAX];
440 
441  if ((getcwd (abs_oldpath, sizeof abs_oldpath) == NULL) ||
442  (strlen (abs_oldpath) + 1 + strlen (oldpath) + 1 > sizeof abs_oldpath))
443  return -1;
444 
445  strcat (abs_oldpath, "/"); /* __STRCAT_CHECKED__ */
446  strcat (abs_oldpath, oldpath); /* __STRCAT_CHECKED__ */
447  if (symlink (abs_oldpath, newpath) == -1)
448  return -1;
449  }
450 
451  if(stat(oldpath, &osb) == -1 || stat(newpath, &nsb) == -1
452  || compare_stat(&osb, &nsb) == -1)
453  {
454  unlink(newpath);
455  return -1;
456  }
457 
458  return 0;
459 }

463 /*
464  * This function is supposed to do nfs-safe renaming of files.
465  *
466  * Warning: We don't check whether src and target are equal.
467  */
468 
469 int safe_rename (const char *src, const char *target)
470 {
471  struct stat ssb, tsb;
472 
473  if (!src || !target)
474  return -1;
475 
476  if (link (src, target) != 0)
477  {
478 
479  /*
480  * Coda does not allow cross-directory links, but tells
481  * us it's a cross-filesystem linking attempt.
482  *
483  * However, the Coda rename call is allegedly safe to use.
484  *
485  * With other file systems, rename should just fail when
486  * the files reside on different file systems, so it's safe
487  * to try it here.
488  *
489  */
490 
491  dprint (1, (debugfile, "safe_rename: link (%s, %s) failed: %s (%d)\n", src, target, strerror (errno), errno));
492 
493  /*
494  * FUSE may return ENOSYS. VFAT may return EPERM. FreeBSD's
495  * msdosfs may return EOPNOTSUPP. ENOTSUP can also appear.
496  */
497  if (errno == EXDEV || errno == ENOSYS || errno == EPERM
498 #ifdef ENOTSUP
499  || errno == ENOTSUP
500 #endif
501 #ifdef EOPNOTSUPP
502  || errno == EOPNOTSUPP
503 #endif
504  )
505  {
506  dprint (1, (debugfile, "safe_rename: trying rename...\n"));
507  if (rename (src, target) == -1)
508  {
509  dprint (1, (debugfile, "safe_rename: rename (%s, %s) failed: %s (%d)\n", src, target, strerror (errno), errno));
510  return -1;
511  }
512  dprint (1, (debugfile, "safe_rename: rename succeeded.\n"));
513 
514  return 0;
515  }
516 
517  return -1;
518  }
519 
520  /*
521  * Stat both links and check if they are equal.
522  */
523 
524  if (lstat (src, &ssb) == -1)
525  {
526  dprint (1, (debugfile, "safe_rename: can't stat %s: %s (%d)\n",
527  src, strerror (errno), errno));
528  return -1;
529  }
530 
531  if (lstat (target, &tsb) == -1)
532  {
533  dprint (1, (debugfile, "safe_rename: can't stat %s: %s (%d)\n",
534  src, strerror (errno), errno));
535  return -1;
536  }
537 
538  /*
539  * pretend that the link failed because the target file
540  * did already exist.
541  */
542 
543  if (compare_stat (&ssb, &tsb) == -1)
544  {
545  dprint (1, (debugfile, "safe_rename: stat blocks for %s and %s diverge; pretending EEXIST.\n", src, target));
546  errno = EEXIST;
547  return -1;
548  }
549 
550  /*
551  * Unlink the original link. Should we really ignore the return
552  * value here? XXX
553  */
554 
555  if (unlink (src) == -1)
556  {
557  dprint (1, (debugfile, "safe_rename: unlink (%s) failed: %s (%d)\n",
558  src, strerror (errno), errno));
559  }
560 
561 
562  return 0;
563 }

568 static int mutt_mkwrapdir (const char *path, char *newfile, size_t nflen,
569  char *newdir, size_t ndlen)
570 {
571  const char *basename;
572  char parent[_POSIX_PATH_MAX];
573  char *p;
574 
575  strfcpy (parent, NONULL (path), sizeof (parent));
576 
577  if ((p = strrchr (parent, '/')))
578  {
579  *p = '\0';
580  basename = p + 1;
581  }
582  else
583  {
584  strfcpy (parent, ".", sizeof (parent));
585  basename = path;
586  }
587 
588  snprintf (newdir, ndlen, "%s/%s", parent, ".muttXXXXXX");
589  if (mkdtemp(newdir) == NULL)
590  {
591  dprint(1, (debugfile, "mutt_mkwrapdir: mkdtemp() failed\n"));
592  return -1;
593  }
594 
595  if (snprintf (newfile, nflen, "%s/%s", newdir, NONULL(basename)) >= nflen)
596  {
597  rmdir(newdir);
598  dprint(1, (debugfile, "mutt_mkwrapdir: string was truncated\n"));
599  return -1;
600  }
601  return 0;

604 /* remove a directory and everything under it */
605 int mutt_rmtree (const char* path)
606 {
607  DIR* dirp;
608  struct dirent* de;
609  char cur[_POSIX_PATH_MAX];
610  struct stat statbuf;
611  int rc = 0;
612 
613  if (!(dirp = opendir (path)))
614  {
615  dprint (1, (debugfile, "mutt_rmtree: error opening directory %s\n", path));
616  return -1;
617  }
618  while ((de = readdir (dirp)))
619  {
620  if (!strcmp (".", de->d_name) || !strcmp ("..", de->d_name))
621  continue;
622 
623  snprintf (cur, sizeof (cur), "%s/%s", path, de->d_name);
624  /* XXX make nonrecursive version */
625 
626  if (stat(cur, &statbuf) == -1)
627  {
628  rc = 1;
629  continue;
630  }
631 
632  if (S_ISDIR (statbuf.st_mode))
633  rc |= mutt_rmtree (cur);
634  else
635  rc |= unlink (cur);
636  }
637  closedir (dirp);
638 
639  rc |= rmdir (path);
640 
641  return rc;
642 }

655 {
656  struct stat osb, nsb;
657  int fd;
658 
659  if (flags & O_EXCL)
660  {
661  char safe_file[_POSIX_PATH_MAX];
662  char safe_dir[_POSIX_PATH_MAX];
663 
664  if (mutt_mkwrapdir (path, safe_file, sizeof (safe_file),
665  safe_dir, sizeof (safe_dir)) == -1)
666  return -1;
667 
668  if ((fd = open (safe_file, flags, 0600)) < 0)
669  {
670  rmdir (safe_dir);
671  return fd;
672  }
673 
674  /* NFS and I believe cygwin do not handle movement of open files well */
675  close (fd);
676  if (mutt_put_file_in_place (path, safe_file, safe_dir) == -1)
677  return -1;

679

681  return fd;
682 
683  /* make sure the file is not symlink */
684  if (lstat (path, &osb) < 0 || fstat (fd, &nsb) < 0 ||
685  compare_stat(&osb, &nsb) == -1)
686  {
687 /* dprint (1, (debugfile, "safe_open(): %s is a symlink!\n", path)); */
688  close (fd);
689  return (-1);
690  }
691 
692  return (fd);

760 /* Read a line from ``fp'' into the dynamically allocated ``s'',
761  * increasing ``s'' if necessary. The ending "\n" or "\r\n" is removed.
762  * If a line ends with "\", this char and the linefeed is removed,
763  * and the next line is read too.
764  */
765 char *mutt_read_line (char *s, size_t *size, FILE *fp, int *line, int flags)
766 {
767  size_t offset = 0;
768  char *ch;
769 
770  if (!s)
771  {
772  s = safe_malloc (STRING);
773  *size = STRING;
774  }
775 
776  FOREVER
777  {
778  if (fgets (s + offset, *size - offset, fp) == NULL)
779  {
780  FREE (&s);
781  return NULL;
782  }
783  if ((ch = strchr (s + offset, '\n')) != NULL)
784  {
785  if (line)
786  (*line)++;
787  if (flags & MUTT_EOL)
788  return s;
789  *ch = 0;
790  if (ch > s && *(ch - 1) == '\r')
791  *--ch = 0;
792  if (!(flags & MUTT_CONT) || ch == s || *(ch - 1) != '\\')
793  return s;
794  offset = ch - s - 1;
795  }
796  else
797  {
798  int c;
799  c = getc (fp); /* This is kind of a hack. We want to know if the
800  char at the current point in the input stream is EOF.
801  feof() will only tell us if we've already hit EOF, not
802  if the next character is EOF. So, we need to read in
803  the next character and manually check if it is EOF. */
804  if (c == EOF)
805  {
806  /* The last line of fp isn't \n terminated */
807  if (line)
808  (*line)++;
809  return s;
810  }

965  const char *dir, size_t dirlen, const char *fname, size_t fnamelen)
966 {

972

975  if (dirlen && fnamelen)
