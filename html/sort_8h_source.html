41 /* Sort and sort_aux are shorts, and are a composite of a
42  * constant sort operation number and a set of compounded
43  * bitflags.
44  *
45  * Everything below SORT_MASK is a constant. There's room for
46  * SORT_MASK constant SORT_ values.
47  *
48  * Everything above is a bitflag. It's OK to move SORT_MASK
49  * down by powers of 2 if we need more, so long as we don't
50  * collide with the constants above. (Or we can just expand
51  * sort and sort_aux to uint32_t.)
52  */
53 #define SORT_MASK ((1<<8) - 1)
54 #define SORT_REVERSE (1<<8)
55 #define SORT_LAST (1<<9)
