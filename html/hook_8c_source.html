50 int mutt_parse_hook (BUFFER *buf, BUFFER *s, unsigned long data, BUFFER *err)
51 {
52  HOOK *ptr;
53  BUFFER command, pattern;
54  int rc, not = 0;
55  regex_t *rx = NULL;
56  pattern_t *pat = NULL;
57  char path[_POSIX_PATH_MAX];
58 
59  mutt_buffer_init (&pattern);
60  mutt_buffer_init (&command);
61 
62  if (~data & MUTT_GLOBALHOOK)
63  {
64  if (*s->dptr == '!')
65  {
66  s->dptr++;
67  SKIPWS (s->dptr);
68  not = 1;
69  }
70 
71  mutt_extract_token (&pattern, s, 0);
72 
73  if (!MoreArgs (s))
74  {
75  strfcpy (err->data, _("too few arguments"), err->dsize);
76  goto error;
77  }
78  }
79  else
80  {
81  mutt_extract_token (&pattern, s, 0);
82  }
83 
84  mutt_extract_token (&command, s, (data & (MUTT_FOLDERHOOK | MUTT_SENDHOOK | MUTT_SEND2HOOK | MUTT_ACCOUNTHOOK | MUTT_REPLYHOOK)) ? MUTT_TOKEN_SPACE : 0);
85 
86  if (!command.data)
87  {
88  strfcpy (err->data, _("too few arguments"), err->dsize);
89  goto error;
90  }
91 
92  if (MoreArgs (s))
93  {
94  strfcpy (err->data, _("too many arguments"), err->dsize);
95  goto error;
96  }
97 
98  if (data & (MUTT_FOLDERHOOK | MUTT_MBOXHOOK))
99  {
100  /* Accidentally using the ^ mailbox shortcut in the .muttrc is a
101  * common mistake */
102  if ((*pattern.data == '^') && (! CurrentFolder))
103  {
104  strfcpy (err->data, _("current mailbox shortcut '^' is unset"), err->dsize);
105  goto error;
106  }
107 
108  strfcpy (path, pattern.data, sizeof (path));
109  _mutt_expand_path (path, sizeof (path), 1);
110 
111  /* Check for other mailbox shortcuts that expand to the empty string.
112  * This is likely a mistake too */
113  if (!*path && *pattern.data)
114  {
115  strfcpy (err->data, _("mailbox shortcut expanded to empty regexp"), err->dsize);
116  goto error;
117  }
118 
119  FREE (&pattern.data);
120  memset (&pattern, 0, sizeof (pattern));
121  pattern.data = safe_strdup (path);
122  }
123 #ifdef USE_COMPRESSED
124  else if (data & (MUTT_APPENDHOOK | MUTT_OPENHOOK | MUTT_CLOSEHOOK)) {
125  if (mutt_comp_valid_command (command.data) == 0) {
126  strfcpy (err->data, _("badly formatted command string"), err->dsize);
127  return -1;
128  }
129  }
130 #endif
131  else if (DefaultHook && !(data & (MUTT_CHARSETHOOK | MUTT_ICONVHOOK | MUTT_ACCOUNTHOOK))
132  && (!WithCrypto || !(data & MUTT_CRYPTHOOK))
133  )
134  {
135  char tmp[HUGE_STRING];
136 
137  /* At this stage remain only message-hooks, reply-hooks, send-hooks,
138  * send2-hooks, save-hooks, and fcc-hooks: All those allowing full
139  * patterns. If given a simple regexp, we expand $default_hook.
140  */
141  strfcpy (tmp, pattern.data, sizeof (tmp));
142  mutt_check_simple (tmp, sizeof (tmp), DefaultHook);
143  FREE (&pattern.data);
144  memset (&pattern, 0, sizeof (pattern));
145  pattern.data = safe_strdup (tmp);
146  }
147 
148  if (data & (MUTT_MBOXHOOK | MUTT_SAVEHOOK | MUTT_FCCHOOK))
149  {
150  strfcpy (path, command.data, sizeof (path));
151  mutt_expand_path (path, sizeof (path));
152  FREE (&command.data);
153  memset (&command, 0, sizeof (command));
154  command.data = safe_strdup (path);
155  }
156 
157  /* check to make sure that a matching hook doesn't already exist */
158  for (ptr = Hooks; ptr; ptr = ptr->next)
159  {
160  if (ptr->type == data &&
161  ptr->rx.not == not &&
162  !mutt_strcmp (pattern.data, ptr->rx.pattern))
163  {
164  if (data & (MUTT_FOLDERHOOK | MUTT_SENDHOOK | MUTT_SEND2HOOK | MUTT_MESSAGEHOOK | MUTT_ACCOUNTHOOK | MUTT_REPLYHOOK | MUTT_CRYPTHOOK | MUTT_TIMEOUTHOOK | MUTT_STARTUPHOOK | MUTT_SHUTDOWNHOOK))
165  {
166  /* these hooks allow multiple commands with the same
167  * pattern, so if we've already seen this pattern/command pair, just
168  * ignore it instead of creating a duplicate */
169  if (!mutt_strcmp (ptr->command, command.data))
170  {
171  FREE (&command.data);
172  FREE (&pattern.data);
173  return 0;
174  }
175  }
176  else
177  {
178  /* other hooks only allow one command per pattern, so update the
179  * entry with the new command. this currently does not change the
180  * order of execution of the hooks, which i think is desirable since
181  * a common action to perform is to change the default (.) entry
182  * based upon some other information. */
183  FREE (&ptr->command);
184  ptr->command = command.data;
185  FREE (&pattern.data);
186  return 0;
187  }
188  }
189  if (!ptr->next)
190  break;
191  }
192 
193  if (data & (MUTT_SENDHOOK | MUTT_SEND2HOOK | MUTT_SAVEHOOK | MUTT_FCCHOOK | MUTT_MESSAGEHOOK | MUTT_REPLYHOOK))
194  {
195  if ((pat = mutt_pattern_comp (pattern.data,
196  (data & (MUTT_SENDHOOK | MUTT_SEND2HOOK | MUTT_FCCHOOK)) ? 0 : MUTT_FULL_MSG,
197  err)) == NULL)
198  goto error;
199  }
200  else
201  {
202  /* Hooks not allowing full patterns: Check syntax of regexp */
203  rx = safe_malloc (sizeof (regex_t));
204 #ifdef MUTT_CRYPTHOOK
205  if ((rc = REGCOMP (rx, NONULL(pattern.data), ((data & (MUTT_CRYPTHOOK|MUTT_CHARSETHOOK|MUTT_ICONVHOOK)) ? REG_ICASE : 0))) != 0)
206 #else
207  if ((rc = REGCOMP (rx, NONULL(pattern.data), (data & (MUTT_CHARSETHOOK|MUTT_ICONVHOOK)) ? REG_ICASE : 0)) != 0)
208 #endif /* MUTT_CRYPTHOOK */
209  {
210  regerror (rc, rx, err->data, err->dsize);
211  FREE (&rx);
212  goto error;
213  }
214  }
215 
216  if (ptr)
217  {
218  ptr->next = safe_calloc (1, sizeof (HOOK));
219  ptr = ptr->next;
220  }
221  else
222  Hooks = ptr = safe_calloc (1, sizeof (HOOK));
223  ptr->type = data;
224  ptr->command = command.data;
225  ptr->pattern = pat;
226  ptr->rx.pattern = pattern.data;
227  ptr->rx.rx = rx;
228  ptr->rx.not = not;
229  return 0;
230 
231 error:
232  FREE (&pattern.data);
233  FREE (&command.data);
234  return (-1);
235 }

249 /* Deletes all hooks of type ``type'', or all defined hooks if ``type'' is 0 */
250 static void delete_hooks (int type)
251 {
252  HOOK *h;
253  HOOK *prev;
254 
255  while (h = Hooks, h && (type == 0 || type == h->type))
256  {
257  Hooks = h->next;
258  delete_hook (h);
259  }
260 
261  prev = h; /* Unused assignment to avoid compiler warnings */
262 
263  while (h)
264  {
265  if (type == h->type)
266  {
267  prev->next = h->next;
268  delete_hook (h);
269  }
270  else
271  prev = h;
272  h = prev->next;
273  }
274 }

276 int mutt_parse_unhook (BUFFER *buf, BUFFER *s, unsigned long data, BUFFER *err)
277 {
278  while (MoreArgs (s))
279  {
280  mutt_extract_token (buf, s, 0);
281  if (mutt_strcmp ("*", buf->data) == 0)
282  {
283  if (current_hook_type)
284  {
285  snprintf (err->data, err->dsize,
286  _("unhook: Can't do unhook * from within a hook."));
287  return -1;
288  }
289  delete_hooks (0);
290  }
291  else
292  {
293  int type = mutt_get_hook_type (buf->data);
294 
295  if (!type)
296  {
297  snprintf (err->data, err->dsize,
298  _("unhook: unknown hook type: %s"), buf->data);
299  return (-1);
300  }
301  if (current_hook_type == type)
302  {
303  snprintf (err->data, err->dsize,
304  _("unhook: Can't delete a %s from within a %s."),
305  buf->data, buf->data);
306  return -1;
307  }
308  delete_hooks (type);
309  }
310  }
311  return 0;
312 }

314 void mutt_folder_hook (char *path)
315 {
316  HOOK *tmp = Hooks;
317  BUFFER err, token;
318 
319  current_hook_type = MUTT_FOLDERHOOK;
320 
321  mutt_buffer_init (&err);
322  err.dsize = STRING;
323  err.data = safe_malloc (err.dsize);
324  mutt_buffer_init (&token);
325  for (; tmp; tmp = tmp->next)
326  {
327  if(!tmp->command)
328  continue;
329 
330  if (tmp->type & MUTT_FOLDERHOOK)
331  {
332  if ((regexec (tmp->rx.rx, path, 0, NULL, 0) == 0) ^ tmp->rx.not)
333  {
334  if (mutt_parse_rc_line (tmp->command, &token, &err) == -1)
335  {
336  mutt_error ("%s", err.data);
337  FREE (&token.data);
338  mutt_sleep (1); /* pause a moment to let the user see the error */
339  current_hook_type = 0;
340  FREE (&err.data);
341 
342  return;
343  }
344  }
345  }
346  }
347  FREE (&token.data);

350  current_hook_type = 0;

423 void mutt_default_save (char *path, size_t pathlen, HEADER *hdr)
424 {
425  *path = 0;
426  if (mutt_addr_hook (path, pathlen, MUTT_SAVEHOOK, Context, hdr) != 0)
427  {
428  char tmp[_POSIX_PATH_MAX];
429  ADDRESS *adr;
430  ENVELOPE *env = hdr->env;
431  int fromMe = mutt_addr_is_user (env->from);
432 
433  if (!fromMe && env->reply_to && env->reply_to->mailbox)
434  adr = env->reply_to;
435  else if (!fromMe && env->from && env->from->mailbox)
436  adr = env->from;
437  else if (env->to && env->to->mailbox)
438  adr = env->to;
439  else if (env->cc && env->cc->mailbox)
440  adr = env->cc;
441  else
442  adr = NULL;
443  if (adr)
444  {
445  mutt_safe_path (tmp, sizeof (tmp), adr);
446  snprintf (path, pathlen, "=%s", tmp);
447  }
448  }
