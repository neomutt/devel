144 static void redraw_crypt_lines (HEADER *msg)
145 {
146  mutt_window_mvprintw (MuttIndexWindow, HDR_CRYPT, 0, TITLE_FMT, "Security: ");
147 
148  if ((WithCrypto & (APPLICATION_PGP | APPLICATION_SMIME)) == 0)
149  {
150  addstr(_("Not supported"));
151  return;
152  }
153 
154  if ((msg->security & (ENCRYPT | SIGN)) == (ENCRYPT | SIGN))
155  addstr (_("Sign, Encrypt"));
156  else if (msg->security & ENCRYPT)
157  addstr (_("Encrypt"));
158  else if (msg->security & SIGN)
159  addstr (_("Sign"));
160  else
161  /* L10N: This refers to the encryption of the email, e.g. "Security: None" */
162  addstr (_("None"));
163 
164  if ((msg->security & (ENCRYPT | SIGN)))
165  {
166  if ((WithCrypto & APPLICATION_PGP) && (msg->security & APPLICATION_PGP))
167  {
168  if ((msg->security & INLINE))
169  addstr (_(" (inline PGP)"));
170  else
171  addstr (_(" (PGP/MIME)"));
172  }
173  else if ((WithCrypto & APPLICATION_SMIME) &&
174  (msg->security & APPLICATION_SMIME))
175  addstr (_(" (S/MIME)"));
176  }
177 
178  if (option (OPTCRYPTOPPORTUNISTICENCRYPT) && (msg->security & OPPENCRYPT))
179  addstr (_(" (OppEnc mode)"));
180 
181  mutt_window_clrtoeol (MuttIndexWindow);
182  mutt_window_move (MuttIndexWindow, HDR_CRYPTINFO, 0);
183  mutt_window_clrtoeol (MuttIndexWindow);
184 
185  if ((WithCrypto & APPLICATION_PGP)
186  && (msg->security & APPLICATION_PGP) && (msg->security & SIGN))
187  printw (TITLE_FMT "%s", _("sign as: "), PgpSignAs ? PgpSignAs : _("<default>"));
188 
189  if ((WithCrypto & APPLICATION_SMIME)
190  && (msg->security & APPLICATION_SMIME) && (msg->security & SIGN)) {
191  printw (TITLE_FMT "%s", _("sign as: "), SmimeDefaultKey ? SmimeDefaultKey : _("<default>"));
192  }
193 
194  if ((WithCrypto & APPLICATION_SMIME)
195  && (msg->security & APPLICATION_SMIME)
196  && (msg->security & ENCRYPT)
197  && SmimeCryptAlg
198  && *SmimeCryptAlg) {
199  mutt_window_mvprintw (MuttIndexWindow, HDR_CRYPTINFO, 40, "%s%s", _("Encrypt with: "),
200  NONULL(SmimeCryptAlg));
201  }
202 }

240 static int
241 check_attachments(ATTACHPTR **idx, short idxlen)
242 {
243  int i, r;
244  struct stat st;
245  char pretty[_POSIX_PATH_MAX], msg[_POSIX_PATH_MAX + SHORT_STRING];
246 
247  for (i = 0; i < idxlen; i++)
248  {
249  strfcpy(pretty, idx[i]->content->filename, sizeof(pretty));
250  if(stat(idx[i]->content->filename, &st) != 0)
251  {
252  mutt_pretty_mailbox(pretty, sizeof (pretty));
253  mutt_error(_("%s [#%d] no longer exists!"),
254  pretty, i+1);
255  return -1;
256  }
257 
258  if(idx[i]->content->stamp < st.st_mtime)
259  {
260  mutt_pretty_mailbox(pretty, sizeof (pretty));
261  snprintf(msg, sizeof(msg), _("%s [#%d] modified. Update encoding?"),
262  pretty, i+1);
263 
264  if((r = mutt_yesorno(msg, MUTT_YES)) == MUTT_YES)
265  mutt_update_encoding(idx[i]->content);
266  else if(r == MUTT_ABORT)
267  return -1;
268  }
269  }
270 
271  return 0;
272 }

330 {

332  char *err = NULL;
333

335  rfc822_write_address (buf, sizeof (buf), *addr, 0);

337  {
338  rfc822_free_address (addr);

342 
343  if (option (OPTNEEDREDRAW))
344  {
345  unset_option (OPTNEEDREDRAW);
346  return (REDRAW_FULL);

348

350  {
351  mutt_error (_("Warning: '%s' is a bad IDN."), err);
352  mutt_refresh();
353  FREE (&err);
354  }
355 
356  /* redraw the expanded list so the user can see the result */
357  buf[0] = 0;
358  rfc822_write_address (buf, sizeof (buf), *addr, 1);
359  mutt_window_move (MuttIndexWindow, line, HDR_XOFFSET);
360  mutt_paddstr (W, buf);

362  return 0;

365 static int delete_attachment (MUTTMENU *menu, short *idxlen, int x)
366 {
367  ATTACHPTR **idx = (ATTACHPTR **) menu->data;
368  int y;
369 
370  menu->redraw = REDRAW_INDEX | REDRAW_STATUS;
371 
372  if (x == 0 && menu->max == 1)
373  {
374  mutt_error _("You may not delete the only attachment.");
375  idx[x]->content->tagged = 0;
376  return (-1);
377  }
378 
379  for (y = 0; y < *idxlen; y++)
380  {
381  if (idx[y]->content->next == idx[x]->content)
382  {
383  idx[y]->content->next = idx[x]->content->next;
384  break;
385  }
386  }
387 
388  idx[x]->content->next = NULL;
389  idx[x]->content->parts = NULL;
390  mutt_free_body (&(idx[x]->content));
391  FREE (&idx[x]->tree);
392  FREE (&idx[x]);
393  for (; x < *idxlen - 1; x++)
394  idx[x] = idx[x+1];
395  idx[*idxlen - 1] = NULL;
396  menu->max = --(*idxlen);
397 
398  return (0);

414 /*
415  * cum_attachs_size: Cumulative Attachments Size
416  *
417  * Returns the total number of bytes used by the attachments in the
418  * attachment list _after_ content-transfer-encodings have been
419  * applied.
420  *
421  */
422 
423 static unsigned long cum_attachs_size (MUTTMENU *menu)
424 {
425  size_t s;
426  unsigned short i;
427  ATTACHPTR **idx = menu->data;
428  CONTENT *info;
429  BODY *b;
430 
431  for (i = 0, s = 0; i < menu->max; i++)
432  {
433  b = idx[i]->content;
434 
435  if (!b->content)
436  b->content = mutt_get_content_info (b->filename, b);
437 
438  if ((info = b->content))
439  {
440  switch (b->encoding)
441  {
442  case ENCQUOTEDPRINTABLE:
443  s += 3 * (info->lobin + info->hibin) + info->ascii + info->crlf;
444  break;
445  case ENCBASE64:
446  s += (4 * (info->lobin + info->hibin + info->ascii + info->crlf)) / 3;
447  break;
448  default:
449  s += info->lobin + info->hibin + info->ascii + info->crlf;
450  break;
451  }
452  }
453  }
454 
455  return s;
456 }

462 /*
463  * compose_format_str()
464  *
465  * %a = total number of attachments
466  * %h = hostname [option]
467  * %l = approx. length of current message (in bytes)
468  * %v = Mutt version
469  *
470  * This function is similar to status_format_str(). Look at that function for
471  * help when modifying this function.
472  */
473 
474 static const char *
475 compose_format_str (char *buf, size_t buflen, size_t col, int cols, char op, const char *src,
476  const char *prefix, const char *ifstring,
477  const char *elsestring,
478  unsigned long data, format_flag flags)
479 {
480  char fmt[SHORT_STRING], tmp[SHORT_STRING];
481  int optional = (flags & MUTT_FORMAT_OPTIONAL);
482  MUTTMENU *menu = (MUTTMENU *) data;
483 
484  *buf = 0;
485  switch (op)
486  {
487  case 'a': /* total number of attachments */
488  snprintf (fmt, sizeof (fmt), "%%%sd", prefix);
489  snprintf (buf, buflen, fmt, menu->max);
490  break;
491 
492  case 'h': /* hostname */
493  snprintf (fmt, sizeof (fmt), "%%%ss", prefix);
494  snprintf (buf, buflen, fmt, NONULL(Hostname));
495  break;
496 
497  case 'l': /* approx length of current message in bytes */
498  snprintf (fmt, sizeof (fmt), "%%%ss", prefix);
499  mutt_pretty_size (tmp, sizeof (tmp), menu ? cum_attachs_size(menu) : 0);
500  snprintf (buf, buflen, fmt, tmp);
501  break;
502 
503  case 'v':
504  snprintf (fmt, sizeof (fmt), "NeoMutt %%s%%s");
505  snprintf (buf, buflen, fmt, PACKAGE_VERSION, GitVer);
506  break;
507 
508  case 0:
509  *buf = 0;
510  return (src);
511 
512  default:
513  snprintf (buf, buflen, "%%%s%c", prefix, op);
514  break;
515  }
516 
517  if (optional)
518  compose_status_line (buf, buflen, col, cols, menu, ifstring);
519  else if (flags & MUTT_FORMAT_OPTIONAL)
520  compose_status_line (buf, buflen, col, cols, menu, elsestring);
521 
522  return (src);
523 }

566 #endif

616  break;

620  {

623  }

627 #ifdef USE_NNTP
628  if (news)

630 #endif
