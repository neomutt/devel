50 /* Find NNTP_DATA for given newsgroup or add it */
51 static NNTP_DATA *nntp_data_find (NNTP_SERVER *nserv, const char *group)
52 {
53  NNTP_DATA *nntp_data = hash_find (nserv->groups_hash, group);
54 
55  if (!nntp_data)
56  {
57  int len = strlen (group) + 1;
58  /* create NNTP_DATA structure and add it to hash */
59  nntp_data = safe_calloc (1, sizeof (NNTP_DATA) + len);
60  nntp_data->group = (char *)nntp_data + sizeof (NNTP_DATA);
61  strfcpy (nntp_data->group, group, len);
62  nntp_data->nserv = nserv;
63  nntp_data->deleted = 1;
64  if (nserv->groups_hash->nelem < nserv->groups_hash->curnelem * 2)
65  nserv->groups_hash = hash_resize (nserv->groups_hash,
66  nserv->groups_hash->nelem * 2, 0);
67  hash_insert (nserv->groups_hash, nntp_data->group, nntp_data, 0);
68 
69  /* add NNTP_DATA to list */
70  if (nserv->groups_num >= nserv->groups_max)
71  {
72  nserv->groups_max *= 2;
73  safe_realloc (&nserv->groups_list,
74  nserv->groups_max * sizeof (nntp_data));
75  }
76  nserv->groups_list[nserv->groups_num++] = nntp_data;
77  }
78  return nntp_data;
79 }

121 /* Parse .newsrc file:
122  * 0 - not changed
123  * 1 - parsed
124  * -1 - error */
125 int nntp_newsrc_parse (NNTP_SERVER *nserv)
126 {
127  unsigned int i;
128  char *line;
129  struct stat sb;
130 
131  if (nserv->newsrc_fp)
132  {
133  /* if we already have a handle, close it and reopen */
134  safe_fclose (&nserv->newsrc_fp);
135  }
136  else
137  {
138  /* if file doesn't exist, create it */
139  nserv->newsrc_fp = safe_fopen (nserv->newsrc_file, "a");
140  safe_fclose (&nserv->newsrc_fp);
141  }
142 
143  /* open .newsrc */
144  nserv->newsrc_fp = safe_fopen (nserv->newsrc_file, "r");
145  if (!nserv->newsrc_fp)
146  {
147  mutt_perror (nserv->newsrc_file);
148  mutt_sleep (2);
149  return -1;
150  }
151 
152  /* lock it */
153  dprint (1, (debugfile, "Locking %s\n", nserv->newsrc_file));
154  if (mx_lock_file (nserv->newsrc_file, fileno (nserv->newsrc_fp), 0, 0, 1))
155  {
156  safe_fclose (&nserv->newsrc_fp);
157  return -1;
158  }
159 
160  if (stat (nserv->newsrc_file, &sb))
161  {
162  mutt_perror (nserv->newsrc_file);
163  nntp_newsrc_close (nserv);
164  mutt_sleep (2);
165  return -1;
166  }
167 
168  if (nserv->size == sb.st_size && nserv->mtime == sb.st_mtime)
169  return 0;
170 
171  nserv->size = sb.st_size;
172  nserv->mtime = sb.st_mtime;
173  nserv->newsrc_modified = 1;
174  dprint (1, (debugfile, "Parsing %s\n", nserv->newsrc_file));
175 
176  /* .newsrc has been externally modified or hasn't been loaded yet */
177  for (i = 0; i < nserv->groups_num; i++)
178  {
179  NNTP_DATA *nntp_data = nserv->groups_list[i];
180 
181  if (!nntp_data)
182  continue;
183 
184  nntp_data->subscribed = 0;
185  nntp_data->newsrc_len = 0;
186  FREE (&nntp_data->newsrc_ent);
187  }
188 
189  line = safe_malloc (sb.st_size + 1);
190  while (sb.st_size && fgets (line, sb.st_size + 1, nserv->newsrc_fp))
191  {
192  char *b, *h, *p;
193  unsigned int subs = 0, i = 1;
194  NNTP_DATA *nntp_data;
195 
196  /* find end of newsgroup name */
197  p = strpbrk (line, ":!");
198  if (!p)
199  continue;
200 
201  /* ":" - subscribed, "!" - unsubscribed */
202  if (*p == ':')
203  subs++;
204  *p++ = '\0';
205 
206  /* get newsgroup data */
207  nntp_data = nntp_data_find (nserv, line);
208  FREE (&nntp_data->newsrc_ent);
209 
210  /* count number of entries */
211  b = p;
212  while (*b)
213  if (*b++ == ',')
214  i++;
215  nntp_data->newsrc_ent = safe_calloc (i, sizeof (NEWSRC_ENTRY));
216  nntp_data->subscribed = subs;
217 
218  /* parse entries */
219  i = 0;
220  while (p)
221  {
222  b = p;
223 
224  /* find end of entry */
225  p = strchr (p, ',');
226  if (p)
227  *p++ = '\0';
228 
229  /* first-last or single number */
230  h = strchr (b, '-');
231  if (h)
232  *h++ = '\0';
233  else
234  h = b;
235 
236  if (sscanf (b, ANUM, &nntp_data->newsrc_ent[i].first) == 1 &&
237  sscanf (h, ANUM, &nntp_data->newsrc_ent[i].last) == 1)
238  i++;
239  }
240  if (i == 0)
241  {
242  nntp_data->newsrc_ent[i].first = 1;
243  nntp_data->newsrc_ent[i].last = 0;
244  i++;
245  }
246  if (nntp_data->lastMessage == 0)
247  nntp_data->lastMessage = nntp_data->newsrc_ent[i - 1].last;
248  nntp_data->newsrc_len = i;
249  safe_realloc (&nntp_data->newsrc_ent, i * sizeof (NEWSRC_ENTRY));
250  nntp_group_unread_stat (nntp_data);
251  dprint (2, (debugfile, "nntp_newsrc_parse: %s\n", nntp_data->group));
252  }
253  FREE (&line);
254  return 1;
255 }

257 /* Generate array of .newsrc entries */
258 void nntp_newsrc_gen_entries (CONTEXT *ctx)
259 {
260  NNTP_DATA *nntp_data = ctx->data;
261  anum_t last = 0, first = 1;
262  int series, i;
263  int save_sort = SORT_ORDER;
264  unsigned int entries;
265 
266  if (Sort != SORT_ORDER)
267  {
268  save_sort = Sort;
269  Sort = SORT_ORDER;
270  mutt_sort_headers (ctx, 0);
271  }
272 
273  entries = nntp_data->newsrc_len;
274  if (!entries)
275  {
276  entries = 5;
277  nntp_data->newsrc_ent = safe_calloc (entries, sizeof (NEWSRC_ENTRY));
278  }
279 
280  /* Set up to fake initial sequence from 1 to the article before the
281  * first article in our list */
282  nntp_data->newsrc_len = 0;
283  series = 1;
284  for (i = 0; i < ctx->msgcount; i++)
285  {
286  /* search for first unread */
287  if (series)
288  {
289  /* We don't actually check sequential order, since we mark
290  * "missing" entries as read/deleted */
291  last = NHDR (ctx->hdrs[i])->article_num;
292  if (last >= nntp_data->firstMessage && !ctx->hdrs[i]->deleted &&
293  !ctx->hdrs[i]->read)
294  {
295  if (nntp_data->newsrc_len >= entries)
296  {
297  entries *= 2;
298  safe_realloc (&nntp_data->newsrc_ent, entries * sizeof (NEWSRC_ENTRY));
299  }
300  nntp_data->newsrc_ent[nntp_data->newsrc_len].first = first;
301  nntp_data->newsrc_ent[nntp_data->newsrc_len].last = last - 1;
302  nntp_data->newsrc_len++;
303  series = 0;
304  }
305  }
306 
307  /* search for first read */
308  else
309  {
310  if (ctx->hdrs[i]->deleted || ctx->hdrs[i]->read)
311  {
312  first = last + 1;
313  series = 1;
314  }
315  last = NHDR (ctx->hdrs[i])->article_num;
316  }
317  }
318 
319  if (series && first <= nntp_data->lastLoaded)
320  {
321  if (nntp_data->newsrc_len >= entries)
322  {
323  entries++;
324  safe_realloc (&nntp_data->newsrc_ent, entries * sizeof (NEWSRC_ENTRY));
325  }
326  nntp_data->newsrc_ent[nntp_data->newsrc_len].first = first;
327  nntp_data->newsrc_ent[nntp_data->newsrc_len].last = nntp_data->lastLoaded;
328  nntp_data->newsrc_len++;
329  }
330  safe_realloc (&nntp_data->newsrc_ent,
331  nntp_data->newsrc_len * sizeof (NEWSRC_ENTRY));
332 
333  if (save_sort != Sort)
334  {
335  Sort = save_sort;
336  mutt_sort_headers (ctx, 0);
337  }
338 }

340 /* Update file with new contents */
341 static int update_file (char *filename, char *buf)
342 {
343  FILE *fp;
344  char tmpfile[_POSIX_PATH_MAX];
345  int rc = -1;
346 
347  while (1)
348  {
349  snprintf (tmpfile, sizeof (tmpfile), "%s.tmp", filename);
350  fp = safe_fopen (tmpfile, "w");
351  if (!fp)
352  {
353  mutt_perror (tmpfile);
354  *tmpfile = '\0';
355  break;
356  }
357  if (fputs (buf, fp) == EOF)
358  {
359  mutt_perror (tmpfile);
360  break;
361  }
362  if (safe_fclose (&fp) == EOF)
363  {
364  mutt_perror (tmpfile);
365  fp = NULL;
366  break;
367  }
368  fp = NULL;
369  if (rename (tmpfile, filename) < 0)
370  {
371  mutt_perror (filename);
372  break;
373  }
374  *tmpfile = '\0';
375  rc = 0;
376  break;
377  }
378  if (fp)
379  safe_fclose (&fp);
380  if (*tmpfile)
381  unlink (tmpfile);
382  if (rc)
383  mutt_sleep (2);
384  return rc;
385 }

387 /* Update .newsrc file */
388 int nntp_newsrc_update (NNTP_SERVER *nserv)
389 {
390  char *buf;
391  size_t buflen, off;
392  unsigned int i;
393  int rc = -1;
394 
395  if (!nserv)
396  return -1;
397 
398  buflen = 10 * LONG_STRING;
399  buf = safe_calloc (1, buflen);
400  off = 0;
401 
402  /* we will generate full newsrc here */
403  for (i = 0; i < nserv->groups_num; i++)
404  {
405  NNTP_DATA *nntp_data = nserv->groups_list[i];
406  unsigned int n;
407 
408  if (!nntp_data || !nntp_data->newsrc_ent)
409  continue;
410 
411  /* write newsgroup name */
412  if (off + strlen (nntp_data->group) + 3 > buflen)
413  {
414  buflen *= 2;
415  safe_realloc (&buf, buflen);
416  }
417  snprintf (buf + off, buflen - off, "%s%c ", nntp_data->group,
418  nntp_data->subscribed ? ':' : '!');
419  off += strlen (buf + off);
420 
421  /* write entries */
422  for (n = 0; n < nntp_data->newsrc_len; n++)
423  {
424  if (off + LONG_STRING > buflen)
425  {
426  buflen *= 2;
427  safe_realloc (&buf, buflen);
428  }
429  if (n)
430  buf[off++] = ',';
431  if (nntp_data->newsrc_ent[n].first == nntp_data->newsrc_ent[n].last)
432  snprintf (buf + off, buflen - off, "%d", nntp_data->newsrc_ent[n].first);
433  else if (nntp_data->newsrc_ent[n].first < nntp_data->newsrc_ent[n].last)
434  snprintf (buf + off, buflen - off, "%d-%d",
435  nntp_data->newsrc_ent[n].first, nntp_data->newsrc_ent[n].last);
436  off += strlen (buf + off);
437  }
438  buf[off++] = '\n';
439  }
440  buf[off] = '\0';
441 
442  /* newrc being fully rewritten */
443  dprint (1, (debugfile, "Updating %s\n", nserv->newsrc_file));
444  if (nserv->newsrc_file && update_file (nserv->newsrc_file, buf) == 0)
445  {
446  struct stat sb;
447 
448  rc = stat (nserv->newsrc_file, &sb);
449  if (rc == 0)
450  {
451  nserv->size = sb.st_size;
452  nserv->mtime = sb.st_mtime;
453  }
454  else
455  {
456  mutt_perror (nserv->newsrc_file);
457  mutt_sleep (2);
458  }
459  }
460  FREE (&buf);
461  return rc;
462 }

502 /* Parse newsgroup */
503 int nntp_add_group (char *line, void *data)
504 {
505  NNTP_SERVER *nserv = data;
506  NNTP_DATA *nntp_data;
507  char group[LONG_STRING];
508  char desc[HUGE_STRING] = "";
509  char mod;
510  anum_t first, last;
511 
512  if (!nserv || !line)
513  return 0;
514 
515  if (sscanf (line, "%s " ANUM " " ANUM " %c %[^\n]", group,
516  &last, &first, &mod, desc) < 4)
517  return 0;
518 
519  nntp_data = nntp_data_find (nserv, group);
520  nntp_data->deleted = 0;
521  nntp_data->firstMessage = first;
522  nntp_data->lastMessage = last;
523  nntp_data->allowed = mod == 'y' || mod == 'm' ? 1 : 0;
524  mutt_str_replace (&nntp_data->desc, desc);
525  if (nntp_data->newsrc_ent || nntp_data->lastCached)
526  nntp_group_unread_stat (nntp_data);
527  else if (nntp_data->lastMessage &&
528  nntp_data->firstMessage <= nntp_data->lastMessage)
529  nntp_data->unread = nntp_data->lastMessage - nntp_data->firstMessage + 1;
530  else
531  nntp_data->unread = 0;
532  return 0;
533 }

635 /* Remove stale cached headers */
636 void nntp_hcache_update (NNTP_DATA *nntp_data, header_cache_t *hc)
637 {
638  char buf[16];
639  int old = 0;
640  void *hdata;
641  anum_t first, last, current;
642 
643  if (!hc)
644  return;
645 
646  /* fetch previous values of first and last */
647  hdata = mutt_hcache_fetch_raw (hc, "index", 5);
648  if (hdata)
649  {
650  dprint (2, (debugfile,
651  "nntp_hcache_update: mutt_hcache_fetch index: %s\n", hdata));
652  if (sscanf (hdata, ANUM " " ANUM, &first, &last) == 2)
653  {
654  old = 1;
655  nntp_data->lastCached = last;
656 
657  /* clean removed headers from cache */
658  for (current = first; current <= last; current++)
659  {
660  if (current >= nntp_data->firstMessage &&
661  current <= nntp_data->lastMessage)
662  continue;
663 
664  snprintf (buf, sizeof (buf), "%d", current);
